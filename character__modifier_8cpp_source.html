166 float Character::manipulator_score( const std::map<bodypart_str_id, bodypart> &body,
167  body_part_type::type type, int override_encumb, int override_wounds ) const
168 {
169  std::map<body_part_type::type, std::vector<std::pair<bodypart, float>>> bodypart_groups;
170  std::vector<float> score_groups;
171  const bool required_type = type != body_part_type::type::num_types;
172  const bool local_effect = has_flag( flag_EFFECT_LIMB_SCORE_MOD_LOCAL );
173  for( const std::pair<const bodypart_str_id, bodypart> &id : body ) {
174  if( required_type ) {
175  for( const auto &bp_type : id.first->limbtypes ) {
176  if( bp_type.first == type ) {
177  bodypart_groups[ bp_type.first ].emplace_back( id.second, bp_type.second );
178  }
179  }
180  } else if( id.first->primary_limb_type() != body_part_type::type::num_types ) {
181  bodypart_groups[ id.first->primary_limb_type() ].emplace_back( id.second,
182  id.first->limbtypes.at( id.first->primary_limb_type() ) );
183  }
184  }
185  for( auto &part : bodypart_groups ) {
186  float total = 0.0f;
187  std::sort( part.second.begin(), part.second.end(),
188  []( const std::pair<bodypart, float> &a, const std::pair<bodypart, float> &b ) {
189  return a.first.get_limb_score_max( limb_score_manip ) * a.second <
190  b.first.get_limb_score_max( limb_score_manip ) * b.second;
191  } );
192  for( const std::pair<bodypart, float> &id : part.second ) {
193  float local_mul = 1.0f;
194  // Calculate local effect modifiers
195  if( local_effect ) {
196  for( const effect &local : get_effects_from_bp( id.first.get_id() ) ) {
197  if( local.has_flag( flag_EFFECT_LIMB_SCORE_MOD_LOCAL ) ) {
198  float temp = local.get_limb_score_mod( limb_score_manip, resists_effect( local ) );
199  local_mul *= temp;
200  if( temp != 1.0f ) {
201  add_msg_debug( debugmode::DF_CHARACTER,
202  "Local limb score modifier %s for manipulation score on BP %s found, effect multiplier %.1f",
203  local.disp_name(), id.first.get_id()->name, local_mul );
204  }
205  }
206  }
207  }
208  total = std::min( total + id.first.get_limb_score( limb_score_manip, -1, override_encumb,
209  override_wounds ) * id.second * local_mul,
210  id.first.get_limb_score_max( limb_score_manip ) * local_mul * id.second );
211  }
212  add_msg_debug( debugmode::DF_CHARACTER,
213  "Manipulation score of bodypart group %s %.1f",
214  io::enum_to_string<body_part_type::type>( part.first ), total );
215  score_groups.emplace_back( total );
216  }
217  const auto score_groups_max = std::max_element( score_groups.begin(), score_groups.end() );
218 
219  if( score_groups_max == score_groups.end() ) {
220  return 0.0f;
221  } else {
222  return std::max( 0.0f, *score_groups_max );
223  }
224 }

226 float Character::get_limb_score( const limb_score_id &score, const body_part_type::type &bp,
227  int override_encumb, int override_wounds ) const
228 {
229  int skill = -1;
230  float effect_mul = 1.0f;
231  // Check for any global limb score modifier
232  for( const effect &eff : get_effects_with_flag( flag_EFFECT_LIMB_SCORE_MOD ) ) {
233  effect_mul *= eff.get_limb_score_mod( score, resists_effect( eff ) );
234  }
235  // manipulator/swim scores are treated a little special for now
236  if( score == limb_score_manip ) {
237  return manipulator_score( body, bp, override_encumb, override_wounds ) * effect_mul;
238  } else if( score == limb_score_swim ) {
239  skill = round( get_skill_level( skill_swimming ) );
240  }
241  float total = 0.0f;
242  // Avoid call has_flag() in a loop to improve performance
243  bool cache_flag_EFFECT_LIMB_SCORE_MOD_LOCAL = has_flag( flag_EFFECT_LIMB_SCORE_MOD_LOCAL );
244  for( const std::pair<const bodypart_str_id, bodypart> &id : body ) {
245  float mod = 0.0f;
246  if( bp == body_part_type::type::num_types ) {
247  mod = id.second.get_limb_score( score, skill, override_encumb, override_wounds );
248  } else if( id.first->has_type( bp ) ) {
249  mod = id.second.get_limb_score( score, skill, override_encumb,
250  override_wounds ) * id.first->limbtypes.at( bp );
251  }
252  if( cache_flag_EFFECT_LIMB_SCORE_MOD_LOCAL ) {
253  for( const effect &local : get_effects_from_bp( id.first ) ) {
254  float local_mul = 1.0f;
255  // Second filter to only apply the local effects at this step (non-local modifiers are already calulated)
256  if( local.has_flag( flag_EFFECT_LIMB_SCORE_MOD_LOCAL ) ) {
257  local_mul = local.get_limb_score_mod( score, resists_effect( local ) );
258  mod *= local_mul;
259  if( local_mul != 1.0f ) {
260  add_msg_debug( debugmode::DF_CHARACTER,
261  "Local limb score modifier %s for limb score %s on BP %s found, effect multiplier %.1f, score contribution after modifier %.1f",
262  local.disp_name(),
263  score.c_str(), id.first.c_str(), local_mul, mod );
264  }
265  }
266  }
267  }
268  total += mod;
269  }
270  return std::max( 0.0f, total * effect_mul );
271 }

332 // Generally for cost mods: mod = min( max_val, ( nominator / limb_score ) - subtractor )
333 // for melee attack roll: mod = max( min_val, limb_score )
334 // for straight limb_score: nominator == 0, subtractor == 0, max_val == 0, min_val == 0
335 float character_modifier::modifier( const Character &c, const skill_id &skill ) const
336 {
337  // use builtin to calculate modifier
338  if( !builtin.empty() ) {
339  return call_builtin( builtin, c, skill );
340  }
341 
342  float score = 0.0f;
343  bool sc_assigned = false;
344  for( const auto &sc : limbscores ) {
345  float mod_sc = c.get_limb_score( sc.first, limbtype, override_encumb, override_wounds );
346  mod_sc = limbscore_modop == MULT ? pow( mod_sc, sc.second ) : mod_sc * sc.second;
347  if( !sc_assigned ) {
348  score = mod_sc;
349  sc_assigned = true;
350  continue;
351  }
352  switch( limbscore_modop ) {
353  case ADD:
354  score += mod_sc;
355  break;
356  case NONE:
357  score = mod_sc;
358  break;
359  case MULT:
360  default:
361  score *= mod_sc;
362  }
363  }
364 
365  // score == 0
366  if( score < std::numeric_limits<float>::epsilon() ) {
367  return min_val > std::numeric_limits<float>::epsilon() ? min_val :
368  max_val > std::numeric_limits<float>::epsilon() ? max_val : 0.0f;
369  }
370  if( nominator > std::numeric_limits<float>::epsilon() ) {
371  score = ( nominator / denominator ) / score;
372  } else {
373  score /= denominator;
374  }
375  if( subtractor > std::numeric_limits<float>::epsilon() ) {
376  score -= subtractor;
377  }
378  if( max_val > std::numeric_limits<float>::epsilon() ) {
379  score = std::min( max_val, score );
380  }
381  if( min_val > std::numeric_limits<float>::epsilon() ) {
382  score = std::max( min_val, score );
383  }
384  return score;
385 }
