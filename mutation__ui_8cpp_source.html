561  if( ret.evt.type == input_event_t::keyboard_char && !ret.evt.sequence.empty() ) {
562  const int newch = ret.evt.get_first_input();
563  if( mutation_chars.valid( newch ) ) {
564  const trait_id other_mut_id = trait_by_invlet( newch );
565  if( !other_mut_id.is_null() ) {
566  std::swap( my_mutations[mut_id].key, my_mutations[other_mut_id].key );
567  } else {
568  my_mutations[mut_id].key = newch;
569  }
570  pop_exit = true;
571  pop_handled = true;
572  } else if( newch == ' ' ) {
573  my_mutations[mut_id].key = newch;
574  pop_exit = true;
575  pop_handled = true;
576  }
577  }
578  if( !pop_handled ) {
579  if( ret.action == "QUIT" ) {
580  pop_exit = true;
581  } else if( ret.action != "HELP_KEYBINDINGS" &&
582  ret.evt.type == input_event_t::keyboard_char ) {
583  popup( _( "Invalid mutation letter. Only those characters are valid:\n\n%s" ),
584  mutation_chars.get_allowed_chars() );
585  }
586  }
587  }
588 
589  menu_mode = mutation_menu_mode::activating;
590  examine_id = std::nullopt;
591  // TODO: show a message like when reassigning a key to an item?
592  break;
593  }
594  case mutation_menu_mode::activating: {
595  if( mut_data.activated ) {
596  if( my_mutations[mut_id].powered ) {
597  add_msg_if_player( m_neutral, _( "You stop using your %s." ), mutation_name( mut_data.id ) );
598  // Reset menu in advance
599  ui.reset();
600  deactivate_mutation( mut_id );
601  // Action done, leave screen
602  exit = true;
603  } else if( ( !mut_data.hunger || get_kcal_percent() >= 0.8f ) &&
604  ( !mut_data.thirst || get_thirst() <= 400 ) &&
605  ( !mut_data.sleepiness || get_sleepiness() <= 400 ) ) {
606  add_msg_if_player( m_neutral, _( "You activate your %s." ), mutation_name( mut_data.id ) );
607  // Reset menu in advance
608  ui.reset();
609  activate_mutation( mut_id );
610  // Action done, leave screen
611  exit = true;
612  } else {
613  popup( _( "You don't have enough in you to activate your %s!" ), mutation_name( mut_data.id ) );
614  }
615  } else {
616  popup( _( "You cannot activate %1$s! To read a description of "
617  "%1$s, press '!', then '%2$c'." ),
618  mutation_name( mut_data.id ), my_mutations[mut_id].key );
619  }
620  break;
621  }
622  case mutation_menu_mode::examining:
623  // Describing mutations, not activating them!
624  examine_id = mut_id;
625  break;
626  case mutation_menu_mode::hiding:
627  my_mutations[mut_id].show_sprite = !my_mutations[mut_id].show_sprite;
628  break;
629  }
630  }
631  } else if( action == "REASSIGN" ) {
632  menu_mode = mutation_menu_mode::reassigning;
633  examine_id = std::nullopt;
634  } else if( action == "TOGGLE_EXAMINE" ) {
635  // switches between activation and examination
636  menu_mode = menu_mode == mutation_menu_mode::activating ?
637  mutation_menu_mode::examining : mutation_menu_mode::activating;
638  if( menu_mode == mutation_menu_mode::examining ) {
639  examine_id = GetTrait( active, passive, cursor, tab_mode );
640  } else {
641  examine_id = std::nullopt;
642  }
643  examine_pos = 0;
644  } else if( action == "TOGGLE_SPRITE" ) {
645  menu_mode = mutation_menu_mode::hiding;
646  examine_id = std::nullopt;
647  } else if( action == "QUIT" ) {
648  exit = true;
649  }
650  }
