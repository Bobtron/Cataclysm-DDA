316 void mission_type::load( const JsonObject &jo, const std::string &src )
317 {
318  const bool strict = src == "dda";
319 
320  mandatory( jo, was_loaded, "name", name );
321 
322  mandatory( jo, was_loaded, "difficulty", difficulty );
323  mandatory( jo, was_loaded, "value", value );
324 
325  if( jo.has_member( "origins" ) ) {
326  origins.clear();
327  for( const std::string &m : jo.get_tags( "origins" ) ) {
328  origins.emplace_back( io::string_to_enum<mission_origin>( m ) );
329  }
330  }
331 
332  if( std::any_of( origins.begin(), origins.end(), []( mission_origin origin ) {
333  return origin == ORIGIN_ANY_NPC || origin == ORIGIN_OPENER_NPC || origin == ORIGIN_SECONDARY;
334 } ) ) {
335  JsonObject djo = jo.get_object( "dialogue" );
336  // TODO: There should be a cleaner way to do it
337  mandatory( djo, was_loaded, "describe", dialogue[ "describe" ] );
338  mandatory( djo, was_loaded, "offer", dialogue[ "offer" ] );
339  mandatory( djo, was_loaded, "accepted", dialogue[ "accepted" ] );
340  mandatory( djo, was_loaded, "rejected", dialogue[ "rejected" ] );
341  mandatory( djo, was_loaded, "advice", dialogue[ "advice" ] );
342  mandatory( djo, was_loaded, "inquire", dialogue[ "inquire" ] );
343  mandatory( djo, was_loaded, "success", dialogue[ "success" ] );
344  mandatory( djo, was_loaded, "success_lie", dialogue[ "success_lie" ] );
345  mandatory( djo, was_loaded, "failure", dialogue[ "failure" ] );
346  }
347 
348  optional( jo, was_loaded, "description", description );
349  optional( jo, was_loaded, "urgent", urgent );
350  optional( jo, was_loaded, "item", item_id );
351  optional( jo, was_loaded, "item_group", group_id );
352  optional( jo, was_loaded, "count", item_count, 1 );
353  optional( jo, was_loaded, "required_container", container_id );
354  optional( jo, was_loaded, "remove_container", remove_container );
355  //intended for situations where closed and open container are different
356  optional( jo, was_loaded, "empty_container", empty_container );
357  optional( jo, was_loaded, "has_generic_rewards", has_generic_rewards, true );
358 
359  goal = jo.get_enum_value<decltype( goal )>( "goal" );
360 
361  assign_function( jo, "place", place, tripoint_function_map );
362  const auto parse_phase = [&]( const std::string & phase,
363  std::function<void( mission * )> &phase_func ) {
364  if( jo.has_string( phase ) ) {
365  assign_function( jo, phase, phase_func, mission_function_map );
366  } else if( jo.has_member( phase ) ) {
367  JsonObject j_start = jo.get_object( phase );
368  if( !parse_funcs( j_start, phase_func ) ) {
369  deferred.emplace_back( jo, src );
370  jo.allow_omitted_members();
371  j_start.allow_omitted_members();
372  return false;
373  }
374  }
375  return true;
376  };
377  if( !parse_phase( "start", start ) ) {
378  return;
379  }
380  if( !parse_phase( "end", end ) ) {
381  return;
382  }
383  if( !parse_phase( "fail", fail ) ) {
384  return;
