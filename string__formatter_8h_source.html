31  * @defgroup string_formatter_convert Convert functions for @ref string_formatter
32  *
33  * The `convert` functions here are used to convert the input value of
34  * @ref string_formatter::parse into the requested type, as defined by the format specifiers.
35  *
36  * @tparam T the input type, as given by the call to `string_format`.
37  * @tparam RT the requested type. The `convert` functions return such a value or they throw
38  * an exception via @ref throw_error.
39  *
40  * Each function has the same parameters:
41  * First parameter defined the requested type. The value of the pointer is ignored, callers
42  * should use a (properly casted) `nullptr`. It is required to "simulate" overloading the
43  * return value. E.g. `long convert(long*, int)` and `short convert(short*, int)` both convert
44  * a input value of type `int`, but the first converts to `long` and the second converts to
45  * `short`. Without the first parameters their signature would be identical.
46  * The second parameter is used to call @ref throw_error / @ref string_formatter_set_temp_buffer.
47  * The third parameter is the input value that is to be converted.
48  * The fourth parameter is a dummy value, it is always ignored, callers should use `0` here.
49  * It is used so the fallback with the variadic arguments is *only* chosen when no other
50  * overload matches.
51  */

161 /**
162  * Type-safe and undefined-behavior free wrapper over `sprintf`.
163  * See @ref string_format for usage.
164  * Basically it extracts the format specifiers and calls `sprintf` for each one separately
165  * and with proper conversion of the input type.
166  * For example `printf("%f", 7)` would yield undefined behavior as "%f" requires a `double`
167  * as argument. This class detects the format specifier and converts the input to `double`
168  * before calling `sprintf`. Similar for `printf("%d", "foo")` (yields UB again), but this
169  * class will just throw an exception.
170  */
171 // Note: argument index is always 0-based *in this code*, but `printf` has 1-based arguments.

182  /// so it's only valid after the parsing has completed.
183  std::string output;

262  template<typename ...Args>
263  void read_conversion( const int format_arg_index, Args &&... args ) {
264  // Removes the prefix "ll", "l", "h" and "hh", "z", and "t".
265  // We later add "ll" again and that
266  // would interfere with the existing prefix. We convert *all* input to (un)signed
267  // long long int and use the "ll" modifier all the time. This will print the
268  // expected value all the time, even when the original modifier did not match.
269  if( consume_next_input_if( 'l' ) ) {
270  consume_next_input_if( 'l' );
271  } else if( consume_next_input_if( 'h' ) ) {
272  consume_next_input_if( 'h' );
273  } else if( consume_next_input_if( 'z' ) ) { // NOLINT(bugprone-branch-clone)
274  // done with it
275  } else if( consume_next_input_if( 't' ) ) {
276  // done with it
277  }
278  const char c = consume_next_input();
279  current_format.push_back( c );
280  switch( c ) {
281  case 'c':
282  return do_formatting( get_nth_arg_as<int, 0>( format_arg_index, std::forward<Args>( args )... ) );
283  case 'd':
284  case 'i':
285  add_long_long_length_modifier();
286  return do_formatting( get_nth_arg_as<signed long long int, 0>( format_arg_index,
287  std::forward<Args>( args )... ) );
288  case 'o':
289  case 'u':
290  case 'x':
291  case 'X':
292  add_long_long_length_modifier();
293  return do_formatting( get_nth_arg_as<unsigned long long int, 0>( format_arg_index,
294  std::forward<Args>( args )... ) );
295  case 'a':
296  case 'A':
297  case 'g':
298  case 'G':
299  case 'f':
300  case 'F':
301  case 'e':
302  case 'E':
303  return do_formatting( get_nth_arg_as<double, 0>( format_arg_index,
304  std::forward<Args>( args )... ) );
305  case 'p':
306  return do_formatting( get_nth_arg_as<void *, 0>( format_arg_index,
307  std::forward<Args>( args )... ) );
308  case 's':
309  return do_formatting( get_nth_arg_as<const char *, 0>( format_arg_index,
310  std::forward<Args>( args )... ) );
311  default:
312  throw_error( "Unsupported format conversion: " + std::string( 1, c ) );
313  }
314  }

321  public:
322  /// @param format The format string as required by `sprintf`.
323  explicit string_formatter( std::string_view format ) : format( format ) { }
324  /// Does the actual `sprintf`. It uses @ref format and puts the formatted
325  /// string into @ref output.
326  /// Note: use @ref get_output to get the formatted string after a successful
327  /// call to this function.
328  /// @throws Exceptions when the arguments do not match the format specifiers,
329  /// see @ref get_nth_arg_as, or when the format is invalid for whatever reason.
330  /// Note: @ref string_format is a wrapper that handles those exceptions.
331  template<typename ...Args>
332  void parse( Args &&... args ) {
333  output.reserve( format.size() );
334  output.resize( 0 );
335  current_index_in_format = 0;
336  current_argument_index = 0;
337  while( const char c = consume_next_input() ) {
338  if( c != '%' ) {
339  output.push_back( c );
340  continue;
341  }
342  if( consume_next_input_if( '%' ) ) {
343  output.push_back( '%' );
344  continue;
345  }
346  current_format = "%";
347  const std::optional<int> format_arg_index = read_argument_index();
348  read_flags();
349  if( const std::optional<int> width_argument_index = read_width() ) {
350  const int w = get_nth_arg_as<int, 0>( *width_argument_index, std::forward<Args>( args )... );
351  current_format += std::to_string( w );
352  }
353  if( const std::optional<int> precision_argument_index = read_precision() ) {
354  const int p = get_nth_arg_as<int, 0>( *precision_argument_index, std::forward<Args>( args )... );
355  current_format += std::to_string( p );
356  }
357  const int arg = format_arg_index ? *format_arg_index : current_argument_index++;
358  read_conversion( arg, std::forward<Args>( args )... );
359  }
360  }
361  std::string get_output() const {
362  return output;
363  }

393 /**
394  * Simple wrapper over @ref string_formatter::parse. It catches any exceptions and returns
395  * some error string. Otherwise it just returns the formatted string.
396  *
397  * These functions perform string formatting according to the rules of the `printf` function,
398  * see `man 3 printf` or any other documentation.
399  *
400  * In short: the \p format parameter is a string with optional placeholders, which will be
401  * replaced with formatted data from the further arguments. The further arguments must have
402  * a type that matches the type expected by the placeholder.
403  * The placeholders look like this:
404  * - `%s` expects an argument of type `const char*` or `std::string` or numeric (which is
405  * converted to a string via `to_string`), which is inserted as is.
406  * - `%d` expects an argument of an integer type (int, short, ...), which is formatted as
407  * decimal number.
408  * - `%f` expects a numeric argument (integer / floating point), which is formatted as
409  * decimal number.
410  *
411  * There are more placeholders and options to them (see documentation of `printf`).
412  * Note that this wrapper (via @ref string_formatter) automatically converts the arguments
413  * to match the given format specifier (if possible) - see @ref string_formatter_convert.
414  */
415 /**@{*/
416 template<typename ...Args>
417 inline std::string string_format( std::string_view format, Args &&...args )
418 {
419  try {
420  cata::string_formatter formatter( format );
421  formatter.parse( std::forward<Args>( args )... );
422  return formatter.get_output();
423  } catch( ... ) {
424  return cata::handle_string_format_error();
425  }
426 }
427 template<typename T, typename ...Args>
428 inline std::enable_if_t<cata::is_translation<T>, std::string>
429 string_format( T &&format, Args &&...args )
430 {
431  return string_format( format.translated(), std::forward<Args>( args )... );
432 }
