400 bool vehicle_part::can_reload( const item &obj ) const
401 {
402  // first check part is not destroyed and can contain ammo
403  if( !is_fuel_store() ) {
404  return false;
405  }
406 
407  if( is_battery() ) {
408  return false;
409  }
410 
411  if( !obj.is_null() ) {
412  const itype_id obj_type = obj.typeId();
413  if( is_reactor() ) {
414  return base.can_reload_with( obj, true );
415  }
416 
417  // forbid filling tanks with solids or non-material things
418  if( is_tank() && ( obj.made_of( phase_id::SOLID ) || obj.made_of( phase_id::PNULL ) ) ) {
419  return false;
420  }
421  // forbid putting liquids, gasses, and plasma in things that aren't tanks
422  if( !obj.made_of( phase_id::SOLID ) && !is_tank() ) {
423  return false;
424  }
425  // prevent mixing of different ammo
426  if( !ammo_current().is_null() && ammo_current() != obj_type ) {
427  return false;
428  }
429  // For storage with set type, prevent filling with different types
430  if( info().fuel_type != fuel_type_none && info().fuel_type != obj_type ) {
431  return false;
432  }
433  // don't fill magazines with inappropriate fuel
434  if( !is_tank() && !base.can_reload_with( obj, true ) ) {
435  return false;
436  }
437  }
438  if( base.is_gun() ) {
439  return false;
440  }
441 
442  if( is_reactor() ) {
443  return true;
444  }
445 
446  if( ammo_current().is_null() ) {
447  return true; // empty tank
448  }
449 
450  // Despite checking for an empty tank, item::find_type can still turn up with an empty ammo pointer
451  if( cata::value_ptr<islot_ammo> a_val = item::find_type( ammo_current() )->ammo ) {
452  return ammo_remaining() < ammo_capacity( a_val->type );
453  }
454 
455  // Nothing in tank
456  return ammo_capacity( obj.ammo_type() ) > 0;
457 }
