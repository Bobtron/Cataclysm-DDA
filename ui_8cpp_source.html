461 /**
462  * Find the minimum width between max( min_width, 1 ) and
463  * max( max_width, min_width, 1 ) to fold the string to no more than max_lines,
464  * or no more than the minimum number of lines possible, assuming that
465  * foldstring( width ).size() decreases monotonously with width.
466  **/
467 static int find_minimum_fold_width( const std::string &str, int max_lines,
468  int min_width, int max_width )
469 {
470  if( str.empty() ) {
471  return std::max( min_width, 1 );
472  }
473  min_width = std::max( min_width, 1 );
474  // max_width could be further limited by the string width, but utf8_width is
475  // not handling linebreaks properly.
476 
477  if( min_width < max_width ) {
478  // If with max_width the string still folds to more than max_lines, find the
479  // minimum width that folds the string to such number of lines instead.
480  max_lines = std::max<int>( max_lines, foldstring( str, max_width ).size() );
481  while( min_width < max_width ) {
482  int width = ( min_width + max_width ) / 2;
483  // width may equal min_width, but will always be less than max_width.
484  int lines = foldstring( str, width ).size();
485  // If the current width folds the string to no more than max_lines
486  if( lines <= max_lines ) {
487  // The minimum width is between min_width and width.
488  max_width = width;
489  } else {
490  // The minimum width is between width + 1 and max_width.
491  min_width = width + 1;
492  }
493  // The new interval will always be smaller than the previous one,
494  // so the loop is guaranteed to end.
495  }
496  }
497  return min_width;
498 }

915 {

917  return false;
918  }
919 
920  bool looparound = scrollby == -1 || scrollby == 1;
921  bool backwards = scrollby < 0;
922  int recmax = static_cast<int>( fentries.size() );
923 
924  fselected += scrollby;
925  if( !looparound ) {
926  if( backwards && fselected < 0 ) {
927  fselected = 0;

931  }
932

935  fselected = fentries.size() - 1;

939  break;
940  }
941  --fselected;
942  if( fselected < 0 ) {
943  fselected = fentries.size() - 1;
944  }
945  }
946  } else {
947  if( fselected >= recmax ) {
948  fselected = 0;

952  break;
953  }
954  ++fselected;
955  if( fselected >= recmax ) {
956  fselected = 0;
957  }
958  }
959  }
