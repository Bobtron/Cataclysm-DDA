12 {
13  // We need different implementations on each platform. MacOS seems to not
14  // support localized comparison of strings via the standard library at all,
15  // so resort to MacOS-specific solution. Windows cannot be expected to be
16  // using a UTF-8 locale (whereas our strings are always UTF-8) and so we
17  // must convert to wstring for comparison there. Linux seems to work as
18  // expected on regular strings; no workarounds needed.
19  // See https://github.com/CleverRaven/Cataclysm-DDA/pull/40041 for further
20  // discussion.
21 #if defined(__APPLE__) // macOS and iOS
22  CFStringRef lr = CFStringCreateWithCStringNoCopy( kCFAllocatorDefault, l.c_str(),
23  kCFStringEncodingUTF8, kCFAllocatorNull );
24  CFStringRef rr = CFStringCreateWithCStringNoCopy( kCFAllocatorDefault, r.c_str(),
25  kCFStringEncodingUTF8, kCFAllocatorNull );
26  bool result = CFStringCompare( lr, rr, kCFCompareLocalized ) < 0;
27  CFRelease( lr );
28  CFRelease( rr );
29  return result;
30 #elif defined(_WIN32)
31  return ( *this )( utf8_to_wstr( l ), utf8_to_wstr( r ) );
32 #else

34 #endif
35 }
