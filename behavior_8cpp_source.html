137 void node_t::load( const JsonObject &jo, const std::string_view )
138 {
139  // We don't initialize the node unless it has no children (opportunistic optimization).
140  // Instead we initialize a parallel struct that holds the labels until finalization.
141  if( jo.has_array( "children" ) ) {
142  temp_node_data.emplace_back();
143  node_data &new_data = temp_node_data.back();
144  new_data.id = id;
145  optional( jo, was_loaded, "children", new_data.children );
146  }
147  if( jo.has_string( "strategy" ) ) {
148  std::unordered_map<std::string, const strategy_t *>::iterator new_strategy =
149  behavior::strategy_map.find( jo.get_string( "strategy" ) );
150  if( new_strategy != strategy_map.end() ) {
151  strategy = new_strategy->second;
152  } else {
153  debugmsg( "While loading %s, failed to find strategy %s.",
154  id.str(), jo.get_string( "strategy" ) );
155  jo.throw_error( "Invalid strategy in behavior." );
156  }
157  }
158  for( const JsonObject predicate_object : jo.get_array( "conditions" ) ) {
159  const std::string predicate_id = predicate_object.get_string( "predicate" );
160  auto new_predicate = predicate_map.find( predicate_id );
161  if( new_predicate == predicate_map.end() ) {
162  debugmsg( "While loading %s, failed to find predicate %s.",
163  id.str(), predicate_id );
164  jo.throw_error( "Invalid predicate in behavior." );
165  }
166  const std::string predicate_argument = predicate_object.get_string( "argument", "" );
167  const bool invert_result = predicate_object.get_bool( "invert_result", false );
168  conditions.emplace_back( new_predicate->second, predicate_argument,
169  invert_result );
170  }
171  optional( jo, was_loaded, "goal", _goal );
