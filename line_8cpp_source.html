654 // Cardinals are cardinals. Result is cardinal and adjacent sub-cardinals.
655 // Sub-Cardinals are sub-cardinals && abs(x) == abs(y). Result is sub-cardinal and adjacent cardinals.
656 // Sub-sub-cardinals are direction && abs(x) > abs(y) or vice versa.
657 // Result is adjacent cardinal and sub-cardinals, plus the nearest other cardinal.
658 // e.g. if the direction is NNE, also include E.
659 std::vector<tripoint> squares_closer_to( const tripoint &from, const tripoint &to )
660 {
661  std::vector<tripoint> adjacent_closer_squares;
662  adjacent_closer_squares.reserve( 5 );
663  const tripoint d( -from + to );
664  const point a( std::abs( d.x ), std::abs( d.y ) );
665  if( d.z != 0 ) {
666  adjacent_closer_squares.push_back( from + tripoint( sgn( d.x ), sgn( d.y ), sgn( d.z ) ) );
667  }
668  if( a.x > a.y ) {
669  // X dominant.
670  adjacent_closer_squares.push_back( from + point( sgn( d.x ), 0 ) );
671  adjacent_closer_squares.push_back( from + point( sgn( d.x ), 1 ) );
672  adjacent_closer_squares.push_back( from + point( sgn( d.x ), -1 ) );
673  if( d.y != 0 ) {
674  adjacent_closer_squares.push_back( from + point( 0, sgn( d.y ) ) );
675  }
676  } else if( a.x < a.y ) {
677  // Y dominant.
678  adjacent_closer_squares.push_back( from + point( 0, sgn( d.y ) ) );
679  adjacent_closer_squares.push_back( from + point( 1, sgn( d.y ) ) );
680  adjacent_closer_squares.push_back( from + point( -1, sgn( d.y ) ) );
681  if( d.x != 0 ) {
682  adjacent_closer_squares.push_back( from + point( sgn( d.x ), 0 ) );
683  }
684  } else if( d.x != 0 ) {
685  // Pure diagonal.
686  adjacent_closer_squares.push_back( from + point( sgn( d.x ), sgn( d.y ) ) );
687  adjacent_closer_squares.push_back( from + point( sgn( d.x ), 0 ) );
688  adjacent_closer_squares.push_back( from + point( 0, sgn( d.y ) ) );
689  }
690 
691  return adjacent_closer_squares;
692 }

694 // Returns a vector of the adjacent square in the direction of the target,
695 // and the two squares flanking it.
696 std::vector<point> squares_in_direction( const point &p1, const point &p2 )
697 {
698  int junk = 0;
699  point center_square = line_to( p1, p2, junk )[0];
700  std::vector<point> adjacent_squares;
701  adjacent_squares.reserve( 3 );
702  adjacent_squares.push_back( center_square );
703  if( p1.x == center_square.x ) {
704  // Horizontally adjacent.
705  adjacent_squares.emplace_back( p1.x + 1, center_square.y );
706  adjacent_squares.emplace_back( p1.x - 1, center_square.y );
707  } else if( p1.y == center_square.y ) {
708  // Vertically adjacent.
709  adjacent_squares.emplace_back( center_square.x, p1.y + 1 );
710  adjacent_squares.emplace_back( center_square.x, p1.y - 1 );
711  } else {
712  // Diagonally adjacent.
713  adjacent_squares.emplace_back( p1.x, center_square.y );
714  adjacent_squares.emplace_back( center_square.x, p1.y );
715  }

911 {

924  mult = -1;
