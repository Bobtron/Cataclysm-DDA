62  }

64  if( it == edges.end() ) {
65  return;
66  }

75  if( previous_match != on_current_branch.end() ) {

77  // reach this vertex, so reconstruct the loop and save it.
78  std::vector<T> loop;

80  while( true ) {

98 // Each T defines a vertex.
99 // For a vertex a, edges[a] is a list of all the vertices connected by edges
100 // from a.
101 // It is acceptable for some vertex keys to be missing from the edges map, if
102 // those vertices have no out-edges.

107 {
