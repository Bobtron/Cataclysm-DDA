50 static weather_gen_common get_common_data( const tripoint &location, const time_point &real_t,
51  unsigned seed )
52 {
53  season_effective_time t( real_t );
54  weather_gen_common result;
55  // Integer x position / widening factor of the Perlin function.
56  result.x = location.x / 2000.0;
57  // Integer y position / widening factor of the Perlin function.
58  result.y = location.y / 2000.0;
59  // Integer turn / widening factor of the Perlin function.
60  result.z = to_days<double>( real_t - calendar::turn_zero );
61  // Limit the random seed during noise calculation, a large value flattens the noise generator to zero
62  // Windows has a rand limit of 32768, other operating systems can have higher limits
63  result.modSEED = seed % SIMPLEX_NOISE_RANDOM_SEED_LIMIT;
64  const double year_fraction( time_past_new_year( t.t ) /
65  calendar::year_length() ); // [0,1)
66 
67  result.cyf = std::cos( tau * ( year_fraction + .125 ) ); // [-1, 1]
68  // We add one-eighth to line up `cyf` so that 1 is at
69  // midwinter and -1 at midsummer. (Cataclsym DDA years
70  // start when spring starts. Gregorian years start when
71  // winter starts.)
72  result.season = season_of_year( t.t );
73 
74  return result;
75 }

98  wg.base_temperature +
99  seasonal_temp_mod[season] +
100  dayv * daily_magnitude_K +

114 w_point weather_generator::get_weather( const tripoint_abs_ms &location, const time_point &real_t,
115  unsigned seed ) const
116 {
117  season_effective_time t( real_t );
118  const weather_gen_common common = get_common_data( location.raw(), real_t, seed );
119 
120  const double x( common.x );
121  const double y( common.y );
122  const double z( common.z );
123 
124  const unsigned modSEED = common.modSEED;
125  const double cyf( common.cyf );
126  const double seasonality = -common.cyf;
127  // -1 in midwinter, +1 in midsummer
128  const season_type season = common.season;
129 
130  // Noise factors
131  const units::temperature T( weather_temperature_from_common_data( *this, common, t ) );
132  double W( raw_noise_4d( x / 2.5, y / 2.5, z / 200, modSEED ) * 10.0 );
133 
134  // Humidity variation
135  double mod_h( 0 );
136  if( season == WINTER ) {
137  mod_h += winter_humidity_manual_mod;
138  } else if( season == SPRING ) {
139  mod_h += spring_humidity_manual_mod;
140  } else if( season == SUMMER ) {
141  mod_h += summer_humidity_manual_mod;
142  } else if( season == AUTUMN ) {
143  mod_h += autumn_humidity_manual_mod;
144  }
145  // Relative humidity, a percentage.
146  double H = std::min( 100., std::max( 0.,
147  base_humidity + mod_h + 100 * (
148  .15 * seasonality +
149  raw_noise_4d( x, y, z, modSEED + 101 ) *
150  .2 * ( -seasonality + 2 ) ) ) );
151 
152  // Pressure
153  double P =
154  base_pressure +
155  raw_noise_4d( x, y, z, modSEED + 211 ) *
156  10 * ( -seasonality + 2 );
157 
158  // Wind power
159  W = std::max( 0, static_cast<int>( base_wind * rng( 1, 2 ) / std::pow( ( P + W ) / 1014.78, rng( 9,
160  base_wind_distrib_peaks ) ) +
161  -cyf / base_wind_season_variation * rng( 1, 2 ) ) );
162  // Initial static variable
163  if( current_winddir == 1000 ) {
164  current_winddir = get_wind_direction( season );
165  current_winddir = convert_winddir( current_winddir );
166  } else {
167  // When wind strength is low, wind direction is more variable
168  bool changedir = one_in( W * 2160 );
169  if( changedir ) {
170  current_winddir = get_wind_direction( season );
171  current_winddir = convert_winddir( current_winddir );
172  }
173  }
174  std::string wind_desc = get_wind_desc( W );
175  return w_point{ T, H, P, W, wind_desc, current_winddir, t, location };
176 }

250 {
251  /**
252  WATER TEMPERATURE
253  source : http://echo2.epfl.ch/VICAIRE/mod_2/chapt_5/main.htm
254  source : http://www.grandriver.ca/index/document.cfm?Sec=2&Sub1=7&sub2=1
255  **/
256 
257  season_effective_time t( calendar::turn );
258  int season_length = to_days<int>( calendar::season_length() );
259  int day = to_days<int>( time_past_new_year( t.t ) );
260  int hour = hour_of_day<int>( t.t );
261 
262  float water_temperature = 0;
263 
264  if( season_length == 0 ) {
265  season_length = 1;
266  }
267 
268  // Temperature varies between 33.8F and 75.2F depending on the time of year. Day = 0 corresponds to the start of spring.
269  float annual_mean_water_temperature = 54.5 + 20.7 * std::sin( tau * ( day - season_length * 0.5 ) /
270  ( season_length * 4.0 ) );
271  // Temperature varies between +2F and -2F depending on the time of day. Hour = 0 corresponds to midnight.
272  float daily_water_temperature_variation = 2.0 + 2.0 * std::sin( tau * ( hour - 6.0 ) / 24.0 );
273 
274  water_temperature = annual_mean_water_temperature + daily_water_temperature_variation;
275 
276  return units::from_fahrenheit( water_temperature );
