395 void proficiency_set::remove( const proficiency_id &lost )
396 {
397  // Removes from proficiencies you are learning
398  for( std::vector<learning_proficiency>::iterator it = learning.begin(); it != learning.end(); ) {
399  if( it->id == lost ) {
400  it = learning.erase( it );
401  } else {
402  ++it;
403  }
404  }
405 
406  // No unintended side effects
407  if( !known.count( lost ) ) {
408  return;
409  }
410  // Loop over all the proficiencies requiring lost, to remove them too
411  // Then the ones requiring it, etc.

414  size_t cached_size = 0;
415  // While we have new proficiencies we need to check for anything requiring them
416  while( to_remove.size() != cached_size ) {
417  cached_size = to_remove.size();
418  std::set<proficiency_id> additional = proficiencies_requiring( to_remove, known );
419  to_remove.insert( additional.begin(), additional.end() );
420  }
421 
422  for( const proficiency_id &gone : to_remove ) {
423  known.erase( gone );
424  }
