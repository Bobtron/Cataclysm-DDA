516 std::list<item> profession::items( bool male, const std::vector<trait_id> &traits ) const
517 {
518  std::list<item> result;
519  auto add_legacy_items = [&result]( const itypedecvec & vec ) {
520  for( const itypedec &elem : vec ) {
521  item it( elem.type_id, advanced_spawn_time(), item::default_charges_tag {} );
522  if( !elem.snip_id.is_null() ) {
523  it.set_snippet( elem.snip_id );
524  }
525  it = it.in_its_container();
526  result.push_back( it );
527  }
528  };
529 
530  add_legacy_items( legacy_starting_items );
531  add_legacy_items( male ? legacy_starting_items_male : legacy_starting_items_female );
532 
533  std::vector<item> group_both = item_group::items_from( _starting_items,
534  advanced_spawn_time() );
535  std::vector<item> group_gender = item_group::items_from( male ? _starting_items_male :
536  _starting_items_female, advanced_spawn_time() );
537  result.insert( result.begin(), std::make_move_iterator( group_both.begin() ),
538  std::make_move_iterator( group_both.end() ) );
539  result.insert( result.begin(), std::make_move_iterator( group_gender.begin() ),
540  std::make_move_iterator( group_gender.end() ) );
541 
542  if( !has_flag( "NO_BONUS_ITEMS" ) ) {
543  const std::vector<item> &items = item_substitutions.get_bonus_items( traits );
544  for( const item &it : items ) {
545  if( it.typeId() != no_bonus ) {
546  result.push_back( it );
547  }
548  }
549  }
550  for( auto iter = result.begin(); iter != result.end(); ) {
551  const std::vector<item> sub = item_substitutions.get_substitution( *iter, traits );
552  if( !sub.empty() ) {
553  result.insert( result.begin(), sub.begin(), sub.end() );
554  iter = result.erase( iter );
555  } else {
556  ++iter;
557  }
558  }
559  for( item &it : result ) {
560  it.visit_items( []( item * it, item * ) {
561  clear_faults( *it );
562  return VisitResponse::NEXT;
563  } );
564  if( it.has_flag( flag_VARSIZE ) ) {
565  it.set_flag( flag_FIT );
566  }
567  }
568 
569  if( result.empty() ) {
570  // No need to do the below stuff. Plus it would cause said below stuff to crash
571  return result;
572  }
573 
574  // Merge charges for items that stack with each other
575  for( auto outer = result.begin(); outer != result.end(); ++outer ) {
576  if( !outer->count_by_charges() ) {
577  continue;
578  }
579  for( auto inner = std::next( outer ); inner != result.end(); ) {
580  if( outer->stacks_with( *inner ) ) {
581  outer->merge_charges( *inner );
582  inner = result.erase( inner );
583  } else {
584  ++inner;
585  }
586  }
587  }
588 
589  result.sort( []( const item & first, const item & second ) {
590  return first.get_layer() < second.get_layer();
591  } );
592  return result;
593 }

760 void json_item_substitution::load( const JsonObject &jo )
761 {
762  auto check_duplicate_item = [&]( const itype_id & it ) {
763  return substitutions.find( it ) != substitutions.end();
764  };
765 
766  if( jo.has_member( "item" ) ) {
767  // items mode
768  if( check_duplicate_item( itype_id( jo.get_string( "item" ) ) ) ) {
769  jo.throw_error( "Duplicate definition of item" );
770  }
771 
772  for( const JsonValue sub : jo.get_array( "sub" ) ) {
773  substitution s;
774  JsonObject obj = sub.get_object();
775  s.trait_reqs = trait_requirements( obj );
776  for( const JsonValue info : obj.get_array( "new" ) ) {
777  s.infos.emplace_back( info );
778  }
779  substitutions[itype_id( jo.get_string( "item" ) )].push_back( s );
780  }
781  } else if( jo.has_member( "trait" ) ) {
782  // traits mode
783  for( const JsonObject sub : jo.get_array( "sub" ) ) {
784  substitution s;
785  itype_id old_it;
786  sub.read( "item", old_it, true );
787  if( check_duplicate_item( old_it ) ) {
788  sub.throw_error( "Duplicate definition of item" );
789  }
790  s.trait_reqs.present.emplace_back( jo.get_string( "trait" ) );
791  for( const JsonValue info : sub.get_array( "new" ) ) {
792  s.infos.emplace_back( info );
793  }
794  substitutions[old_it].push_back( s );
795  }
796  } else if( jo.has_member( "bonus" ) ) {
797  // bonuses mode
798  const item_group_id &bonus_items = item_group::load_item_group( jo.get_member( "group" ),
799  "collection", "bonus items" );
800  bonuses.emplace_back( bonus_items, trait_requirements( jo.get_object( "bonus" ) ) );
801  }
802 }

804 void json_item_substitution::check_consistency() const
805 {
806  auto check_if_trait = []( const trait_id & t ) {
807  if( !t.is_valid() ) {
808  debugmsg( "%s is not a trait", t.c_str() );
809  }
810  };
811  auto check_if_itype = []( const itype_id & i ) {
812  if( !item::type_is_defined( i ) ) {
813  debugmsg( "%s is not an itype_id", i.c_str() );
814  }
815  };
816  auto check_if_igroup = []( const item_group_id & gr ) {
817  if( !item_group::group_is_defined( gr ) ) {
818  debugmsg( "%s is not an item_group_id", gr.c_str() );
819  }
820  };
821  auto check_trait_reqs = [&check_if_trait]( const trait_requirements & tr ) {
822  for( const trait_id &str : tr.present ) {
823  check_if_trait( str );
824  }
825  for( const trait_id &str : tr.absent ) {
826  check_if_trait( str );
827  }
828  };
829 
830  for( const auto &pair : substitutions ) {
831  check_if_itype( pair.first );
832  for( const substitution &s : pair.second ) {
833  check_trait_reqs( s.trait_reqs );
834  for( const substitution::info &inf : s.infos ) {
835  check_if_itype( inf.new_item );
836  }
837  }
838  }
839  for( const auto &pair : bonuses ) {
840  check_if_igroup( pair.first );
841  check_trait_reqs( pair.second );
842  }
843 }

845 bool json_item_substitution::trait_requirements::meets_condition( const std::vector<trait_id>
846  &traits ) const
847 {
848  const auto pred = [&traits]( const trait_id & s ) {
849  return std::find( traits.begin(), traits.end(), s ) != traits.end();
850  };
851  return std::all_of( present.begin(), present.end(), pred ) &&
852  std::none_of( absent.begin(), absent.end(), pred );
853 }

855 std::vector<item> json_item_substitution::get_substitution( const item &it,
856  const std::vector<trait_id> &traits ) const
857 {
858  auto iter = substitutions.find( it.typeId() );
859  std::vector<item> ret;
860  if( iter == substitutions.end() ) {
861  for( const item *con : it.all_items_top() ) {
862  const auto sub = get_substitution( *con, traits );
863  ret.insert( ret.end(), sub.begin(), sub.end() );
864  }
865  return ret;
866  }
867 
868  const auto sub = std::find_if( iter->second.begin(),
869  iter->second.end(), [&traits]( const substitution & s ) {
870  return s.trait_reqs.meets_condition( traits );
871  } );
872  if( sub == iter->second.end() ) {
873  return ret;
874  }
875 
876  const int old_amt = it.count();
877  for( const substitution::info &inf : sub->infos ) {
878  item result( inf.new_item, advanced_spawn_time() );
879  int new_amount = std::max( 1, static_cast<int>( std::round( inf.ratio * old_amt ) ) );
880 
881  if( !result.count_by_charges() ) {
882  for( int i = 0; i < new_amount; i++ ) {
883  ret.push_back( result.in_its_container() );
884  }
885  } else {
886  while( new_amount > 0 ) {
887  const item pushed = result.in_its_container( new_amount );
888  new_amount -= pushed.charges_of( inf.new_item );
889  ret.push_back( pushed );
890  }
891  }
892  }
893  return ret;
894 }

896 std::vector<item> json_item_substitution::get_bonus_items( const std::vector<trait_id> &traits )
