344 int armor_portion_data::calc_encumbrance( units::mass weight, bodypart_id bp ) const
345 {
346  // this function takes some fixed points for mass to encumbrance and interpolates them to get results for head encumbrance
347  // TODO: Generalize this for other body parts (either with a modifier or seperated point graphs)
348  // TODO: Handle distributed weight
349 
350  int encumbrance = 0;
351 
352  std::map<units::mass, int> mass_to_encumbrance = bp->encumbrance_per_weight;
353 
354  std::map<units::mass, int>::iterator itt = mass_to_encumbrance.lower_bound( weight );
355 
356  if( itt == mass_to_encumbrance.begin() || itt == mass_to_encumbrance.end() ) {
357  debugmsg( "Can't find a notable point to match this with" );
358  return 100;
359  }
360 
361  // get the bound bellow our given weight
362  --itt;
363 
364  std::map<units::mass, int>::iterator next_itt = std::next( itt );
365 
366  // between itt and next_itt need to figure out how much and scale values
367  float scale = static_cast<float>( weight.value() - itt->first.value() ) / static_cast<float>
368  ( next_itt->first.value() - itt->first.value() );
369 
370  // encumbrance is scaled by range between the two values
371  encumbrance = itt->second + std::roundf( static_cast<float>( next_itt->second - itt->second ) *
372  scale );
373 
374  // then add some modifiers
375  int multiplier = 100;
376  int additional_encumbrance = 0;
377  for( const encumbrance_modifier &em : encumber_modifiers ) {
378  std::tuple<encumbrance_modifier_type, int> modifier = armor_portion_data::convert_descriptor_to_val(
379  em );
380  if( std::get<0>( modifier ) == encumbrance_modifier_type::FLAT ) {
381  additional_encumbrance += std::get<1>( modifier );
382  } else if( std::get<0>( modifier ) == encumbrance_modifier_type::MULT ) {
383  multiplier += std::get<1>( modifier );
384  }
385  }
386  // modify by multiplier
387  encumbrance = std::roundf( static_cast<float>( encumbrance ) * static_cast<float>
388  ( multiplier ) / 100.0f );
389  // modify by flat
390  encumbrance += additional_encumbrance;
391 
392  // cap encumbrance at at least 1
393  return std::max( encumbrance, 1 );
394 }
