38 static Trait_group_tag get_unique_trait_group_id()
39 {

41  // name is already used, we simply go the next id.

44  // (if actually printed somewhere) stands out. Theoretically those auto-generated group

48  Trait_group_tag new_group( unique_prefix + std::to_string( next_id++ ) );
49  if( !new_group.is_valid() ) {
