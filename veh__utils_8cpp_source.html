33 int calc_xp_gain( const vpart_info &vp, const skill_id &sk, const Character &who )
34 {
35  const auto iter = vp.install_skills.find( sk );
36  if( iter == vp.install_skills.end() ) {
37  return 0;
38  }
39 
40  // how many levels are we above the requirement?
41  const int lvl = std::max( static_cast<int>( who.get_skill_level( sk ) ) - iter->second, 1 );
42 
43  // scale xp gain per hour according to relative level
44  // 0-1: 60 xp /h
45  // 2: 15 xp /h
46  // 3: 6 xp /h
47  // 4: 4 xp /h
48  // 5: 3 xp /h
49  // 6: 2 xp /h
50  // 7+: 1 xp /h
51  return std::ceil( to_moves<double>( vp.install_moves ) /
52  to_moves<int>( 1_minutes * std::pow( lvl, 2 ) ) );
53 }

93 {

95 
96  const requirement_data reqs = pt.is_broken()
97  ? vp.install_requirements()
98  : vp.repair_requirements() * pt.get_base().repairable_levels();
99 
100  const inventory &inv = who.crafting_inventory( who.pos(), PICKUP_RANGE, !who.is_npc() );
101  inventory map_inv;
102  // allow NPCs to use welding rigs they can't see ( on the other side of a vehicle )
103  // as they have the handicap of not being able to use the veh interaction menu
104  // or able to drag a welding cart etc.
105  map_inv.form_from_map( who.pos(), PICKUP_RANGE, &who, false, !who.is_npc() );
106  if( !reqs.can_make_with_inventory( inv, is_crafting_component ) ) {
107  who.add_msg_if_player( m_info, _( "You don't meet the requirements to repair the %s." ),
108  pt.name() );
109  return false;
110  }
111

113  item base( vp.base_item );
114  for( const auto &e : reqs.get_components() ) {

116  is_crafting_component ) ) {
117  if( obj.typeId() == vp.base_item ) {
118  base = obj;
119  }
120  }
121  }
122 
123  for( const auto &e : reqs.get_tools() ) {
124  who.consume_tools( who.select_tool_component( e, 1, map_inv ), 1 );
125  }
126 
127  who.invalidate_crafting_inventory();
128 
129  for( const auto &sk : pt.is_broken() ? vp.install_skills : vp.repair_skills ) {
130  who.practice( sk.first, calc_xp_gain( vp, sk.first, who ) );
131  }
132 
133  // If part is broken, it will be destroyed and references invalidated
134  const std::string partname = pt.name( false );
135  const std::string startdurability = pt.get_base().damage_indicator();
136  if( pt.is_broken() ) {
137  const vpart_id vpid = pt.info().id;
138  const point mount = pt.mount;
139  const units::angle direction = pt.direction;
140  const std::string variant = pt.variant;
141  get_map().spawn_items( who.pos(), pt.pieces_for_broken_part() );
142  veh.remove_part( pt );
143  const int partnum = veh.install_part( mount, vpid, std::move( base ) );
144  if( partnum >= 0 ) {
145  vehicle_part &vp = veh.part( partnum );
146  vp.direction = direction;
147  vp.variant = variant;
148  }
149  veh.part_removal_cleanup();
150  who.add_msg_if_player( m_good, _( "You replace the %1$s's %2$s. (was %3$s)" ),
151  veh.name, partname, startdurability );
152  } else {
153  veh.set_hp( pt, pt.info().durability, true );
154  who.add_msg_if_player( m_good, _( "You repair the %1$s's %2$s. (was %3$s)" ),
155  veh.name, partname, startdurability );
156  }
157  return true;
158 }
