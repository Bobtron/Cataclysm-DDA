59 static void advance_jsin( TextJsonIn *jsin, flexbuffers::Reference root, const JsonPath &path )
60 {
61  for( JsonPath::index_type idx : path ) {
62  if( root.IsMap() ) {
63  std::string member = root.AsMap().Keys()[ idx ].AsString().str();
64  jsin->start_object();
65  while( !jsin->end_object() ) {
66  if( jsin->get_member_name() == member ) {
67  break;
68  }
69  jsin->skip_value();
70  }
71  root = root.AsVector()[ idx ];
72  } else if( root.IsVector() ) {
73  jsin->start_array();
74  for( size_t i = 0; i < idx && !jsin->end_array(); ++i ) {
75  jsin->skip_value();
76  }
77  root = root.AsVector()[ idx ];
78  } else {
79  cata_unreachable();
80  }
81  }
