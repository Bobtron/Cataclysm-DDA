359  return rhs.settings.priority() > this->settings.priority();
360  }
361

363  if( !rhs.settings.accepts_item( it ) ) {
364  return false;
365  } else if( !this->settings.accepts_item( it ) ) {
366  return true;

370  !rhs.settings.get_category_whitelist().empty();

372  !this->settings.get_category_whitelist().empty();

375  }

378  if( has_item_stacks_with( it ) != rhs_it_stack ) {

385  }
386

388  // pockets restricted by flag should try to get filled first
389  return !rhs.data->get_flag_restrictions().empty();
390  }

393  // a lower spoil multiplier is better
394  return rhs.spoil_multiplier() < spoil_multiplier();

400  }

405  return rhs.data->watertight;
406  }
407  }

410  if( nested ) {
411  return false;
412  }

415  // pockets with less extra encumbrance should be prioritized
416  return rhs.data->extra_encumbrance < data->extra_encumbrance;
417  }

422  }

2603  * Check to see if the given item is accepted by this pocket based on player configured rules.
2604  *
2605  * The rules are defined in two list:
2606  *
2607  * - whitelist - when one or more items are on the list then only those items are accepted.
2608  * - blacklist - those items listed here are never accepted.
2609  *
2610  * When a whitelist is empty and the item is not blacklisted then it will be accepted.
2611  * Container items will be accepted only when all items inside are accepted.
2612  *
2613  * Note that the rules take into account both item id and category which have
2614  * to be accepted by the rules outlined above for an item to be accepted.
2615  *
2616  * @param it item to check.
2617  * @return true if the item is accepted by this pocket, false otherwise.
2618  * @see item_pocket::favorite_settings::accepts_container(const item&)
2619  */

2621 {

2623  if( disabled ) {
2624  return false;
2625  }
2626  const itype_id &id = it.typeId();
2627  const item_category_id &cat = category_blacklist.empty() && category_whitelist.empty()
2628  ? item_category_id{} :
2629  it.get_category_of_contents().id;
2630 
2631  // if the item is explicitly listed in either of the lists, then it's clear what to do with it
2632  if( item_blacklist.count( id ) ) {
2633  return false;
2634  }
2635  if( item_whitelist.count( id ) ) {
2636  return true;
2637  }
2638 
2639  // otherwise check the category, the same way
2640  if( category_blacklist.count( cat ) ) {
2641  return false;
2642  }
2643  if( category_whitelist.count( cat ) ) {
2644  return true;
2645  }

2652  return accepts_item( *it );

2656  // and either allow or deny everything that's fallen through to here.

2659  }

2663  }
