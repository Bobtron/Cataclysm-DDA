42 directed_path<point> greedy_path( const point &source, const point &dest, const point &max,
43  const two_node_scoring_fn<point> &scorer )
44 {
45  using Node = point_node;
46  const auto inbounds = [ max ]( const point & p ) {
47  return p.x >= 0 && p.x < max.x && p.y >= 0 && p.y < max.y;
48  };
49  const auto map_index = [ max ]( const point & p ) {
50  return p.y * max.x + p.x;
51  };
52 
53  directed_path<point> res;
54  if( source == dest ) {
55  return res;
56  }
57  if( !inbounds( source ) || !inbounds( dest ) ) {
58  return res;
59  }
60  const Node first_node( source, om_direction::type::invalid, 1000 );
61  if( scorer( directed_node<point>( first_node.pos, first_node.dir ),
62  std::nullopt ).node_cost < 0 ) {
63  return res;
64  }
65  const size_t map_size = static_cast<size_t>( max.x ) * max.y;
66  std::vector<bool> closed( map_size, false );
67  std::vector<int> open( map_size, 0 );
68  std::vector<short> dirs( map_size, 0 );
69  std::array<std::priority_queue<Node, std::vector<Node>>, 2> nodes;
70 
71  int i = 0;
72  nodes[i].push( first_node );
73  open[map_index( source )] = std::numeric_limits<int>::max();
74  while( !nodes[i].empty() ) {
75  const Node mn( nodes[i].top() ); // get the best-looking node
76  nodes[i].pop();
77  // mark it visited
78  closed[map_index( mn.pos )] = true;
79  // if we've reached the end, draw the path and return
80  if( mn.pos == dest ) {
81  point p = mn.pos;
82  res.nodes.reserve( nodes[i].size() );
83  while( p != source ) {
84  const int n = map_index( p );
85  const om_direction::type dir = static_cast<om_direction::type>( dirs[n] );
86  res.nodes.emplace_back( p, dir );
87  p += om_direction::displace( dir );
88  }
89  res.nodes.emplace_back( p );
90  return res;
91  }
92  for( om_direction::type dir : om_direction::all ) {
93  const point p = mn.pos + om_direction::displace( dir );
94  const int n = map_index( p );
95  // don't allow out of bounds or already traversed tiles
96  if( !inbounds( p ) || closed[n] ) {
97  continue;
98  }
99  const node_score score = scorer( directed_node<point>( p, dir ), directed_node<point>( mn.pos,
100  mn.dir ) );
101  if( score.node_cost < 0 ) {
102  continue;
103  }
104  const int priority = score.node_cost + score.estimated_dest_cost;
105  // record direction to shortest path
106  if( open[n] == 0 || open[n] > priority ) {
107  dirs[n] = ( static_cast<int>( dir ) + 2 ) % 4;
108  if( open[n] != 0 ) {
109  while( nodes[i].top().pos != p ) {
110  nodes[1 - i].push( nodes[i].top() );
111  nodes[i].pop();
112  }
113  nodes[i].pop();
114  if( nodes[i].size() > nodes[1 - i].size() ) {
115  i = 1 - i;
116  }
117  while( !nodes[i].empty() ) {
118  nodes[1 - i].push( nodes[i].top() );
119  nodes[i].pop();
120  }
121  i = 1 - i;
122  }
123  open[n] = priority;
124  nodes[i].emplace( p, dir, priority );
125  }
126  }
127  }
128  return res;

258 int adjust_omt_cost( int base_cost, direction dir_in, direction dir_out )
259 {
260  // Adjust cost for 90-degree turns. We travel from the midpoint of one edge
261  // to the midpoint of an adjacent edge in a square, which is a diagonal
262  // line with length = sqrt(2) / 2 for a unit square.
263  if( dir_in != dir_out && is_horizontal( dir_in ) && is_horizontal( dir_out ) ) {
264  // Note: sqrt(2) is approximately equal to 99 / 70.
265  return base_cost * 99 / 140;
266  }
267  return base_cost;
268 }

360  }
361  }
362  }
363  return ret;
