479 {
480  widget *w = nullptr;
481  // Get the original widget from the widget factory.
482  for( const widget &wgt : widget::get_all() ) {
483  if( wgt.getId() == id ) {
484  w = const_cast<widget *>( &wgt );
485  break;
486  }
487  }
488  // None found, return 0 as the label width.
489  if( w == nullptr ) {
490  return 0;
491  } else if( w->_widgets.empty() ) {
492  // No more nested layouts, we've found an individual widget.
493  // Return the widget's label width, or 0 if the label or pad_labels is disabled.
494  return w->_label.empty() || w->has_flag( json_flag_W_LABEL_NONE ) ||
495  ! w->_pad_labels ? 0 : w->_label_width;
496  }
497  // If we get here, we have a layout that contains nested widgets.
498

502  // Dive deeper to retrieve the nested element's label width.
503  int tmpw = widget::finalize_label_width_recursive( wid );
504  if( tmpw > width ) {
505  width = tmpw;
506  }
507  }
508 
509  if( w->_pad_labels ) {
510  // Update this layout's label width to reflect the longest label within.
511  w->_label_width = width;
512  } else {
513  w->_label_width = 0;
514  }
515  return w->_label_width;
516 }

912 // Drawing function, provided as a callback to the window_panel constructor.
913 // Handles rendering a widget's content into a window panel.
914 static int custom_draw_func( const draw_args &args )
915 {
916  const avatar &u = args._ava;
917  const catacurses::window &w = args._win;
918  widget *wgt = args.get_widget();
919 
920  // Get full window width
921  const int width = catacurses::getmaxx( w );
922  // Leave 1 character space for margin on left and right
923  const int margin = 1;
924  const int widt = width - 2 * margin;
925  // Whether to subtract height lines from the drawn panel space
926  const bool disable_empty = wgt->has_flag( json_flag_W_DISABLED_WHEN_EMPTY );
927 
928  // Quit if there is nothing to draw or no space to draw it
929  if( wgt == nullptr || width <= 0 ) {
930  return 0;
931  }
932 
933  int height_diff = 0;
934  const bool skip_pad = wgt->has_flag( json_flag_W_NO_PADDING );
935 
936  werase( w );
937  if( wgt->_style == "sidebar" ) {
938  // noop
939  } else if( wgt->_style == "layout" ) {
940  if( wgt->_arrange == "rows" ) {
941  // Layout widgets in rows
942  std::vector<string_id<widget>> widgets = wgt->widgets( !wgt->_clauses.empty() );
943  int row_num = 0;
944  for( const widget_id &row_wid : widgets ) {
945  widget row_widget = row_wid.obj();
946 
947  const std::string txt = row_widget.layout( u, widt, wgt->_label_width,
948  skip_pad || row_wid->has_flag( json_flag_W_NO_PADDING ) );
949  if( row_wid->has_flag( json_flag_W_DISABLED_WHEN_EMPTY ) && txt.empty() ) {
950  // reclaim the skipped height in the sidebar
951  height_diff -= row_widget._height;
952  } else {
953  // draw normally
954  row_num = widget::custom_draw_multiline( txt, w, margin, widt, row_num );
955  }
956  }
957 
958  if( wgt->has_flag( json_flag_W_DYNAMIC_HEIGHT ) ) {
959  wgt->_height = row_num;
960  }
961  if( disable_empty && widgets.empty() ) {
962  // reclaim the skipped height in the sidebar
963  height_diff -= wgt->_height;
964  }
965  } else {
966  // Layout widgets in columns
967  // For now, this is the default when calling layout()
968  // So, just layout self on a single line
969 
970  const std::string txt = wgt->layout( u, widt, wgt->_label_width, skip_pad );
971  if( disable_empty && txt.empty() ) {
972  // reclaim the skipped height in the sidebar
973  height_diff -= wgt->_height;
974  } else {
975  // draw normally
976  widget::custom_draw_multiline( txt, w, margin, widt, 0 );
977  }
978  }
979  } else {
980  // No layout, just a widget
981  const std::string txt = wgt->layout( u, widt, 0, skip_pad );
982  if( disable_empty && txt.empty() ) {
983  // reclaim the skipped height in the sidebar
984  height_diff -= wgt->_height;
985  } else {
986  // draw normally
987  widget::custom_draw_multiline( txt, w, margin, widt, 0 );
988  }
989  }
990  wnoutrefresh( w );
991 
992  return height_diff;
993 }

995 window_panel widget::get_window_panel( const int width, const int req_height )
996 {
997  // Width is fixed, but height may vary depending on child widgets
998  int height = req_height;
999 
1000  // For layout with rows, height will be the combined
1001  // height of all child widgets.
1002  if( _style == "layout" && _arrange == "rows" ) {
1003  height = 0;
1004  for( const widget_id &wid : _widgets ) {
1005  height += wid->_height > 0 ? wid->_height : 1;
1006  }
1007  for( const widget_clause &clause : _clauses ) {
1008  int clause_height = 0;
1009  for( const widget_id &wid : clause.widgets ) {
1010  clause_height += wid->_height > 0 ? wid->_height : 1;
1011  }
1012  if( clause_height > height ) {
1013  height = clause_height;
1014  }
1015  }
1016  } else if( _style == "widget" || _style == "text" ) {
1017  height = _height > 1 ? _height : req_height;

1019  // Minimap and log do not have a predetermined height
1020  // (or they should allow caller to customize height)
1021 
1022  window_panel win( _label.translated(), _label, height, width,
1023  !has_flag( json_flag_W_DISABLED_BY_DEFAULT ) );
1024  win.set_widget( this->id );

1291 nc_color widget::value_color( int value )
1292 {
1293  // Maps the range of values from min to max to the list of "colors", 0-indexed
1294  // [ _var_min ... ( _var_norm ) ... _var_max ]
1295  // [ 0, 1, 2, ... ... num_colors-1 ]
1296  //
1297  if( _colors.empty() ) {
1298  return c_unset;
1299  }
1300  // Index of last color
1301  const int color_max = _colors.size() - 1;
1302  // Whether min, max, and normal are defined
1303  const bool min_max_defined = INT_MIN < _var_min && _var_max < INT_MAX;
1304  const bool normal_defined = INT_MIN < _var_norm.first && _var_norm.second < INT_MAX;
1305 
1306  // Get range of values from min to max
1307  const int var_range = _var_max - _var_min;
1308 
1309  if( ! _breaks.empty() ) {
1310  const int value_offset = ( 100 * ( value - _var_min ) ) / var_range;
1311  for( int i = 0; i < color_max; i++ ) {
1312  if( value_offset < _breaks[i] ) {
1313  return _colors[i];
1314  }
1315  }
1316  return _colors[color_max];
1317  }
1318 
1319  // Convert value to a positive offset within the range
1320  const int value_offset = std::max( value - _var_min, 0 );
1321 
1322  // If (min, max) is a valid nonzero range, fit value to a color using that range
1323  if( min_max_defined && var_range > 0 ) {
1324  // If value is within defined _var_norm range, the color is c_white
1325  if( normal_defined && _var_norm.first <= value && value <= _var_norm.second ) {
1326  return c_white;
1327  } else if( value <= _var_min ) {
1328  // If smaller than lower limit, return first color
1329  return _colors.front();
1330  } else if( value >= _var_max ) {
1331  // If larger than upper limit, return last color
1332  return _colors.back();
1333  } else {
1334  // If value is within the range, map it to an appropriate color
1335  // Scale value offset within range from [0, 1] to map color range
1336  const double scale = static_cast<double>( value_offset ) / var_range;
1337  // The +0.5 offset makes top and bottom color slots more equitable
1338  // (without the offset, only the max value gets the top color)
1339  const int color_index = std::floor( scale * color_max + 0.5 );
1340  return _colors[color_index];
1341  }
1342  }
1343  // No scaling by min-max range; assume colors map to 0, 1, 2 ...
1344  // Truncate below 0 and above color_max
1345  if( value < 0 ) {
1346  return _colors[0];
1347  } else if( value <= color_max ) {
1348  return _colors[value];
1349  } else {
1350  return _colors.back();
1351  }
1352 }
