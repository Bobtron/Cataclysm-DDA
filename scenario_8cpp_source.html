509 const profession *scenario::weighted_random_profession() const
510 {
511  // Strategy: 1/3 of the time, return the generic profession (if it's permitted).
512  // Otherwise, the weight of each permitted profession is 2 / ( |point cost| + 2 )
513  const auto choices = permitted_professions();
514  if( one_in( 3 ) && choices.front() == profession::generic()->ident() ) {
515  return profession::generic();
516  }
517 
518  while( true ) {
519  const string_id<profession> &candidate = random_entry_ref( choices );
520  if( candidate->can_pick().success() && x_in_y( 2, 2 + std::abs( candidate->point_cost() ) ) ) {
521  return &candidate.obj();
522  }
523  }
