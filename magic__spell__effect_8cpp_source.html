664 // Adds node to a search tree. Returns true if new node is allocated.
665 bool area_expander::enqueue( const tripoint &from, const tripoint &to, float cost )
666 {
667  if( contains( to ) ) {
668  // We will modify existing node if its cost is lower.
669  int index = area_search[to];
670  node &node = area[index];
671  if( cost < node.cost ) {
672  node.from = from;
673  node.cost = cost;
674  }
675  return false;
676  }
677  int index = area.size();
678  area.push_back( {to, from, cost} );
679  frontier.push( index );
680  area_search[to] = index;
681  return true;

684 // Run wave propagation
685 int area_expander::run( const tripoint &center )
686 {
687  enqueue( center, center, 0.0 );
688 
689  static constexpr std::array<int, 8> x_offset = {{ -1, 1, 0, 0, 1, -1, -1, 1 }};
690  static constexpr std::array<int, 8> y_offset = {{ 0, 0, -1, 1, -1, 1, -1, 1 }};
691 
692  // Number of nodes expanded.
693  int expanded = 0;
694 
695  map &here = get_map();
696  while( !frontier.empty() ) {
697  int best_index = frontier.top();
698  frontier.pop();
699 
700  for( size_t i = 0; i < 8; i++ ) {
701  node &best = area[best_index];
702  const tripoint &pt = best.position + point( x_offset[ i ], y_offset[ i ] );
703 
704  if( here.impassable( pt ) ) {
705  continue;
706  }
707 
708  float center_range = static_cast<float>( rl_dist( center, pt ) );
709  if( max_range > 0 && center_range > max_range ) {
710  continue;
711  }
712 
713  if( max_expand > 0 && expanded > max_expand && contains( pt ) ) {
714  continue;
715  }
716 
717  float delta_range = trig_dist( best.position, pt );
718 
719  if( enqueue( best.position, pt, best.cost + delta_range ) ) {
720  expanded++;
721  }
722  }
723  }
724  return expanded;
725 }

983 {

985  // this group of variables is for deferring movement of the avatar
986  int pushed_distance = 0;
987  tripoint push_to;
988  std::vector<tripoint> pushed_vec;
989  bool player_pushed = false;
990 
991  ::map &here = get_map();
992 
993  // whether it's push or pull, so how the multimap is sorted
994  // -1 is push and 1 is pull
995  const int sign = sp.damage( caster ) > 0 ? -1 : 1;
996 
997  std::multimap<int, tripoint> targets_ordered_by_range;
998  for( const tripoint &pt : area ) {
999  targets_ordered_by_range.emplace( sign * rl_dist( pt, caster.pos() ), pt );
1000  }
1001 
1002  creature_tracker &creatures = get_creature_tracker();
1003  for( const std::pair<const int, tripoint> &pair : targets_ordered_by_range ) {
1004  const tripoint &push_point = pair.second;
1005  const units::angle angle = coord_to_angle( caster.pos(), target );
1006  // positive is push, negative is pull
1007  int push_distance = sp.damage( caster );
1008  const int prev_distance = rl_dist( caster.pos(), target );
1009  if( push_distance < 0 ) {
1010  push_distance = std::max( -std::abs( push_distance ), -std::abs( prev_distance ) );
1011  }
1012  if( push_distance == 0 ) {
1013  continue;
1014  }

1017  calc_ray_end( angle, push_distance, push_point, push_dest );

1019 
1020  const Creature *critter = creatures.creature_at<Creature>( push_point );
1021  if( critter != nullptr ) {
1022  const Creature::Attitude attitude_to_target =
1023  caster.attitude_to( *creatures.creature_at<Creature>( push_point ) );
1024 
1025  monster *mon = creatures.creature_at<monster>( push_point );
1026  npc *guy = creatures.creature_at<npc>( push_point );
