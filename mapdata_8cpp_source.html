533 template<typename C, typename F>
534 void load_season_array( const JsonObject &jo, const std::string &key, const std::string &context,
535  C &container, F load_func )
536 {
537  if( jo.has_string( key ) ) {
538  container.fill( load_func( jo.get_string( key ) ) );
539 
540  } else if( jo.has_array( key ) ) {
541  JsonArray arr = jo.get_array( key );
542  if( arr.size() == 1 ) {
543  container.fill( load_func( arr.get_string( 0 ) ) );
544 
545  } else if( arr.size() == container.size() ) {
546  for( auto &e : container ) {
547  e = load_func( arr.next_string() );
548  }
549 
550  } else {
551  jo.throw_error_at( key, "Incorrect number of entries" );
552  }
553 
554  } else if( jo.has_member( key ) ) {
555  jo.throw_error_at(
556  key, string_format( "Expected '%s' member to be string or array", key ) );
557  } else {
558  jo.throw_error(
559  string_format( "Expected '%s' member in %s but none was found", key, context ) );
560  }

984 void ter_t::check() const
985 {
986  map_data_common_t::check();
987  check_bash_items( bash, id.str(), true );
988  check_decon_items( deconstruct, id.str(), true );
989 
990  if( !transforms_into.is_valid() ) {
991  debugmsg( "invalid transforms_into %s for %s", transforms_into.c_str(), id.c_str() );
992  }
993 
994  // Validate open/close transforms
995  if( !open.is_valid() ) {
996  debugmsg( "invalid terrain %s for opening %s", open.c_str(), id.c_str() );
997  }
998  if( !close.is_valid() ) {
999  debugmsg( "invalid terrain %s for closing %s", close.c_str(), id.c_str() );
1000  }
1001  // Check transition consistency for opening/closing terrain. Has an obvious
1002  // exception for locked terrains - those aren't expected to be locked again
1003  if( open && open->close && open->close != id && !has_flag( ter_furn_flag::TFLAG_LOCKED ) ) {
1004  debugmsg( "opening terrain %s for %s doesn't reciprocate", open.c_str(), id.c_str() );
1005  }
1006  if( close && close->open && close->open != id && !has_flag( ter_furn_flag::TFLAG_LOCKED ) ) {
1007  debugmsg( "closing terrain %s for %s doesn't reciprocate", close.c_str(), id.c_str() );
1008  }
1009 
1010  // Validate curtain transforms
1011  if( has_examine( iexamine::curtains ) && !has_curtains() ) {
1012  debugmsg( "%s is a curtain, but has no curtain_transform", id.c_str() );
1013  }
1014  if( !has_examine( iexamine::curtains ) && has_curtains() ) {
1015  debugmsg( "%s is not a curtain, but has curtain_transform", id.c_str() );
1016  }
1017  if( !curtain_transform.is_empty() && !curtain_transform.is_valid() ) {
1018  debugmsg( "%s has invalid curtain transform target %s", id.c_str(), curtain_transform.c_str() );
1019  }

1021  // Validate generic transforms
1022  if( transforms_into && transforms_into == id ) {
1023  debugmsg( "%s transforms_into itself", id.c_str() );
1024  }
1025 
1026  for( const emit_id &e : emissions ) {
1027  if( !e.is_valid() ) {
1028  debugmsg( "ter %s has invalid emission %s set", id.c_str(), e.str().c_str() );
1029  }
1030  }
1031 }
