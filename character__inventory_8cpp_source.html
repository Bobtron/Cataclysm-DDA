739 bool Character::dispose_item( item_location &&obj, const std::string &prompt )
740 {
741  uilist menu;
742  menu.text = prompt.empty() ? string_format( _( "Dispose of %s" ), obj->tname() ) : prompt;
743  std::vector<dispose_option> opts;
744 
745  const bool bucket = obj->will_spill() && !obj->is_container_empty();
746 
747  opts.emplace_back( dispose_option{
748  bucket ? _( "Spill contents and store in inventory" ) : _( "Store in inventory" ),
749  can_stash( *obj ), '1',
750  item_handling_cost( *obj ),
751  [this, bucket, &obj] {
752  if( bucket && !obj->spill_open_pockets( *this, obj.get_item() ) )
753  {
754  return false;
755  }
756 
757  mod_moves( -item_handling_cost( *obj ) );
758  this->i_add( *obj, true, &*obj, &*obj );
759  obj.remove_item();

761  }
762  } );
763 
764  opts.emplace_back( dispose_option{
765  _( "Drop item" ), true, '2', 0, [this, &obj] {
766  put_into_vehicle_or_drop( *this, item_drop_reason::deliberate, { *obj } );
767  obj.remove_item();
768  return true;
769  }
770  } );
771 
772  opts.emplace_back( dispose_option{
773  bucket ? _( "Spill contents and wear item" ) : _( "Wear item" ),
774  can_wear( *obj ).success(), '3', item_wear_cost( *obj ),
775  [this, bucket, &obj] {
776  if( bucket && !obj->spill_contents( *this ) )
777  {
778  return false;
779  }
780 
781  item it = *obj;
782  obj.remove_item();
783  return !!wear_item( it );
784  }
785  } );
786 
787  worn.holster_opts( opts, obj, *this );
788 
789  int w = utf8_width( menu.text, true ) + 4;
790  for( const dispose_option &e : opts ) {
791  w = std::max( w, utf8_width( e.prompt, true ) + 4 );
792  }
793  for( dispose_option &e : opts ) {
794  e.prompt += std::string( w - utf8_width( e.prompt, true ), ' ' );
795  }
796 
797  menu.text.insert( 0, 2, ' ' ); // add space for UI hotkeys
798  menu.text += std::string( w + 2 - utf8_width( menu.text, true ), ' ' );
799  menu.text += _( " | Moves " );
800 
801  for( const dispose_option &e : opts ) {
802  menu.addentry( -1, e.enabled, e.invlet, string_format( e.enabled ? "%s | %-7d" : "%s |",
803  e.prompt, e.moves ) );
804  }
805 
806  menu.query();
807  if( menu.ret >= 0 ) {
