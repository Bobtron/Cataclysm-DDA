798 bool input_context::resolve_conflicts( const std::vector<input_event> &events,
799  const std::string &ignore_action )
800 {
801  // FIND CONFLICTS
802  std::map<std::string, std::vector<input_event>> conflicting_actions_events;
803  // for events we want to add
804  for( const input_event &event : events ) {
805  const std::string conflicting_action = get_conflicts( event, ignore_action );
806  const bool has_conflicts = !conflicting_action.empty();
807  if( has_conflicts ) {
808  conflicting_actions_events[conflicting_action].emplace_back( event );
809  }
810  }
811  // HUMAN READABLE
812  std::string conflict_list;
813  for( const auto &[action, events] : conflicting_actions_events ) {
814  conflict_list += action + ": ";
815  std::string sep;
816  for( const input_event &event : events ) {
817  conflict_list += sep + event.long_description();
818  sep = ", ";
819  }
820  conflict_list += "\n";
821  }
822  // RESOLVE CONFLICTS
823  if( !conflicting_actions_events.empty() &&
824  !query_yn(
825  _( "Conflict. The following keybinding(s) will be removed from the respective action(s):\n" )
826  + conflict_list + "\n"
827  + _( "Continue?" ) )
828  ) {
829  return false;
830  }
831  for( const auto &action_events : conflicting_actions_events ) {
832  for( const input_event &event : action_events.second ) {
833  clear_conflicting_keybindings( event, ignore_action );
834  }
835  }
836  return true;
837 }

839 bool input_context::action_reset( const std::string &action_id )
840 {
841  // FIND CONFLICTS
842  std::vector<input_event> conflicting_events;
843  std::array<std::reference_wrapper<const std::string>, 2> contexts = { default_context_id, category };
844  for( const std::string &context : contexts ) {
845  const input_manager::t_actions &def = inp_mngr.basic_action_contexts.at( context );
846 
847  bool is_in_def = def.find( action_id ) != def.end();
848  if( is_in_def ) {
849  for( const input_event &event : def.at( action_id ).input_events ) {
850  conflicting_events.emplace_back( event );
851  }
852  }
853  }
854  if( !resolve_conflicts( conflicting_events, action_id ) ) {
855  return false;
856  }
857 
858  // RESET KEY BINDINGS
859  for( const std::string &context : contexts ) {
860  const input_manager::t_actions &def = inp_mngr.basic_action_contexts.at( context );
861  const input_manager::t_actions &cus = inp_mngr.action_contexts.at( context );
862

864  bool is_in_cus = cus.find( action_id ) != cus.end();
865

867  inp_mngr.remove_input_for_action( action_id, context );

870  if( is_in_def ) {
871  for( const input_event &event : def.at( action_id ).input_events ) {
