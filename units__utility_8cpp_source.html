49 int angle_to_dir8( const units::angle direction )
50 {
51  // Direction is snapped to "notches" of 15-degrees on a circle, we bump direction towards the
52  // middle of each notch, and lookup the 8-way direction. Cardinal directions get 5 "notches", diagonals
53  // get 1 at the boundary of each 2 cardinals. This can be shorter but this way is easier to read,
54  // the compilers will optimize into lookup table. This is mostly used in curses mode.
55  constexpr units::angle bump = 15_degrees / 2.0;
56  const int snap = normalize( direction - bump ) / 15_degrees + 1;
57  // NOLINTBEGIN(bugprone-branch-clone)
58  switch( snap ) {
59  // *INDENT-OFF*
60  case 22: return 0; // north
61  case 23: return 0; // north
62  case 24: return 0; // north ( 0 degrees )
63  case 1: return 0; // north
64  case 2: return 0; // north
65  case 3: return 1; // north-west ( 45 degrees )
66  case 4: return 2; // west
67  case 5: return 2; // west
68  case 6: return 2; // west ( 90 degrees )
69  case 7: return 2; // west
70  case 8: return 2; // west
71  case 9: return 3; // south-west ( 135 degrees )
72  case 10: return 4; // south
