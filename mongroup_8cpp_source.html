129 static bool is_spawn_valid(
130  const MonsterGroupEntry &entry, const time_point &sunset, const time_point &sunrise,
131  const season_type season, const bool can_spawn_events )
132 {
133  // If an event was specified for this entry, check if it matches the current holiday
134  if( entry.event != holiday::none && ( !can_spawn_events ||
135  entry.event != get_holiday_from_time() ) ) {
136  return false;
137  }
138 
139  //Insure that the time is not before the spawn first appears or after it stops appearing
140  if( calendar::turn < calendar::start_of_cataclysm + entry.starts ) {
141  return false;
142  }
143  if( !entry.lasts_forever() && calendar::turn >= calendar::start_of_cataclysm + entry.ends ) {
144  return false;
145  }
146 
147  std::vector<std::pair<time_point, time_point> > valid_times_of_day;
148  bool season_limited = false;
149  bool season_matched = false;
150  //Collect the various spawn conditions, and then insure they are met appropriately
151  for( const std::string &elem : entry.conditions ) {
152  //Collect valid time of day ranges
153  if( elem == "DAY" || elem == "NIGHT" || elem == "DUSK" || elem == "DAWN" ) {
154  if( elem == "DAY" ) {
155  valid_times_of_day.emplace_back( sunrise, sunset );
156  } else if( elem == "NIGHT" ) {
157  valid_times_of_day.emplace_back( sunset, sunrise );
158  } else if( elem == "DUSK" ) {
159  valid_times_of_day.emplace_back( sunset - 1_hours, sunset + 1_hours );
160  } else if( elem == "DAWN" ) {
161  valid_times_of_day.emplace_back( sunrise - 1_hours, sunrise + 1_hours );
162  }
163  }
164 
165  //If we have any seasons listed, we know to limit by season, and if any season matches this season, we are good to spawn
166  if( elem == "SUMMER" || elem == "WINTER" || elem == "SPRING" || elem == "AUTUMN" ) {
167  season_limited = true;
168  if( ( season == SUMMER && elem == "SUMMER" ) ||
169  ( season == WINTER && elem == "WINTER" ) ||
170  ( season == SPRING && elem == "SPRING" ) ||
171  ( season == AUTUMN && elem == "AUTUMN" ) ) {
172  season_matched = true;
173  }
174  }
175  }
176 
177  //Make sure the current time of day is within one of the valid time ranges for this spawn
178  bool is_valid_time_of_day = false;
179  if( valid_times_of_day.empty() ) {
180  //Then it can spawn whenever, since no times were defined
181  is_valid_time_of_day = true;
182  } else {
183  //Otherwise, it's valid if it matches any of the times of day
184  for( auto &elem : valid_times_of_day ) {
185  if( calendar::turn > elem.first && calendar::turn < elem.second ) {
186  is_valid_time_of_day = true;
187  }
188  }
189  }
190  if( !is_valid_time_of_day ) {
191  return false;
192  }
193 
194  //If we are limited by season, make sure we matched a season
195  if( season_limited && !season_matched ) {
196  return false;
197  }
198 
199  return true;
200 }

204 std::vector<MonsterGroupResult> MonsterGroupManager::GetResultFromGroup(

206  bool *returned_default, bool use_pack_size )

210  //Our spawn details specify, by default, a single instance of the default monster
211  std::vector<MonsterGroupResult> spawn_details;
212 
213  bool monster_found = false;
214  // Loop invariant values
215  const time_point sunset = ::sunset( calendar::turn );
216  const time_point sunrise = ::sunrise( calendar::turn );
217  const season_type season = season_of_year( calendar::turn );
218  std::string opt = get_option<std::string>( "EVENT_SPAWNS" );
219  const bool can_spawn_events = opt == "monsters" || opt == "both";
220 
221  // Step through spawn definitions from the monster group until one is found or
222  for( const MonsterGroupEntry &entry : group.monsters ) {
223  if( !is_spawn_valid( entry, sunset, sunrise, season, can_spawn_events ) ) {
224  continue;
225  }
226

228  if( entry.frequency < spawn_chance ) {
229  spawn_chance -= entry.frequency;
230  continue;
231  }

235

238  for( int i = 0; i < pack_size; i++ ) {

240  GetResultFromGroup( entry.group, quantity, mon_found, true );

242  }
243  } else if( use_pack_size ) {

245  spawn_details.emplace_back( entry.name, pack_size, entry.data );

247  // value as a side effect. We will reduce it by the spawn rule's cost multiplier.
248  if( quantity ) {
249  *quantity -= std::max( 1, entry.cost_multiplier * pack_size );
250  }
251  }
252  } else {
253  spawn_details.emplace_back( entry.name, pack_size, entry.data );
254  // And if a quantity pointer with remaining value was passed, will modify the external
255  // value as a side effect. We will reduce it by the spawn rule's cost multiplier.
256  if( quantity ) {
257  *quantity -= std::max( 1, entry.cost_multiplier * pack_size );
258  }
259  }
260  monster_found = true;
261  break;
262  }
263 
264  // Force quantity to decrement regardless of whether we found a monster.
265  if( quantity && !monster_found && !is_recursive ) {
266  ( *quantity )--;
267  }
268  if( mon_found ) {
269  ( *mon_found ) = monster_found;
270  }
271 
272  if( !is_recursive && spawn_details.empty() ) {
273  spawn_details.emplace_back( group.defaultMonster, 1, spawn_data() );
274  if( returned_default ) {
275  ( *returned_default ) = true;
276  }
277  }
278 
279  return spawn_details;
280 }
