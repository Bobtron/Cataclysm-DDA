93 void move_mode::finalize()

99  // Sort it fastest to slowest

102  };
103

105 
106  // Cycle to the move mode above ours

108  const move_mode &curr = *move_modes_sorted[i - 1];
109  if( i == move_modes_sorted.size() ) {
110  curr.set_cycle( move_modes_sorted[0] );

112  curr.set_cycle( move_modes_sorted[i] );
113  }
114  }

116  // Cycle to the move mode below ours

119  if( i == 1 ) {
