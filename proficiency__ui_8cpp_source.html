70 void prof_window::filter()
71 {
72  enum class _prof_filter_prefix {
73  LEARNED,
74  UNLEARNED,
75  AS_PROGRESS,
76  NONE
77  };
78  std::string qry = trim( filter_str );
79  filtered_profs.clear();
80  if( qry.empty() ) {
81  return;
82  }
83  _prof_filter_prefix prefix = _prof_filter_prefix::NONE;
84  if( qry.size() > 2 && qry[1] == ':' ) {
85  switch( qry[0] ) {
86  case 'l':
87  prefix = _prof_filter_prefix::LEARNED;
88  qry = qry.substr( 2 );
89  break;
90  case 'i':
91  prefix = _prof_filter_prefix::UNLEARNED;
92  qry = qry.substr( 2 );
93  break;
94  case 'p':
95  prefix = _prof_filter_prefix::AS_PROGRESS;
96  qry = qry.substr( 2 );
97  break;
98  default:
99  break;
100  }
101  }
102  float prog_val = 0.0f;
103  if( prefix == _prof_filter_prefix::AS_PROGRESS ) {
104  try {
105  prog_val = std::stof( qry );
106  } catch( const std::invalid_argument &e ) {
107  // User mistyped query. Not severe enough for debugmsg.
108  DebugLog( DebugLevel::D_WARNING, DebugClass::D_GAME ) <<
109  "Malformed proficiency query \"" << filter_str << "\": " << e.what();
110  return;
111  }
112  }
113  for( display_prof_deps &dp : all_profs ) {
114  if( prefix == _prof_filter_prefix::AS_PROGRESS ) {
115  if( prog_val < dp.first.practice * 100.0f ) {
116  filtered_profs.push_back( &dp );
117  }
118  continue;
119  }
120  if( ( prefix == _prof_filter_prefix::LEARNED && !dp.first.known ) ||
121  ( prefix == _prof_filter_prefix::UNLEARNED && dp.first.known ) ) {
122  continue;
123  }
124  if( qry == "*" || lcmatch( dp.first.id->name(), qry ) ) {
125  filtered_profs.push_back( &dp );
126  }
127  }
