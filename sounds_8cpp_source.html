248 // This is an attempt to handle attenuation of sound for underground areas.
249 // The main issue it addresses is that you can hear activity
250 // relatively deep underground while on the surface.
251 // My research indicates that attenuation through soil-like materials is as
252 // high as 100x the attenuation through air, plus vertical distances are
253 // roughly five times as large as horizontal ones.
254 static int sound_distance( const tripoint &source, const tripoint &sink )
255 {
256  const int lower_z = std::min( source.z, sink.z );
257  const int upper_z = std::max( source.z, sink.z );
258  const int vertical_displacement = upper_z - lower_z;
259  int vertical_attenuation = vertical_displacement;
260  if( lower_z < 0 && vertical_displacement > 0 ) {
261  // Apply a moderate bonus attenuation (5x) for the first level of vertical displacement.
262  vertical_attenuation += 4;
263  // At displacements greater than one, apply a large additional attenuation (100x) per level.
264  const int underground_displacement = std::min( -lower_z, vertical_displacement );
265  vertical_attenuation += ( underground_displacement - 1 ) * 20;
266  }
267  // Regardless of underground effects, scale the vertical distance by 5x.
268  vertical_attenuation *= 5;
269  return rl_dist( source.xy(), sink.xy() ) + vertical_attenuation;
270 }

378 static std::vector<centroid> cluster_sounds( std::vector<std::pair<tripoint, monster_sound_event>>
379  input_sounds )
380 {
381  // If there are too many monsters and too many noise sources (which can be monsters, go figure),
382  // applying sound events to monsters can dominate processing time for the whole game,
383  // so we cluster sounds and apply the centroids of the sounds to the monster AI
384  // to fight the combinatorial explosion.
385  std::vector<centroid> sound_clusters;
386  if( input_sounds.empty() ) {
387  return sound_clusters;
388  }
389  const int num_seed_clusters =
390  std::max( std::min( input_sounds.size(), static_cast<size_t>( 10 ) ),
391  static_cast<size_t>( std::log( input_sounds.size() ) ) );
392  const size_t stopping_point = input_sounds.size() - num_seed_clusters;
393  const size_t max_map_distance = sound_distance( tripoint( point_zero, OVERMAP_DEPTH ),
394  tripoint( MAPSIZE_X, MAPSIZE_Y, OVERMAP_HEIGHT ) );
395  // Randomly choose cluster seeds.
396  for( size_t i = input_sounds.size(); i > stopping_point; i-- ) {
397  size_t index = rng( 0, i - 1 );
398  // The volume and cluster weight are the same for the first element.
399  sound_clusters.push_back(
400  // Assure the compiler that these int->float conversions are safe.
401  {
402  static_cast<float>( input_sounds[index].first.x ), static_cast<float>( input_sounds[index].first.y ),
403  static_cast<float>( input_sounds[index].first.z ),
404  static_cast<float>( input_sounds[index].second.volume ), static_cast<float>( input_sounds[index].second.volume ),
405  input_sounds[index].second.provocative
406  } );
407  vector_quick_remove( input_sounds, index );
408  }
409  for( const auto &sound_event_pair : input_sounds ) {
410  auto found_centroid = sound_clusters.begin();
411  float dist_factor = max_map_distance;
412  const auto cluster_end = sound_clusters.end();
413  for( auto centroid_iter = sound_clusters.begin(); centroid_iter != cluster_end;
414  ++centroid_iter ) {
415  // Scale the distance between the two by the max possible distance.
416  tripoint centroid_pos { static_cast<int>( centroid_iter->x ), static_cast<int>( centroid_iter->y ), static_cast<int>( centroid_iter->z ) };
417  const int dist = sound_distance( sound_event_pair.first, centroid_pos );
418  if( dist * dist < dist_factor ) {
419  found_centroid = centroid_iter;
420  dist_factor = dist * dist;
421  }
422  }
423  const float volume_sum = static_cast<float>( sound_event_pair.second.volume ) +
424  found_centroid->weight;
425  // Set the centroid location to the average of the two locations, weighted by volume.
426  found_centroid->x = static_cast<float>( ( sound_event_pair.first.x *
427  sound_event_pair.second.volume ) +
428  ( found_centroid->x * found_centroid->weight ) ) / volume_sum;
429  found_centroid->y = static_cast<float>( ( sound_event_pair.first.y *
430  sound_event_pair.second.volume ) +
431  ( found_centroid->y * found_centroid->weight ) ) / volume_sum;
432  found_centroid->z = static_cast<float>( ( sound_event_pair.first.z *
433  sound_event_pair.second.volume ) +
434  ( found_centroid->z * found_centroid->weight ) ) / volume_sum;
435  // Set the centroid volume to the larger of the volumes.
436  found_centroid->volume = std::max( found_centroid->volume,
437  static_cast<float>( sound_event_pair.second.volume ) );
438  // Set the centroid weight to the sum of the weights.
439  found_centroid->weight = volume_sum;
440  // Set and keep provocative if any sound in the centroid is provocative
441  found_centroid->provocative |= sound_event_pair.second.provocative;
442  }
443  return sound_clusters;
444 }

446 static int get_signal_for_hordes( const centroid &centr )
447 {
448  //Volume in tiles. Signal for hordes in submaps
449  //modify vol using weather vol.Weather can reduce monster hearing
450  const int vol = centr.volume - get_weather().weather_id->sound_attn;
451  const int min_vol_cap = 60; //Hordes can't hear volume lower than this
452  const int underground_div = 2; //Coefficient for volume reduction underground
453  const int hordes_sig_div = SEEX; //Divider coefficient for hordes
454  const int min_sig_cap = 8; //Signal for hordes can't be lower that this if it pass min_vol_cap
455  const int max_sig_cap = 26; //Signal for hordes can't be higher that this
456  //Lower the level - lower the sound
457  int vol_hordes = ( ( centr.z < 0 ) ? vol / ( underground_div * std::abs( centr.z ) ) : vol );
458  if( vol_hordes > min_vol_cap ) {
459  //Calculating horde hearing signal
460  int sig_power = std::ceil( static_cast<float>( vol_hordes ) / hordes_sig_div );
461  //Capping minimum horde hearing signal
462  sig_power = std::max( sig_power, min_sig_cap );
463  //Capping extremely high signal to hordes
464  sig_power = std::min( sig_power, max_sig_cap );
465  add_msg_debug( debugmode::DF_SOUND, "vol %d vol_hordes %d sig_power %d ", vol, vol_hordes,
466  sig_power );
467  return sig_power;
468  }
469  return 0;
470 }

476  for( const centroid &this_centroid : sound_clusters ) {
477  // Since monsters don't go deaf ATM we can just use the weather modified volume
478  // If they later get physical effects from loud noises we'll have to change this
479  // to use the unmodified volume for those effects.
480  const int vol = this_centroid.volume - weather_vol;
481  const tripoint source = tripoint( this_centroid.x, this_centroid.y, this_centroid.z );
482  // --- Monster sound handling here ---
483  // Alert all hordes
484  int sig_power = get_signal_for_hordes( this_centroid );
485  if( sig_power > 0 ) {
486

488  // TODO: fix point types
489  const point_abs_sm abs_sm( ms_to_sm_copy( abs_ms ) );
490  const tripoint_abs_sm target( abs_sm, source.z );
491  overmap_buffer.signal_hordes( target, sig_power );
492  }
493  // Alert all monsters (that can hear) to the sound.
494  for( monster &critter : g->all_monsters() ) {
495  // TODO: Generalize this to Creature::hear_sound
496  const int dist = sound_distance( source, critter.pos() );
497  if( vol * 2 > dist ) {
498  // Exclude monsters that certainly won't hear the sound
499  critter.hear_sound( source, vol, dist, this_centroid.provocative );

501  }

507  // Exclude traps that certainly won't hear the sound
508  if( vol * 2 > dist ) {
509  if( tr.triggered_by_sound( vol, dist ) ) {
510  tr.trigger( tp );
511  }
512  }
513  }
514  }
515  }
516  recent_sounds.clear();

569 void sounds::process_sound_markers( Character *you )
570 {
571  bool is_deaf = you->is_deaf();
572  const float volume_multiplier = you->hearing_ability();
573  const int weather_vol = get_weather().weather_id->sound_attn;
574  // NOLINTNEXTLINE(modernize-loop-convert)
575  for( std::size_t i = 0; i < sounds_since_last_turn.size(); i++ ) {
576  // copy values instead of making references here to fix use-after-free error
577  // sounds_since_last_turn may be inserted with new elements inside the loop
578  // so the references may become invalid after the vector enlarged its internal buffer
579  const tripoint pos = sounds_since_last_turn[i].first;
580  const sound_event sound = sounds_since_last_turn[i].second;
581  const int distance_to_sound = sound_distance( you->pos(), pos );
582  const int raw_volume = sound.volume;
583 
584  // The felt volume of a sound is not affected by negative multipliers, such as already
585  // deafened players or players with sub-par hearing to begin with.
586  int felt_volume = static_cast<int>( raw_volume * std::min( 1.0f,
587  volume_multiplier ) ) - distance_to_sound;
588  if( you->has_flag( json_flag_HEARING_PROTECTION ) ) {
589  felt_volume /= 2;
590  }
591  // Deafening is based on the felt volume, as a player may be too deaf to
592  // hear the deafening sound but still suffer additional hearing loss.
593  const bool is_sound_deafening = rng( felt_volume / 2, felt_volume ) >= 150;
594 
595  // Deaf players hear no sound, but still are at risk of additional hearing loss.
596  if( is_deaf ) {
597  if( is_sound_deafening && !you->is_immune_effect( effect_deaf ) ) {
598  you->add_effect( effect_deaf, std::min( 4_minutes,
599  time_duration::from_turns( felt_volume - 130 ) / 8 ) );
600  if( !you->has_flag( json_flag_PAIN_IMMUNE ) ) {
601  you->add_msg_if_player( m_bad, _( "Your eardrums suddenly ache!" ) );
602  if( you->get_pain() < 10 ) {
603  you->mod_pain( rng( 0, 2 ) );
604  }
605  }
606  }
607  continue;
608  }
609 
610  if( is_sound_deafening && !you->is_immune_effect( effect_deaf ) ) {
611  const time_duration deafness_duration = time_duration::from_turns( felt_volume - 130 ) / 4;
612  you->add_effect( effect_deaf, deafness_duration );
613  if( you->is_deaf() && !is_deaf ) {
614  is_deaf = true;
615  continue;
616  }
617  }
618 
619  // The heard volume of a sound is the player heard volume, regardless of true volume level.
620  const int heard_volume = static_cast<int>( ( raw_volume - weather_vol ) *
621  volume_multiplier ) - distance_to_sound;
622 
623  if( heard_volume <= 0 && pos != you->pos() ) {
624  continue;
625  }
626 
627  // Player volume meter includes all sounds from their tile and adjacent tiles
628  if( distance_to_sound <= 1 ) {
629  you->volume = std::max( you->volume, heard_volume );
630  }
631 
632  // Noises from vehicle player is in.
633  if( you->controlling_vehicle ) {
634  vehicle *veh = veh_pointer_or_null( get_map().veh_at( you->pos() ) );
635  const int noise = veh ? static_cast<int>( veh->vehicle_noise ) : 0;
636 
637  you->volume = std::max( you->volume, noise );
638  }
639 
640  // Secure the flag before wake_up() clears the effect
641  bool slept_through = you->has_effect( effect_slept_through_alarm );
642  // See if we need to wake someone up
643  if( you->has_effect( effect_sleep ) ) {
644  if( ( ( !( you->has_trait( trait_HEAVYSLEEPER ) ||
645  you->has_trait( trait_HEAVYSLEEPER2 ) ) && dice( 2, 15 ) < heard_volume ) ||
646  ( you->has_trait( trait_HEAVYSLEEPER ) && dice( 3, 15 ) < heard_volume ) ||
647  ( you->has_trait( trait_HEAVYSLEEPER2 ) && dice( 6, 15 ) < heard_volume ) ) &&
648  !you->has_effect( effect_narcosis ) &&
649  !you->has_bionic( bio_sleep_shutdown ) ) {
650  //Not kidding about sleep-through-firefight
651  you->wake_up();
652  you->add_msg_if_player( m_warning, _( "Something is making noise." ) );
653  } else {
654  continue;
655  }
656  }
657  const std::string &description = sound.description.empty() ? _( "a noise" ) : sound.description;
658  if( you->is_npc() ) {
659  if( !sound.ambient ) {
660  npc *guy = dynamic_cast<npc *>( you );
661  guy->handle_sound( sound.category, description, heard_volume, pos );
662  }
663  continue;
664  }
665 
666  if( sound.category == sound_t::music ) {
667  music::activate_music_id( music::music_id::sound );
668  }
669 
670  // don't print our own noise or things without descriptions
671  if( !sound.ambient && ( pos != you->pos() ) && !get_map().pl_sees( pos, distance_to_sound ) ) {
672  if( uistate.distraction_noise &&
673  !you->activity.is_distraction_ignored( distraction_type::noise ) &&
674  !get_safemode().is_sound_safe( sound.description, distance_to_sound, you->controlling_vehicle ) ) {
675  const std::string query = string_format( _( "Heard %s!" ),
676  trim_trailing_punctuations( description ) );
677  g->cancel_activity_or_ignore_query( distraction_type::noise, query );
678  }
679  }
680 
681  // skip some sounds to avoid message spam
682  const bool from_player = pos == you->pos() || ( sound.category == sound_t::movement &&
683  distance_to_sound <= 1 );
684  if( describe_sound( sound.category, from_player ) ) {
685  game_message_type severity = m_info;
686  if( sound.category == sound_t::combat || sound.category == sound_t::alarm ) {
687  severity = m_warning;
688  }
689  // if we can see it, don't print a direction
690  if( pos == you->pos() ) {
691  add_msg( severity, _( "From your position you hear %1$s" ), description );
692  } else if( you->sees( pos ) ) {
693  add_msg( severity, _( "You hear %1$s" ), description );
694  } else {
695  std::string direction = direction_name( direction_from( you->pos(), pos ) );
696  add_msg( severity, _( "From the %1$s you hear %2$s" ), direction, description );
697  }
698  }
699 
700  if( !you->has_effect( effect_sleep ) && you->has_effect( effect_alarm_clock ) &&
701  !you->has_flag( STATIC( json_character_flag( "ALARMCLOCK" ) ) ) ) {
702  // if we don't have effect_sleep but we're in_sleep_state, either
703  // we were trying to fall asleep for so long our alarm is now going
704  // off or something disturbed us while trying to sleep
705  const bool trying_to_sleep = you->in_sleep_state();
706  if( you->get_effect( effect_alarm_clock ).get_duration() == 1_turns ) {
707  if( slept_through ) {
708  add_msg( _( "Your alarm clock finally wakes you up." ) );
709  } else if( !trying_to_sleep ) {
710  add_msg( _( "Your alarm clock wakes you up." ) );
711  } else {
712  add_msg( _( "Your alarm clock goes off and you haven't slept a wink." ) );
713  you->activity.set_to_null();
714  }
715  add_msg( _( "You turn off your alarm-clock." ) );
716  you->get_effect( effect_alarm_clock ).set_duration( 0_turns );
717  }
718  }
719 
720  const std::string &sfx_id = sound.id;
721  const std::string &sfx_variant = sound.variant;
722  const std::string &sfx_season = sound.season;
723  const bool indoors = !is_creature_outside( get_player_character() );
724  const bool night = is_night( calendar::turn );
725  if( !sfx_id.empty() ) {
726  sfx::play_variant_sound( sfx_id, sfx_variant, sfx_season, indoors, night,
727  sfx::get_heard_volume( pos ) );
728  }
729 
730  // Place footstep markers.
731  if( pos == you->pos() || ( you->sees( pos ) && ( sound.category != sound_t::sensory ) ) ) {
732  // If we are or can see the source, don't draw a marker, except for sonar etc
733  continue;
734  }
735 
736  int err_offset;
737 
738  if( ( heard_volume + distance_to_sound ) / distance_to_sound < 2 ) {
739  err_offset = rng( 0, 3 );
740  } else if( ( heard_volume + distance_to_sound ) / distance_to_sound < 3 ) {
741  err_offset = rng( 0, 2 );
742  } else {
743  err_offset = rng( 0, 1 );
744  }
745 
746  // Echolocation has to be fairly precise or it's worse than useless.
747  // However, it is never perfect.
748  if( sound.category == sound_t::sensory ) {
749  if( ( heard_volume + distance_to_sound ) / distance_to_sound < 2 ) {
750  err_offset = rng( 0, 3 );
751  } else if( ( heard_volume + distance_to_sound ) / distance_to_sound < 3 ) {
752  err_offset = rng( 0, 2 );
753  } else {
754  if( one_in( 3 ) ) {
755  err_offset = rng( 0, 1 );
756  } else {
757  err_offset = 0;
758  }
759  }
760  }
761 
762  // If Z-coordinate is different, draw even when you can see the source
763  const bool diff_z = pos.z != you->posz();
764 
765  // Enumerate the valid points the player *cannot* see.
766  // Unless the source is on a different z-level, then any point is fine
767  // Also show sensory sounds like SONAR even if we can see the point.
768  std::vector<tripoint> unseen_points;
769  for( const tripoint &newp : get_map().points_in_radius( pos, err_offset ) ) {
770  if( diff_z || sound.category == sound_t::sensory || !you->sees( newp ) ) {
771  unseen_points.emplace_back( newp );
772  }
773  }
774 
775  // Then place the sound marker in a random one.
776  if( !unseen_points.empty() ) {
777  sound_markers.emplace( random_entry( unseen_points ), sound );
778  }
779  }
780  if( you->is_avatar() ) {
781  sounds_since_last_turn.clear();
782  }
783 }
