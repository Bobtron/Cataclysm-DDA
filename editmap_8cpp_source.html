1635 /*
1636  * Shift 'var' (i.e., part of a coordinate plane) by 'shift'.
1637  * If the result is not >= min and < 'max', constrain the result and adjust 'shift',
1638  * so it can adjust subsequent points of a set consistently.

1641 {
1642  if( var + shift < min ) {
1643  shift = min - var;
1644  } else if( var + shift >= max ) {
1645  shift = max - 1 - var;
1646  }
