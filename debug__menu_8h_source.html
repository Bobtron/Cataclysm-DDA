130 /* Splits a string by @param delimiter and push_back's the elements into _Container */
131 template<typename Container>
132 Container string_to_iterable( const std::string_view str, const std::string_view delimiter )
133 {
134  Container res;
135 
136  size_t pos = 0;
137  size_t start = 0;
138  while( ( pos = str.find( delimiter, start ) ) != std::string::npos ) {
139  if( pos > start ) {
140  res.emplace_back( str.substr( start, pos - start ) );
141  }
142  start = pos + delimiter.length();
143  }
144  if( start != str.length() ) {
145  res.emplace_back( str.substr( start, str.length() - start ) );
146  }
147 
148  return res;
149 }

153  * @param f is callable that is called to transform each value
154  * */
155 template<typename Container, typename Mapper>
156 std::string iterable_to_string( const Container &values, const std::string_view delimiter,
157  const Mapper &f )
158 {
159  std::string res;
160  for( auto iter = values.begin(); iter != values.end(); ++iter ) {
161  if( iter != values.begin() ) {
162  res += delimiter;

164  res += f( *iter );
