234  if( n_children == 0 ) {

236  case ExprTokenType::Constant:
237  return token.value;

242  ExprTokenTypeToString( token.type ) ) );
243  }
244  } else if( n_children == 1 ) {

249  throw EvaluationError( string_format( "Unexpected token %s at a node with 1 child.",
250  ExprTokenTypeToString( token.type ) ) );
251  }
252  } else if( n_children == 2 ) {

255  switch( token.type ) {
256  case ExprTokenType::Plus:
257  return a + b;
258  case ExprTokenType::Minus:
259  return a - b;
260  case ExprTokenType::Multiplication:
261  return a * b;
262  case ExprTokenType::Division:
263  if( b == 0 ) {
264  throw EvaluationError( "Division by zero" );
265  }
266  return a / b;
267  case ExprTokenType::Remainder:
268  if( b == 0 ) {
269  throw EvaluationError( "Division by zero" );
270  }
271  return a % b;
272  case ExprTokenType::LessThan:
273  return a < b;
274  case ExprTokenType::Equal:
275  return a == b;
276  case ExprTokenType::GreaterThan:
277  return a > b;
278  case ExprTokenType::NotEqual:
279  return a != b;
280  case ExprTokenType::LessOrEqual:
281  return a <= b;
282  case ExprTokenType::GreaterOrEqual:
283  return a >= b;
284  case ExprTokenType::Or:

286  case ExprTokenType::And:

305 void TranslationPluralRulesEvaluator::UnwindStacks( std::stack<ExprNode *> &operands,
306  std::stack<ExprNode *> &operators,
307  int precedence )
308 {
309  while( !operators.empty() &&
310  OperatorPrecedence( operators.top()->token.type ) >= precedence &&
311  ( OperatorPrecedence( operators.top()->token.type ) > precedence ||
312  !IsTernaryToken( operators.top()->token.type ) ) ) {
313  if( IsUnaryToken( operators.top()->token.type ) ) {
314  if( operands.empty() ) {
315  throw ParseError( "Insufficient operands" );
316  }
317  ExprNode *a = operands.top();
318  operands.pop();
319  ExprNode *op = operators.top();
320  operators.pop();
321  op->AddChild( a );
322  operands.push( op );
323  } else if( IsBinaryToken( operators.top()->token.type ) ) {
324  if( operands.size() < 2 ) {
325  throw ParseError( "Insufficient operands" );
326  }
327  ExprNode *b = operands.top();
328  operands.pop();
329  ExprNode *a = operands.top();
330  operands.pop();
331  ExprNode *op = operators.top();
332  operators.pop();
333  op->AddChild( a );
334  op->AddChild( b );
335  operands.push( op );
336  } else if( IsTernaryToken( operators.top()->token.type ) ) {
337  if( operands.size() < 3 || operands.size() < 2 ) {
338  break;
339  }
340  ExprNode *c = operands.top();
341  operands.pop();
342  ExprNode *b = operands.top();
343  operands.pop();
344  ExprNode *a = operands.top();
345  operands.pop();
346  ExprNode *op2 = operators.top();
347  operators.pop();
348  ExprNode *op1 = operators.top();
349  operators.pop();
350  op1->AddChild( a );
351  op1->AddChild( b );
352  op1->AddChild( c );
353  operands.push( op1 );
354  delete op2;
355  }
356  }

359 ExprNode *TranslationPluralRulesEvaluator::ConstructExprTreeRecursive(
360  std::vector<ExprToken>::const_iterator &it,
361  const std::vector<ExprToken>::const_iterator &end )
362 {
363  std::stack<ExprNode *> operands;
364  std::stack<ExprNode *> operators;
365  while( it != end && it->type != ExprTokenType::Termination ) {
366  if( it->type == ExprTokenType::LeftBracket ) {
367  operands.push( ConstructExprTreeRecursive( ++it, end ) );
368  } else if( it->type == ExprTokenType::RightBracket ) {
369  ++it;
370  break;
371  } else {
372  ExprNode *node = new ExprNode;
373  node->token = *it;
374  if( IsOperandToken( it->type ) ) {
375  operands.push( node );
376  } else {
377  UnwindStacks( operands, operators, OperatorPrecedence( it->type ) );
378  operators.push( node );
379  }
380  ++it;
381  }
382  }
383  UnwindStacks( operands, operators );
384  if( operands.size() == 1 && operators.empty() ) {
385  return operands.top();
386  } else if( operators.size() == 1 && operands.empty() ) {
387  return operators.top();
388  }
389  throw ParseError( "Stacks not unwinded during expression tree construction." );
390 }

426 {
427  const std::string nplurals_header = "nplurals=";
428  std::size_t nplurals_header_pos = plural_forms.find( nplurals_header );
429  if( nplurals_header_pos == std::string::npos ) {
430  throw HeaderError( string_format( "Expect \"%s\" in %s", nplurals_header, plural_forms ) );
431  }
432  nplurals_header_pos += nplurals_header.length();
433  std::size_t first_semicolon = plural_forms.find( ';', nplurals_header_pos );
434  if( first_semicolon == std::string::npos ) {
435  throw HeaderError( string_format( "Expect semicolon in %s", plural_forms ) );
436  }
437  ret_val<int> n_plurals_optional = try_parse_integer<int>( plural_forms.substr(
438  nplurals_header_pos, first_semicolon - nplurals_header_pos ), false );
439  if( !n_plurals_optional.success() || n_plurals_optional.value() < 1 ) {
440  throw HeaderError( string_format( "Invalid number of plural forms in %s", plural_forms ) );
441  }
442  n_plurals = static_cast<std::size_t>( n_plurals_optional.value() );
443  const std::string plural_header = "plural=";
444  std::size_t plural_header_pos = plural_forms.find( plural_header );
445  if( plural_header_pos == std::string::npos ) {
446  throw HeaderError( string_format( "Expect \"%s\" in %s", plural_header, plural_forms ) );
447  }
448  plural_header_pos += plural_header.length();
