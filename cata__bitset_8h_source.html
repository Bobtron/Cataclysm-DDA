18  // The layout of this struct is a single uintptr_t type which either holds:
19  // - A pointer to heap allocated data, with size + capacity prepended
20  // - N-1 bytes of inline storage and 1 byte of metadata
21  // The least significant byte of metadata is, in order of most to least significant,
22  // 7 bits of size, and 1 bit indicating inline storage.
23  // Whether 32 or 64 bit, just use 6 bits for inline storage (instead of optionally 5 on 32 bit)
24  static constexpr block_t kInlineSizeMask = 0b11111110;

240  // Op needs to support two functions
241  // void(block_t&) and void(block_t&, block_t). References to bits
242  // are passed and may be read or written directly.
243  // The second form takes an argument which is a bitmask where set
244  // bits correspond to bits to ignore in the data argument. These bits
245  // must not be changed, so write operations must appropriately include
246  // the existing value in what is written.
247  template<typename Op>
248  void map( Op &&op ) {
249  size_t bits_used = size();
250  if( bits_used == 0 ) {
251  return;
252  }
253  if( is_inline() ) {
254  block_t unused_mask = ( kLowBit << ( kBitsPerBlock - bits_used ) ) - 1;
255  op( storage_, unused_mask );
256  return;
257  }
258 
259  // The compiler auto vectorizer is too smart and inserts a bunch of code
260  // to handle various cases of how big the data is. We expect things to be
261  // small, so we can hand roll a loop which doesn't blow up code size or
262  // insert a huge number of conditionals into the hot path, and probably
263  // not measurably slower than the fully expanded vectorized code which handles
264  // 128 bytes (1,024 bits!) in one pass.
265  size_t used_blocks = bits_used / kBitsPerBlock;
266  // The final block requires special handling.
267  size_t used_bits_in_final_block = bits_used % kBitsPerBlock;
268  size_t used_whole_blocks = used_bits_in_final_block > 0 &&
269  used_blocks > 0 ? used_blocks - 1 : used_blocks;
270 
271  // Process 4 at a time for some manual unrolling.
272  block_t *blocks = bits();
273  size_t i = 0;
274  for( size_t end = used_whole_blocks; i + 3 < end; i += 4 ) {
275  op( blocks[i + 0] );
276  op( blocks[i + 1] );
277  op( blocks[i + 2] );
278  op( blocks[i + 3] );
279  }
280  // final whole blocks that don't fit in unrolled loop.
281  for( ; i < used_whole_blocks; ++i ) {
282  op( blocks[i] );
283  }
284  // final, partial block
285  block_t unused_mask;
286  if( used_bits_in_final_block != 0 ) {
287  // Create mask which is all bits set for bits not used in capacity.
288  // If we used N bits, create a mask of the k - N least significant bits by
289  // shifting a 1 k-N times, into the k-N'th bit _index_ (or k-N+1'th bit) then
290  // subtracting one, setting k-N of the least significant bits to 1.
291  unused_mask = ( kLowBit << ( kBitsPerBlock - used_bits_in_final_block ) ) - 1;
292  op( blocks[i], unused_mask );
293  }
294  }
