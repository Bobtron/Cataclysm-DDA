614 bool ma_requirements::is_valid_character( const Character &u ) const
615 {
616  if( !buff_requirements_satisfied( u ) ) {
617  add_msg_debug( debugmode::DF_MELEE, "Buff requirements not satisfied, attack discarded" );
618  return false;
619  }
620 
621  //A technique is valid if it applies to unarmed strikes, if it applies generally
622  //to all weapons (such as Ninjutsu sneak attacks or innate weapon techniques like RAPID)
623  //or if the weapon is flagged as being compatible with the style. Some techniques have
624  //further restrictions on required weapon properties (is_valid_weapon).
625  bool cqb = u.has_active_bionic( bio_cqb );
626  // There are 4 different cases of "armedness":
627  // Truly unarmed, unarmed weapon, style-allowed weapon, generic weapon
628  const item_location weapon = u.get_wielded_item();
629  bool melee_style = u.martial_arts_data->selected_strictly_melee();
630  bool is_armed = u.is_armed();
631  bool forced_unarmed = u.martial_arts_data->selected_force_unarmed();
632  bool weapon_ok = melee_allowed && weapon && is_valid_weapon( *weapon );
633  bool style_weapon = weapon && u.martial_arts_data->selected_has_weapon( weapon->typeId() );
634  bool all_weapons = u.martial_arts_data->selected_allow_all_weapons();
635 
636  bool melee_ok = weapon_ok && ( style_weapon || all_weapons );
637 
638  bool valid_unarmed = !melee_style && unarmed_allowed && !is_armed;
639  bool valid_melee = !strictly_unarmed && ( forced_unarmed || melee_ok );
640 
641  if( !valid_unarmed && !valid_melee ) {
642  return false;
643  }
644 
645  if( wall_adjacent && !get_map().is_wall_adjacent( u.pos() ) ) {
646  return false;
647  }
648 
649  for( const auto &pr : min_skill ) {
650  if( ( cqb ? 5 : static_cast<int>( u.get_skill_level( pr.first ) ) ) < pr.second ) {
651  add_msg_debug( debugmode::DF_MELEE, "Skill level requirement %d not satisfied, attack discarded",
652  pr.second );
653  return false;
654  }
655  }
656 
657  if( !req_char_flags.empty() ) {
658  bool has_flag = false;
659  for( const json_character_flag &flag : req_char_flags ) {
660  if( u.has_flag( flag ) ) {
661  has_flag = true;
662  }
663  }
664  if( !has_flag ) {
665  add_msg_debug( debugmode::DF_MELEE, "Required flag(any) not found, attack discarded" );
666  return false;
667  }
668  }
669 
670  for( const json_character_flag &flag : req_char_flags_all ) {
671  if( !u.has_flag( flag ) ) {
672  add_msg_debug( debugmode::DF_MELEE, "Required flags(all) not found, attack discarded" );
673  return false;
674  }
675  }
676 
677  if( !forbidden_char_flags.empty() ) {
678  bool has_flag = false;
679  for( const json_character_flag &flag : forbidden_char_flags ) {
680  if( u.has_flag( flag ) ) {
681  has_flag = true;
682  }
683  }
684  if( has_flag ) {
685  add_msg_debug( debugmode::DF_MELEE, "Forbidden flag found, attack discarded" );
686  return false;
687  }
688  }
689 
690  if( !weapon_categories_allowed.empty() ) {
691  bool valid_weap_cat = false;
692  for( const weapon_category_id &w_cat : weapon_categories_allowed ) {
693  if( u.used_weapon() && u.used_weapon()->typeId()->weapon_category.count( w_cat ) > 0 ) {
694  valid_weap_cat = true;
695  }
696  }
697  if( !valid_weap_cat ) {
698  add_msg_debug( debugmode::DF_MELEE, "Not using weapon from a valid category, attack discarded" );
699  return false;
700  }
701  }
702 
703  return true;
704 }
