473  avatar &player_character = get_avatar();

531  found_quantity ++;

597  case MGOAL_ASSASSINATE:
598  case MGOAL_KILL_MONSTER:

653  for( item *itm : items ) {
654  bool correct_container = ( required_container == actual_container ) ||
655  !specific_container_required;
656 
657  bool item_in_group = item_group::group_contains_item( grp_type, itm->typeId() );
658 
659  //check whether item itself is target
660  if( item_in_group && correct_container ) {
661  std::map<itype_id, int>::iterator it = matches.find( itm->typeId() );
662  if( it != matches.end() ) {
663  it->second = ( it->second ) + 1;
664  } else {
665  matches.insert( std::make_pair( itm->typeId(), 1 ) );
666  }
667  }
668 
669  //recursively check item contents for target
670  if( itm->is_container() && !itm->is_container_empty() ) {
671  std::list<item *> content_list = itm->all_items_top();
672  std::vector<item *> content = std::vector<item *>();
673 
674  //list of item into list item*
675  std::transform(
676  content_list.begin(), content_list.end(),
677  std::back_inserter( content ),
678  []( item * p ) {
679  return p;
680  } );
681 
682  get_all_item_group_matches(
683  content, grp_type, matches,
684  required_container, itm->typeId(), specific_container_required );
685  }
686  }
