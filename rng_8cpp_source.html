45 // Returns the area under a curve with provided standard deviation and center
46 // from difficulty to positive infinity. That is, the chance that a normal roll on
47 // said curve will return a value of difficulty or greater.
48 float normal_roll_chance( float center, float stddev, float difficulty )
49 {
50  cata_assert( stddev >= 0.f );
51  // We're going to be using them a lot, so let's name our variables.
52  // M = the given "center" of the curve
53  // S = the given standard deviation of the curve
54  // A = the difficulty
55  // So, the equation of the normal curve is...
56  // y = (1.f/(S*std::sqrt(2 * M_PI))) * exp(-(std::pow(x - M, 2))/(2 * std::pow(S, 2)))
57  // Thanks to wolfram alpha, we know the integral of that from A to B to be
58  // 0.5 * (erf((M-A)/(std::sqrt(2) * S)) - erf((M-B)/(std::sqrt(2) * S)))
59  // And since we know B to be infinity, we can simplify that to
60  // 0.5 * (erfc((A-m)/(std::sqrt(2)* S))+sgn(S)-1) (as long as S != 0)
61  // Wait a second, what are erf, erfc and sgn?
62  // Oh, those are the error function, complementary error function, and sign function
63  // Luckily, erf() is provided to us in math.h, and erfc is just 1 - erf
64  // Sign is pretty obvious x > 0 ? x == 0 ? 0 : 1 : -1;
65  // Since we know S will always be > 0, that term vanishes.
66 
67  // With no standard deviation, we will always return center
68  if( stddev == 0.f ) {
69  return ( center > difficulty ) ? 1.f : 0.f;
70  }
71 
72  float numerator = difficulty - center;
73  float denominator = std::sqrt( 2 ) * stddev;
74  float compl_erf = 1.f - std::erf( numerator / denominator );
75  return 0.5 * compl_erf;
76 }
