209 static void prepare_sorted_lists( std::vector<damage_info_order> &list,
210  damage_info_order::info_type sb )
211 {
212  for( auto iter = list.begin(); iter != list.end(); ) {
213  bool erase = false;
214  switch( sb ) {
215  case damage_info_order::info_type::BIO:
216  if( !iter->bionic_info.show_type ) {
217  erase = true;
218  }
219  break;
220  case damage_info_order::info_type::PROT:
221  if( !iter->protection_info.show_type ) {
222  erase = true;
223  }
224  break;
225  case damage_info_order::info_type::PET:
226  if( !iter->pet_prot_info.show_type ) {
227  erase = true;
228  }
229  break;
230  case damage_info_order::info_type::MELEE:
231  if( !iter->melee_combat_info.show_type ) {
232  erase = true;
233  }
234  break;
235  case damage_info_order::info_type::ABLATE:
236  if( !iter->ablative_info.show_type ) {
237  erase = true;
238  }
239  break;
240  case damage_info_order::info_type::NONE:
241  default:
242  break;
243  }
244  if( erase ) {
245  iter = list.erase( iter );
246  } else {
247  iter++;
248  }
249  }
250  if( sb == damage_info_order::info_type::NONE ) {
251  return;
252  }
253  std::sort( list.begin(), list.end(),
254  [sb]( const damage_info_order & a, const damage_info_order & b ) {
255  switch( sb ) {
256  case damage_info_order::info_type::BIO:
257  return a.bionic_info.order < b.bionic_info.order;
258  case damage_info_order::info_type::PROT:
259  return a.protection_info.order < b.protection_info.order;
260  case damage_info_order::info_type::PET:
261  return a.pet_prot_info.order < b.pet_prot_info.order;
262  case damage_info_order::info_type::MELEE:
263  return a.melee_combat_info.order < b.melee_combat_info.order;
264  case damage_info_order::info_type::ABLATE:
265  return a.ablative_info.order < b.ablative_info.order;
266  case damage_info_order::info_type::NONE:
267  default:
268  return false;
269  }
270  } );
271 }

491 void damage_instance::add( const damage_unit &added_du )
492 {
493  auto iter = std::find_if( damage_units.begin(), damage_units.end(),
494  [&added_du]( const damage_unit & du ) {
495  return du.type == added_du.type;
496  } );
497  if( iter == damage_units.end() ) {
498  damage_units.emplace_back( added_du );
499  } else {
500  damage_unit &du = *iter;
501  float mult = added_du.damage_multiplier / du.damage_multiplier;
502  du.amount += added_du.amount * mult;
503  du.res_pen += added_du.res_pen * mult;
504  // Linearly interpolate armor multiplier based on damage proportion contributed
505  float t = added_du.damage_multiplier / ( added_du.damage_multiplier + du.damage_multiplier );
506  du.res_mult = lerp( du.res_mult, added_du.damage_multiplier, t );
507 
508  du.unconditional_res_mult *= added_du.unconditional_res_mult;
509  du.unconditional_damage_mult *= added_du.unconditional_damage_mult;
510 
511  if( added_du.barrels.empty() ) {
512  if( du.barrels.empty() ) {
513  du.barrels = added_du.barrels;
514  } else {
515  debugmsg( "Tried to add two sets of barrel definitions for damage together, this is not currently supported." );
516  }
517  }
518  }

781 void finalize_damage_map( std::unordered_map<damage_type_id, float> &damage_map, bool force_derive,
782  float default_value )
783 {
784  const std::vector<damage_type> &dams = damage_type::get_all();
785  if( dams.empty() ) {
786  debugmsg( "called before damage_types have been loaded." );
787  return;
788  }
789 
790  auto get_and_erase = [&damage_map]( const damage_type_id & did, float def_val ) {
791  auto iter = damage_map.find( did );
792  if( iter == damage_map.end() ) {
793  return def_val;
794  }
795  float val = iter->second;
796  damage_map.erase( iter );
797  return val;
798  };
799 
800  const float all = get_and_erase( STATIC( damage_type_id( "all" ) ), default_value );
801  const float physical = get_and_erase( STATIC( damage_type_id( "physical" ) ), all );
802  const float non_phys = get_and_erase( STATIC( damage_type_id( "non_physical" ) ), all );
803 
804  std::vector<damage_type_id> to_derive;
805  for( const damage_type &dam : dams ) {
806  float val = dam.physical ? physical : non_phys;
807  if( damage_map.count( dam.id ) > 0 ) {
808  val = damage_map.at( dam.id );
809  } else if( force_derive && !dam.derived_from.first.is_null() ) {
810  to_derive.emplace_back( dam.id );
811  continue;
812  }
813  damage_map[dam.id] = val;
814  }
815 
816  for( auto &td : to_derive ) {
817  auto iter = damage_map.find( td->derived_from.first );
818  damage_map[td] = iter == damage_map.end() ? ( td->physical ? physical : non_phys ) :
819  iter->second * td->derived_from.second;
820  }
821 }
