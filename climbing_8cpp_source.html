372 
373  this->examp = examp;
374  height = 0;
375  height_until_furniture = 0;
376  height_until_creature = 0;
377  height_until_vehicle = 0;
378 
379  // Get coordinates just below and at ground level.
380  // Also detect if furniture would block our tools/abilities.
381  tripoint bottom = examp;
382  tripoint just_below = examp;
383  just_below.z--;
384 
385  int hit_furn = false;
386  int hit_crea = false;
387  int hit_veh = false;
388 
389  for( tripoint lower = just_below; here.valid_move( bottom, lower, false, true ); ) {
390  if( !hit_furn ) {
391  if( here.has_furn( lower ) ) {
392  hit_furn = true;
393  } else {
394  ++height_until_furniture;
395  }
396  }
397  if( !hit_veh ) {
398  if( here.veh_at( lower ) ) {
399  hit_veh = true;
400  } else {
401  ++height_until_vehicle;
402  }
403  }
404  if( !hit_crea ) {
405  ++height_until_creature;
406  if( creatures.creature_at( lower, false ) ) {
407  hit_crea = true;
408  } else {
409  ++height_until_creature;
410  }
411  }
