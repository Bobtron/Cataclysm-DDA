108 void auto_note_settings::load( bool bCharacter )

112  const auto parseJson = [ &, bCharacter ]( const JsonValue & jv ) {
113  JsonObject jo = jv;
114 
115  for( JsonMember member : jo ) {
116  const std::string name = member.name();
117 
118  if( name == "enabled" ) {
119  JsonArray enabled_notes = member;
120  for( std::string entry : enabled_notes ) {
121  character_autoNoteEnabled.insert( map_extra_id{ entry } );
122  }
123  } else if( name == "disabled" ) {
124  JsonArray disabled_notes = member;
125  for( std::string entry : disabled_notes ) {
126  global_autoNoteDisabled.insert( map_extra_id{ entry } );
127  }
128  } else if( name == "discovered" ) {
129  JsonArray discovered_array = member;
130  for( std::string entry : discovered_array ) {
131  discovered.insert( map_extra_id {std::move( entry )} );
132  }
133  } else if( name == "custom_symbols" ) {
134  JsonArray symbols_json = member;
135  for( JsonObject joSymbols : symbols_json ) {
136  const std::string entry = joSymbols.get_string( "map_extra" );
137  const std::string custom_symbol_str = joSymbols.get_string( "symbol" );
138  const std::string custom_color = joSymbols.get_string( "color" );
139  custom_symbol sym;
140  sym.set_symbol( custom_symbol_str );
141  sym.set_color( custom_color );
142  ( bCharacter ? character_custom_symbols : global_custom_symbols ).insert( std::make_pair(
143  map_extra_id {entry}, sym ) );
144  }
145  }
146  }
147  };
148 
149  if( bCharacter ) {
150  if( !read_from_file_optional_json( build_save_path(), parseJson ) ) {
151  default_initialize();
152  save( true );
153  }
154  } else {
