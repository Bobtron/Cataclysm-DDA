1510 static std::pair<unsigned, unsigned> calculate_shared_column_win_height(
1511  const unsigned available_height, unsigned first_win_size_y_max, unsigned second_win_size_y_max )
1512 /**
1513  * Calculate max allowed height of two windows sharing column space.
1514  */
1515 {
1516  if( ( second_win_size_y_max + 1 + first_win_size_y_max ) > available_height ) {
1517  // maximum space for either window if they're both the same size
1518  unsigned max_shared_y = ( available_height - 1 ) / 2;
1519  if( std::min( second_win_size_y_max, first_win_size_y_max ) > max_shared_y ) {
1520  // both are larger than the shared size
1521  second_win_size_y_max = max_shared_y;
1522  first_win_size_y_max = available_height - 1 - second_win_size_y_max;
1523  } else if( first_win_size_y_max <= max_shared_y ) {
1524  // first window is less than the shared size, so give space to second window
1525  second_win_size_y_max = available_height - 1 - first_win_size_y_max;
1526  } else {
1527  // second window is less than the shared size
1528  first_win_size_y_max = available_height - 1 - second_win_size_y_max;
1529  }
1530  }
1531  return { first_win_size_y_max, second_win_size_y_max };
1532 }
