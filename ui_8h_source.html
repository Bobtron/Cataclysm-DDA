590 /**
591  * Helper for typical UI list navigation with wrap-around
592  * Add delta to val. If on bounds, wrap to other bound, otherwise clamp to the range [0,size)
593  */
594 template<typename V, typename S>
595 inline std::enable_if_t < !std::is_enum_v<V>, V >
596 inc_clamp_wrap( V val, int delta, S size )
597 {
598  if( size == 0 ) {
599  return 0;
600  }
601  // wrap to last
602  if( val <= 0 && delta < 0 ) {
603  return size - 1;
604  }
605  // wrap to first
606  // Templating of existing `unsigned int` triggers linter rules against `unsigned long`
607  // NOLINTNEXTLINE(cata-no-long)
608  if( val >= static_cast<V>( size ) - 1 && delta > 0 ) {
609  return 0;
610  }
611  return std::clamp<V>( val + delta, 0, size - 1 );
612 }
