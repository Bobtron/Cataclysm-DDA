167

169 
170  if( !cards.empty() && query_yn( _( query_msg ) ) ) {
171  you.mod_moves( -to_moves<int>( 1_seconds ) );

173  for( monster &critter : g->all_monsters() ) {

179  critter.has_flag( mon_flag_ID_CARD_DESPAWN ) &&
180  critter.attitude_to( you ) == Creature::Attitude::HOSTILE ) {
181  g->remove_zombie( critter );

183  }
184  if( open ) {

187  } else {
