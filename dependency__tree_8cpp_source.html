396 void dependency_tree::strong_connect( dependency_node *dnode )
397 {
398  dnode->index = open_index;
399  dnode->lowlink = open_index;
400  ++open_index;
401  connection_stack.push( dnode );
402  dnode->on_stack = true;
403 
404  for( std::vector<dependency_node *>::iterator it = dnode->parents.begin();
405  it != dnode->parents.end(); ++it ) {
406  if( ( *it )->index < 0 ) {
407  strong_connect( *it );
408  dnode->lowlink = std::min( dnode->lowlink, ( *it )->lowlink );
409  } else if( ( *it )->on_stack ) {
410  dnode->lowlink = std::min( dnode->lowlink, ( *it )->index );
411  }
412  }
413 
414  if( dnode->lowlink == dnode->index ) {
415  std::vector<dependency_node *> scc;
416

418  do {
