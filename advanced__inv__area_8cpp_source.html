177 bool advanced_inv_area::is_same( const advanced_inv_area &other ) const
178 {
179  // All locations (sans the below) are compared by the coordinates,
180  // e.g. dragged vehicle (to the south) and AIM_SOUTH are the same.
181  if( id != AIM_INVENTORY && other.id != AIM_INVENTORY &&
182  id != AIM_WORN && other.id != AIM_WORN &&
183  id != AIM_CONTAINER && other.id != AIM_CONTAINER ) {
184  // have a vehicle?... ...do the cargo index and pos match?... ...at least pos?
185  return veh == other.veh ? pos == other.pos && vstor == other.vstor : pos == other.pos;
186  }
187  // ...is the id?
188  return id == other.id;
189 }

274 advanced_inv_area::itemstack advanced_inv_area::i_stacked( T items )
275 {
276  //create a new container for our stacked items
277  advanced_inv_area::itemstack stacks;
278  // used to recall indices we stored `itype_id' item at in itemstack
279  std::unordered_map<itype_id, std::set<int>> cache;
280  // iterate through and create stacks
281  for( item &elem : items ) {
282  const itype_id id = elem.typeId();
283  auto iter = cache.find( id );
284  bool got_stacked = false;
285  // cache entry exists
286  if( iter != cache.end() ) {
287  // check to see if it stacks with each item in a stack, not just front()
288  for( const int &idx : iter->second ) {
289  for( item *&it : stacks[idx] ) {
290  if( ( got_stacked = it->display_stacked_with( elem ) ) ) {
291  stacks[idx].push_back( &elem );
292  break;
293  }
294  }
295  if( got_stacked ) {
296  break;
297  }
298  }
299  }
300  if( !got_stacked ) {
301  cache[id].insert( stacks.size() );
302  stacks.push_back( { &elem } );
303  }
304  }
305  return stacks;
306 }
