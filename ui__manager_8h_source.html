22 /**
23  * Adaptor between UI code and the UI management system. Using this class allows
24  * UIs to be correctly resized and redrawn when the game window is resized or
25  * when exiting from other UIs.
26  *
27  * Usage example:
28  * ```
29  * // Effective in the local scope
30  * ui_adaptor ui;
31  * // Ncurses window for drawing
32  * catacurses::window win;
33  * // Things to do when the game window changes size
34  * ui.on_screen_resize( [&]( ui_adaptor & ui ) {
35  * // Create an ncurses window
36  * win = catacurses::newwin( TERMX / 2, TERMY / 2, point( TERMX / 4, TERMY / 4 ) );
37  * // The window passed to this call must contain all the space the redraw
38  * // callback draws to, to ensure proper refreshing when resizing or exiting
39  * // from other UIs.
40  * ui.position_from_window( win );
41  * } );
42  * // Mark the resize callback to be called on the first redraw
43  * ui.mark_resize();
44  * // Things to do when redrawing the UI
45  * ui.on_redraw( [&]( ui_adaptor & ui ) {
46  * // Clear UI area
47  * werase( win );
48  * // Print things
49  * mvwprintw( win, point_zero, "Hello World!" );
50  * // Record the cursor position for screen readers and IME preview to
51  * // correctly function on curses
52  * ui.record_cursor( win );
53  * // Write to frame buffer
54  * wnoutrefresh( win );
55  * } );
56  *
57  * input_context ctxt( "<CATEGORY_NAME>" );
58  * ctxt.register_action( "QUIT" );
59  * while( true ) {
60  * // Invalidate the top UI (that is, this UI) and redraw all
61  * // invalidated UIs (including lower UIs that calls this UI).
62  * // May call the resize callbacks.
63  * ui_manager::redraw();
64  * // Get user input. Note that this may call the resize and redraw
65  * // callbacks multiple times due to screen resize, rendering target
66  * // reset, etc.
67  * const std::string action = ctxt.handle_input();
68  * if( action == "QUIT" ) {
69  * break;
70  * }
71  * }
72  * ```
73  **/
