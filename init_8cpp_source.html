161 {
162  while( !data.empty() ) {
163  const size_t n = data.size();
164  auto it = data.begin();
165  for( size_t idx = 0; idx != n; ++idx ) {
166  try {

497  loading_ui &ui )
498 {
499  cata_assert( !finalized &&
500  "Can't load additional data after finalization. Must be unloaded first." );
501  // We assume that each folder is consistent in itself,
502  // and all the previously loaded folders.
503  // E.g. the core might provide a vpart "frame-x"
504  // the first loaded mode might provide a vehicle that uses that frame
505  // But not the other way round.
506 
507  std::vector<cata_path> files;
508  if( dir_exist( path.get_unrelative_path() ) ) {
509  const std::vector<cata_path> dir_files = get_files_from_path( ".json", path, true, true );
510  files.insert( files.end(), dir_files.begin(), dir_files.end() );
511  } else if( file_exist( path.get_unrelative_path() ) ) {
512  files.emplace_back( path );
513  }
514 
515  // iterate over each file
516  for( const cata_path &file : files ) {
517  try {
518  // parse it
519  JsonValue jsin = json_loader::from_path( file );
520  load_all_from_json( jsin, src, ui, path, file );
521  } catch( const JsonError &err ) {
522  throw std::runtime_error( err.what() );
523  }
524  }
