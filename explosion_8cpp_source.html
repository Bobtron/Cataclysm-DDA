943 fragment_cloud shrapnel_calc( const fragment_cloud &initial,
944  const fragment_cloud &cloud,
945  const int &distance )
946 {
947  // SWAG coefficient of drag.
948  constexpr float Cd = 1.5f;
949  fragment_cloud new_cloud;
950  new_cloud.velocity = initial.velocity * std::exp( -cloud.velocity * ( (
