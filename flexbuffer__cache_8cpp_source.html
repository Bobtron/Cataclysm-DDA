58  const char *buffer,
59  const char *source_filename_opt ) noexcept( false )
60 {
61  flatbuffers::IDLOptions opts;
62  opts.strict_json = true;
63  opts.use_flexbuffers = true;
64  opts.no_warnings = true;
65  flatbuffers::Parser parser{ opts };
66  flexbuffers::Builder fbb;
67 
68  if( !parser.ParseFlexBuffer( buffer, source_filename_opt, &fbb ) ) {
69  std::istringstream is{ buffer };
70  TextJsonIn jsin{ is, source_filename_opt ? source_filename_opt : "<unknown source file>" };
71 
72  if( fbb.HasDuplicateKeys() ) {
73  // The error from the parser isn't very informative. TextJsonObject can detect the
74  // condition but we have to deep scan to find it.
75  TextJsonValue jv = jsin.get_value();
76  try_find_and_throw_json_error( jv );
77  // If we didn't find and throw it, reset the stream.
78  is.seekg( 0, std::ios_base::beg );
79  }
80 
81  size_t line = 0;
82  size_t col = 0;
83  size_t error_offset = 0;
84  if( ( error_offset = parser.error_.find( "EOF:" ) ) == std::string::npos ) {
85  // Try to extract line and col to position TextJsonIn at an appropriate location.
86  // %n modifier returns number of characters consumed by sscanf to skip the text
87  // in the error that the flexbuffer parser returns, because TextJsonIn will add it.
88  // Format is "(filename:)?(EOF:|line:col:) message"
89  // But filename might be C:something
90  int scanned_chars = 0;
91  // NOLINTNEXTLINE(cert-err34-c)
92  if( sscanf( parser.error_.c_str(), "%*[^:]:%*[^:]:%zu:%zu: %n", &line, &col,
93  &scanned_chars ) != 2 &&
94  // NOLINTNEXTLINE(cert-err34-c)
95  sscanf( parser.error_.c_str(), "%*[^:]:%zu:%zu: %n", &line, &col, &scanned_chars ) != 2 &&
96  // NOLINTNEXTLINE(cert-err34-c)
97  sscanf( parser.error_.c_str(), "%zu:%zu: %n", &line, &col, &scanned_chars ) != 2 ) {
98  line = 0;
99  col = 0;
100  }
101  error_offset = scanned_chars;
102  } else {
103  error_offset += 5; // skip "EOF: " because find returns the offset of the start.
104  }
105 
106  constexpr const char *kErrorPrefix = "error: ";
107  if( strncmp( parser.error_.c_str() + error_offset, kErrorPrefix, strlen( kErrorPrefix ) ) == 0 ) {
108  error_offset += strlen( kErrorPrefix );
109  }
110 
111  if( line != 0 ) {
112  // Lines are 1 indexed.
113  while( line > 1 ) {
114  is.ignore( std::numeric_limits<std::streamsize>::max(), '\n' );
115  --line;
116  }
117  } else {
118  // Seek to end.
119  is.seekg( 0, std::ios_base::end );
120  // Force EOF state
121  is.peek();
122  }
123  jsin.error( col ? col - 1 : 0, parser.error_.substr( error_offset ) );
124  }
125
