70 void Character::apply_persistent_morale()
71 {
72  // Hoarders get a morale penalty if they're not carrying a full inventory.
73  if( has_trait( trait_HOARDER ) ) {
74  hoarder_morale_penalty();
75  }
76  // Nomads get a morale penalty if they stay near the same overmap tiles too long.
77  if( has_trait( trait_NOMAD ) || has_trait( trait_NOMAD2 ) || has_trait( trait_NOMAD3 ) ) {
78  const tripoint_abs_omt ompos = global_omt_location();
79  float total_time = 0.0f;
80  // Check how long we've stayed in any overmap tile within 5 of us.
81  const int max_dist = 5;
82  for( const tripoint_abs_omt &pos : points_in_radius( ompos, max_dist ) ) {
83  const float dist = rl_dist( ompos, pos );
84  if( dist > max_dist ) {
85  continue;
86  }
87  const auto iter = overmap_time.find( pos.xy() );
88  if( iter == overmap_time.end() ) {
89  continue;
90  }
91  // Count time in own tile fully, tiles one away as 4/5, tiles two away as 3/5, etc.
92  total_time += to_moves<float>( iter->second ) * ( max_dist - dist ) / max_dist;
93  }
94  // Characters with higher tiers of Nomad suffer worse morale penalties, faster.
95  int max_unhappiness;
96  float min_time;
97  float max_time;
98  if( has_trait( trait_NOMAD ) ) {
99  max_unhappiness = 20;
100  min_time = to_moves<float>( 12_hours );
101  max_time = to_moves<float>( 1_days );
102  } else if( has_trait( trait_NOMAD2 ) ) {
103  max_unhappiness = 40;
104  min_time = to_moves<float>( 4_hours );
105  max_time = to_moves<float>( 8_hours );
106  } else { // traid_NOMAD3
107  max_unhappiness = 60;
108  min_time = to_moves<float>( 1_hours );
109  max_time = to_moves<float>( 2_hours );
110  }
111  // The penalty starts at 1 at min_time and scales up to max_unhappiness at max_time.
112  const float t = ( total_time - min_time ) / ( max_time - min_time );
113  const int pen = std::ceil( lerp_clamped( 0, max_unhappiness, t ) );
114  if( pen > 0 ) {
115  add_morale( MORALE_PERM_NOMAD, -pen, -pen, 1_minutes, 1_minutes, true );
116  }
117  }
118 
119  if( has_trait( trait_PROF_FOODP ) ) {
120  // Losing your face is distressing
121  if( !( is_wearing( itype_foodperson_mask ) ||
122  is_wearing( itype_foodperson_mask_on ) ) ) {
123  add_morale( MORALE_PERM_NOFACE, -20, -20, 1_minutes, 1_minutes, true );
124  } else if( is_wearing( itype_foodperson_mask ) ||
125  is_wearing( itype_foodperson_mask_on ) ) {
126  rem_morale( MORALE_PERM_NOFACE );
127  }
128 
129  if( is_wearing( itype_foodperson_mask_on ) ) {
130  add_morale( MORALE_PERM_FPMODE_ON, 10, 10, 1_minutes, 1_minutes, true );
131  } else {
132  rem_morale( MORALE_PERM_FPMODE_ON );
133  }
134  }
135 }
