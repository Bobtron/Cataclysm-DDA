635  if( iter == accum.end() ) {
636  accum.push_back( comp ); // component doesn't exist yet, adding it
637  } else {
638  // this is the list of alternatives (a OR b OR c), so if there are two identical components
639  // with the different counts as an alternatives, the one with the lowest count is selected
640  iter->count = std::min( comp.count, iter->count );
641  }
642  };
643 
644  // if `comp` is a component, adds it to the `accum`
645  // if it's a requirement, recursively expands it into simple components and adds them to the vector
646  const auto inline_one = [&]( const T & comp, std::vector<T> &accum ) {
647  std::set<requirement_id> already_nested;
648  std::list<requirement_id> stack;
649 
650  std::function<void( const T &comp )> rec;
651  rec = [&]( const T & comp ) {
652  // add simple component to the vector
653  if( !comp.requirement ) {
654  add_component( comp, accum );
655  return;
656  }
657  // otherwise expand component as requirement
658  const requirement_id r( comp.type.str() );
659  if( !r.is_valid() ) {
660  debugmsg( "Tried to inline unknown requirement %s", r.c_str() );
661  return;
662  }
663  // stack just holds the current path of inlining for debug purposes
664  stack.push_back( r );
665  if( already_nested.count( r ) ) {
666  // print debug msg and skip just this one requirement
667  debugmsg( "Tried to inline requirement %s which forms a cycle: %s",
668  r.c_str(), debug_menu::iterable_to_string( stack, " -> ", []( const requirement_id & r ) {
669  return r.str();
670  } ) );
671  stack.pop_back();
672  return;
673  }
674  already_nested.insert( r );
675 
676  const requirement_data &req = r.obj();
677  const requirement_data multiplied = req * comp.count;
678 
679  const std::vector<std::vector<T>> &to_inline = getter( multiplied );
680  if( !to_inline.empty() ) {
681  const std::vector<T> &nested = to_inline.front();
682  for( const T &nested_comp : nested ) {
683  rec( nested_comp );
684  }
685  }
686  already_nested.erase( r );
687  stack.pop_back();
688  };
689 
690  rec( comp );
691  };
692 
693  for( std::vector<T> &vec : list ) {
694  const std::vector<T> vec_copy = vec;
695  vec.clear();
696  for( const T &comp : vec_copy ) {
697  inline_one( comp, vec );
698  }
699  }

1043 bool requirement_data::check_enough_materials( const item_comp &comp,
1044  const read_only_visitable &crafting_inv,
1045  const std::function<bool( const item & )> &filter, int batch ) const
1046 {
1047  if( comp.available != available_status::a_true ) {
1048  return false;
1049  }
1050  const int cnt = std::abs( comp.count ) * batch;
1051  const tool_comp *tq = find_by_type( tools, comp.type );
1052  if( tq != nullptr && tq->available == available_status::a_true ) {
1053  // The very same item type is also needed as tool!
1054  // Use charges of it, or use it by count?
1055  const int tc = tq->by_charges() ? 1 : std::abs( tq->count );
1056  // Check for components + tool count. Check item amount (excludes
1057  // pseudo items) and tool amount (includes pseudo items)
1058  // Imagine: required = 1 welder (component) + 1 welder (tool),
1059  // available = 1 welder (real item), 1 welding rig (creates
1060  // a pseudo welder item). has_components(welder,2) returns false
1061  // as there is only one real welder available, but has_tools(welder,2)
1062  // returns true.
1063  // Keep in mind that both requirements (tool+component) are checked
1064  // before this. That assures that one real item is actually available,
1065  // two welding rigs (and no real welder) would make this component
1066  // non-available even before this function is called.
1067  // Only ammo and (some) food is counted by charges, both are unlikely
1068  // to appear as tool, but it's possible /-:
1069  const item_comp i_tmp( comp.type, cnt + tc );
1070  const tool_comp t_tmp( comp.type, -( cnt + tc ) ); // not by charges!
1071  // batch factor is explicitly 1, because it's already included in the count.
1072  if( !i_tmp.has( crafting_inv, filter, 1 ) && !t_tmp.has( crafting_inv, filter, 1 ) ) {
1073  comp.available = available_status::a_insufficient;
1074  }
1075  }
1076  const itype *it = item::find_type( comp.type );
1077  for( const auto &ql : it->qualities ) {
1078  const quality_requirement *qr = find_by_type( qualities, ql.first );
1079  if( qr == nullptr || qr->level > ql.second ) {
1080  continue;
1081  }
1082  // This item can be used for the quality requirement, same as above for specific
1083  // tools applies.
1084  if( !crafting_inv.has_quality( qr->type, qr->level, qr->count + std::abs( comp.count ) ) ) {
1085  comp.available = available_status::a_insufficient;
1086  }
1087  }
1088  return comp.available == available_status::a_true;
1089 }

1334 template<typename T, typename Accum>
1335 static std::vector<std::vector<T>> consolidate( std::vector<std::vector<T>> old_vec,
1336  const Accum &accum )
1337 {
1338  const auto type_lt = []( const T & lhs, const T & rhs ) -> bool {
1339  //TODO change to use localized sorting
1340  return std::forward_as_tuple( lhs.type.str(), lhs.requirement )
1341  < std::forward_as_tuple( rhs.type.str(), rhs.requirement );
1342  };
1343  // in order to simplify blueprint requirements, we merge a longer requirement
1344  // list into a shorter requirement list whose types are a subsequence of the
1345  // longer list's types. However, this operation is not symmetric and depends
1346  // on the order of the requirement lists. Thus we sort the lists first, to
1347  // ensure consistent results when the order of construction requirements changes.
1348  for( std::vector<T> &old_inner : old_vec ) {
1349  std::sort( old_inner.begin(), old_inner.end(), type_lt );
1350  }
1351  std::sort( old_vec.begin(), old_vec.end(),
1352  [&type_lt]( const std::vector<T> &lhs, const std::vector<T> &rhs ) -> bool {
1353  return std::lexicographical_compare( lhs.begin(), lhs.end(), rhs.begin(), rhs.end(),
1354  type_lt );
1355  } );
1356 
1357  std::vector<std::vector<T>> new_vec;
1358  for( std::vector<T> &old_inner : old_vec ) {
1359  bool match = false;
1360  for( std::vector<T> &new_inner : new_vec ) {
1361  // in order to simplify blueprint requirements, we merge a longer
1362  // requirement list into a shorter requirement list whose types are
1363  // a subsequence of the longer list's types.
1364  //
1365  // note this actually may make a requirement stricter.
1366  // for example, if the item requirement was
1367  // [ [ [ "a", 1 ], [ "b", 1 ], [ "c", 1 ] ],
1368  // [ [ "a", 1 ], [ "b", 1 ] ] ]
1369  // then you could satisfy it by having one "a" and one "b", one
1370  // "c" and one "a", two "a", or two "b", etc.
1371  //
1372  // but after consolidation, it becomes
1373  // [ [ [ "a", 2 ], [ "b", 2 ] ] ]
1374  // then you can only satisfy it by having either 2 "a" or 2 "b"
1375  if( std::includes( new_inner.begin(), new_inner.end(),
1376  old_inner.begin(), old_inner.end(),
1377  type_lt ) ) {
1378  // old_inner is a subsequence of new_inner
1379  match = true;
1380  std::swap( old_inner, new_inner );
1381  } else if( std::includes( old_inner.begin(), old_inner.end(),
1382  new_inner.begin(), new_inner.end(),
1383  type_lt ) ) {
1384  // new_inner is a subsequence of old_inner
1385  match = true;
1386  }
1387  if( match ) {
1388  for( auto it1 = new_inner.begin(), it2 = old_inner.begin();
1389  it1 < new_inner.end(); ++it2 ) {
1390  if( !type_lt( *it2, *it1 ) ) {
1391  // which means *it2 and *it1 have the same type, since
1392  // we know new_inner is a subsequence of old_inner
1393  *it1 = accum( *it1, *it2 );
1394  ++it1;
1395  }
1396  }
1397  break;
1398  }
1399  }
1400  if( !match ) {
1401  new_vec.emplace_back( old_inner );
1402  }
1403  }
1404  return new_vec;
1405 }

1500 /// Helper function for deduped_requirement_data constructor below.
1501 ///
1502 /// The goal of this function is to consolidate a particular item_comp that
1503 /// would otherwise be duplicated between two requirements.
1504 ///
1505 /// It operates recursively (increasing @p index with the depth of recursion),
1506 /// searching for another item_comp to merge @p leftover with. For each
1507 /// compatible item_comp found it performs that merger and writes out a
1508 /// suitably updated form of the overall requirements to @p result.
1509 ///
1510 /// If it chooses *not* to merge with any particular item_comp, then it deletes
1511 /// that item_comp from the options, to avoid the duplication.
1512 ///
1513 /// Lastly, it also outputs a version of the requirements where @p leftover
1514 /// remains where it was, and all other compatible item_comp entries have been
1515 /// deleted.
1516 ///
1517 /// @param leftover The item_comp needing to be dealt with.
1518 /// @param req_prefix The requirements considered so far; more will be appended
1519 /// to this.
1520 /// @param to_expand The original requirements we are working through to look
1521 /// for a duplicate.
1522 /// @param orig_index The index into the alter_item_comp_vector where @p
1523 /// leftover was originally to be found. If it isn't merged with another item,
1524 /// then it will be re-inserted at this position.
1525 /// @param index The position within @p to_expand where we will next look for
1526 /// duplicates of @p leftover to merge with.
1527 /// @param result The finished requirements should be appended to this.
1528 static void expand_item_in_reqs(
1529  const item_comp &leftover, requirement_data::alter_item_comp_vector req_prefix,
1530  const requirement_data::alter_item_comp_vector &to_expand, size_t orig_index, size_t index,
1531  std::vector<requirement_data::alter_item_comp_vector> &result )
1532 {
1533  cata_assert( req_prefix.size() >= orig_index );
1534  cata_assert( orig_index < index );
1535 
1536  if( index == to_expand.size() ) {
1537  // We reached the end without using the leftovers. So need to add them
1538  // as their own requirement, separate from everything else.
1539  req_prefix.insert( req_prefix.begin() + orig_index, { leftover } );
1540  result.push_back( req_prefix );
1541  return;
1542  }
1543 
1544  std::vector<item_comp> this_requirement = to_expand[index];
1545  auto duplicate = std::find_if( this_requirement.begin(), this_requirement.end(),
1546  [&]( const item_comp & c ) {
1547  return c.type == leftover.type;
1548  } );
1549  if( duplicate == this_requirement.end() ) {
1550  // No match in this one; proceed to next
1551  req_prefix.push_back( this_requirement );
1552  expand_item_in_reqs( leftover, req_prefix, to_expand, orig_index, index + 1, result );
1553  return;
1554  }
1555  // First option: amalgamate the leftovers into this requirement, which
1556  // forces us to pick that specific option:
1557  requirement_data::alter_item_comp_vector req = req_prefix;
1558  req.push_back( { item_comp( leftover.type, leftover.count + duplicate->count ) } );
1559  req.insert( req.end(), to_expand.begin() + index + 1, to_expand.end() );
1560  result.push_back( req );
1561 
1562  // Second option: use a separate option for this requirement, which means
1563  // we need to recurse further to find something into which to amalgamate
1564  // the original requirement
1565  this_requirement.erase( duplicate );
1566  if( !this_requirement.empty() ) {
1567  req_prefix.push_back( this_requirement );
1568  expand_item_in_reqs( leftover, req_prefix, to_expand, orig_index, index + 1, result );
1569  }
1570 }

1572 deduped_requirement_data::deduped_requirement_data( const requirement_data &in,
1573  const recipe_id &context )
1574 {
1575  // This constructor works through a requirement_data, converting it into an
1576  // equivalent set of requirement_data alternatives, where each alternative
1577  // has the property that no item type appears more than once.
1578  //
1579  // We maintain a queue of requirement_data component info to be split.
1580  // Each to_check struct has a vector of component requirements, and an
1581  // index. The index is the position within the vector to be checked next.
1582  struct to_check {
1583  alter_item_comp_vector components;
1584  size_t index;
1585  };
1586  std::stack<to_check, std::vector<to_check>> pending;
1587  pending.push( { in.get_components(), 0 } );
1588 
1589  // Make sure the tools are not duplicated.
1590  requirement_data::alter_tool_comp_vector tools_unique;
1591  for( auto tools : in.get_tools() ) {
1592  std::sort( tools.begin(), tools.end() );
1593  tools.erase( std::unique( tools.begin(), tools.end() ), tools.end() );
1594  tools_unique.push_back( tools );
1595  }
1596 
1597  while( !pending.empty() ) {
1598  to_check next = pending.top();
1599  pending.pop();
1600 
1601  if( next.index == next.components.size() ) {
1602  alternatives_.emplace_back( tools_unique, in.get_qualities(), next.components );
1603  continue;
1604  }
1605 
1606  // Build a set of all the itypes used in later stages of this set of
1607  // requirements.
1608  std::unordered_set<itype_id> later_itypes;
1609  for( size_t i = next.index + 1; i != next.components.size(); ++i ) {
1610  std::transform( next.components[i].begin(), next.components[i].end(),
1611  std::inserter( later_itypes, later_itypes.end() ),
1612  []( const item_comp & c ) {
1613  return c.type;
1614  } );
1615  }
1616 
1617  std::vector<item_comp> this_requirement = next.components[next.index];
1618 
1619  auto first_duplicated = std::stable_partition(
1620  this_requirement.begin(), this_requirement.end(),
1621  [&]( const item_comp & c ) {
1622  return !later_itypes.count( c.type );
1623  }
1624  );
1625 
1626  for( auto comp_it = first_duplicated; comp_it != this_requirement.end(); ++comp_it ) {
1627  // Factor this requirement out into its own separate case
1628 
1629  alter_item_comp_vector req_prefix( next.components.begin(),
1630  next.components.begin() + next.index );
1631  std::vector<alter_item_comp_vector> result;
1632  expand_item_in_reqs( *comp_it, req_prefix, next.components, next.index, next.index + 1,
1633  result );
1634  for( const alter_item_comp_vector &v : result ) {
1635  // When v is smaller, that means the current requirement was
1636  // deleted, in which case we don't advance index.
1637  size_t index_inc = v.size() == next.components.size() ? 1 : 0;
1638  pending.push( { v, next.index + index_inc } );
1639  }
1640  }
1641 
1642  // Deal with all the remaining, non-duplicated ones
1643  this_requirement.erase( first_duplicated, this_requirement.end() );
1644  if( !this_requirement.empty() ) {
1645  alter_item_comp_vector without_dupes = next.components;
1646  without_dupes[next.index] = this_requirement;
1647  pending.push( { without_dupes, next.index + 1 } );
1648  }
1649 
1650  if( alternatives_.empty() && pending.empty() ) {
1651  debugmsg( "Recipe definition %s somehow has no valid recipes!", context.str() );
1652  }
1653 
1654  // Because this algorithm is super-exponential in the worst case, add a
1655  // sanity check to prevent things getting too far out of control.
1656  // The worst case in the core game currently is boots_fur
1657  // with 104 alternatives.
1658  static constexpr size_t max_alternatives = 105;
1659  if( alternatives_.size() + pending.size() > max_alternatives ) {
1660  debugmsg( "Construction of deduped_requirement_data generated too many alternatives. "
1661  "The recipe %1s should be simplified. See the Recipe section in "
1662  "doc/JSON_INFO.md for more details. It has %2s alternatives.", context.str(),
1663  alternatives_.size() + pending.size() );
1664  is_too_complex_ = true;
1665  alternatives_ = { in };
1666  return;
1667  }
1668  }
