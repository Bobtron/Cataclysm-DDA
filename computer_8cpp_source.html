140 void computer::load_legacy_data( const std::string &data )
141 {
142  options.clear();
143  failures.clear();
144 
145  std::istringstream dump( data );
146  dump.imbue( std::locale::classic() );
147 
148  dump >> name >> security >> mission_id;
149 
150  name = string_replace( name, "_", " " );
151 
152  // Pull in options
153  int optsize;
154  dump >> optsize;
155  for( int n = 0; n < optsize; n++ ) {
156  std::string tmpname;
157 
158  int tmpaction;
159  int tmpsec;
160 
161  dump >> tmpname >> tmpaction >> tmpsec;
162  // Legacy missile launch option that got removed before `computer_action` was
163  // refactored to be saved and loaded as string ids. Do not change this number:
164  // `computer_action` now has different underlying values from back then!
165  if( tmpaction == 15 ) {
166  continue;
167  }
168  add_option( string_replace( tmpname, "_", " " ), computer_action_from_legacy_enum( tmpaction ),
169  tmpsec );
170  }
171 
172  // Pull in failures
173  int failsize;
174  dump >> failsize;
175  for( int n = 0; n < failsize; n++ ) {
176  int tmpfail;
177  dump >> tmpfail;
178  add_failure( computer_failure_type_from_legacy_enum( tmpfail ) );
179  }
180 
181  std::string tmp_access_denied;
182  dump >> tmp_access_denied;
183 
184  // For backwards compatibility, only set the access denied message if it
185  // isn't empty. This is to avoid the message becoming blank when people
186  // load old saves.
187  if( !tmp_access_denied.empty() ) {
188  access_denied = string_replace( tmp_access_denied, "_", " " );
189  }
190 }
