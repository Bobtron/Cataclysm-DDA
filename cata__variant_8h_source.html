106 // given value type. e.g. type_for<mtype_id>() == cata_variant_type::mtype_id.

108 template<typename T, size_t... I>
109 constexpr cata_variant_type type_for_impl( std::index_sequence<I...> )
110 {
111  constexpr size_t num_types = static_cast<size_t>( cata_variant_type::num_types );
112  constexpr std::array<bool, num_types> matches = {{
113  std::is_same_v<T, typename convert<static_cast<cata_variant_type>( I )>::type>...
114  }
115  };
116  for( size_t i = 0; i < num_types; ++i ) {
117  if( matches[i] ) {
118  return static_cast<cata_variant_type>( i );
119  }
120  }
121  // No match
122  return cata_variant_type::num_types;
123 }

436  // Default constructor for an 'empty' variant (you can't get a value
437  // from it).
438  cata_variant() : type_( cata_variant_type::void_ ) {}

440  // Constructor that attempts to infer the type from the type of the
441  // value passed.
442  template < typename Value,
443  typename = std::enable_if_t <(
444  cata_variant_type_for<Value>() < cata_variant_type::num_types )>>
445  explicit cata_variant( Value && value ) {
446  constexpr cata_variant_type Type = cata_variant_type_for<Value>();
447  type_ = Type;
448  value_ =
449  cata_variant_detail::convert<Type>::to_string( std::forward<Value>( value ) );
450  }
451 
452  // Call this function to unambiguously construct a cata_variant instance,
453  // pass the type as a template parameter.
454  // In cases where the value type alone is unambiguous, the above
455  // simpler constructor call can be used.
456  template<cata_variant_type Type, typename Value>
457  static cata_variant make( Value &&value ) {
458  return cata_variant(
459  Type, cata_variant_detail::convert<Type>::to_string(
460  std::forward<Value>( value ) ) );
461  }
462 
463  // Call this to construct from a type + string. This should rarely be
464  // necessary, so think twice before using it (that's why the equivalent
465  // constructor is private).
466  static cata_variant from_string( cata_variant_type t, std::string &&v ) {
467  return cata_variant( t, std::move( v ) );
468  }
469 
470  cata_variant_type type() const {
471  return type_;
472  }
