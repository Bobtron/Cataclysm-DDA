59 {
60  Character &player_character = get_player_character();

63  if( !critter.type->in_species( species_FUNGUS ) ) {
64  add_msg_if_player_sees( p, _( "The %s is covered in tiny spores!" ), critter.name() );
65  }
66  if( !critter.make_fungus() ) {
67  // Don't insta-kill non-fungables. Jabberwocks, for example
68  critter.add_effect( effect_stunned, rng( 1_turns, 3_turns ) );
69  critter.apply_damage( origin, bodypart_id( "torso" ), rng( 25, 50 ) );
70  }
71  } else if( player_character.pos() == p ) {
72  // TODO: Make this accept NPCs when they understand fungals
73  ///\EFFECT_DEX increases chance of knocking fungal spores away with your TAIL_CATTLE
74  ///\EFFECT_MELEE increases chance of knocking fungal sports away with your TAIL_CATTLE
75  if( player_character.has_trait( trait_TAIL_CATTLE ) &&
76  one_in( 20 - player_character.dex_cur - player_character.get_skill_level( skill_melee ) ) ) {
77  add_msg( _( "The spores land on you, but you quickly swat them off with your tail!" ) );
78  return;
79  }
80  // Spores hit the player--is there any hope?
81  bool hit = false;

83  player_character.add_env_effect( effect_spores, bodypart_id( "head" ), 3, 9_minutes,
84  bodypart_id( "head" ) );
85  hit |= one_in( 2 ) &&
86  player_character.add_env_effect( effect_spores, bodypart_id( "torso" ), 3, 9_minutes,
87  bodypart_id( "torso" ) );

111  }
112  }
113  } else {

250 void fungal_effects::spread_fungus( const tripoint &p )
251 {
252  int growth = 1;
253  map &here = get_map();
254  for( const tripoint &tmp : here.points_in_radius( p, 1 ) ) {
255  if( tmp == p ) {
256  continue;
257  }
258  if( here.has_flag( ter_furn_flag::TFLAG_FUNGUS, tmp ) ) {
259  growth += 1;
260  }
261  }
262 
263  if( !here.has_flag_ter( ter_furn_flag::TFLAG_FUNGUS, p ) ) {
264  spread_fungus_one_tile( p, growth );
265  } else {
266  // Everything is already fungus
267  if( growth == 9 ) {
268  return;
269  }
270  for( const tripoint &dest : here.points_in_radius( p, 1 ) ) {
271  // One spread on average
272  if( !here.has_flag( ter_furn_flag::TFLAG_FUNGUS, dest ) && one_in( 9 - growth ) ) {
273  //growth chance is 100 in X simplified
274  spread_fungus_one_tile( dest, 10 );
275  }
276  }
277  }
