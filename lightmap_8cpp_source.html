666 map::apparent_light_info map::apparent_light_helper( const level_cache &map_cache,
667  const tripoint &p )
668 {
669  avatar const &u = get_avatar();
670  const int dist = rl_dist( u.pos(), p );
671  const float abs_vis =
672  std::max( map_cache.seen_cache[p.x][p.y], map_cache.camera_cache[p.x][p.y] );
673  const float vis = dist > u.unimpaired_range() ? map_cache.camera_cache[p.x][p.y] : abs_vis;
674  const bool obstructed = vis <= LIGHT_TRANSPARENCY_SOLID + 0.1;
675  const bool abs_obstructed = abs_vis <= LIGHT_TRANSPARENCY_SOLID + 0.1;
676 
677  auto is_opaque = [&map_cache]( const point & p ) {
678  return map_cache.transparency_cache[p.x][p.y] <= LIGHT_TRANSPARENCY_SOLID &&
679  map_cache.vision_transparency_cache[p.x][p.y] <= LIGHT_TRANSPARENCY_SOLID;
680  };
681 
682  // possibly reduce view if aiming (also blocks light)
683  if( get_avatar().recoil < MAX_RECOIL ) {
684  if( get_avatar().cant_see( p ) ) {
685  return { true, true, 0.0 };
686  }
687  }
688 
689  const bool p_opaque = is_opaque( p.xy() );
690  float apparent_light;
691 
692  if( p_opaque && vis > 0 ) {
693  // This is the complicated case. We want to check which quadrants the
694  // player can see the tile from, and only count light values from those
695  // quadrants.
696  struct offset_and_quadrants {
697  point offset;
698  std::array<quadrant, 2> quadrants;
699  };
700  static constexpr std::array<offset_and_quadrants, 8> adjacent_offsets = {{
701  { point_south, {{ quadrant::SE, quadrant::SW }} },
702  { point_north, {{ quadrant::NE, quadrant::NW }} },
703  { point_east, {{ quadrant::SE, quadrant::NE }} },
704  { point_south_east, {{ quadrant::SE, quadrant::SE }} },
705  { point_north_east, {{ quadrant::NE, quadrant::NE }} },
706  { point_west, {{ quadrant::SW, quadrant::NW }} },
707  { point_south_west, {{ quadrant::SW, quadrant::SW }} },
708  { point_north_west, {{ quadrant::NW, quadrant::NW }} },
709  }
710  };
711 
712  four_quadrants seen_from( 0 );
713  for( const offset_and_quadrants &oq : adjacent_offsets ) {
714  const point neighbour = p.xy() + oq.offset;
715 
716  if( !lightmap_boundaries.contains( neighbour ) ) {
717  continue;
718  }
719  if( is_opaque( neighbour ) ) {
720  continue;
721  }
722  if( ( rl_dist( u.pos().xy(), neighbour ) > u.unimpaired_range() &&
723  map_cache.camera_cache[neighbour.x][neighbour.y] == 0 ) ||
724  ( map_cache.seen_cache[neighbour.x][neighbour.y] == 0 &&
725  map_cache.camera_cache[neighbour.x][neighbour.y] == 0 ) ) {
726  continue;
727  }
728  // This is a non-opaque visible neighbour, so count visibility from the relevant
729  // quadrants
730  seen_from[oq.quadrants[0]] = vis;
731  seen_from[oq.quadrants[1]] = vis;
732  }
733  apparent_light = ( seen_from * map_cache.lm[p.x][p.y] ).max();
734  } else {
735  // This is the simple case, for a non-opaque tile light from all
736  // directions is equivalent
737  apparent_light = vis * map_cache.lm[p.x][p.y].max();
738  }
739  return { obstructed, abs_obstructed, apparent_light };
740 }

833 template<int xx, int xy, int yx, int yy, typename T, typename Out,
834  T( *calc )( const T &, const T &, const int & ),
835  bool( *check )( const T &, const T & ),
836  void( *update_output )( Out &, const T &, quadrant ),
837  T( *accumulate )( const T &, const T &, const int & )>
838 void castLight( cata::mdarray<Out, point_bub_ms> &output_cache,
839  const cata::mdarray<T, point_bub_ms> &input_array,
840  const point &offset, const int offsetDistance, const T numerator,
841  const int row, float start, const float end, T cumulative_transparency )
842 {
843  constexpr quadrant quad = quadrant_from_x_y( -xx - xy, -yx - yy );
844  float newStart = 0.0f;
845  float radius = 60.0f - offsetDistance;
846  if( start < end ) {
847  return;
848  }
849  T last_intensity( 0.0 );
850  tripoint delta;
851  for( int distance = row; distance <= radius; distance++ ) {
852  delta.y = -distance;
853  bool started_row = false;
854  T current_transparency( 0.0 );
855  float away = start - ( -distance + 0.5f ) / ( -distance -
856  0.5f ); //The distance between our first leadingEdge and start
857 
858  //We initialize delta.x to -distance adjusted so that the commented start < leadingEdge condition below is never false
859  delta.x = -distance + std::max( static_cast<int>( std::ceil( away * ( -distance - 0.5f ) ) ), 0 );
860 
861  for( ; delta.x <= 0; delta.x++ ) {
862  point current( offset.x + delta.x * xx + delta.y * xy, offset.y + delta.x * yx + delta.y * yy );
863  float trailingEdge = ( delta.x - 0.5f ) / ( delta.y + 0.5f );
864  float leadingEdge = ( delta.x + 0.5f ) / ( delta.y - 0.5f );
865 
866  if( !( current.x >= 0 && current.y >= 0 && current.x < MAPSIZE_X &&
867  current.y < MAPSIZE_Y ) /* || start < leadingEdge */ ) {
868  continue;
869  } else if( end > trailingEdge ) {
870  break;
871  }
872  if( !started_row ) {
873  started_row = true;
874  current_transparency = input_array[ current.x ][ current.y ];
875  }
876 
877  const int dist = rl_dist( tripoint_zero, delta ) + offsetDistance;
878  last_intensity = calc( numerator, cumulative_transparency, dist );
879 
880  T new_transparency = input_array[ current.x ][ current.y ];
881 
882  if( check( new_transparency, last_intensity ) ) {
883  update_output( output_cache[current.x][current.y], last_intensity,
884  quadrant::default_ );
885  } else {
886  update_output( output_cache[current.x][current.y], last_intensity, quad );
887  }
888 
889  if( new_transparency == current_transparency ) {
890  newStart = leadingEdge;
891  continue;
892  }
893  // Only cast recursively if previous span was not opaque.
894  if( check( current_transparency, last_intensity ) ) {
895  castLight<xx, xy, yx, yy, T, Out, calc, check, update_output, accumulate>(
896  output_cache, input_array, offset, offsetDistance,
897  numerator, distance + 1, start, trailingEdge,
898  accumulate( cumulative_transparency, current_transparency, distance ) );
899  }
900  // The new span starts at the leading edge of the previous square if it is opaque,
901  // and at the trailing edge of the current square if it is transparent.
902  if( !check( current_transparency, last_intensity ) ) {
903  start = newStart;
904  } else {
905  // Note this is the same slope as the recursive call we just made.
906  start = trailingEdge;
907  }
908  // Trailing edge ahead of leading edge means this span is fully processed.
909  if( start < end ) {
910  return;
911  }
912  current_transparency = new_transparency;
913  newStart = leadingEdge;
914  }
915  if( !check( current_transparency, last_intensity ) ) {
916  // If we reach the end of the span with terrain being opaque, we don't iterate further.
917  break;
918  }
919  // Cumulative average of the transparency values encountered.
920  cumulative_transparency = accumulate( cumulative_transparency, current_transparency, distance );
921  }
922 }

969  * coordinates. Returns a lightmap for a result where the values represent a
970  * percentage of fully lit.
971  *
972  * A value equal to or below 0 means that cell is not in the
973  * field of view, whereas a value equal to or above 1 means that cell is
974  * in the field of view.
975  *
976  * @param origin the starting location
977  * @param target_z Z-level to draw light map on
978  */
979 void map::build_seen_cache( const tripoint &origin, const int target_z, int extension_range,
980  bool cumulative, bool camera, int penalty )
981 {
982  level_cache &map_cache = get_cache( target_z );
983  using mdarray = cata::mdarray<float, point_bub_ms>;
984  mdarray &transparency_cache = map_cache.vision_transparency_cache;
985  mdarray &seen_cache = map_cache.seen_cache;
986  mdarray &camera_cache = map_cache.camera_cache;
987  mdarray &out_cache = camera ? camera_cache : seen_cache;
988 
989  constexpr float light_transparency_solid = LIGHT_TRANSPARENCY_SOLID;
990  constexpr int map_dimensions = MAPSIZE_X * MAPSIZE_Y;
991  if( !cumulative ) {
992  std::uninitialized_fill_n(
993  &camera_cache[0][0], map_dimensions, light_transparency_solid );
994  }
995 
996  // Cache the caches (pointers to them)
997  array_of_grids_of<const float> transparency_caches;
998  array_of_grids_of<float> seen_caches;
999  array_of_grids_of<const bool> floor_caches;
1000  vertical_direction directions_to_cast = vertical_direction::BOTH;
1001  for( int z = -OVERMAP_DEPTH; z <= OVERMAP_HEIGHT; z++ ) {
1002  level_cache &cur_cache = get_cache( z );
1003  transparency_caches[z + OVERMAP_DEPTH] = &cur_cache.vision_transparency_cache;
1004  seen_caches[z + OVERMAP_DEPTH] = camera ? &cur_cache.camera_cache : &cur_cache.seen_cache;
1005  floor_caches[z + OVERMAP_DEPTH] = &cur_cache.floor_cache;
1006  if( !cumulative ) {
1007  std::uninitialized_fill_n(
1008  &( *seen_caches[z + OVERMAP_DEPTH] )[0][0], map_dimensions, light_transparency_solid );
1009  }
1010  cur_cache.seen_cache_dirty = false;
1011  if( origin.z == z && cur_cache.no_floor_gaps ) {
1012  directions_to_cast = vertical_direction::UP;
1013  }
1014  }
1015  if( origin.z == target_z ) {
1016  ( *seen_caches[ target_z + OVERMAP_DEPTH ] )[origin.x][origin.y] = VISIBILITY_FULL;
1017  }
1018 
1019  cast_zlight<float, sight_calc, sight_check, accumulate_transparency>(
1020  seen_caches, transparency_caches, floor_caches, origin, penalty, 1.0,
1021  directions_to_cast );
1022  seen_cache_process_ledges( seen_caches, floor_caches, std::nullopt );
1023 
1024  const optional_vpart_position vp = veh_at( origin );
1025  if( !vp ) {
1026  return;
1027  }
1028  vehicle *const veh = &vp->vehicle();
1029 
1030  // We're inside a vehicle. Do mirror calculations.
1031  std::vector<int> mirrors;
1032  // Do all the sight checks first to prevent fake multiple reflection
1033  // from happening due to mirrors becoming visible due to processing order.
1034  // Cameras are also handled here, so that we only need to get through all vehicle parts once
1035  int cam_control = -1;
1036  for( const vpart_reference &vp : veh->get_all_parts_with_fakes() ) {
1037  if( vp.part().removed || vp.part().is_broken() || !vp.info().has_flag( VPFLAG_EXTENDS_VISION ) ) {
1038  continue;
1039  }
1040  const tripoint mirror_pos = vp.pos();
1041  if( rl_dist( origin, vp.pos() ) > extension_range ) {
1042  continue;
1043  }
1044  // We can utilize the current state of the seen cache to determine
1045  // if the player can see the mirror from their position.
1046  if( !vp.info().has_flag( "CAMERA" ) &&
1047  out_cache[mirror_pos.x][mirror_pos.y] < LIGHT_TRANSPARENCY_SOLID + 0.1 ) {
1048  continue;
1049  } else if( !vp.info().has_flag( "CAMERA_CONTROL" ) ) {
1050  mirrors.emplace_back( static_cast<int>( vp.part_index() ) );
1051  } else {
1052  if( square_dist( origin, mirror_pos ) <= 1 && veh->camera_on ) {
1053  cam_control = static_cast<int>( vp.part_index() );
1054  }
1055  }
1056  }
1057 
1058  for( const int mirror : mirrors ) {
1059  const vehicle_part &vp_mirror = veh->part( mirror );
1060  const vpart_info &vpi_mirror = vp_mirror.info();
1061  const bool is_camera = vpi_mirror.has_flag( "CAMERA" );
1062  if( is_camera && cam_control < 0 ) {
1063  continue; // Player not at camera control, so cameras don't work
1064  }
1065 
1066  const tripoint mirror_pos = veh->global_part_pos3( vp_mirror );
1067 
1068  // Determine how far the light has already traveled so mirrors
1069  // don't cheat the light distance falloff.
1070  int offsetDistance;
1071  mdarray *mocache = &out_cache;
1072  if( !is_camera ) {
