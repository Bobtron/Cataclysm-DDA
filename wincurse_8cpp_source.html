685 // we can probably add support for keycode mode, but wincurse is deprecated
686 // so we just ignore the mode argument.
687 input_event input_manager::get_input_event( const keyboard_mode /*preferred_keyboard_mode*/ )
688 {
689  if( test_mode ) {
690  // input should be skipped in caller's code
691  throw std::runtime_error( "input_manager::get_input_event called in test mode" );
692  }
693 
694  // standards note: getch is sometimes required to call refresh
695  // see, e.g., http://linux.die.net/man/3/getch
696  // so although it's non-obvious, that refresh() call (and maybe InvalidateRect?) IS supposed to be there
697  uint64_t Frequency;
698  QueryPerformanceFrequency( reinterpret_cast<PLARGE_INTEGER>( &Frequency ) );
699  wrefresh( catacurses::stdscr );
700  InvalidateRect( WindowHandle, nullptr, true );
701  lastchar = ERR;
702  if( inputdelay < 0 ) {
703  for( ; lastchar == ERR; Sleep( 1 ) ) {
704  CheckMessages();
705  }
706  } else if( inputdelay > 0 ) {
707  for( uint64_t t0 = GetPerfCount(), t1 = 0; t1 < ( t0 + inputdelay * Frequency / 1000 );
708  t1 = GetPerfCount() ) {
709  CheckMessages();
710  if( lastchar != ERR ) {
711  break;
712  }
713  Sleep( 1 );
714  }
715  } else {
716  CheckMessages();
717  }
718 
719  if( lastchar != ERR && get_option<std::string>( "HIDE_CURSOR" ) == "hidekb" && CursorVisible ) {
720  CursorVisible = false;
721  ShowCursor( false );
722  }
723 
724  previously_pressed_key = 0;
725  input_event rval;
726  if( lastchar == ERR ) {
727  if( input_timeout > 0 ) {
728  rval.type = input_event_t::timeout;
729  } else {
730  rval.type = input_event_t::error;
731  }
732  } else {
733  // == Unicode DELETE
734  if( lastchar == 127 ) {
735  previously_pressed_key = KEY_BACKSPACE;
736  return input_event( KEY_BACKSPACE, input_event_t::keyboard_char );
737  }
738  rval.type = input_event_t::keyboard_char;
739  rval.text = utf32_to_utf8( lastchar );
740  previously_pressed_key = lastchar;
741  // for compatibility only add the first byte, not the code point
742  // as it would conflict with the special keys defined by ncurses
743  rval.add_input( lastchar );
744  }
745 
746  return rval;
747 }
