129 std::pair<int, int> Character::gunmod_installation_odds( const item_location &gun,
130  const item &mod ) const
131 {
132  // Mods with INSTALL_DIFFICULT have a chance to fail, potentially damaging the gun
133  if( !mod.has_flag( flag_INSTALL_DIFFICULT ) || has_trait( trait_DEBUG_HS ) ) {
134  return std::make_pair( 100, 0 );
135  }
136 
137  int roll = 100; // chance of success (%)
138  int risk = 0; // chance of failure (%)
139  float chances = 1.0f; // start with 1 in 6 (~17% chance)
140 
141  for( const auto &e : mod.type->min_skills ) {
142  // gain an additional chance for every level above the minimum requirement
143  skill_id sk = e.first.str() == "weapon" ? gun->gun_skill() : e.first;
144  chances += std::max( get_greater_skill_or_knowledge_level( sk ) - e.second, 0.0f );
145  }
146  // cap success from skill alone to 1 in 5 (~83% chance)
147  roll = std::min( static_cast<double>( chances ), 5.0 ) / 6.0 * 100;
148  // focus is either a penalty or bonus of at most +/-10%
149  roll += ( std::min( std::max( get_focus(), 140 ), 60 ) - 100 ) / 4;
150  // dexterity and intelligence give +/-2% for each point above or below 12
151  roll += ( get_dex() - 12 ) * 2;
152  roll += ( get_int() - 12 ) * 2;
153  // each level of damage to the base gun reduces success by 10%
154  roll -= gun->damage_level() * 10;
155  roll = std::min( std::max( roll, 0 ), 100 );
156 
157  // risk of causing damage on failure increases with less durable guns
158  risk = ( 100 - roll ) * ( ( 10.0 - std::min( gun->type->gun->durability, 9 ) ) / 10.0 );
159 
160  return std::make_pair( roll, risk );
161 }
