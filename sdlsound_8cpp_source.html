419 void play_music( const std::string &playlist )
420 {
421  // Don't interrupt playlist that's already playing.
422  if( playlist == current_playlist ) {
423  return;
424  } else {
425  stop_music();
426  }
427 
428  const auto iter = playlists.find( playlist );
429  if( iter == playlists.end() ) {
430  return;
431  }
432  const music_playlist &list = iter->second;
433  if( list.entries.empty() ) {
434  return;
435  }
436 
437  for( size_t i = 0; i < list.entries.size(); i++ ) {
438  playlist_indexes.push_back( i );
439  }

442  // affects audio, not game logic.
443  // NOLINTNEXTLINE(cata-determinism)
444  static cata_default_random_engine eng = cata_default_random_engine(
445  std::chrono::steady_clock::now().time_since_epoch().count() );
446  std::shuffle( playlist_indexes.begin(), playlist_indexes.end(), eng );
447  }
448 
449  current_playlist = playlist;
450  current_playlist_at = playlist_indexes.at( absolute_playlist_at );
451 
452  const music_playlist::entry &next = list.entries[current_playlist_at];
453  current_music_track_volume = next.volume;
454  play_music_file( next.file, next.volume );
455 }

678 static Mix_Chunk *do_pitch_shift( Mix_Chunk *s, float pitch )
679 {
680  Uint32 s_in = s->alen / 4;
681  Uint32 s_out = static_cast<Uint32>( static_cast<float>( s_in ) * pitch );
682  float pitch_real = static_cast<float>( s_out ) / static_cast<float>( s_in );
683  Mix_Chunk *result = static_cast<Mix_Chunk *>( malloc( sizeof( Mix_Chunk ) ) );
684  result->allocated = 1;
685  result->alen = s_out * 4;
686  result->abuf = static_cast<Uint8 *>( malloc( result->alen * sizeof( Uint8 ) ) );
687  result->volume = s->volume;
688  for( Uint32 i = 0; i < s_out; i++ ) {
689  Sint16 lt = 0;
690  Sint16 rt = 0;
691  Sint16 lt_out = 0;
692  Sint16 rt_out = 0;
693  Sint64 lt_avg = 0;
694  Sint64 rt_avg = 0;
695  Uint32 begin = static_cast<Uint32>( static_cast<float>( i ) / pitch_real );
696  Uint32 end = static_cast<Uint32>( static_cast<float>( i + 1 ) / pitch_real );
697 
698  // check for boundary case
699  if( end > 0 && ( end >= ( s->alen / 4 ) ) ) {
700  end = begin;
701  }
702 
703  for( Uint32 j = begin; j <= end; j++ ) {
704  lt = ( s->abuf[( 4 * j ) + 1] << 8 ) | ( s->abuf[( 4 * j ) + 0] );
705  rt = ( s->abuf[( 4 * j ) + 3] << 8 ) | ( s->abuf[( 4 * j ) + 2] );
706  lt_avg += lt;
707  rt_avg += rt;
708  }
709  lt_out = static_cast<Sint16>( static_cast<float>( lt_avg ) / static_cast<float>
710  ( end - begin + 1 ) );
711  rt_out = static_cast<Sint16>( static_cast<float>( rt_avg ) / static_cast<float>
712  ( end - begin + 1 ) );
713  result->abuf[( 4 * i ) + 1] = static_cast<Uint8>( ( lt_out >> 8 ) & 0xFF );
714  result->abuf[( 4 * i ) + 0] = static_cast<Uint8>( lt_out & 0xFF );
715  result->abuf[( 4 * i ) + 3] = static_cast<Uint8>( ( rt_out >> 8 ) & 0xFF );
716  result->abuf[( 4 * i ) + 2] = static_cast<Uint8>( rt_out & 0xFF );
717  }
718  return result;
719 }
