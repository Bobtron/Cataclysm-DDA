136  aligned_pointer_type
137  last_endpoint; // The address that is one past the highest cell number that's been used so far in this group - does not change with erase command but may change with insert (if no previously-erased locations are available) - is necessary because an iterator cannot access the colony's end_iterator. Most-used variable in colony use (operator ++, --) so first in struct
138  group_pointer_type
139  next_group; // Next group in the intrusive list of all groups. NULL if no next group
140  const aligned_pointer_type elements; // Element storage
141  const skipfield_pointer_type
142  skipfield; // Skipfield storage. The element and skipfield arrays are allocated contiguously, hence the skipfield pointer also functions as a 'one-past-end' pointer for the elements array. There will always be one additional skipfield node allocated compared to the number of elements. This is to ensure a faster ++ iterator operation (fewer checks are required when this is present). The extra node is unused and always zero, but checked, and not having it will result in out-of-bounds memory errors.
143  group_pointer_type
144  previous_group; // previous group in the intrusive list of all groups. NULL if no preceding group
145  skipfield_type
146  free_list_head; // The index of the last erased element in the group. The last erased element will, in turn, contain the number of the index of the next erased element, and so on. If this is == maximum skipfield_type value then free_list is empty i.e. no erasures have occurred in the group (or if they have, the erased locations have then been reused via insert()).
147  const skipfield_type
148  capacity; // The element capacity of this particular group
149  skipfield_type

151  group_pointer_type

278  colony_iterator &operator++() {
279  // covers uninitialized colony_iterator
280  cata_assert( group_pointer != nullptr );
281  // Assert that iterator is not already at end()
282  cata_assert( !( element_pointer == group_pointer->last_endpoint &&
283  group_pointer->next_group != nullptr ) );
284 
285  skipfield_type skip = *( ++skipfield_pointer );
286 
287  // ie. beyond end of available data
288  if( ( element_pointer += skip + 1 ) == group_pointer->last_endpoint &&
289  group_pointer->next_group != nullptr ) {
290  group_pointer = group_pointer->next_group;
291  skip = *( group_pointer->skipfield );
292  element_pointer = group_pointer->elements + skip;
293  skipfield_pointer = group_pointer->skipfield;
294  }
295 
296  skipfield_pointer += skip;
297  return *this;
298  }

464  // In this case we have to redefine the algorithm, rather than using the internal iterator's -- operator, in order for the reverse_iterator to be allowed to reach rend() ie. begin_iterator - 1
465  colony_reverse_iterator &operator++() {
466  colony::group_pointer_type &group_pointer = it.group_pointer;
467  colony::aligned_pointer_type &element_pointer = it.element_pointer;
468  colony::skipfield_pointer_type &skipfield_pointer = it.skipfield_pointer;
469 
470  cata_assert( group_pointer != nullptr );
471  cata_assert( !( element_pointer == group_pointer->elements - 1 &&
472  group_pointer->previous_group == nullptr ) ); // Assert that we are not already at rend()
473 
474  if( element_pointer != group_pointer->elements ) { // ie. not already at beginning of group
475  element_pointer -= *( --skipfield_pointer ) + 1;
476  skipfield_pointer -= *skipfield_pointer;
477 
478  if( !( element_pointer == group_pointer->elements - 1 &&
479  group_pointer->previous_group == nullptr ) ) { // i.e. iterator is not == rend()
480  return *this;
481  }
482  }
483 
484  if( group_pointer->previous_group != nullptr ) { // i.e. not first group in colony
485  group_pointer = group_pointer->previous_group;
486  skipfield_pointer = group_pointer->skipfield + group_pointer->capacity - 1;
487  element_pointer = ( reinterpret_cast<colony::aligned_pointer_type>( group_pointer->skipfield ) - 1 )
488  - *skipfield_pointer;
489  skipfield_pointer -= *skipfield_pointer;
490  } else { // necessary so that reverse_iterator can end up == rend(), if we were already at first element in colony
491  --element_pointer;
492  --skipfield_pointer;

886  if( ( total_number_of_elements != 0 ) & !std::is_trivially_destructible<element_type>::value ) {

890  const aligned_pointer_type end_pointer = begin_iterator.group_pointer->last_endpoint;
891 
892  do {
893  COLONY_DESTROY( element_allocator_type, ( *this ),
894  reinterpret_cast<pointer>( begin_iterator.element_pointer ) );
895  ++begin_iterator.skipfield_pointer;
896  begin_iterator.element_pointer += *begin_iterator.skipfield_pointer + 1;
897  begin_iterator.skipfield_pointer += *begin_iterator.skipfield_pointer;
898  } while( begin_iterator.element_pointer != end_pointer ); // ie. beyond end of available data
899 
900  const group_pointer_type next_group = begin_iterator.group_pointer->next_group;
901  COLONY_DESTROY( group_allocator_type, group_allocator_pair, begin_iterator.group_pointer );
902  COLONY_DEALLOCATE( group_allocator_type, group_allocator_pair, begin_iterator.group_pointer, 1 );
903  begin_iterator.group_pointer =
904  next_group; // required to be before if statement in order for first_group to be NULL and avoid potential double-destruction in future
905 
906  if( next_group == nullptr ) {
907  return;
908  }
909 
910  begin_iterator.element_pointer = next_group->elements + *( next_group->skipfield );
911  begin_iterator.skipfield_pointer = next_group->skipfield + *( next_group->skipfield );
912  }
913  } else { // Avoid iteration for both empty groups and trivially-destructible types eg. POD, structs, classes with empty destructors
914  // Technically under a type-traits-supporting compiler total_number_of_elements could be non-zero at this point, but since begin_iterator.group_pointer would already be NULL in the case of double-destruction, it's unnecessary to zero total_number_of_elements
915  while( begin_iterator.group_pointer != nullptr ) {
916  const group_pointer_type next_group = begin_iterator.group_pointer->next_group;
917  COLONY_DESTROY( group_allocator_type, group_allocator_pair, begin_iterator.group_pointer );
918  COLONY_DEALLOCATE( group_allocator_type, group_allocator_pair, begin_iterator.group_pointer, 1 );
919  begin_iterator.group_pointer = next_group;

947  /**
948  * Inserts the element supplied to the colony, using the object's copy-constructor. Will
949  * insert the element into a previously erased element slot if one exists, otherwise will
950  * insert to back of colony. Returns iterator to location of inserted element.
951  */
952  iterator insert( const element_type &element ) {
953  if( end_iterator.element_pointer != nullptr ) {
954  switch( ( ( groups_with_erasures_list_head != nullptr ) << 1 ) | ( end_iterator.element_pointer ==
955  reinterpret_cast<aligned_pointer_type>( end_iterator.group_pointer->skipfield ) ) ) {
956  case 0: { // ie. there are no erased elements and end_iterator is not at end of current final group
957  // Make copy for return before modifying end_iterator
958  iterator return_iterator = end_iterator;
959 
960  if COLONY_CONSTEXPR( std::is_nothrow_copy_constructible<element_type>::value ) {
961  // For no good reason this compiles to faster code under GCC:
962  COLONY_CONSTRUCT( element_allocator_type, ( *this ),
963  reinterpret_cast<pointer>( end_iterator.element_pointer++ ), element );
964  end_iterator.group_pointer->last_endpoint = end_iterator.element_pointer;
965  } else {
966  COLONY_CONSTRUCT( element_allocator_type, ( *this ),
967  reinterpret_cast<pointer>( end_iterator.element_pointer ), element );
968  // Shift the addition to the second operation, avoiding problems if an exception is thrown during construction
969  end_iterator.group_pointer->last_endpoint = ++end_iterator.element_pointer;
970  }
971 
972  ++( end_iterator.group_pointer->number_of_elements );
973  ++end_iterator.skipfield_pointer;
974  ++total_number_of_elements;
975 
976  return return_iterator; // return value before incrementation
977  }
978  case 1: { // ie. there are no erased elements and end_iterator is at end of current final group - ie. colony is full - create new group
979  end_iterator.group_pointer->next_group = COLONY_ALLOCATE( group_allocator_type,
980  group_allocator_pair, 1, end_iterator.group_pointer );
981  group &next_group = *( end_iterator.group_pointer->next_group );
982  const skipfield_type new_group_size = ( total_number_of_elements < static_cast<size_type>
983  ( group_allocator_pair.max_elements_per_group ) ) ? static_cast<skipfield_type>
984  ( total_number_of_elements ) : group_allocator_pair.max_elements_per_group;
985 
986  try {
987  COLONY_CONSTRUCT( group_allocator_type, group_allocator_pair, &next_group, new_group_size,
988  end_iterator.group_pointer );
989  } catch( ... ) {
990  COLONY_DEALLOCATE( group_allocator_type, group_allocator_pair, &next_group, 1 );
991  end_iterator.group_pointer->next_group = nullptr;
992  throw;
993  }
994 
995  if COLONY_CONSTEXPR( std::is_nothrow_copy_constructible<element_type>::value ) {
996  COLONY_CONSTRUCT( element_allocator_type, ( *this ),
997  reinterpret_cast<pointer>( next_group.elements ), element );
998  } else {
999  try {
1000  COLONY_CONSTRUCT( element_allocator_type, ( *this ),
1001  reinterpret_cast<pointer>( next_group.elements ), element );
1002  } catch( ... ) {
1003  COLONY_DESTROY( group_allocator_type, group_allocator_pair, &next_group );
1004  COLONY_DEALLOCATE( group_allocator_type, group_allocator_pair, &next_group, 1 );
1005  end_iterator.group_pointer->next_group = nullptr;
1006  throw;
1007  }
1008  }
1009 
1010  end_iterator.group_pointer = &next_group;
1011  end_iterator.element_pointer = next_group.last_endpoint;
1012  end_iterator.skipfield_pointer = next_group.skipfield + 1;
1013  ++total_number_of_elements;
1014  total_capacity += new_group_size;
1015 
1016  // returns value before incrementation
1017  return iterator( end_iterator.group_pointer, next_group.elements, next_group.skipfield );
1018  }
1019  default: { // ie. there are erased elements, reuse previous-erased element locations
1020  iterator new_location;
1021  new_location.group_pointer = groups_with_erasures_list_head;
1022  new_location.element_pointer = groups_with_erasures_list_head->elements +
1023  groups_with_erasures_list_head->free_list_head;
1024  new_location.skipfield_pointer = groups_with_erasures_list_head->skipfield +
1025  groups_with_erasures_list_head->free_list_head;
1026 
1027  // always at start of skipblock, update skipblock:
1028  const skipfield_type prev_free_list_index = *( reinterpret_cast<skipfield_pointer_type>
1029  ( new_location.element_pointer ) );
1030 
1031  COLONY_CONSTRUCT( element_allocator_type, ( *this ),
1032  reinterpret_cast<pointer>( new_location.element_pointer ), element );
1033 
1034  // Update skipblock:
1035  const skipfield_type new_value = *( new_location.skipfield_pointer ) - 1;
1036 
1037  // ie. skipfield was not 1, ie. a single-node skipblock, with no additional nodes to update
1038  if( new_value != 0 ) {
1039  // set (new) start and (original) end of skipblock to new value:
1040  *( new_location.skipfield_pointer + new_value ) = *( new_location.skipfield_pointer + 1 ) =
1041  new_value;
1042 
1043  // transfer free list node to new start node:
1044  ++( groups_with_erasures_list_head->free_list_head );
1045 
1046  // i.e. not the tail free list node
1047  if( prev_free_list_index != std::numeric_limits<skipfield_type>::max() ) {
1048  *( reinterpret_cast<skipfield_pointer_type>( new_location.group_pointer->elements +
1049  prev_free_list_index ) + 1 ) = groups_with_erasures_list_head->free_list_head;
1050  }
1051 
1052  *( reinterpret_cast<skipfield_pointer_type>( new_location.element_pointer + 1 ) ) =
1053  prev_free_list_index;
1054  *( reinterpret_cast<skipfield_pointer_type>( new_location.element_pointer + 1 ) + 1 ) =
1055  std::numeric_limits<skipfield_type>::max();
1056  } else {
1057  groups_with_erasures_list_head->free_list_head = prev_free_list_index;
1058 
1059  // i.e. not the last free list node
1060  if( prev_free_list_index != std::numeric_limits<skipfield_type>::max() ) {
1061  *( reinterpret_cast<skipfield_pointer_type>( new_location.group_pointer->elements +
1062  prev_free_list_index ) + 1 ) = std::numeric_limits<skipfield_type>::max();
1063  } else {
1064  groups_with_erasures_list_head = groups_with_erasures_list_head->erasures_list_next_group;
1065  }
1066  }

1740  /**
1741  * Removes the element pointed to by the supplied iterator, from the colony. Returns an
1742  * iterator pointing to the next non-erased element in the colony (or to end() if no more
1743  * elements are available). This must return an iterator because if a colony group becomes
1744  * entirely empty, it will be removed from the colony, invalidating the existing iterator.
1745  * Attempting to erase a previously-erased element results in undefined behavior (this is
1746  * checked for via an assert in debug mode).
1747  *
1748  * must return iterator to subsequent non-erased element (or end()), in case the group
1749  * containing the element which the iterator points to becomes empty after the erasure, and
1750  * is thereafter removed from the colony chain, making the current iterator invalid and
1751  * unusable in a ++ operation:
1752  *
1753  * if uninitialized/invalid iterator supplied, function could generate an exception
1754  */
1755  iterator erase( const const_iterator &it ) {
1756  cata_assert( !empty() );
1757  const group_pointer_type group_pointer = it.group_pointer;
1758  // not uninitialized iterator
1759  cata_assert( group_pointer != nullptr );
1760  // != end()
1761  cata_assert( it.element_pointer != group_pointer->last_endpoint );
1762  // element pointed to by iterator has not been erased previously
1763  cata_assert( *( it.skipfield_pointer ) == 0 );
1764

1766  if COLONY_CONSTEXPR( !std::is_trivially_destructible<element_type>::value ) {
1767  COLONY_DESTROY( element_allocator_type, ( *this ),
1768  reinterpret_cast<pointer>( it.element_pointer ) ); // Destruct element
1769  }

1773  // ie. non-empty group at this point in time, don't consolidate - optimization note: GCC optimizes postfix + 1 comparison better than prefix + 1 comparison in many cases.
1774  if( group_pointer->number_of_elements-- != 1 ) {
1775  // Code logic for following section:
1776  // ---------------------------------
1777  // If current skipfield node has no skipped node on either side, continue as usual
1778  // If node only has skipped node on left, set current node and start node of the skipblock to left node value + 1.
1779  // If node only has skipped node on right, make this node the start node of the skipblock and update end node
1780  // If node has skipped nodes on left and right, set start node of left skipblock and end node of right skipblock to the values of the left + right nodes + 1
1781 
1782  // Optimization explanation:
1783  // The contextual logic below is the same as that in the insert() functions but in this case the value of the current skipfield node will always be
1784  // zero (since it is not yet erased), meaning no additional manipulations are necessary for the previous skipfield node comparison - we only have to check against zero
1785  const char prev_skipfield = *( it.skipfield_pointer - ( it.skipfield_pointer !=
1786  group_pointer->skipfield ) ) != 0;
1787  // NOTE: boundary test (checking against end-of-elements) is able to be skipped due to the extra skipfield node (compared to element field) - which is present to enable faster iterator operator ++ operations
1788  const char after_skipfield = *( it.skipfield_pointer + 1 ) != 0;
1789  skipfield_type update_value = 1;

1792  case 0: { // no consecutive erased elements
1793  *it.skipfield_pointer = 1; // solo skipped node
1794  const skipfield_type index = static_cast<skipfield_type>( it.element_pointer -
1795  group_pointer->elements );
1796

1798  if( group_pointer->free_list_head != std::numeric_limits<skipfield_type>::max() ) {
1799  // set prev free list head's 'next index' number to the index of the current element
1800  *( reinterpret_cast<skipfield_pointer_type>( group_pointer->elements +
1801  group_pointer->free_list_head ) + 1 ) = index;
1802  } else {

1804  group_pointer->erasures_list_next_group = groups_with_erasures_list_head;
1805  groups_with_erasures_list_head = group_pointer;

1809  *( reinterpret_cast<skipfield_pointer_type>( it.element_pointer ) + 1 ) =
1810  std::numeric_limits<skipfield_type>::max();

1812  break;
1813  }
1814  case 1: { // previous erased consecutive elements, none following
1815  *( it.skipfield_pointer - * ( it.skipfield_pointer - 1 ) ) = *it.skipfield_pointer = *
1816  ( it.skipfield_pointer - 1 ) + 1;
1817  break;
1818  }
1819  case 2: { // following erased consecutive elements, none preceding
1820  const skipfield_type following_value = *( it.skipfield_pointer + 1 ) + 1;
1821  *( it.skipfield_pointer + following_value - 1 ) = *( it.skipfield_pointer ) = following_value;
1822

1824  ( it.element_pointer + 1 ) );

1826  ( it.element_pointer + 1 ) + 1 );

1828  *( reinterpret_cast<skipfield_pointer_type>( it.element_pointer ) + 1 ) = following_next;

1835  *( reinterpret_cast<skipfield_pointer_type>( group_pointer->elements + following_previous ) + 1 ) =
1836  index;
1837  }
1838 
1839  if( following_next != std::numeric_limits<skipfield_type>::max() ) {
1840  // Set previous index of next free list node to this node's 'previous' index
1841  *( reinterpret_cast<skipfield_pointer_type>( group_pointer->elements + following_next ) ) = index;
1842  } else {
1843  group_pointer->free_list_head = index;
1844  }
1845

1848  }
1849  case 3: { // both preceding and following consecutive erased elements

1854  *( it.skipfield_pointer - preceding_value ) = *( it.skipfield_pointer + following_value - 1 ) =
1855  preceding_value + following_value;

1859  ( it.element_pointer + 1 ) );
1860  const skipfield_type following_next = *( reinterpret_cast<skipfield_pointer_type>
1861  ( it.element_pointer + 1 ) + 1 );

1865  *( reinterpret_cast<skipfield_pointer_type>( group_pointer->elements + following_previous ) + 1 ) =
1866  following_next;
1867  }
1868 
1869  if( following_next != std::numeric_limits<skipfield_type>::max() ) {
1870  // Set previous index of next free list node to this node's 'previous' index
1871  *( reinterpret_cast<skipfield_pointer_type>( group_pointer->elements + following_next ) ) =
1872  following_previous;
1873  } else {
1874  group_pointer->free_list_head = following_previous;
1875  }

1883  it.skipfield_pointer + update_value );

1885

1890

1896  begin_iterator.group_pointer )
1897  << 1 ) ) {
1898  case 0: { // ie. group_pointer == begin_iterator.group_pointer && group_pointer->next_group == NULL; only group in colony

1981  * Erases all elements of a given colony from first to the element before the last iterator.
1982  * This function is optimized for multiple consecutive erasures and will always be faster
1983  * than sequential single-element erase calls in that scenario.
1984  */
1985  void erase( const const_iterator &iterator1, const const_iterator &iterator2 ) {
1986  // if uninitialized/invalid iterators supplied, function could generate an exception. If iterator1 > iterator2, behavior is undefined.
1987  cata_assert( iterator1 <= iterator2 );
1988 
1989  iterator current = iterator1;
1990 
1991  if( current.group_pointer != iterator2.group_pointer ) {
1992  if( current.element_pointer != current.group_pointer->elements + *
1993  ( current.group_pointer->skipfield ) ) {
1994  // if iterator1 is not the first non-erased element in it's group - most common case
1995  size_type number_of_group_erasures = 0;
1996 
1997  // Now update skipfield:
1998  const aligned_pointer_type end = iterator1.group_pointer->last_endpoint;
1999 
2000  // Schema: first erase all non-erased elements until end of group & remove all skipblocks post-iterator1 from the free_list. Then, either update preceding skipblock or create new one:
2001 
2002  // if trivially-destructible, and no erasures in group, skip while loop below and just jump straight to the location
2003  if( std::is_trivially_destructible<element_type>::value &
2004  ( current.group_pointer->free_list_head == std::numeric_limits<skipfield_type>::max() ) ) {
2005  number_of_group_erasures += static_cast<size_type>( end - current.element_pointer );
2006  } else {
2007  while( current.element_pointer != end ) {
2008  if( *current.skipfield_pointer == 0 ) {
2009  if COLONY_CONSTEXPR( !std::is_trivially_destructible<element_type>::value ) {
2010  COLONY_DESTROY( element_allocator_type, ( *this ),
2011  reinterpret_cast<pointer>( current.element_pointer ) ); // Destruct element
2012  }
2013 
2014  ++number_of_group_erasures;
2015  ++current.element_pointer;
2016  ++current.skipfield_pointer;
2017  } else { // remove skipblock from group:
2018  const skipfield_type prev_free_list_index = *( reinterpret_cast<skipfield_pointer_type>
2019  ( current.element_pointer ) );
2020  const skipfield_type next_free_list_index = *( reinterpret_cast<skipfield_pointer_type>
2021  ( current.element_pointer ) + 1 );
2022 
2023  current.element_pointer += *( current.skipfield_pointer );
2024  current.skipfield_pointer += *( current.skipfield_pointer );
2025 
2026  // if this is the last skipblock in the free list
2027  if( next_free_list_index == std::numeric_limits<skipfield_type>::max() &&

3008  * if the second iterator supplied is before the first iterator supplied in terms of its

3015  // If iterators are the same, return 0
3016  // Otherwise, find which iterator is later in colony, copy that to iterator2. Copy the lower to iterator1.
3017  // If they are not pointing to elements in the same group, process the intermediate groups and add distances,
3018  // skipping manual incrementation in all but the initial and final groups.
3019  // In the initial and final groups, manual incrementation must be used to calculate distance, if there have been no prior erasures in those groups.
3020  // If there are no prior erasures in either of those groups, we can use pointer arithmetic to calculate the distances for those groups.
3021

3023  !( last.group_pointer == nullptr ) ); // Check that they are initialized
3024
