688  //

707  // If the last predecessor is the same type as the current type, overwrite.
708  // Else push the current type.
709  oter_id &last_predecessor = om_predecessors.back();
710  if( last_predecessor->get_type_id() == current_type_id ) {
711  last_predecessor = current_oter;
712  } else {
713  om_predecessors.push_back( current_oter );
714  }
715  }
716  current_oter = id;
717  };
718 
719  current_oter = serialized_predecessors.front();
720  for( size_t i = 1; i < serialized_predecessors.size(); ++i ) {
721  local_set_ter( serialized_predecessors[i] );
722  }
723  local_set_ter( layer[p.z() + OVERMAP_DEPTH].terrain[p.xy()] );
724  }
725  predecessors_.insert_or_assign( p, std::move( om_predecessors ) );
726 
727  // Reuse allocations because it's a good habit.
728  om_predecessors = std::move( serialized_predecessors );
729  om_predecessors.clear();
730  }
731  }
