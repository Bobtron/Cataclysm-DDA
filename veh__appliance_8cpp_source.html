326 // Helper function for selecting a part in the parts list.
327 // If only one part is available, don't prompt the player.
328 static vehicle_part *pick_part( const std::vector<vehicle_part *> &parts,
329  const std::string &query_msg )
330 {
331  if( parts.empty() ) {
332  return nullptr;
333  }
334  vehicle_part *pt = parts.front();
335  if( parts.size() > 1 ) {
336  uilist pt_sel( query_msg, {} );
337  for( const vehicle_part *vpr : parts ) {
338  std::string enttxt;
339  std::string vname = vpr->name();
340  if( !vpr->ammo_current().is_null() && vpr->ammo_current() != fuel_type_battery &&
341  !vpr->get_base().empty() ) {
342  units::volume mult = units::legacy_volume_factor / item::find_type(
343  vpr->ammo_current() )->stack_size;
344  double vcur = to_liter( vpr->ammo_remaining() * mult );
345  double vmax = to_liter( vpr->ammo_capacity( vpr->get_base().only_item().ammo_type() ) * mult );
346  //~ Vehicle part name, capacity (current/max L) and name of contents
347  enttxt = string_format( _( "%1$s (%2$.1f/%3$.1fL %4$s)" ), vname, round_up( vcur, 1 ),
348  round_up( vmax, 1 ), item::nname( vpr->ammo_current() ) );
349  } else {
350  enttxt = vname;
351  }
352  pt_sel.addentry( enttxt );
353  }
354  pt_sel.query();
355  int sel = pt_sel.ret;
356  if( sel < 0 || static_cast<size_t>( sel ) >= parts.size() ) {
357  return nullptr;
358  }
359  pt = parts[sel];
360  }
361  return pt;
362 }
