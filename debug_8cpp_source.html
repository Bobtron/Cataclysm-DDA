854 static std::string debug_resolve_binary( const std::string &binary, std::ostream &out )
855 {
856  if( binary.find( '/' ) != std::string::npos ) {
857  // The easy case, where we have a path to the binary
858  return binary;
859  }
860  // If the binary name has no slashes then it was found via PATH
861  // lookup, and we need to do the same to pass the correct name
862  // to addr2line. An alternative would be to use /proc/self/exe,
863  // but that's Linux-specific.
864  // Obviously this will not work in all situations, but it will
865  // usually do the right thing.

888 static std::optional<uintptr_t> debug_compute_load_offset(
889  const std::string &binary, const std::string &symbol,
890  const std::string &offset_within_symbol_s, void *address, std::ostream &out )
891 {
892  // I don't know a good way to compute this offset. This
893  // seems to work, but I'm not sure how portable it is.
894  //
895  // backtrace_symbols has provided the address of a symbol as loaded
896  // in memory. We use nm to compute the address of the same symbol
897  // in the binary file, and take the difference of the two.
898  //
899  // There are platform-specific functions which can do similar
900  // things (e.g. dladdr1 in GNU libdl) but this approach might
901  // perhaps be more portable and adds no link-time dependencies.
902 
903  uintptr_t offset_within_symbol = std::stoull( offset_within_symbol_s, nullptr, 0 );
904  std::string string_sought = " " + symbol;
905 
906  // We need to try calling nm in two different ways, because one
907  // works for executables and the other for libraries.
908  std::array<const char *, 2> nm_variants = { "nm", "nm -D" };
909  for( const char *nm_variant : nm_variants ) {
910  std::ostringstream cmd;
911  cmd << nm_variant << ' ' << binary << " 2>&1";
912  FILE *nm = popen( cmd.str().c_str(), "re" );
913  if( !nm ) {
914  out << " backtrace: popen(nm) failed: " << strerror( errno ) << "\n";
915  return std::nullopt;
916  }
917 
918  std::array<char, 1024> buf;
919  while( fgets( buf.data(), buf.size(), nm ) ) {
920  std::string line( buf.data() );
921  while( !line.empty() && isspace( line.end()[-1] ) ) {
922  line.erase( line.end() - 1 );
923  }
924  if( string_ends_with( line, string_sought ) ) {
925  std::istringstream line_is( line );
926  uintptr_t symbol_address;
927  line_is >> std::hex >> symbol_address;
928  if( line_is ) {
929  pclose( nm );
930  return reinterpret_cast<uintptr_t>( address ) -
931  ( symbol_address + offset_within_symbol );
932  }
933  }
934  }
935 
936  pclose( nm );
937  }
938 
939  return std::nullopt;
940 }
