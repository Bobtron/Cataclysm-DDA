677 void item_contents::combine( const item_contents &read_input, const bool convert,
678  const bool into_bottom, bool restack_charges, bool ignore_contents )
679 {
680  std::list<item_pocket> mismatched_pockets;
681  std::vector<item> uninserted_items;
682  size_t pocket_index = 0;
683 
684  for( const item &pocket : read_input.additional_pockets ) {
685  add_pocket( pocket );
686  }
687 
688  for( const item_pocket &pocket : read_input.contents ) {
689  if( pocket_index < contents.size() ) {
690  if( convert ) {
691  if( pocket.is_type( pocket_type::MIGRATION ) ||
692  pocket.is_type( pocket_type::CORPSE ) ||
693  pocket.is_type( pocket_type::MAGAZINE ) ||
694  pocket.is_type( pocket_type::MAGAZINE_WELL ) ||
695  pocket.is_type( pocket_type::SOFTWARE ) ||
696  pocket.is_type( pocket_type::EBOOK ) ) {
697  ++pocket_index;
698  for( const item *it : pocket.all_items_top() ) {
699  insert_item( *it, pocket.get_pocket_data()->type, ignore_contents );
700  }
701  continue;
702  } else if( pocket.is_type( pocket_type::MOD ) ) {
703  // skipping mod type pocket because using combine this way requires mods to come first
704  // and to call update_mod_pockets
705  ++pocket_index;
706  continue;
707  }
708  } else {
709  if( pocket.saved_type() == pocket_type::MOD ) {
710  // this is already handled in item_contents::read_mods
711  ++pocket_index;
712  continue;
713  } else if( pocket.saved_type() == pocket_type::MIGRATION ||
714  pocket.saved_type() == pocket_type::CORPSE ) {
715  for( const item *it : pocket.all_items_top() ) {
716  insert_item( *it, pocket.saved_type(), ignore_contents );
717  }
718  ++pocket_index;
719  continue;
720  }
721  }
722  auto current_pocket_iter = contents.begin();
723  std::advance( current_pocket_iter, pocket_index );
724 
725  if( !current_pocket_iter->is_type( pocket.saved_type() ) ) {
726  mismatched_pockets.push_back( pocket );
727  continue;
728  }
729 
730  for( const item *it : pocket.all_items_top() ) {
731  const ret_val<item *> inserted = current_pocket_iter->insert_item( *it,
732  into_bottom, restack_charges, ignore_contents );
733  if( !inserted.success() ) {
734  uninserted_items.push_back( *it );
735  debugmsg( "error: item %s cannot fit into pocket while loading: %s",
736  it->typeId().str(), inserted.str() );
737  }
738  }
739 
740  if( pocket.saved_sealed() ) {
741  current_pocket_iter->seal();
742  }
743  current_pocket_iter->settings = pocket.settings;
744  } else {
745  for( const item *it : pocket.all_items_top() ) {
746  uninserted_items.push_back( *it );
747  }
748  }
749  ++pocket_index;
750  }
751 
752  for( const item_pocket &pocket : mismatched_pockets ) {
753  for( const item *it : pocket.all_items_top() ) {
754  const ret_val<item *> inserted = insert_item( *it, pocket.saved_type(), ignore_contents );
755  if( !inserted.success() ) {
756  uninserted_items.push_back( *it );
757  debugmsg( "error: item %s cannot fit into any pocket while loading: %s",
758  it->typeId().str(), inserted.str() );
759  }
760  }
761  }
762 
763  for( const item &uninserted_item : uninserted_items ) {
764  insert_item( uninserted_item, pocket_type::MIGRATION, ignore_contents );
765  }
766 }

768 struct item_contents::item_contents_helper {
769  // Static helper function to implement the const and non-const versions of
770  // find_pocket_for with less code duplication
771  template<typename ItemContents>
772  using item_pocket_type = std::conditional_t <
773  std::is_const_v<ItemContents>,
774  const item_pocket,
775  item_pocket
776  >;
777 
778  template<typename ItemContents>
779  static ret_val<item_pocket_type<ItemContents>*> find_pocket_for(
780  ItemContents &contents, const item &it, pocket_type pk_type ) {
781  using my_pocket_type = item_pocket_type<ItemContents>;
782  static constexpr item_pocket *null_pocket = nullptr;
783 
784  std::vector<std::string> failure_messages;
785  int num_pockets_of_type = 0;
786 
787  for( my_pocket_type &pocket : contents.contents ) {
788  if( !pocket.is_type( pk_type ) ) {
789  continue;
790  }
791  if( pk_type != pocket_type::CONTAINER &&
792  pk_type != pocket_type::MAGAZINE &&
793  pk_type != pocket_type::MAGAZINE_WELL &&
794  pocket.is_type( pk_type ) ) {
795  return ret_val<my_pocket_type *>::make_success(
796  &pocket, "special pocket type override" );
797  }
798  ++num_pockets_of_type;
799  ret_val<item_pocket::contain_code> ret_contain = pocket.can_contain( it );
800  if( ret_contain.success() ) {
801  return ret_val<my_pocket_type *>::make_success( &pocket, ret_contain.str() );
802  } else {
803  failure_messages.push_back( ret_contain.str() );
804  }
805  }
806 
807  if( failure_messages.empty() ) {
808  return ret_val<my_pocket_type *>::make_failure( null_pocket, _( "pocket with type (%s) not found" ),
809  io::enum_to_string( pk_type ) );
810  }
811  std::sort( failure_messages.begin(), failure_messages.end(), localized_compare );
812  failure_messages.erase(
813  std::unique( failure_messages.begin(), failure_messages.end() ),
814  failure_messages.end() );
815  return ret_val<my_pocket_type *>::make_failure(
816  null_pocket,
817  n_gettext( "pocket unacceptable because %s", "pockets unacceptable because %s",
818  num_pockets_of_type ),
819  enumerate_as_string( failure_messages, enumeration_conjunction::or_ ) );
820  }
821 };

893 std::pair<item_location, item_pocket *> item_contents::best_pocket( const item &it,
894  item_location &this_loc, const item *avoid, const bool allow_sealed, const bool ignore_settings,
895  const bool nested, bool ignore_rigidity )
896 {
897  // @TODO: this could be made better by doing a plain preliminary volume check.
898  // if the total volume of the parent is not sufficient, a child won't have enough either.
899  std::pair<item_location, item_pocket *> ret = { this_loc, nullptr };
900  std::vector<item_pocket *> valid_pockets;
901  for( item_pocket &pocket : contents ) {
902  if( pocket.is_forbidden() ) {
903  continue;
904  }
905  if( !pocket.is_type( pocket_type::CONTAINER ) ) {
906  // best pocket is for picking stuff up.
907  // containers are the only pockets that are available for such
908  continue;
909  }
910  if( !allow_sealed && pocket.sealed() ) {
911  // we don't want to unseal a pocket to put something in it automatically
912  // that needs to be something a player explicitly does
913  continue;
914  }
915  valid_pockets.emplace_back( &pocket );
916  if( !ignore_settings && !pocket.settings.accepts_item( it ) ) {
917  // Item forbidden by whitelist / blacklist
918  continue;
919  }
920  if( !pocket.rigid() && (
921  !pocket.settings.get_item_whitelist().empty() ||
922  !pocket.settings.get_category_whitelist().empty() ||
923  pocket.settings.priority() > 0 ) ) {
924  ignore_rigidity = true;
925  }
926  if( ( !ignore_rigidity && nested && !pocket.rigid() ) || !pocket.can_contain( it ).success() ) {
927  // non-rigid nested pocket makes no sense, item should also be able to fit in parent.
928  continue;
929  }
930  if( ret.second == nullptr || ret.second->better_pocket( pocket, it, /*nested=*/nested ) ) {
931  ret.second = &pocket;
932  }
933  }
934  const bool parent_pkt_selected = !!ret.second;
935  for( item_pocket *pocket : valid_pockets ) {
936  std::pair<item_location, item_pocket *const> nested_content_pocket =
937  pocket->best_pocket_in_contents( this_loc, it, avoid, allow_sealed, ignore_settings );
938  if( !nested_content_pocket.second ||
939  ( !nested_content_pocket.second->rigid() && pocket->remaining_volume() < it.volume() ) ) {
940  // no nested pocket found, or the nested pocket is soft and the parent is full
941  continue;
942  }
943  if( parent_pkt_selected ) {
944  if( ret.second->better_pocket( *nested_content_pocket.second, it, true ) ) {
945  // item is whitelisted in nested pocket, prefer that over parent pocket
946  ret = nested_content_pocket;
947  }
948  continue;
949  }
950  ret = nested_content_pocket;
951  }
952  return ret;
953 }

1242 {

1244  for( item_pocket &pocket : contents ) {
1245  if( pocket.is_type( pocket_type::MAGAZINE_WELL ) ) {
1246  // we are assuming only one magazine per well

1248  return 0;
1249  }

1252  const int res = mag.ammo_consume( qty, pos, nullptr );
