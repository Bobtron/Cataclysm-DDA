100 int64_t recipe::batch_time( const Character &guy, int batch, float multiplier,
101  size_t assistants ) const
102 {
103  // 1.0f is full speed
104  // 0.33f is 1/3 speed
105  if( multiplier == 0.0f ) {
106  // If an item isn't craftable in the dark, show the time to complete as if you could craft it
107  multiplier = 1.0f;
108  }
109 
110  const double local_time = static_cast<double>( time_to_craft_moves( guy ) ) / multiplier;
111 
112  // if recipe does not benefit from batching and we have no assistants, don't do unnecessary additional calculations
113  if( batch_rscale == 0.0 && assistants == 0 ) {
114  return static_cast<int64_t>( local_time ) * batch;
115  }
116 
117  double total_time = 0.0;
118  // if recipe does not benefit from batching but we do have assistants, skip calculating the batching scale factor
119  if( batch_rscale == 0.0f ) {
120  total_time = local_time * batch;
121  } else {
122  // recipe benefits from batching, so batching scale factor needs to be calculated
123  // At batch_rsize, incremental time increase is 99.5% of batch_rscale
124  const double scale = batch_rsize / 6.0f;
125  for( int x = 0; x < batch; x++ ) {
126  // scaled logistic function output
127  const double logf = ( 2.0 / ( 1.0 + std::exp( -( x / scale ) ) ) ) - 1.0;
128  total_time += local_time * ( 1.0 - ( batch_rscale * logf ) );
129  }
130  }
131 
132  //Assistants can decrease the time for production but never less than that of one unit
133  if( assistants == 1 ) {
134  total_time = total_time * .75;
135  } else if( assistants >= 2 ) {
136  total_time = total_time * .60;
137  }
138  if( total_time < local_time ) {
139  total_time = local_time;
140  }
141 
142  return static_cast<int64_t>( total_time );
143 }

1038 static float proficiency_time_malus( const Character &crafter, const recipe_proficiency &prof )
1039 {
1040  if( !crafter.has_proficiency( prof.id )
1041  && !helpers_have_proficiencies( crafter, prof.id )
1042  && prof.time_multiplier > 1.0f
1043  ) {
1044  double malus = prof.time_multiplier - 1.0;
1045  malus *= 1.0 - crafter.crafting_inventory().get_book_proficiency_bonuses().time_factor( prof.id );
1046  double pl = get_aided_proficiency_level( crafter, prof.id );
1047  // Sigmoid function that mitigates 100% of the time malus as pl approaches 1.0
1048  // but has little effect at pl < 0.5. See #49198
1049  malus *= 1.0 - std::pow( 0.5 - 0.5 * std::cos( pl * M_PI ), 2 );
1050  return static_cast<float>( 1.0 + malus );
1051  }
1052  return 1.0f;
1053 }
