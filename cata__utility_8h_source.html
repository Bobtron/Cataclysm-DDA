567 /**
568  * Checks if two sets are equal, ignoring specified elements.
569  * Works as if `ignored_elements` were temporarily erased from both sets before comparison.
570  * @tparam Set type of the set (must be ordered, i.e. std::set, cata::flat_set)
571  * @param set first set to compare
572  * @param set2 second set to compare
573  * @param ignored_elements elements from both sets to ignore
574  * @return true, if sets without ignored elements are equal, false otherwise
575  */
576 template<typename Set, typename T = std::decay_t<decltype( *std::declval<const Set &>().begin() )>>
577 bool equal_ignoring_elements( const Set &set, const Set &set2, const Set &ignored_elements )
578 {
579  // general idea: splits both sets into the ranges bounded by elements from `ignored_elements`
580  // and checks that these ranges are equal
581 
582  // traverses ignored elements in
583  if( ignored_elements.empty() ) {
584  return set == set2;
585  }
586 
587  using Iter = typename Set::iterator;
588  Iter end = ignored_elements.end();
589  Iter cur = ignored_elements.begin();
590  Iter prev = cur;
591  cur++;
592 
593  // first comparing the sets range [set.begin() .. ignored_elements.begin()]
594  if( !std::equal( set.begin(), set.lower_bound( *prev ),
595  set2.begin(), set2.lower_bound( *prev ) ) ) {
596  return false;
597  }
598 
599  // compare ranges bounded by two elements: [`prev` .. `cur`]
600  while( cur != end ) {
601  if( !std::equal( set.upper_bound( *prev ), set.lower_bound( *cur ),
602  set2.upper_bound( *prev ), set2.lower_bound( *cur ) ) ) {
603  return false;
604  }
605  prev = cur;
606  cur++;
607  }
608 
609  // compare the range after the last element of ignored_elements: [ignored_elements.back() .. set.end()]
610  return static_cast<bool>( std::equal( set.upper_bound( *prev ), set.end(),
611  set2.upper_bound( *prev ), set2.end() ) );
612 }

650  while( rbegin != rend && ignore_keys.count( rbegin->first ) == 1 ) {
651  ++rbegin;
652  }
653  if( rbegin == rend ) {
654  break;
655  }
656  }
657  // If we've skipped ignored keys and the keys still don't match,
658  // then the maps are unequal.
659  if( lbegin->first != rbegin->first ) {
660  return false;
661  }
662  }
663  if( lbegin->second != rbegin->second && ignore_keys.count( lbegin->first ) != 1 ) {
664  return false;
665  }
666  // Either the values were equal, or the key was ignored.
667  }
668  // At least one map ran out of keys. The other may still have ignored keys in it.
669  while( lbegin != lend && ignore_keys.count( lbegin->first ) ) {
670  ++lbegin;
671  }
672  while( rbegin != rend && ignore_keys.count( rbegin->first ) ) {
673  ++rbegin;
674  }
675  return lbegin == lend && rbegin == rend;
676 }

720  * Convenience function to get the aggregate value for a list of values.
721  */
722 template<typename T, std::enable_if_t<std::is_integral_v<T>, int> = 0>
723 T aggregate( const std::vector<T> &values, aggregate_type agg_func )
724 {
725  if( values.empty() ) {
726  return T();
727  }
728  switch( agg_func ) {
729  case aggregate_type::FIRST:
730  return values.front();
731  case aggregate_type::LAST:
732  return *values.rbegin();
733  case aggregate_type::MIN:
734  return *std::min_element( values.begin(), values.end() );
735  case aggregate_type::MAX:
736  return *std::max_element( values.begin(), values.end() );
737  case aggregate_type::AVERAGE:
738  case aggregate_type::SUM: {
739  T agg_sum = std::accumulate( values.begin(), values.end(), 0 );
740  return agg_func == aggregate_type::SUM ? agg_sum : agg_sum / values.size();
741  }
742  default:
743  return T();
