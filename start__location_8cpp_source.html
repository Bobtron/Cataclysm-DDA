158 static void add_boardable( const tinymap &m, const tripoint &p, std::vector<tripoint> &vec )
159 {
160  if( m.has_furn( p ) ) {
161  // Don't need to board this up, is already occupied
162  return;
163  }
164  if( m.ter( p ) != ter_t_floor ) {
165  // Other terrain (door, wall, ...), not boarded either
166  return;

179 static void board_up( tinymap &m, const tripoint_range<tripoint> &range )
180 {
181  std::vector<tripoint> furnitures1;
182  std::vector<tripoint> furnitures2;
183  std::vector<tripoint> boardables;
184  for( const tripoint &p : range ) {
185  bool must_board_around = false;
186  const ter_id t = m.ter( p );
187  if( t == ter_t_window_domestic || t == ter_t_window || t == ter_t_window_no_curtains ) {
188  // Windows are always to the outside and must be boarded
189  must_board_around = true;
190  m.ter_set( p, ter_t_window_boarded );
191  } else if( t == ter_t_door_c || t == ter_t_door_locked || t == ter_t_door_c_peep ) {
192  // Only board up doors that lead to the outside
193  if( m.is_outside( p + tripoint_north ) || m.is_outside( p + tripoint_south ) ||
194  m.is_outside( p + tripoint_east ) || m.is_outside( p + tripoint_west ) ) {
195  m.ter_set( p, ter_t_door_boarded );
196  must_board_around = true;
197  } else {
198  // internal doors are opened instead
199  m.ter_set( p, ter_t_door_o );
200  }
201  }
202  if( must_board_around ) {
203  // Board up the surroundings of the door/window
204  for( const tripoint &neigh : points_in_radius( p, 1 ) ) {
205  if( neigh == p ) {
206  continue;
207  }
208  add_boardable( m, neigh, boardables );
209  }
210  }
211  }
212  // Find all furniture that can be used to board up some place
213  for( const tripoint &p : range ) {
214  if( std::find( boardables.begin(), boardables.end(), p ) != boardables.end() ) {
215  continue;
216  }
217  if( !m.has_furn( p ) ) {
218  continue;
219  }
220  // If the furniture is movable and the character can move it, use it to barricade
221  // is workable here as NPCs by definition are not starting the game. (Let's hope.)
222  ///\EFFECT_STR determines what furniture might be used as a starting area barricade
223  if( m.furn( p ).obj().is_movable() &&
224  m.furn( p ).obj().move_str_req < get_player_character().get_str() ) {
225  if( m.furn( p ).obj().movecost == 0 ) {
226  // Obstacles are better, prefer them
227  furnitures1.push_back( p );
228  } else {
229  furnitures2.push_back( p );
230  }
231  }
232  }
233  while( ( !furnitures1.empty() || !furnitures2.empty() ) && !boardables.empty() ) {
234  const tripoint fp = random_entry_removed( furnitures1.empty() ? furnitures2 : furnitures1 );
235  const tripoint bp = random_entry_removed( boardables );
236  m.furn_set( bp, m.furn( fp ) );
237  m.furn_set( fp, furn_str_id::NULL_ID() );
238  map_stack destination_items = m.i_at( bp );
239  for( const item &moved_item : m.i_at( fp ) ) {
240  destination_items.insert( moved_item );
241  }
242  m.i_clear( fp );
243  }
244 }

371 /** Helper for place_player
372  * Flood-fills the area from a given point, then returns the area it found.
373  * Considers unpassable tiles passable if they can be bashed down or opened.
374  * Will return INT_MAX if it reaches upstairs or map edge.
375  * We can't really use map::route here, because it expects a destination
376  * Maybe TODO: Allow "picking up" items or parts of bashable furniture
377  * and using them to help with bash attempts.
378  */
379 static int rate_location( map &m, const tripoint &p,
380  const bool must_be_inside, const bool accommodate_npc,
381  const int bash_str, const int attempt,
382  cata::mdarray<int, point_bub_ms> &checked )
383 {
384  const auto invalid_char_pos = [&]( const tripoint & tp ) -> bool {
385  return ( must_be_inside && m.is_outside( tp ) ) ||
386  m.impassable( tp ) || m.is_divable( tp ) ||
387  m.has_flag( ter_furn_flag::TFLAG_NO_FLOOR, tp );
388  };
389 
390  if( checked[p.x][p.y] > 0 || invalid_char_pos( p ) ||
391  ( accommodate_npc && invalid_char_pos( p + point_north_west ) ) ) {
392  return 0;
393  }
394 
395  // Vector that will be used as a stack
396  std::vector<tripoint> st;
397  st.reserve( MAPSIZE_X * MAPSIZE_Y );
398  st.push_back( p );
399 
400  // If not checked yet and either can be moved into, can be bashed down or opened,
401  // add it on the top of the stack.
402  const auto maybe_add = [&]( const point & add_p, const tripoint & from ) {
403  if( checked[add_p.x][add_p.y] >= attempt ) {
404  return;
405  }
406 
407  const tripoint pt( add_p, p.z );
408  if( m.passable( pt ) ||
409  m.bash_resistance( pt ) <= bash_str ||
410  m.open_door( get_avatar(), pt, !m.is_outside( from ), true ) ) {
411  st.push_back( pt );
412  }
413  };
414 
415  int area = 0;
416  while( !st.empty() ) {
417  area++;
418  const tripoint cur = st.back();
419  st.pop_back();
420 
421  checked[cur.x][cur.y] = attempt;
422  if( cur.x == 0 || cur.x == MAPSIZE_X - 1 ||
423  cur.y == 0 || cur.y == MAPSIZE_Y - 1 ||
424  m.has_flag( ter_furn_flag::TFLAG_GOES_UP, cur ) ) {
425  return INT_MAX;
426  }
427 
428  maybe_add( cur.xy() + point_west, cur );
429  maybe_add( cur.xy() + point_north, cur );
430  maybe_add( cur.xy() + point_east, cur );
431  maybe_add( cur.xy() + point_south, cur );
432  maybe_add( cur.xy() + point_north_west, cur );
433  maybe_add( cur.xy() + point_north_east, cur );
434  maybe_add( cur.xy() + point_south_west, cur );
435  maybe_add( cur.xy() + point_south_east, cur );
436  }
437 
438  return area;
439 }

441 void start_location::place_player( avatar &you, const tripoint_abs_omt &omtstart ) const
442 {
443  // Need the "real" map with it's inside/outside cache and the like.
444  map &here = get_map();
445  // Start us off somewhere in the center of the map
446  you.move_to( midpoint( project_bounds<coords::ms>( omtstart ) ) );
447  here.invalidate_map_cache( here.get_abs_sub().z() );
448  here.build_map_cache( here.get_abs_sub().z() );
449  const bool must_be_inside = flags().count( "ALLOW_OUTSIDE" ) == 0;
450  const bool accommodate_npc = flags().count( "LONE_START" ) == 0;
451  ///\EFFECT_STR allows player to start behind less-bashable furniture and terrain
452  // TODO: Allow using items here
453  const int bash = you.get_str();
454 
455  // Remember biggest found location
456  // Sometimes it may be impossible to automatically found an ideal location
457  // but the player may be more creative than this algorithm and do away with just "good"
458  int best_rate = 0;
459  tripoint best_spot = you.pos();
460  // In which attempt did this area get checked?
461  // We can overwrite earlier attempts, but not start in them
462  cata::mdarray<int, point_bub_ms> checked = {};
463 
464  bool found_good_spot = false;
465 
466  //Check if a start_point zone exists first
467  const zone_manager &mgr = zone_manager::get_manager();
468  for( const auto &i : mgr.get_zones() ) {
469  const zone_data &zone = i.get();
470  if( zone.get_type() == zone_type_ZONE_START_POINT ) {
471  if( here.inbounds( zone.get_center_point() ) ) {
472  found_good_spot = true;
473  best_spot = here.getlocal( zone.get_center_point() );
474  break;
475  }
476  }
477  }
478 
479  // Otherwise, find a random starting spot
480 
481  int tries = 0;
482  const auto check_spot = [&]( const tripoint & pt ) {
483  ++tries;
484  const int rate = rate_location( here, pt, must_be_inside, accommodate_npc, bash, tries, checked );
485  if( best_rate < rate ) {
486  best_rate = rate;
487  best_spot = pt;
488  if( rate == INT_MAX ) {
489  return true;
490  }
491  }
492  return false;
493  };
494 
495  while( !found_good_spot && tries < 100 ) {
496  tripoint rand_point( HALF_MAPSIZE_X + rng( 0, SEEX * 2 - 1 ),
497  HALF_MAPSIZE_Y + rng( 0, SEEY * 2 - 1 ),
498  you.posz() );
499  found_good_spot = check_spot( rand_point );
500  }
501  // If we haven't got a good location by now, screw it and brute force it
502  // This only happens in exotic locations (deep of a science lab), but it does happen
503  if( !found_good_spot ) {
504  tripoint tmp = you.pos();
505  int &x = tmp.x;
506  int &y = tmp.y;
507  for( x = 0; x < MAPSIZE_X; x++ ) {
508  for( y = 0; y < MAPSIZE_Y && !found_good_spot; y++ ) {
509  found_good_spot = check_spot( tmp );
510  }
511  }
512  }
513 
514  you.setpos( best_spot );
515 
516  if( !found_good_spot ) {
517  debugmsg( "Could not find a good starting place for character" );
518  }
519 }
