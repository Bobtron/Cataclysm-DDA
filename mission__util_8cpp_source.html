179 {

188  std::pair<std::string, ot_match_type> temp_pair;

201  if( params.random ) {

203  } else {

205  }

208  // have to see the location beforehand, then we can attempt to create the new terrain.
209  if( target_pos == overmap::invalid_tripoint && params.create_if_necessary &&

231  find_params.existing_only = true;

233  target_pos = overmap_buffer.find_random( origin_pos, find_params );
234  }

237  // to our desired mission terrain.
238  if( target_pos != overmap::invalid_tripoint ) {
239  overmap_buffer.ter_set( target_pos, oter_id( params.overmap_terrain.evaluate( d ) ) );
240  }
241  }
242  }
243  };
244 
245  // First try to get the position where we only allow existing overmaps.
246  get_target_position();
247 
248  if( target_pos == overmap::invalid_tripoint ) {
249  // If it's invalid, then that means we couldn't find it or create it (if allowed) on
250  // our current overmap. We'll now go ahead and try again but allow it to create new overmaps.
251  find_params.existing_only = false;
252  get_target_position();
253  }
254 
255  // If we got here and this is still invalid, it means that we couldn't find it nor create it
256  // on any overmap (new or existing) within the allowed search range.
257  if( target_pos == overmap::invalid_tripoint ) {
258  debugmsg( "Unable to find and assign mission target %s.", params.overmap_terrain.evaluate( d ) );
259  return std::nullopt;
260  }
261  return target_pos;
