48 /* Used for selecting which part to target in a projectile attack
49  * Primarily a template for ease of testing, but can be reused!
50  * To use:
51  * 1. Create a 'Wrapper' class implementing two (static) functions:
52  * T Wrapper::connection(const T &)
53  * given an element, produce the element it is connected to.
54  * double Wrapper::weight(const T &)
55  * given an element, produce the weight of this element relative to other elements in the graph
56  *
57  * 2. Create a targeting_graph<T, Wrapper>, and call generate()
58  * provide generate with a central part,
59  * and a vector of parts which all connect (eventually) to the center part
60  *
61  * 3. Select a part by calling select()
62  * Provide select() with a value for how accurate the hit was,
63  * and a range (min/max) of possible values for accuracy (ensure max > min)
64  * Lower values are more accurate.
65  */
66 template<typename T, typename Wrapper>
67 class targeting_graph

90  add_node( center );
91  root = center;

93  // First, enumerate all of the connections
94  std::map<T, std::set<T>> connects_to;
95  for( const T &cursor : parts ) {
96  // The root part connects to itself - don't add that
97  if( Wrapper::connection( cursor ) == cursor ) {
98  continue;
99  }
100  // Add the connection to the map, if it's not there already
101  auto result = connects_to.emplace( Wrapper::connection( cursor ), std::set<T>() );
102  // And then add the connected to the vector stored there
103  result.first->second.emplace( cursor );
104  }
105 
106  std::list<T> next;
107  next.push_back( center );
108  while( !next.empty() ) {
109  const T current = next.front();
110  next.pop_front();
111  // Look for what connects to me
112  const auto connected_iter = connects_to.find( current );
113  if( connected_iter != connects_to.end() ) {
114  for( const T &cursor : connected_iter->second ) {
115  add( current, cursor );
116  next.push_back( cursor );
117  }
118  connects_to.erase( connected_iter );
119  }
120  // Look for what I connect to
121  for( std::pair<const T, std::set<T>> &entry : connects_to ) {
122  if( entry.second.find( current ) == entry.second.end() ) {
123  continue;
124  }
125  next.push_back( entry.first );
126  add( current, entry.first );

128  }
129  }

132  T select( const double range_min, const double range_max, double value ) const {
133  // First, find the path we will follow
134  // That is, what body parts we will hit with less and less accurate shots
135  std::list<const node *> path;
136  // Of course, start with the center of mass
137  path.push_back( &graph.at( root ) );
138  // While there's still a body part to hit with less accurate shots
139  while( !path.back()->connections.empty() ) {
140  // Select the next body part to hit based on the relative size of the options
141  weighted_float_list<size_t> next;
142  const std::vector<const node *> &connections = path.back()->connections;
143  for( size_t i = 0; i < connections.size(); ++i ) {
144  next.add( i, connections[i]->weight );
145  }
146  path.push_back( connections[*next.pick()] );
147  }
148 
149  // This will make the lowest possible value of 'value' 0,
150  // and the highest possible value 'range_max' - 'range_min'
151  // Or, if range_min is -0.5, range_max is 1.5, this will put value into the range 0-2
152  // And below, we'll have a scale factor of 2 / total_weight
153  value -= range_min;
154 
155  // Now, find the total weight along our path
156  double total_weight = 0.0;
157  for( const node *nd : path ) {
158  total_weight += nd->weight;
159  }
160 
161  // And using this, and the min/max ranges of our target value,
162  // find how to appropriate scale the weight of each node in our path
163  double scale_factor = ( range_max - range_min ) / total_weight;
164  // Then, just walk along the path
165  double accumulated_weight = 0.0;
166  for( const node *nd : path ) {
167  accumulated_weight += nd->weight * scale_factor;
168  // And quit when we've gone far enough that we can't make it to the next part
169  if( accumulated_weight > value ) {
170  return nd->val;
171  }
172  }
173 
174  // And if we made it all the way here, we (somehow) hit with a very inaccurate shot
175  return path.back()->val;
176  }
