20 /**
21  * Transparency 101:
22  * Transparency usually ranges between 0.038 (open air) and 0.38 (regular smoke).
23  * The bigger the value, the more opaque it is (less light goes through).
24  * To make sense of the specific value:
25  * For transparency t, vision becomes "obstructed" (see map::apparent_light_helper) after
26  * ≈ (2.3 / t) consequent tiles of transparency `t` ( derived from 1 / e^(dist * t) = 0.1 )
27  * e.g. for smoke with effective transparency 0.38 it's 2.3/0.38 ≈ 6 tiles
28  * for clean air with t=0.038 dist = 2.3/0.038 ≈ 60 tiles
29  *
30  * Note: LIGHT_TRANSPARENCY_SOLID=0 is a special case (it indicates completely opaque tile)
31  * */
32 constexpr float LIGHT_TRANSPARENCY_SOLID = 0.0f;
33 // Calculated to run out at 60 squares.
34 // Cumulative transparency should drop to 0.1 or lower over 60 squares,
35 // Bright sunlight should drop to LIGHT_AMBIENT_LOW over 60 squares.
