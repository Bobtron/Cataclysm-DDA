166 double logarithmic_range( int min, int max, int pos )
167 {
168  const double LOGI_CUTOFF = 4;
169  const double LOGI_MIN = logarithmic( -LOGI_CUTOFF );
170  const double LOGI_MAX = logarithmic( +LOGI_CUTOFF );
171  const double LOGI_RANGE = LOGI_MAX - LOGI_MIN;
172 
173  if( min >= max ) {
174  debugmsg( "Invalid interval (%d, %d).", min, max );
175  return 0.0;
176  }
177 
178  // Anything beyond (min,max) gets clamped.
179  if( pos <= min ) {
180  return 1.0;
181  } else if( pos >= max ) {
182  return 0.0;
183  }
184 
185  // Normalize the pos to [0,1]
186  double range = max - min;
187  double unit_pos = ( pos - min ) / range;
188 
189  // Scale and flip it to [+LOGI_CUTOFF,-LOGI_CUTOFF]
190  double scaled_pos = LOGI_CUTOFF - 2 * LOGI_CUTOFF * unit_pos;
191 
192  // Get the raw logistic value.
193  double raw_logistic = logarithmic( scaled_pos );
194 
195  // Scale the output to [0,1]
196  return ( raw_logistic - LOGI_MIN ) / LOGI_RANGE;
197 }

773 {
774  static holiday cached_holiday = holiday::none;

784  is_cached = true;

786  bool success = false;

789  std::time_t current_time = time == 0 ? std::time( nullptr ) : time;
790

792 #if defined(_WIN32)

795  if( err == 0 ) {

798 
799 #else

802 
803 #endif

806 
807  const int month = local_time.tm_mon + 1;
808  const int day = local_time.tm_mday;
809  const int wday = local_time.tm_wday;
810  const int year = local_time.tm_year + 1900;
811 
812  /* check date against holidays */
813  if( month == 1 && day == 1 ) {
814  cached_holiday = holiday::new_year;
815  return cached_holiday;
816  }
817  // only run easter date calculation if currently March or April
818  else if( ( month == 3 || month == 4 ) && is_easter( day, month, year ) ) {
819  cached_holiday = holiday::easter;
820  return cached_holiday;
821  } else if( month == 7 && day == 4 ) {
822  cached_holiday = holiday::independence_day;
823  return cached_holiday;
824  }
825  // 13 days seems appropriate for Halloween
826  else if( month == 10 && day >= 19 ) {
827  cached_holiday = holiday::halloween;
828  return cached_holiday;
829  } else if( month == 11 && ( day >= 22 && day <= 28 ) && wday == 4 ) {
830  cached_holiday = holiday::thanksgiving;
831  return cached_holiday;
832  }
833  // For the 12 days of Christmas, my true love gave to me...
834  else if( month == 12 && ( day >= 14 && day <= 25 ) ) {
835  cached_holiday = holiday::christmas;
836  return cached_holiday;
837  }
838  }
839  // fall through to here if localtime fails, or none of the day tests hit
840  cached_holiday = holiday::none;
