264 bool computer_session::hack_attempt( Character &you, int Security ) const
265 {
266  if( Security == -1 ) {
267  Security = comp.security; // Set to main system security if no value passed
268  }
269  const int hack_skill = round( you.get_skill_level( skill_computer ) );
270 
271  // Every time you dig for lab notes, (or, in future, do other suspicious stuff?)
272  // +2 dice to the system's hack-resistance
273  // So practical max files from a given terminal = 5, at 10 Computer
274  if( comp.alerts > 0 ) {
275  Security += ( comp.alerts * 2 );
276  }
277 
278  you.mod_moves( -( 10 * ( 5 + Security * 2 ) / std::max( 1, hack_skill + 1 ) ) );
279 
280  int player_roll = round( you.get_greater_skill_or_knowledge_level(
281  skill_computer ) ); //this relates to the success of the roll for hacking - the practical skill covers the time.
282  ///\EFFECT_INT <8 randomly penalizes hack attempts, 50% of the time
283  if( you.int_cur < 8 && one_in( 2 ) ) {
284  player_roll -= rng( 0, 8 - you.int_cur );
285  ///\EFFECT_INT >8 randomly benefits hack attempts, 33% of the time
286  } else if( you.int_cur > 8 && one_in( 3 ) ) {
287  player_roll += rng( 0, you.int_cur - 8 );
288  }
289 
290  ///\EFFECT_COMPUTER increases chance of successful hack attempt, vs Security level
291  bool successful_attempt = dice( player_roll, 6 ) >= dice( Security, 6 );
