663  const tripoint_abs_omt &omp, int &rota, int &subtile )

668  if( !uistate.overmap_show_forest_trails &&
669  ( cur_ter->get_type_id() == oter_type_forest_trail ) ) {

681  if( ot_type.has_connections() ) {
682  // This would be for connected terrain

687  oter_at( omp + point_east )->get_type_id(),
