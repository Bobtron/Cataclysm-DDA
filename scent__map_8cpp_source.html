136 bool scent_map::inbounds( const tripoint &p ) const
137 {
138  // HACK: This weird long check here is a hack around the fact that scentmap is 2D
139  // A z-level can access scentmap if it is within SCENT_MAP_Z_REACH flying z-level move from player's z-level
140  // That is, if a flying critter could move directly up or down (or stand still) and be on same z-level as player
141  const int levz = get_map().get_abs_sub().z();
142  const bool scent_map_z_level_inbounds = ( p.z == levz ) ||
143  ( std::abs( p.z - levz ) == SCENT_MAP_Z_REACH &&
144  get_map().valid_move( p, tripoint( p.xy(), levz ), false, true ) );
145  if( !scent_map_z_level_inbounds ) {
146  return false;
147  }
148  return inbounds( p.xy() );
149 }

162 void scent_map::update( const tripoint &center, map &m )
163 {
164  // Stop updating scent after X turns of the player not moving.
165  // Once wind is added, need to reset this on wind shifts as well.
166  if( !player_last_position || center != *player_last_position ) {
167  player_last_position.emplace( center );
168  player_last_moved = calendar::turn;
169  } else if( player_last_moved + 1000_turns < calendar::turn ) {
170  return;
171  }
172 
173  // note: the next four intermediate matrices need to be at least
174  // [2*SCENT_RADIUS+3][2*SCENT_RADIUS+1] in size to hold enough data
175  // The code I'm modifying used [MAPSIZE_X]. I'm staying with that to avoid new bugs.
176 
177  // These two matrices are transposed so that x addresses are contiguous in memory
178  scent_array<int> sum_3_scent_y;
179  scent_array<int> squares_used_y;
180 
181  // these are for caching flag lookups
182  scent_array<bool> blocks_scent; // currently only ter_furn_flag::TFLAG_NO_SCENT blocks scent
183  scent_array<bool> reduces_scent;
184 
185  // for loop constants
186  const int scentmap_minx = center.x - SCENT_RADIUS;
187  const int scentmap_maxx = center.x + SCENT_RADIUS;
188  const int scentmap_miny = center.y - SCENT_RADIUS;
189  const int scentmap_maxy = center.y + SCENT_RADIUS;
190 
191  // decrease this to reduce gas spread. Keep it under 125 for
192  // stability. This is essentially a decimal number * 1000.
193  const int diffusivity = 100;
194 
195  // The new scent flag searching function. Should be wayyy faster than the old one.
196  m.scent_blockers( blocks_scent, reduces_scent, point( scentmap_minx - 1, scentmap_miny - 1 ),
197  point( scentmap_maxx + 1, scentmap_maxy + 1 ) );
198  // Sum neighbors in the y direction. This way, each square gets called 3 times instead of 9
199  // times. This cost us an extra loop here, but it also eliminated a loop at the end, so there
200  // is a net performance improvement over the old code. Could probably still be better.
201  // note: this method needs an array that is one square larger on each side in the x direction
202  // than the final scent matrix. I think this is fine since SCENT_RADIUS is less than
203  // MAPSIZE_X, but if that changes, this may need tweaking.
204  for( int x = scentmap_minx - 1; x <= scentmap_maxx + 1; ++x ) {
205  for( int y = scentmap_miny; y <= scentmap_maxy; ++y ) {
206  // remember the sum of the scent val for the 3 neighboring squares that can defuse into
207  sum_3_scent_y[y][x] = 0;
208  squares_used_y[y][x] = 0;
209  for( int i = y - 1; i <= y + 1; ++i ) {
210  if( !blocks_scent[x][i] ) {
211  if( reduces_scent[x][i] ) {
212  // only 20% of scent can diffuse on REDUCE_SCENT squares
213  sum_3_scent_y[y][x] += 2 * grscent[x][i];
214  squares_used_y[y][x] += 2;
215  } else {
216  sum_3_scent_y[y][x] += 10 * grscent[x][i];
217  squares_used_y[y][x] += 10;
218  }
219  }
220  }
221  }
222  }
223 
224  // Rest of the scent map
225  for( int x = scentmap_minx; x <= scentmap_maxx; ++x ) {
226  for( int y = scentmap_miny; y <= scentmap_maxy; ++y ) {
227  int &scent_here = grscent[x][y];
228  if( !blocks_scent[x][y] ) {
229  // to how many neighboring squares do we diffuse out? (include our own square
230  // since we also include our own square when diffusing in)
231  const int squares_used = squares_used_y[y][x - 1]
232  + squares_used_y[y][x]
233  + squares_used_y[y][x + 1];
234 
235  int this_diffusivity;
236  if( !reduces_scent[x][y] ) {
237  this_diffusivity = diffusivity;
238  } else {
239  this_diffusivity = diffusivity / 5; //less air movement for REDUCE_SCENT square
240  }
241  // take the old scent and subtract what diffuses out
242  int temp_scent = scent_here * ( 10 * 1000 - squares_used * this_diffusivity );
243  // neighboring REDUCE_SCENT squares absorb some scent
244  temp_scent -= scent_here * this_diffusivity * ( 90 - squares_used ) / 5;
245  // we've already summed neighboring scent values in the y direction in the previous
246  // loop. Now we do it for the x direction, multiply by diffusion, and this is what
247  // diffuses into our current square.
248  scent_here =
249  ( temp_scent
250  + this_diffusivity * ( sum_3_scent_y[y][x - 1]
251  + sum_3_scent_y[y][x]
252  + sum_3_scent_y[y][x + 1] )
253  ) / ( 1000 * 10 );
254  } else {
255  // this cell blocks scent via NO_SCENT (in json)
256  scent_here = 0;
257  }
258  }
259  }
