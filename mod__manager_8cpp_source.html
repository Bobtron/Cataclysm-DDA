293 bool mod_manager::copy_mod_contents( const t_mod_list &mods_to_copy,
294  const cata_path &output_base_path )
295 {
296  if( mods_to_copy.empty() ) {
297  // nothing to copy, so technically we succeeded already!
298  return true;
299  }
300  std::vector<std::string> search_extensions;
301  search_extensions.emplace_back( ".json" );
302 
303  DebugLog( D_INFO, DC_ALL ) << "Copying mod contents into directory: " << output_base_path;
304 
305  if( !assure_dir_exist( output_base_path ) ) {
306  DebugLog( D_ERROR, DC_ALL ) << "Unable to create or open mod directory at [" << output_base_path <<
307  "] for saving";
308  return false;
309  }
310 
311  for( size_t i = 0; i < mods_to_copy.size(); ++i ) {
312  const MOD_INFORMATION &mod = *mods_to_copy[i];
313 
314  // now to get all of the json files inside of the mod and get them ready to copy
315  auto input_files = get_files_from_path( ".json", mod.path, true, true );
316  auto input_dirs = get_directories_with( search_extensions, mod.path, true );
317 
318  cata_path mod_base_path = mod.path;
319 
320  if( input_files.empty() && mod.path.get_relative_path().filename().u8string() == MOD_SEARCH_FILE ) {
321  // Self contained mod, all data is inside the modinfo.json file
322  input_files.push_back( mod.path );
323  mod_base_path = mod.path.parent_path();
324  }
325 
326  if( input_files.empty() ) {
327  continue;
328  }
329 
330  // create needed directories
331  // NOLINTNEXTLINE(cata-translate-string-literal)
332  const cata_path cur_mod_dir = output_base_path / string_format( "mod_%05d", i + 1 );
333 
334  std::queue<cata_path> dir_to_make;
335  dir_to_make.push( cur_mod_dir );
336  for( cata_path &input_dir : input_dirs ) {
337  dir_to_make.push( cur_mod_dir / input_dir.get_relative_path().lexically_relative(
338  mod_base_path.get_relative_path() ) );
339  }
340 
341  while( !dir_to_make.empty() ) {
342  if( !assure_dir_exist( dir_to_make.front() ) ) {
343  DebugLog( D_ERROR, DC_ALL ) << "Unable to create or open mod directory at [" <<
344  dir_to_make.front() << "] for saving";
345  }
346 
347  dir_to_make.pop();
348  }
349 
350  // trim file paths from full length down to just /data forward
351  for( cata_path &input_file : input_files ) {
352  cata_path output_path = cur_mod_dir / ( input_file.get_relative_path().lexically_relative(
353  mod_base_path.get_relative_path() ) );
354  copy_file( input_file, output_path );
355  }
356  }
357  return true;
358 }
