253 void SkillLevel::train( int amount, float catchup_modifier, float knowledge_modifier,
254  bool allow_multilevel )
255 {
256  if( amount < 0 ) {
257  debugmsg( "train() called with negative xp: %d", amount );
258  return;
259  }
260  // catchup gets faster the higher the level gap gets.
261  float level_gap = 1.0f * std::max( knowledgeLevel(), 1 ) / std::max( level(), 1 );
262  float catchup_amount = amount * catchup_modifier;
263  float knowledge_amount = amount * knowledge_modifier;
264  if( knowledgeLevel() > level() ) {
265  catchup_amount *= level_gap;
266  } else if( knowledgeLevel() == level() && _knowledgeExperience > _exercise ) {
267  // when you're in the same level, the catchup starts to slow down.
268  catchup_amount = amount * std::max( catchup_modifier - 1.0f * _exercise / _knowledgeExperience,
269  1.0f );
270  knowledge_amount = amount * std::max( knowledge_modifier - 0.1f * _exercise / _knowledgeExperience,
271  1.0f );
272  } else {
273  // When your two xp's are equal just do the basic thing.
274  catchup_amount = amount * 1.0f;
275  knowledge_amount = amount * 1.0f;
276  }
277 
278  // Learning knowledge faster than skill, when you're actually practicing, will generate some annoying problems.
279  if( knowledge_amount > catchup_amount * 0.9f ) {
280  knowledge_amount = catchup_amount * 0.9f;
281  }
282 
283  if( unadjustedKnowledgeLevel() >= MAX_SKILL ) {
284  knowledge_amount = 0;
285  }
286 
287  const double scaling = get_option<float>( "SKILL_TRAINING_SPEED" );
288  if( scaling > 0.0 ) {
289  catchup_amount *= scaling;
290  knowledge_amount *= scaling;
291  }
292  _exercise += catchup_amount;
293  if( _exercise < 0 ) {
294  debugmsg( "integer overflow in train() amount=%d catchup_modifier=%g knowledge_modifier=%g level_gap=%g catchup_amount=%g knowledge_amount=%g scaling=%g _exercise=%d",
295  amount, catchup_modifier, knowledge_modifier, level_gap, catchup_amount, knowledge_amount, scaling,
296  _exercise );
297  _exercise -= catchup_amount;
298  return;
299  }
300  _rustAccumulator -= catchup_amount;
301  _knowledgeExperience += knowledge_amount;
302 
303  on_exercise_change( allow_multilevel );
304  practice();
305 }

307 void SkillLevel::knowledge_train( int amount, int npc_knowledge )
308 {
309  float level_gap = 1.0f;
310  // when your _level is the same or 1 level below your knowledge, gain xp at the normal rate.
311  // as your practical skill lags behind your knowledge, it gets harder to contextualize that
312  // theoretical knowledge, and your ability to learn the theory gets slower.
313 
314  // The same formula applies to NPCs teaching you, but in that case the level decreases as their knowledge
315  // level exceeds your own. The best teacher is one who is only somewhat more knowledgeable than you.
316  if( npc_knowledge > 0 ) {
317  // This should later be modified by NPC teaching proficiencies.
318  level_gap = std::max<float>( npc_knowledge - unadjustedKnowledgeLevel(), 1.0f );
319  } else {
320  // Some day this should be affected by json specific to the skill, some skills are more amenable
321  // to book learning.
322  level_gap = std::max<float>( unadjustedKnowledgeLevel() - unadjustedLevel(), 1.0f );
323  }
324  float level_mult = 2.0f / ( level_gap + 1.0f );
325  amount *= level_mult;
326 
327  const double scaling = get_option<float>( "SKILL_TRAINING_SPEED" );
328  if( scaling > 0.0 ) {
329  amount = std::ceil( amount * scaling );
330  }
331  _knowledgeExperience += amount;
332 
333  if( _knowledgeExperience >= 10000 * pow( unadjustedKnowledgeLevel() + 1, 2U ) ) {
334  _knowledgeExperience = 0;
335  ++_knowledgeLevel;
336  }
337 
338 }

347 bool SkillLevel::rust( int rust_resist, float rust_multiplier )
348 {
349  if( ( calendar::turn - _lastPracticed ) < 24_hours ) {
350  // don't rust within the grace period
351  return false;
352  }
353 
354  if( unadjustedLevel() >= MAX_SKILL ) {
355  // don't rust any more once you hit the level cap, at least until we have a way to "pause" rust for a while.
356  return false;
357  }
358 
359  const int level_multiplier = pow( unadjustedLevel() + 1, 2U );
360  float level_exp = level_multiplier * 10000.0f;
361  if( _rustAccumulator > level_exp * 3 ) {
362  // at this point the numbers ahead will be too small to bother. Just cap it off.
363  return false;
364  }
365 
366  // Future plans: Have rust_slowdown impacted by intelligence and other memory-affecting things
367  float rust_slowdown = std::max( static_cast<float>( std::sqrt( _rustAccumulator / level_exp ) ),
368  0.04f );
369 
370  // rust amount starts at 4% of a level's xp, run every 24 hours.
371  // Once the accumulated rust exceeds 16% of a level, rust_amount starts to drop.
372  int rust_amount = level_multiplier * rust_multiplier * 16 / rust_slowdown;
373 
374  if( rust_resist > 0 ) {
375  rust_amount = std::lround( rust_amount * ( std::max( ( 100 - rust_resist ), 0 ) / 100.0 ) );
376  }
377 
378  if( level() == 0 ) {
379  rust_amount = std::min( rust_amount, _exercise );
380  }
381 
382  if( rust_amount <= 0 ) {
383  return false;
384  }
385 
386  _rustAccumulator += rust_amount;
387  _exercise -= rust_amount;
388  on_exercise_change();
389 
390  return false;
391 }

419 {
420  if( _exercise < 0 ) {
