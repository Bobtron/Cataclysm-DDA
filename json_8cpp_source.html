1073 static bool get_escaped_or_unicode( std::istream &stream, std::string &s, std::string &err )
1074 {
1075  if( !stream.good() ) {
1076  err = "stream not good";
1077  return false;
1078  }
1079  char ch;
1080  stream.get( ch );
1081  if( !stream.good() ) {
1082  err = "read operation failed";
1083  return false;
1084  }
1085  if( ch == '\\' ) {
1086  // converting \", \\, \/, \b, \f, \n, \r, \t and \uxxxx according to JSON spec.
1087  stream.get( ch );
1088  if( !stream.good() ) {
1089  err = "read operation failed";
1090  return false;
1091  }
1092  switch( ch ) {
1093  case '\\':
1094  s += '\\';
1095  break;
1096  case '"':
1097  s += '"';
1098  break;
1099  case '/':
1100  s += '/';
1101  break;
1102  case 'b':
1103  s += '\b';
1104  break;
1105  case 'f':
1106  s += '\f';
1107  break;
1108  case 'n':
1109  s += '\n';
1110  break;
1111  case 'r':
1112  s += '\r';
1113  break;
1114  case 't':
1115  s += '\t';
1116  break;
1117  case 'u': {
1118  uint32_t u = 0;
1119  for( int i = 0; i < 4; ++i ) {
1120  stream.get( ch );
1121  if( !stream.good() ) {
1122  err = "read operation failed";
1123  return false;
1124  }
1125  if( ch >= '0' && ch <= '9' ) {
1126  u = ( u << 4 ) | ( ch - '0' );
1127  } else if( ch >= 'a' && ch <= 'f' ) {
1128  u = ( u << 4 ) | ( ch - 'a' + 10 );
1129  } else if( ch >= 'A' && ch <= 'F' ) {
1130  u = ( u << 4 ) | ( ch - 'A' + 10 );
1131  } else {
1132  err = "expected hex digit";
1133  return false;
1134  }
1135  }
1136  s += utf32_to_utf8( u );
1137  }
1138  break;
1139  default:
1140  err = "invalid escape sequence";
1141  return false;
1142  }
1143  } else if( ch == '\r' || ch == '\n' ) {
1144  err = "reached end of line without closing string";
1145  return false;
1146  } else if( ch == '"' ) {
1147  // the caller is supposed to handle the ending quote
1148  err = "unexpected ending quote";
1149  return false;
1150  } else if( static_cast<unsigned char>( ch ) < 0x20 ) {
1151  err = "invalid character inside string";
1152  return false;
1153  } else {
1154  unsigned char uc = static_cast<unsigned char>( ch );
1155  uint32_t unicode = 0;
1156  int n = 0;
1157  if( uc >= 0xFC ) {
1158  unicode = uc & 0x01;
1159  n = 5;
1160  } else if( uc >= 0xF8 ) {
1161  unicode = uc & 0x03;
1162  n = 4;
1163  } else if( uc >= 0xF0 ) {
1164  unicode = uc & 0x07;
1165  n = 3;
1166  } else if( uc >= 0xE0 ) {
1167  unicode = uc & 0x0f;
1168  n = 2;
1169  } else if( uc >= 0xC0 ) {
1170  unicode = uc & 0x1f;
1171  n = 1;
1172  } else if( uc >= 0x80 ) {
1173  err = "invalid utf8 sequence";
1174  return false;
1175  } else {
1176  unicode = uc;
1177  n = 0;
1178  }
1179  s += ch;
1180  for( ; n > 0; --n ) {
1181  stream.get( ch );
1182  if( !stream.good() ) {
1183  err = "read operation failed";
1184  return false;
1185  }
1186  uc = static_cast<unsigned char>( ch );
1187  if( uc < 0x80 || uc >= 0xC0 ) {
1188  err = "invalid utf8 sequence";
1189  return false;
1190  }
1191  unicode = ( unicode << 6 ) | ( uc & 0x3f );
1192  s += ch;
1193  }
1194  if( unicode > 0x10FFFF ) {
1195  err = "invalid unicode codepoint";
1196  return false;
1197  }
1198  }
1199  return true;
1200 }

1758 /**
1759  * Get the normal form of a relative path. Does not work on absolute paths.
1760  * Slash and backslash are both treated as path separators and replaced with
1761  * slash. Trailing slashes are always removed.
1762  */
1763 static std::string normalize_relative_path( const std::string &path )
1764 {
1765  if( path.empty() ) {
1766  // normal form of an empty path is an empty path
1767  return path;
1768  }
1769  std::vector<std::string> names;
1770  for( size_t name_start = 0; name_start < path.size(); ) {
1771  const size_t name_end = std::min( path.find_first_of( "\\/", name_start ),
1772  path.size() );
1773  if( name_start < name_end ) {
1774  const std::string name = path.substr( name_start, name_end - name_start );
1775  if( name == "." ) {
1776  // do nothing
1777  } else if( name == ".." ) {
1778  if( names.empty() || names.back() == ".." ) {
1779  names.emplace_back( name );
1780  } else {
1781  names.pop_back();
1782  }
1783  } else {
1784  names.emplace_back( name );
1785  }
1786  }
1787  name_start = std::min( path.find_first_not_of( "\\/", name_end ),
1788  path.size() );
1789  }
1790  if( names.empty() ) {
1791  return ".";
1792  } else {
1793  std::string normpath;
1794  for( auto it = names.begin(); it != names.end(); ++it ) {
1795  if( it != names.begin() ) {
1796  normpath += "/";
1797  }
1798  normpath += *it;
1799  }
1800  return normpath;
1801  }

1902 // see note at utf8_width_raw(...
1903 static uint32_t UTF8_getch_raw( const char **src, int *srclen )
1904 {
1905  const unsigned char *p = *reinterpret_cast<const unsigned char **>( src );
1906  int left = 0;
1907  bool overlong = false;
1908  bool underflow = false;
1909  uint32_t ch = UNKNOWN_UNICODE;
1910 
1911  if( *srclen == 0 ) {
1912  return UNKNOWN_UNICODE;
1913  }
1914  if( p[0] >= 0xFC ) {
1915  if( ( p[0] & 0xFE ) == 0xFC ) {
1916  if( p[0] == 0xFC && ( p[1] & 0xFC ) == 0x80 ) {
1917  overlong = true;
1918  }
1919  ch = static_cast<uint32_t>( p[0] & 0x01 );
1920  left = 5;
1921  }
1922  } else if( p[0] >= 0xF8 ) {
1923  if( ( p[0] & 0xFC ) == 0xF8 ) {
1924  if( p[0] == 0xF8 && ( p[1] & 0xF8 ) == 0x80 ) {
1925  overlong = true;
1926  }
1927  ch = static_cast<uint32_t>( p[0] & 0x03 );
1928  left = 4;
1929  }
1930  } else if( p[0] >= 0xF0 ) {
1931  if( ( p[0] & 0xF8 ) == 0xF0 ) {
1932  if( p[0] == 0xF0 && ( p[1] & 0xF0 ) == 0x80 ) {
1933  overlong = true;
1934  }
1935  ch = static_cast<uint32_t>( p[0] & 0x07 );
1936  left = 3;
1937  }
1938  } else if( p[0] >= 0xE0 ) {
1939  if( ( p[0] & 0xF0 ) == 0xE0 ) {
1940  if( p[0] == 0xE0 && ( p[1] & 0xE0 ) == 0x80 ) {
1941  overlong = true;
1942  }
1943  ch = static_cast<uint32_t>( p[0] & 0x0F );
1944  left = 2;
1945  }
1946  } else if( p[0] >= 0xC0 ) {
1947  if( ( p[0] & 0xE0 ) == 0xC0 ) {
1948  if( ( p[0] & 0xDE ) == 0xC0 ) {
1949  overlong = true;
1950  }
1951  ch = static_cast<uint32_t>( p[0] & 0x1F );
1952  left = 1;
1953  }
1954  } else {
1955  if( ( p[0] & 0x80 ) == 0x00 ) {
1956  ch = static_cast<uint32_t>( p[0] );
1957  }
1958  }
1959  ++*src;
1960  --*srclen;
1961  while( left > 0 && *srclen > 0 ) {
1962  ++p;
1963  if( ( p[0] & 0xC0 ) != 0x80 ) {
1964  ch = UNKNOWN_UNICODE;
1965  break;
1966  }
1967  ch <<= 6;
1968  ch |= ( p[0] & 0x3F );
1969  ++*src;
1970  --*srclen;
1971  --left;
1972  }
1973  if( left > 0 ) {
1974  underflow = true;
1975  }
1976  if( overlong || underflow ||
1977  ( ch >= 0xD800 && ch <= 0xDFFF ) ||
1978  ( ch == 0xFFFE || ch == 0xFFFF ) || ch > 0x10FFFF ) {
1979  ch = UNKNOWN_UNICODE;
1980  }
1981  return ch;
1982 }

2023  std::string color_error;
2024  std::string color_highlight;
2025  std::string color_end;
2026  switch( json_error_output_colors ) {
2027  case json_error_output_colors_t::color_tags:
2028  color_normal = "<color_white>";
2029  color_error = "<color_light_red>";
2030  color_highlight = "<color_cyan>";
2031  color_end = "</color>";
2032  break;
2033  case json_error_output_colors_t::ansi_escapes:
2034  color_normal = "\033[0m";
2035  color_error = "\033[0;31m";
2036  color_highlight = "\033[0;36m";
2037  color_end = "\033[0m";
2038  break;
2039  case json_error_output_colors_t::unset:
2040  err_header << "( json_error_output_colors is unset, defaulting to no colors ) ";
2041  break;
2042  default:
2043  // leave color strings empty
2044  break;
2045  }
2046 
2047  // if we can't get more info from the stream don't try
2048  if( !stream->good() ) {
2049  throw JsonError( err_header.str() + escape_data( message ) );
2050  }
2051  // Seek to eof after throwing to avoid continue reading from the incorrect
2052  // location. The calling code of json error methods is supposed to restore
2053  // the stream location if it wishes to recover from the error.
2054  on_out_of_scope seek_to_eof( [this]() {
2055  stream->seekg( 0, std::istream::end );
2056  } );
2057  std::ostringstream err;
2058  err << color_normal << color_highlight << message << color_end << "\n\n";
2059  const int max_context_lines = 3; // limits context length to this many lines
2060  const int max_context_chars = 480; // limits context length to this many chars
2061  stream->seekg( offset, std::istream::cur );
2062 
2063  // if offset points to middle of a codepoint then rewind stream to codepoint start
2064  // this is so utf8_width below has correct utf-8 to count length.
2065  // pattern of 10xxxxxx means the byte is utf-8 continuation byte.
2066  while( ( stream->tellg() > 0 ) && ( ( stream->peek() & 0xC0 ) == 0x80 ) ) {
2067  stream->seekg( -1, std::istream::cur );
2068  }
2069 
2070  // remember positions of several places to print a few lines of context
2071  const size_t cursor_pos = tell(); // exact position of error
2072  rewind( 1, max_context_chars ); // rewind to start of line
2073  const size_t start_of_line = tell(); // start of error line
2074  rewind( max_context_lines, max_context_chars ); // rewind a couple lines to show context
2075  const size_t start_of_context = tell(); // start of context (couple lines above)
2076  size_t end_of_line; // end of error line (without \n)
2077 
2078  {
2079  // find end of line and store in end_of_line
2080  seek( cursor_pos );
2081  end_of_line = tell();
2082  while( stream->peek() != EOF ) {
2083  if( stream->peek() == '\n' ) {
2084  break;
2085  }
2086  stream->get();
2087  end_of_line = tell();
2088  }
2089  }
2090  {
2091  // print context lines to start of error line
2092  seek( start_of_context );
2093  std::string buffer( start_of_line - start_of_context, '\0' );
2094  stream->read( buffer.data(), start_of_line - start_of_context );
2095  err << buffer;
2096  }

2099  std::string buffer( end_of_line - start_of_line, '\0' );
2100  stream->read( buffer.data(), end_of_line - start_of_line );
2101  err << color_error << buffer << color_end << "\n";
2102  }

2104  seek( start_of_line );
2105  std::string buffer( cursor_pos - start_of_line, '\0' );
2106  stream->read( buffer.data(), cursor_pos - start_of_line );
2107

2113  }

2116  int line_count = 0;
2117  char ch = 0;
2118  for( int i = 0; line_count < max_context_lines && stream->good() && i < max_context_chars; ++i ) {
