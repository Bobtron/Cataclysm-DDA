168  load_keyboard_settings( keymap, keymap_file_loaded_from, unbound_keymap );
169  init_keycode_mapping();
170  reset_timeout();
171 
172  try {
173  load( PATH_INFO::keybindings(), false );
174  } catch( const JsonError &err ) {
175  throw std::runtime_error( err.what() );
176  }
177  try {
178  load( PATH_INFO::keybindings_vehicle(), false );
179  } catch( const JsonError &err ) {
180  throw std::runtime_error( err.what() );
181  }
182  // save non customized keybindings
183  basic_action_contexts = action_contexts;
184  try {
185  load( PATH_INFO::user_keybindings(), true );
186  } catch( const JsonError &err ) {
187  throw std::runtime_error( err.what() );
188  }
189 
190  if( keymap_file_loaded_from.empty() || ( keymap.empty() && unbound_keymap.empty() ) ) {
191  // No keymap file was loaded, or the file has no mappings and no unmappings,
192  // we can skip the remaining part of the function, especially the save function
193  return;
194  }
195  t_actions &actions = action_contexts["DEFAULTMODE"];
196  std::set<action_id> touched;
197  for( const std::pair<const char, action_id> &a : keymap ) {
198  const std::string action_id = action_ident( a.second );
199  // Put the binding from keymap either into the global context
200  // (if an action with that ident already exists there - think movement keys)
201  // or otherwise to the DEFAULTMODE context.
202  std::string context = "DEFAULTMODE";
203  if( action_contexts[default_context_id].count( action_id ) > 0 ) {
204  context = default_context_id;
205  } else if( touched.count( a.second ) == 0 ) {
206  // Note: movement keys are somehow special as the default in keymap
207  // does not contain the arrow keys, so we don't clear existing keybindings
208  // for them.
209  // If the keymap contains a binding for this action, erase all the
210  // previously (default!) existing bindings, to only keep the bindings,
211  // the user is used to
212  action_contexts[action_id].clear();
213  touched.insert( a.second );
214  }
215  add_input_for_action( action_id, context, input_event( a.first, input_event_t::keyboard_char ) );
216  }
217  // Unmap actions that are explicitly not mapped
218  for( const action_id &elem : unbound_keymap ) {
219  const std::string action_id = action_ident( elem );
220  actions[action_id].input_events.clear();
221  }
222  // Imported old bindings from old keymap file, save those to the new
223  // keybindings.json file.
224  try {
225  save();
226  } catch( std::exception &err ) {
227  debugmsg( "Could not write imported keybindings: %s", err.what() );
228  return;
229  }
230  // Finally if we did import a file, and saved it to the new keybindings
231  // file, delete the old keymap file to prevent re-importing it.
232  if( !keymap_file_loaded_from.empty() ) {
233  remove_file( keymap_file_loaded_from );
234  }

239 void input_manager::load( const cata_path &file_name, bool is_user_preferences )
240 {

242 
243  if( !jsin_opt.has_value() ) {
244  // Only throw if this is the first file to load, that file _must_ exist,
245  // otherwise the keybindings can not be read at all.
246  if( action_contexts.empty() ) {
247  throw std::runtime_error( std::string( "Could not read " ) + file_name.generic_u8string() );
248  }
249  return;
250  }

253 
254  //Crawl through once and create an entry for every definition
255  for( JsonObject action : actions_json ) {
256  // JSON object representing the action
257 
258  int version = current_keybinding_version;
259  if( is_user_preferences ) {
260  // if there isn't a "version" value it means the object was written before
261  // introduction of keybinding version, which is denoted by version 0.
262  version = action.get_int( "version", 0 );
263  }
264 
265  const std::string type = action.get_string( "type", "keybinding" );
266  if( type != "keybinding" ) {
267  debugmsg( "Only objects of type 'keybinding' (not %s) should appear in the "
268  "keybindings file '%s'", type, file_name.generic_u8string() );
269  continue;
270  }
271 
272  const std::string action_id = action.get_string( "id" );
273  const std::string context = action.get_string( "category", default_context_id );
274  t_actions &actions = action_contexts[context];
275  if( action.has_member( "name" ) ) {
276  action.read( "name", actions[action_id].name );
277  }
278 
279  t_input_event_list events;
280  for( const JsonObject keybinding : action.get_array( "bindings" ) ) {
281  std::string input_method = keybinding.get_string( "input_method" );
282  std::vector<input_event> new_events( 1 );
283  if( input_method == "keyboard_any" ) {
284  new_events.resize( 2 );
285  new_events[0].type = input_event_t::keyboard_char;
286  new_events[1].type = input_event_t::keyboard_code;
287  } else if( input_method == "keyboard_char" || input_method == "keyboard" ) {
288  new_events[0].type = input_event_t::keyboard_char;
289  } else if( input_method == "keyboard_code" ) {
290  new_events[0].type = input_event_t::keyboard_code;
291  } else if( input_method == "gamepad" ) {
292  new_events[0].type = input_event_t::gamepad;
293  } else if( input_method == "mouse" ) {
294  new_events[0].type = input_event_t::mouse;
295  } else {
296  keybinding.throw_error_at( "input_method", "unknown input_method" );
297  }
298 
299  if( keybinding.has_member( "mod" ) ) {
300  for( const JsonValue val : keybinding.get_array( "mod" ) ) {
301  const std::string str = val;
302  keymod_t mod = keymod_t::ctrl;
303  if( str == "ctrl" ) {
304  mod = keymod_t::ctrl;
305  } else if( str == "alt" ) {
306  mod = keymod_t::alt;
307  } else if( str == "shift" ) {
308  mod = keymod_t::shift;
309  } else {
310  val.throw_error( "unknown modifier name" );
311  }
312  for( input_event &new_event : new_events ) {
313  new_event.modifiers.emplace( mod );
314  }
315  }
316  }
317 
318  if( keybinding.has_array( "key" ) ) {
319  for( const std::string line : keybinding.get_array( "key" ) ) {
320  for( input_event &new_event : new_events ) {
321  new_event.sequence.push_back( get_keycode( new_event.type, line ) );
322  }
323  }
324  } else { // assume string if not array, and throw if not string
325  for( input_event &new_event : new_events ) {
326  new_event.sequence.push_back(
327  get_keycode( new_event.type, keybinding.get_string( "key" ) )
328  );
329  }
330  }
331 
332  for( const input_event &evt : new_events ) {
333  if( std::find( events.begin(), events.end(), evt ) == events.end() ) {
334  events.emplace_back( evt );
335  }
336  }
337 
338  if( is_user_preferences && version <= 1 ) {
339  // Add keypad enter to old keybindings with return key
340  for( const input_event &evt : new_events ) {
341  input_event new_evt = evt;
342  bool has_return = false;
343  // As of version 2 the key sequence actually only supports
344  // one key, so we just replace all return with enter
345  if( new_evt.type == input_event_t::keyboard_char ) {
346  for( int &key : new_evt.sequence ) {
347  if( key == '\n' ) {
348  key = KEY_ENTER;
349  has_return = true;
350  }
351  }
352  }
353  if( has_return && std::find( events.begin(), events.end(), new_evt ) == events.end() ) {
354  events.emplace_back( new_evt );
355  }
356  }
357  }
358  }
359 
360  // In case this is the second file containing user preferences,
361  // this replaces the default bindings with the user's preferences.
362  action_attributes &attributes = actions[action_id];
363  if( is_user_preferences && version == 0 ) {
364  // version 0 means the keybinding was written prior to the division
365  // of `input_event_t::keyboard_char` and `input_event_t::keyboard_code`,
366  // so we copy any `input_event_t::keyboard_code` event from the default
367  // keybindings to be compatible with old user keybinding files.
368  for( const input_event &evt : attributes.input_events ) {
369  if( evt.type == input_event_t::keyboard_code ) {
370  events.emplace_back( evt );
371  }
372  }
373  }
374  attributes.input_events = events;
375  if( action.has_member( "is_user_created" ) ) {
376  attributes.is_user_created = action.get_bool( "is_user_created" );
377  }
378  }

394  jsout.member( "version", current_keybinding_version );

399  }
400

403  jsout.member( "bindings" );
404  jsout.start_array();

406  jsout.start_object();
407  switch( event.type ) {
408  case input_event_t::keyboard_char:
409  jsout.member( "input_method", "keyboard_char" );
410  break;
411  case input_event_t::keyboard_code:
412  jsout.member( "input_method", "keyboard_code" );
413  break;
414  case input_event_t::gamepad:
415  jsout.member( "input_method", "gamepad" );
416  break;
417  case input_event_t::mouse:
418  jsout.member( "input_method", "mouse" );
419  break;
420  default:
421  throw std::runtime_error( "unknown input_event_t" );
422  }
423 
424  // optimization: no writing if empty
425  if( !event.modifiers.empty() ) {
426  jsout.member( "mod" );
427  jsout.start_array();

441  }
442  }
443  jsout.end_array();
444  }
445 
446  jsout.member( "key" );
447  jsout.start_array();

451  jsout.end_array();

453  }
454  jsout.end_array();

662 int input_manager::get_keycode( const input_event_t inp_type, const std::string &name ) const
663 {
664  const t_name_to_key_map *map = nullptr;
665  switch( inp_type ) {
666  default:
667  break;
668  case input_event_t::keyboard_char:
669  map = &keyboard_char_keyname_to_keycode;
670  break;
671  case input_event_t::keyboard_code:
672  map = &keyboard_code_keyname_to_keycode;
673  break;
674  case input_event_t::gamepad:
675  map = &gamepad_keyname_to_keycode;
676  break;
677  case input_event_t::mouse:
678  map = &mouse_keyname_to_keycode;
679  break;
680  }
681  if( map ) {
682  const auto it = map->find( name );
683  if( it != map->end() ) {
684  return it->second;
685  }
686  }
687  // Not found in map, try to parse as int
688  if( name.compare( 0, 8, "UNKNOWN_" ) == 0 ) {
689  return str_to_int( name.substr( 8 ) );
690  }
691  return 0;

694 std::string input_manager::get_keyname( int ch, input_event_t inp_type, bool portable ) const
695 {
696  const t_key_to_name_map *map = nullptr;
697  switch( inp_type ) {
698  default:
699  break;
700  case input_event_t::keyboard_char:
701  map = &keyboard_char_keycode_to_keyname;
702  break;
703  case input_event_t::keyboard_code:
704  map = &keyboard_code_keycode_to_keyname;
705  break;
706  case input_event_t::gamepad:
707  map = &gamepad_keycode_to_keyname;
708  break;
709  case input_event_t::mouse:
710  map = &mouse_keycode_to_keyname;
711  break;
712  }
713  if( map ) {
714  const auto it = map->find( ch );
715  if( it != map->end() ) {
716  switch( inp_type ) {
717  case input_event_t::keyboard_char:
718  if( IS_F_KEY( ch ) ) {
719  // special case it since F<num> key names are generated using loop
720  // and not marked individually for translation
721  if( portable ) {
722  return it->second;
723  } else {
724  return string_format( pgettext( "function key name", "F%d" ), F_KEY_NUM( ch ) );
725  }
726  } else if( IS_CTRL_CHAR( ch ) && !IS_NAMED_CTRL_CHAR( ch ) ) {
727  if( portable ) {
728  return it->second;
729  } else {
730  return string_format( pgettext( "control key name", "CTRL+%c" ), ch + 64 );
731  }
732  } else if( ch >= char_key_beg && ch <= char_key_end && ch != ' ' ) {
733  // character keys except space need no translation
734  return it->second;
735  }
736  break;
737  case input_event_t::keyboard_code:
738  if( ch >= char_key_beg && ch < char_key_end && ch != ' ' ) {
739  // character keys except space need no translation
740  return it->second;
741  }
742  break;
743  default:
744  break;
745  }
746  return portable ? it->second : pgettext( "key name", it->second.c_str() );
747  }
748  }
749  if( portable ) {
750  return std::string( "UNKNOWN_" ) + int_to_str( ch );
751  } else {
752  return string_format( _( "unknown key %ld" ), ch );
753  }
754 }

771 const action_attributes &input_manager::get_action_attributes(
772  const std::string &action_id,
773  const std::string &context,
774  bool *overwrites_default,
775  bool use_basic_action_contexts )
776 {
777  t_action_contexts &action_contexts_ref = use_basic_action_contexts
778  ? basic_action_contexts
779  : action_contexts;
780 
781  if( context != default_context_id ) {
782  // Check if the action exists in the provided context
783  const t_action_contexts::const_iterator action_context = action_contexts_ref.find( context );
784  if( action_context != action_contexts_ref.end() ) {
785  const t_actions::const_iterator action = action_context->second.find( action_id );
786  if( action != action_context->second.end() ) {
787  if( overwrites_default ) {
788  *overwrites_default = true;
789  }
790 
791  return action->second;
792  }
793  }
794  }
795 
796  // If not, we use the default binding.
797  if( overwrites_default ) {
798  *overwrites_default = false;
799  }
800 
801  t_actions &default_action_context = action_contexts_ref[default_context_id];
802  const t_actions::const_iterator default_action = default_action_context.find( action_id );
803  if( default_action == default_action_context.end() ) {
804  // A new action is created in the event that the requested action is
805  // not in the keybindings configuration e.g. the entry is missing.
806  default_action_context[action_id].name = get_default_action_name( action_id );
807  }
808 
809  return default_action_context[action_id];

828 input_manager::t_input_event_list &input_manager::get_or_create_event_list(
829  const std::string &action_descriptor, const std::string &context )
830 {
831  // A new context is created in the event that the user creates a local
832  // keymapping in a context that doesn't yet exist e.g. a context without
833  // any pre-existing keybindings.
834  t_actions &actions = action_contexts[context];
835 
836  // A new action is created in the event that the user creates a local
837  // keymapping that masks a global one.
838  if( actions.find( action_descriptor ) == actions.end() ) {
839  action_attributes &attributes = actions[action_descriptor];
840  attributes.name = get_default_action_name( action_descriptor );
841  attributes.is_user_created = true;
842  }
843 
844  return actions[action_descriptor].input_events;

847 bool input_manager::remove_input_for_action(
848  const std::string &action_descriptor, const std::string &context )
849 {
850  const t_action_contexts::iterator action_context = action_contexts.find( context );
851  if( action_context == action_contexts.end() ) {
852  return false;
853  }
854  t_actions &actions = action_context->second;
855  t_actions::iterator action = actions.find( action_descriptor );
856  if( action == actions.end() ) {
857  return false;
858  }
859  if( action->second.is_user_created ) {
860  // Since this is a user created hotkey, remove it so that the
861  // user will fallback to the hotkey in the default context.
862  actions.erase( action );
863  return true;
864  }
865  // If a context no longer has any keybindings remaining for an action but
866  // there's an attempt to remove bindings anyway, presumably the user wants
867  // to fully remove the binding from that context.
868  if( action->second.input_events.empty() ) {
869  actions.erase( action );
870  return true;
871  }
872  action->second.input_events.clear();
873  return false;
874 }
