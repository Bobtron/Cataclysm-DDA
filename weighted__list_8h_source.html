228  void precalc() {
229  precalc_array.clear();
230  // to avoid additional reallocations
231  precalc_array.reserve( this->total_weight );
232  // weights [3,1,5] will produce vector of indices [0,0,0,1,2,2,2,2,2]
233  for( size_t i = 0; i < this->objects.size(); i++ ) {
234  precalc_array.resize( precalc_array.size() + this->objects[i].weight, i );
235  }
236  }

238  protected:

240  size_t pick_ent( unsigned int randi ) const override {
241  if( this->objects.size() == 1 ) {
242  return 0;
243  }
244  size_t i;
245  const int picked = ( randi % ( this->total_weight ) ) + 1;
246  if( !precalc_array.empty() ) {
247  // if the precalc_array is populated, use it for O(1) lookup
248  i = precalc_array[picked - 1];
249  } else {
250  // otherwise do O(N) search through items
251  int accumulated_weight = 0;
252  for( i = 0; i < this->objects.size(); i++ ) {
253  accumulated_weight += this->objects[i].weight;
254  if( accumulated_weight >= picked ) {
255  break;
256  }
257  }
258  }
259  return i;
260  }

263  precalc_array.clear();
