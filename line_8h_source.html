27 //! This compile-time usable function combines the sign of each (x, y, z) component into a single integer
28 //! to allow simple runtime and compile-time mapping of (x, y, z) tuples to @ref direction enumerators.
29 //! Specifically, (0, -, +) => (0, 1, 2); a base-3 number.
30 //! This only works correctly for inputs between -1,-1,-1 and 1,1,1.
31 //! For numbers outside that range, use make_xyz().
32 inline constexpr unsigned make_xyz_unit( const tripoint &p ) noexcept
33 {
34  return ( ( p.x > 0 ) ? 2u : ( p.x < 0 ) ? 1u : 0u ) * 1u +
35  ( ( p.y > 0 ) ? 2u : ( p.y < 0 ) ? 1u : 0u ) * 3u +
36  ( ( p.z > 0 ) ? 2u : ( p.z < 0 ) ? 1u : 0u ) * 9u;
37 }
