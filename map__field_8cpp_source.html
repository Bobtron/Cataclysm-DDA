644 static void field_processor_fd_electricity( const tripoint &p, field_entry &cur,
645  field_proc_data &pd )
646 {
647  // Higher chance of spreading for intense fields
648  int current_intensity = cur.get_field_intensity();
649  if( one_in( current_intensity * 2 ) ) {
650  return;
651  }
652 
653  const int spread_intensity_cap = 3 + std::max( ( current_intensity - 3 ) / 2, 0 );
654 
655  std::vector<tripoint> grounded_tiles;
656  std::vector<tripoint> tiles_with_creatures;
657  std::vector<tripoint> other_tiles;
658 
659  bool valid_candidates = false;
660  for( const tripoint &dst : points_in_radius( p, 1 ) ) {
661  if( !pd.here.inbounds( dst ) ) {
662  continue;
663  }
664  // Skip tiles with intense fields
665  const field_type_str_id &field_type = pd.here.get_applicable_electricity_field( dst );
666  if( field_entry *field = pd.here.get_field( dst, field_type ) ) {
667  if( field->get_field_intensity() >= spread_intensity_cap ) {
668  continue;
669  }
670  }
671 
672  if( pd.here.impassable( dst ) ) {
673  grounded_tiles.push_back( dst );
674  } else {
675  if( get_creature_tracker().creature_at( dst ) ) {
676  tiles_with_creatures.push_back( dst );
677  } else {
678  other_tiles.push_back( dst );
679  }
680  }
681  valid_candidates = true;
682  }
683 
684  if( !valid_candidates ) {
685  return;
686  }
687 
688  const bool here_impassable = pd.here.impassable( p );
689  int grounded_weight = here_impassable ? 1 : 6;
690  int creature_weight = here_impassable ? 5 : 6;
691  int other_weight = here_impassable ? 0 : 1;
692 
693  std::vector<tripoint> *target_vector = nullptr;
694  while( current_intensity > 0 ) {
695 
696  if( here_impassable && one_in( 3 ) ) {
697  // Electricity in impassable tiles will find a way to the ground sometimes
698  cur.set_field_intensity( --current_intensity );
699  continue;
700  }
701 
702  const int vector_choice = bucket_index_from_weight_list( std::vector<int>( {
703  grounded_tiles.empty() ? 0 : grounded_weight,
704  tiles_with_creatures.empty() ? 0 : creature_weight,
705  other_tiles.empty() ? 0 : other_weight
706  } ) );
707 
708  switch( vector_choice ) {
709  default:
710  case 0:
711  if( here_impassable && !one_in( 5 ) ) {
712  return;
713  }
714  target_vector = &grounded_tiles;
715  break;
716  case 1:
717  target_vector = &tiles_with_creatures;
718  break;
719  case 2:
720  target_vector = &other_tiles;
721  break;
722  }
723 
724  if( target_vector->empty() ) {
725  return;
726  }
727 
728  int vector_index = rng( 0, target_vector->size() - 1 );
729  auto target_it = target_vector->begin() + vector_index;
730  tripoint target_point = *target_it;
731 
732  const field_type_str_id &field_type = pd.here.get_applicable_electricity_field( target_point );
733 
734  // Intensify target field if it exists, create a new one otherwise
735  if( field_entry *target_field = pd.here.get_field( target_point, field_type ) ) {
736  int target_field_intensity = target_field->get_field_intensity();
737  target_field->set_field_intensity( ++target_field_intensity );
738  if( target_field_intensity >= spread_intensity_cap ) {
739  target_vector->erase( target_it );
740  }
741  } else {
742  pd.here.add_field( target_point, field_type, 1, cur.get_field_age() + 1_turns );
743  }
744 
745  cur.set_field_intensity( --current_intensity );
746 
747  if( one_in( current_intensity * 2 ) ) {
748  // Weaker fields have a harder time spreading
749  break;
750  }
751  }
752 }

932 void field_processor_fd_fire( const tripoint &p, field_entry &cur, field_proc_data &pd )

935  map &here = pd.here;
936  maptile &map_tile = pd.map_tile;
937  const oter_id om_ter = pd.om_ter;
938 
939  cur.set_field_age( std::max( -24_hours, cur.get_field_age() ) );
940  // Entire objects for ter/frn for flags
941  bool sheltered = g->is_sheltered( p );
942  weather_manager &weather = get_weather();
943  int winddirection = weather.winddirection;
944  int windpower = get_local_windpower( weather.windspeed, om_ter, tripoint_abs_ms( p ), winddirection,
945  sheltered );
946  const ter_t &ter = map_tile.get_ter_t();
947  const furn_t &frn = map_tile.get_furn_t();
948 
949  // We've got ter/furn cached, so let's use that
950  const bool is_sealed = ter_furn_has_flag( ter, frn, ter_furn_flag::TFLAG_SEALED ) &&
951  !ter_furn_has_flag( ter, frn, ter_furn_flag::TFLAG_ALLOW_FIELD_EFFECT );
952  // Smoke generation probability, consumed items count
953  int smoke = 0;
954  int consumed = 0;
955  // How much time to add to the fire's life due to burned items/terrain/furniture
956  time_duration time_added = 0_turns;
957  // Checks if the fire can spread
958  const bool can_spread = !ter_furn_has_flag( ter, frn, ter_furn_flag::TFLAG_FIRE_CONTAINER );
959  // If the flames are in furniture with fire_container flag like brazier or oven,
960  // they're fully contained, so skip consuming terrain
961  const bool can_burn = ( ter.is_flammable() || frn.is_flammable() ) &&
962  !ter_furn_has_flag( ter, frn, ter_furn_flag::TFLAG_FIRE_CONTAINER );
963  // The huge indent below should probably be somehow moved away from here
964  // without forcing the function to use i_at( p ) for fires without items
965  if( !is_sealed && map_tile.get_item_count() > 0 ) {
966  map_stack items_here = here.i_at( p );
967  std::vector<item> new_content;
968  for( auto it = items_here.begin(); it != items_here.end(); ) {
969  if( it->will_explode_in_fire() ) {
970  // We need to make a copy because the iterator validity is not predictable
971  item copy = *it;
972  it = items_here.erase( it );
973  if( copy.detonate( p, new_content ) ) {
974  // Need to restart, iterators may not be valid
975  it = items_here.begin();
976  }
977  } else {
978  ++it;
979  }
980  }
981 
982  fire_data frd( cur.get_field_intensity(), !can_spread );
983  // The highest # of items this fire can remove in one turn
984  int max_consume = cur.get_field_intensity() * 2;
985 
986  for( auto fuel = items_here.begin(); fuel != items_here.end() && consumed < max_consume; ) {
987  // `item::burn` modifies the charges in order to simulate some of them getting
988  // destroyed by the fire, this changes the item weight, but may not actually
989  // destroy it. We need to spawn products anyway.
990  const units::mass old_weight = fuel->weight( false );
991  bool destroyed = fuel->burn( frd );
992  // If the item is considered destroyed, it may have negative charge count,
993  // see `item::burn?. This in turn means `item::weight` returns a negative value,
994  // which we can not use, so only call `weight` when it's still an existing item.
995  const units::mass new_weight = destroyed ? 0_gram : fuel->weight( false );
996  if( old_weight != new_weight ) {
997  here.create_burnproducts( p, *fuel, old_weight - new_weight );
998  }
999 
1000  if( destroyed ) {
1001  // If we decided the item was destroyed by fire, remove it.
1002  // But remember its contents, except for irremovable stuff, if any
1003  if( !fuel->has_flag( json_flag_NO_UNLOAD ) ) {
1004  const std::list<item *> content_list = fuel->all_items_top();
1005  for( item *it : content_list ) {
1006  if( !it->is_irremovable() ) {
1007  new_content.emplace_back( *it );
1008  }
1009  }
1010  }
1011  fuel = items_here.erase( fuel );
1012  consumed++;
1013  } else {
1014  ++fuel;
1015  }
1016  }
1017 
1018  here.spawn_items( p, new_content );
1019  smoke = roll_remainder( frd.smoke_produced );
1020  time_added = 1_turns * roll_remainder( frd.fuel_produced );
1021  }
1022 
1023  int part;
1024  // Get the part of the vehicle in the fire (_internal skips the boundary check)
1025  vehicle *veh = here.veh_at_internal( p, part );
1026  if( veh != nullptr ) {
1027  veh->damage( here, part, cur.get_field_intensity() * 10, damage_heat, true );
1028  // Damage the vehicle in the fire.
1029  }

1031  if( ter.has_flag( ter_furn_flag::TFLAG_SWIMMABLE ) ) {
1032  // Flames die quickly on water
1033  cur.set_field_age( cur.get_field_age() + 4_minutes );

1038  // The fire feeds on the ground itself until max intensity.
1039  time_added += 1_turns * ( 5 - cur.get_field_intensity() );
1040  smoke += 2;
1041  smoke += static_cast<int>( windpower / 5 );
1042  if( cur.get_field_intensity() > 1 &&
1043  one_in( 200 - cur.get_field_intensity() * 50 ) ) {
1044  here.destroy( p, false );
1045  }
1046 
1047  } else if( ter_furn_has_flag( ter, frn, ter_furn_flag::TFLAG_FLAMMABLE_HARD ) &&
1048  one_in( 3 ) ) {
1049  // The fire feeds on the ground itself until max intensity.
1050  time_added += 1_turns * ( 4 - cur.get_field_intensity() );
1051  smoke += 2;
1052  smoke += static_cast<int>( windpower / 5 );
1053  if( cur.get_field_intensity() > 1 &&
1054  one_in( 200 - cur.get_field_intensity() * 50 ) ) {
1055  here.destroy( p, false );
1056  }
1057 
1058  } else if( ter.has_flag( ter_furn_flag::TFLAG_FLAMMABLE_ASH ) ) {
1059  // The fire feeds on the ground itself until max intensity.
1060  time_added += 1_turns * ( 5 - cur.get_field_intensity() );
1061  smoke += 2;
1062  smoke += static_cast<int>( windpower / 5 );
1063  if( cur.get_field_intensity() > 1 &&
1064  one_in( 200 - cur.get_field_intensity() * 50 ) ) {
1065  here.spawn_item( p, "ash", 1, rng( 10, 1000 ) );
1066  if( p.z > 0 ) {
1067  // We're in the air. Need to invalidate the furniture otherwise it'll cause problems
1068  here.furn_set( p, furn_str_id::NULL_ID() );
1069  here.ter_set( p, ter_t_open_air );
1070  } else if( p.z < -1 ) {
1071  // We're deep underground, in bedrock. Whatever terrain was here is burned to the ground, leaving only the carved out rock (including ceiling)
1072  here.ter_set( p, ter_t_rock_floor );
1073  } else {
1074  // Need to invalidate the furniture otherwise it'll cause problems when supporting terrain collapses
1075  here.furn_set( p, furn_str_id::NULL_ID() );
1076  here.ter_set( p, ter_t_dirt );
1077  }
1078  }

1080  } else if( frn.has_flag( ter_furn_flag::TFLAG_FLAMMABLE_ASH ) ) {
1081  // The fire feeds on the ground itself until max intensity.
1082  time_added += 1_turns * ( 5 - cur.get_field_intensity() );

1086  one_in( 200 - cur.get_field_intensity() * 50 ) ) {
1087  here.furn_set( p, furn_f_ash );
1088  here.add_item_or_charges( p, item( "ash" ) );
1089  }
1090 
1091  } else if( ter.has_flag( ter_furn_flag::TFLAG_NO_FLOOR ) && p.z > -OVERMAP_DEPTH ) {
1092  // We're hanging in the air - let's fall down
