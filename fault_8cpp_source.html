188  const requirement_id req_id( "fault_fix_" + f.id_.str() + "_inline_req" );
189  requirement_data::load_requirement( req, req_id );

191  } else {

193  }

204 void fault_fix::finalize()
205 {
206  for( auto& [fix_id, fix] : fault_fixes_all ) {
207  const auto range = reqs_temp_storage.equal_range( fix_id );
208  for( auto it = range.first; it != range.second; ++it ) {
209  const requirement_id req_id( it->second.first );
210  const int amount = it->second.second;
211  if( !req_id.is_valid() ) {
212  debugmsg( "fault_fix '%s' has invalid requirement_id '%s'", fix_id.str(), req_id.str() );
213  continue;
214  }
215  *fix.requirements = *fix.requirements + ( *req_id ) * amount;
216  }
217  fix.requirements->consolidate();
218  for( const fault_id &fid : fix.faults_removed ) {
219  const_cast<fault &>( *fid ).fixes.emplace( fix_id );
220  }
221  }
222  reqs_temp_storage.clear();
223 }
