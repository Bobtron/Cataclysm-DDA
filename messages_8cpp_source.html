208  return;
209  }

245  void modify_msg_with_exclamations( std::string &msg, game_message_type const type ) {
246  // add smileys to messages if the player is high on weed
247  Character *p_char = &get_player_character();
248  if( !p_char->has_effect( effect_weed_high ) ) {
249  return;
250  }
251 
252  const time_duration weed_dur = p_char->get_effect_dur( effect_weed_high );
253  const float in_minutes = to_minutes<float>( weed_dur );
254  // we will now calculate how often smiley faces should appear in messages.
255  // Slowly rises. 0 at 0, 1 at 22.5, 2 at 45 mins. Asymptote at 3.
256  const float max_per_msg = 3 - 3 / ( 1 + 0.001 * in_minutes * in_minutes );
257  //possible values for smiles_for_this_msg are 0,1,2,3.
258  int smiles_for_this_msg = static_cast<int>( round( rng_normal( 0, max_per_msg ) ) );
259 
260  // now insert the smiley face somewhere in the msg string. We want to avoid breaking up words, so we will only replace spaces.
261  // find all suitable locations for smiley. That is, end of message + all spaces (except after certain words)
262  std::vector<size_t> smiley_locations;
263  // we also need to track space locations to extract words properly. Annoying but necessary.
264  std::vector<size_t> space_locations;
265 
266  // end of msg
267  smiley_locations.push_back( msg.size() );
268  // now all spaces
269  for( size_t j = 0; j < msg.size(); j++ ) {
270  if( msg[j] == ' ' ) {
271  bool wordskip = false;
272  // fancy logic. It's weird if the smiley comes after "to" or "that" or "a". Let's try and skip words like those.
273  // We skip all short words (<=3 letters), and all words in the forbidden_words array.
274  if( !space_locations.empty() ) {
275  const size_t prev_space = space_locations.back();
276  // if prev_space is the previous char, skip
277  if( prev_space == j - 1 ) {
278  wordskip = true;
279  }
280  // check that smiley_locations.back() did not return none
281  if( prev_space != std::string::npos ) {
282  const std::string word = msg.substr( prev_space, j - prev_space );
283  const std::array<std::string, 6> forbidden_words = { " that", " with", " this", " over", " your", " onto" };
284  for( const std::string &forbidden_word : forbidden_words ) {
285  if( word == forbidden_word ) {
286  wordskip = true;
287  break;
288  }
289  }
290  if( word.size() > 1 && word.size() <= 4 ) {
291  wordskip = true;
292  }
293  }
294  } else {
295  // skip very first word
296  wordskip = true;
297  }
298  space_locations.push_back( j );
299  if( !wordskip ) {
300  smiley_locations.push_back( j );
301  }
302  }
303  }
304  // first, check if smiles_for_this_msg == len(smiley_locations)
305  if( smiles_for_this_msg >= static_cast<int>( smiley_locations.size() ) ) {
306  smiles_for_this_msg = static_cast<int>( smiley_locations.size() ) - 1;
307  }
308  // then pick one at random
309  for( int i = 0; i < smiles_for_this_msg; i++ ) {
310  const size_t smiley_location = random_entry_removed( smiley_locations );
311  std::string smiley_string = " :)";
312  if( type == m_good || type == m_critical ) {
313  if( one_in( 5 ) ) {
314  smiley_string = " :D";
315  }
316  }
317  msg.insert( smiley_location, smiley_string );
318  //for( size_t j = 0; j < smiley_locations.size(); j++ ) {
319  // modernize-loop-convert. Use range-based for loop instead.
320  for( size_t &smil_loc : smiley_locations ) {
321  if( smil_loc > smiley_location ) {
322  // we increment to account for the insertion of the smiley string
323  smil_loc += 3;
324  }
325  }
326  }
327 
328  // if we are *very* high, there will be a chance to finish a long message with an exclamation
329  if( smiles_for_this_msg > 1 ) {
330  int woah_chance = static_cast<int>( 10 / smiles_for_this_msg );
331  if( msg.size() > 16 && one_in( woah_chance ) ) {
332  std::vector<std::string> exclamations = {
333  "Woah.", "Dude."
334  };
335  std::vector<std::string> good_exclamations = {
336  "Cool!", "Wicked!", "Awesome!"
337  };
338  std::vector<std::string> bad_exclamations = {
339  "Not cool.", "Oof.", "Bummer."
340  };
341  if( type == m_good || type == m_critical || type == m_headshot ) {
342  msg += " " + random_entry( good_exclamations );
343  } else if( type == m_bad || type == m_grazing ) {
344  msg += " " + random_entry( bad_exclamations );

369  /** Refresh the cooldown timers, removing elapsed ones and making new ones if needed.
370  * @param message The current message that needs to be checked.
371  * @param flags Flags pertaining to the message.
372  */
373  void refresh_cooldown( const game_message &message, const game_message_flags flags ) {
374  // is cooldown used? (also checks for messages arriving here at game initialization: we don't care about them).
375  if( message_cooldown <= 0 || message.turn() <= calendar::turn_zero ) {
376  return;
377  }
378 
379  // housekeeping: remove any cooldown message with an expired cooldown time from the cooldown queue.
380  const time_point now = calendar::turn;
381  for( auto it = cooldown_templates.begin(); it != cooldown_templates.end(); ) {
382  // number of turns elapsed since the cooldown started.
383  const int turns = to_turns<int>( now - it->turn() );
384  if( turns >= message_cooldown ) {
385  // time elapsed! remove it.
386  it = cooldown_templates.erase( it );
387  } else {
388  ++it;
389  }
390  }
391 
392  // do not hide messages which bypasses cooldown.
393  if( ( flags & gmf_bypass_cooldown ) != 0 ) {
394  return;
395  }
396 
397  // Is the message string already in the cooldown queue?
398  // If it's not we must put it in the cooldown queue now, otherwise just increment the number of times we have seen it.
399  const auto cooldown_message_it = std::find_if( cooldown_templates.begin(),
400  cooldown_templates.end(), [&message]( game_message & cooldown_message ) -> bool {
401  return cooldown_message.message == message.message;
402  } );
403  if( cooldown_message_it == cooldown_templates.end() ) {
404  // push current message to cooldown message templates.
405  cooldown_templates.emplace_back( message );
406  } else {
407  // increment the number of time we have seen this message.
408  cooldown_message_it->cooldown_seen++;
409  }
410  }

770 void Messages::dialog::do_filter( const std::string &filter_str )
771 {
772  // Split the search string into type and text
773  bool has_type_filter = false;
774  game_message_type filter_type = m_neutral;
775  std::string filter_text;
776  const size_t colon = filter_str.find( ':' );
777  if( colon != std::string::npos ) {
778  has_type_filter = msg_type_from_name( filter_type, filter_str.substr( 0, colon ) );
779  filter_text = filter_str.substr( colon + 1 );
780  } else {
781  filter_text = filter_str;
782  }
783 
784  // Start filtering the log
785  folded_filtered.clear();
786  for( size_t folded_ind = 0; folded_ind < folded_all.size(); ) {
787  const size_t msg_ind = folded_all[folded_ind].first;
788  const game_message &msg = player_messages.history( msg_ind );
789  const bool match = ( !has_type_filter || filter_type == msg.type ) &&
790  ci_find_substr( remove_color_tags( msg.get_with_count() ), filter_text ) >= 0;
791 
792  // Always advance the index, but only add to filtered list if the original message matches
793  for( ; folded_ind < folded_all.size() && folded_all[folded_ind].first == msg_ind; ++folded_ind ) {
794  if( match ) {
795  folded_filtered.emplace_back( folded_ind );
796  }
797  }
798  }
799 
800  // Reset view
801  if( log_from_top || max_lines > folded_filtered.size() ) {
802  offset = 0;
803  } else {
804  offset = folded_filtered.size() - max_lines;
805  }
806 }

808 void Messages::dialog::input()
809 {
810  canceled = false;
811  if( filtering ) {
812  filter.query( false );
813  if( filter.confirmed() || filter.canceled() ) {
814  filtering = false;
815  }
816  if( !filter.canceled() ) {
817  const std::string &new_filter_str = filter.text();
818  if( new_filter_str != filter_str ) {
819  filter_str = new_filter_str;
820 
821  do_filter( filter_str );
822  }
823  } else {
824  filter.text( filter_str );
825  }
826  } else {
827  const std::string &action = ctxt.handle_input();
828  if( ( action == "DOWN" || action == "SCROLL_DOWN" ) &&
829  offset + max_lines < folded_filtered.size() ) {
830  ++offset;
831  } else if( ( action == "UP" || action == "SCROLL_UP" ) && offset > 0 ) {
832  --offset;
833  } else if( action == "PAGE_DOWN" ) {
834  if( offset + max_lines * 2 <= folded_filtered.size() ) {
835  offset += max_lines;
836  } else if( max_lines <= folded_filtered.size() ) {
837  offset = folded_filtered.size() - max_lines;
838  } else {
839  offset = 0;
840  }
841  } else if( action == "PAGE_UP" ) {
842  if( offset >= max_lines ) {
843  offset -= max_lines;
844  } else {
845  offset = 0;
846  }
847  } else if( action == "FILTER" ) {
848  filtering = true;
849  } else if( action == "RESET_FILTER" ) {
850  filter_str.clear();
851  filter.text( filter_str );
852  do_filter( filter_str );
853  } else if( action == "QUIT" ) {
854  canceled = true;
855  }
856  }
