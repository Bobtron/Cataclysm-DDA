169 std::vector<const recipe *> recipe_subset::search(
170  const std::string_view txt, const search_type key,
171  const std::function<void( size_t, size_t )> &progress_callback ) const
172 {
173  auto predicate = [&]( const recipe * r ) {
174  if( !*r || r->obsolete ) {
175  return false;
176  }
177  switch( key ) {
178  case search_type::name:
179  return lcmatch( r->result_name(), txt );
180 
181  case search_type::exclude_name:
182  return !lcmatch( r->result_name(), txt );
183 
184  case search_type::skill: {
185  if( r->skill_used && lcmatch( r->skill_used->name(), txt ) ) {
186  return true;
187  }
188  const auto &skills = r->required_skills;
189  return std::any_of( skills.begin(), skills.end(), [&]( const std::pair<skill_id, int> &e ) {
190  return lcmatch( e.first->name(), txt );
191  } );
192  }
193 
194  case search_type::primary_skill:
195  return lcmatch( r->skill_used->name(), txt );
196 
197  case search_type::component:
198  return search_reqs( r->simple_requirements().get_components(), txt );
199 
200  case search_type::tool:
201  return search_reqs( r->simple_requirements().get_tools(), txt );
202 
203  case search_type::quality:
204  return search_reqs( r->simple_requirements().get_qualities(), txt );
205 
206  case search_type::quality_result: {
207  return item::find_type( r->result() )->has_any_quality( txt );
208  }
209 
210  case search_type::description_result: {
211  if( r->is_practice() ) {
212  return lcmatch( r->description.translated(), txt );
213  } else {
214  const item result( r->result() );
215  return lcmatch( remove_color_tags( result.info( true ) ), txt );
216  }
217  }
218 
219  case search_type::proficiency:
220  return lcmatch( r->recipe_proficiencies_string(), txt );
221 
222  case search_type::difficulty: {
223  std::string range_start;
224  std::string range_end;
225  bool use_range = false;
226  for( const char &chr : txt ) {
227  if( std::isdigit( chr ) ) {
228  if( use_range ) {
229  range_end += chr;
230  } else {
231  range_start += chr;
232  }
233  } else if( chr == '~' ) {
234  use_range = true;
235  } else {
236  // unexpected character
237  return true;
238  }
239  }
240 
241  int start = 0;
242  int end = INT_MAX;
243 
244  if( use_range ) {
245  if( !range_start.empty() ) {
246  start = std::stoi( range_start );
247  }
248 
249  if( !range_end.empty() ) {
250  end = std::stoi( range_end );
251  }
252 
253  if( range_start.empty() && range_end.empty() ) {
254  return true;
255  }
256  } else {
257  if( !range_start.empty() ) {
258  start = std::stoi( range_start );
259  }
260 
261  if( range_start.empty() && range_end.empty() ) {
262  return true;
263  }
264  }
265 
266  if( use_range && start > end ) {
267  std::swap( start, end );
268  }
269 
270  if( use_range ) {
271  // check if number is between two numbers inclusive
272  return r->difficulty == clamp( r->difficulty, start, end );
273  } else {
274  return r->difficulty == start;
275  }
276  }
277 
278  case search_type::activity_level:
279  return lcmatch( display::activity_level_str( r->exertion_level() ), txt );
280 
281  default:
282  return false;
283  }
284  };
285 
286  std::vector<const recipe *> res;
287  size_t i = 0;
288  for( const recipe *r : recipes ) {
289  if( progress_callback ) {
290  progress_callback( i, recipes.size() );
291  }
292  if( predicate( r ) ) {
293  res.push_back( r );
294  }

296  }

533 void recipe_dictionary::find_items_on_loops()
534 {
535  // Check for infinite recipe loops in food (which are problematic for
536  // nutrient calculations).
537  //
538  // Start by building a directed graph of itypes to potential components of
539  // those itypes.
540  items_on_loops.clear();
541  std::unordered_map<itype_id, std::vector<itype_id>> potential_components_of;
542  for( const itype *i : item_controller->all() ) {
543  if( !i->comestible || i->has_flag( STATIC( flag_id( "NUTRIENT_OVERRIDE" ) ) ) ) {
544  continue;
545  }
546  std::vector<itype_id> &potential_components = potential_components_of[i->get_id()];
547  for( const recipe_id &rec : i->recipes ) {
548  const requirement_data requirements = rec->simple_requirements();
549  const requirement_data::alter_item_comp_vector &component_requirements =
550  requirements.get_components();
551 
552  for( const std::vector<item_comp> &component_options : component_requirements ) {
553  for( const item_comp &component_option : component_options ) {
554  potential_components.push_back( component_option.type );
555  }
556  }
557  }
558  }
559 
560  // Now check that graph for loops
561  std::vector<std::vector<itype_id>> loops = cata::find_cycles( potential_components_of );
562  for( const std::vector<itype_id> &loop : loops ) {
563  std::string error_message =
564  "loop in comestible recipes detected: " + loop.back().str();
565  for( const itype_id &i : loop ) {
566  error_message += " -> " + i.str();
567  items_on_loops.insert( i );
568  }
569  error_message += ". Such loops can be broken by either removing or altering "
570  "recipes or marking one of the items involved with the NUTRIENT_OVERRIDE "
571  "flag";
572  debugmsg( error_message );
573  }

734 void recipe_subset::include( const recipe *r, int custom_difficulty )
735 {
736  if( custom_difficulty < 0 ) {
737  custom_difficulty = r->difficulty;
738  }
739  // We always prefer lower difficulty for the subset, but we save it only if it's not default
740  if( recipes.count( r ) > 0 ) {
741  const auto iter = difficulties.find( r );
742  // See if we need to lower the difficulty of the existing recipe
743  if( iter != difficulties.end() && custom_difficulty < iter->second ) {
744  if( custom_difficulty != r->difficulty ) {
745  iter->second = custom_difficulty; // Added again with lower difficulty
746  } else {
747  difficulties.erase( iter ); // No need to keep the default difficulty. Free some memory
748  }
749  } else if( custom_difficulty < r->difficulty ) {
750  difficulties[r] = custom_difficulty; // Added again with lower difficulty
751  }
752  } else {
753  // add recipe to category and component caches
754  for( const auto &opts : r->simple_requirements().get_components() ) {
755  for( const item_comp &comp : opts ) {
756  component[comp.type].insert( r );
757  }
758  }
759  category[r->category].insert( r );
760  // Set the difficulty is it's not the default
761  if( custom_difficulty != r->difficulty ) {
762  difficulties[r] = custom_difficulty;
763  }
764  // insert the recipe
765  recipes.insert( r );
766  }
