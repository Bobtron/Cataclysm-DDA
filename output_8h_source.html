122 /**
123  * @anchor color_tags
124  * @name color tags
125  *
126  * Most print function have only one color parameter (or none at all), therefore they would
127  * print the whole text in one color. To print some parts of a text in a different color,
128  * one would have to write separate calls to the print functions.
129  *
130  * Color tags allow embedding coloring instructions directly into the text, which allows
131  * a single call to the print function with the full text. The color tags are removed
132  * (not printed), but the text inside the tags is printed with a specific color.
133  *
134  * Example: Print "You need a tool to do this." in white and the word "tool" in red.
135  * \code
136  * wprintz( w, c_white, "You need a " );
137  * wprintz( w, c_red, "tool" );
138  * wprintz( w, c_white, " to do this." );
139  * \endcode
140  * Same code with color tags:
141  * \code
142  * print_colored_text( w, c_white, "You need a <color_red>tool</color> to do this." );
143  * \endcode
144  *
145  * Color tags must appear in pairs, first `<color_XXX>`, followed by text, followed by `</color>`.
146  * The text in between is colored according to the color `XXX`. `XXX` must be a valid color name,
147  * see @ref color_from_string. Color tags do *not* stack, the text inside the color tags should
148  * not contain any other color tags.
149  *
150  * Functions that handle color tags should contain a note about this in their documentation. If
151  * they have no such note, they probably don't handle color tags (which means they just print the
152  * string as is).
153  *
154  * Note: use @ref colorize to add color tags to a string.
155  * \code
156  * nc_color color = ...;
157  * text = colorize( "some text", color );
158  * \endcode
159  *
160  * One can use @ref utf8_width with the second parameter set to `true` to determine the printed
161  * length of a string containing color tags. The parameter instructs `utf8_width` to ignore the
162  * color tags. For example `utf8_width("<color_red>text</color>")` would return 23, but
163  * `utf8_width("<color_red>text</color>", true)` returns 4 (the length of "text").
164  */

641 /**
642  * @return String containing enumerated elements in format: "a, b, c, ..., and z". Uses the Oxford comma.
643  * @param values A vector of strings
644  * @param conj Choose how to separate the last elements.
645  */
646 template<typename Container>
647 std::string enumerate_as_string( const Container &values,
648  enumeration_conjunction conj = enumeration_conjunction::and_ )
649 {
650  const std::string final_separator = [&]() {
651  switch( conj ) {
652  case enumeration_conjunction::none:
653  return _( ", " );
654  case enumeration_conjunction::and_:
655  return values.size() > 2 ? _( ", and " ) : _( " and " );
656  case enumeration_conjunction::or_:
657  return values.size() > 2 ? _( ", or " ) : _( " or " );
658  case enumeration_conjunction::arrow:
659  return _( " > " );
660  }
661  debugmsg( "Unexpected conjunction" );
662  return _( ", " );
663  }
664  ();
665  const std::string separator = [&conj]() {
666  switch( conj ) {
667  case enumeration_conjunction::arrow:
668  return _( " > " );
669  default:
670  return _( ", " );
671  }
672  }
673  ();
674  std::string res;
675  for( auto iter = values.begin(); iter != values.end(); ++iter ) {
676  if( iter != values.begin() ) {
677  if( std::next( iter ) == values.end() ) {
678  res += final_separator;
679  } else {
680  res += separator;
681  }
682  }
683  res += *iter;
684  }
685  return res;
686 }

692  * @param string_for Function that accepts an element and returns a representing string.
693  * May return an empty string to omit the element.

703  const std::string str( string_for( *iter ) );
704  if( !str.empty() ) {
705  values.push_back( str );
706  }
707  }
