309 void explosion_handler::draw_custom_explosion( const tripoint &,
310  const std::map<tripoint, nc_color> &all_area, const std::optional<std::string> &tile_id )
311 {
312  if( test_mode ) {
313  // avoid segfault from null tilecontext in tests
314  return;
315  }
316 
317  constexpr explosion_neighbors all_neighbors = N_NORTH | N_SOUTH | N_WEST | N_EAST;
318  // We will "shell" the explosion area
319  // Each phase will strip a single layer of points
320  // A layer contains all points that have less than 4 neighbors in cardinal directions
321  // Layers will first be generated, then drawn in inverse order
322 
323  // Start by getting rid of everything except current z-level
324  std::map<tripoint, explosion_tile> neighbors;
325  avatar &player_character = get_avatar();
326 #if defined(TILES)
327  if( !use_tiles ) {
328  for( const auto &pr : all_area ) {
329  const tripoint relative_point = relative_view_pos( player_character, pr.first );
330  if( relative_point.z == 0 ) {
331  neighbors[pr.first] = explosion_tile{ N_NO_NEIGHBORS, pr.second, tile_id };
332  }
333  }
334  } else {
335  // In tiles mode, the coordinates have to be absolute
336  const tripoint view_center = relative_view_pos( player_character, player_character.pos() );
337  for( const auto &pr : all_area ) {
338  // Relative point is only used for z level check
339  const tripoint relative_point = relative_view_pos( player_character, pr.first );
340  if( relative_point.z == view_center.z ) {
341  neighbors[pr.first] = explosion_tile{ N_NO_NEIGHBORS, pr.second, tile_id };
342  }
343  }
344  }
345 #else
346  for( const auto &pr : all_area ) {
347  const tripoint relative_point = relative_view_pos( player_character, pr.first );
348  if( relative_point.z == 0 ) {
349  neighbors[pr.first] = explosion_tile{ N_NO_NEIGHBORS, pr.second, tile_id };
350  }
351  }
352 #endif
353 
354  // Searches for a neighbor, sets the neighborhood flag on current point and on the neighbor
355  const auto set_neighbors = [&]( const tripoint & pos,
356  explosion_neighbors & ngh,
357  explosion_neighbors here,
358  explosion_neighbors there ) {
359  if( ( ngh & here ) == N_NO_NEIGHBORS ) {
360  auto other = neighbors.find( pos );
361  if( other != neighbors.end() ) {
362  ngh = ngh | here;
363  other->second.neighborhood = other->second.neighborhood | there;
364  }
365  }
366  };
367 
368  // If the point we are about to remove has a neighbor in a given direction
369  // unset that neighbor's flag that our current point is its neighbor
370  const auto unset_neighbor = [&]( const tripoint & pos,
371  const explosion_neighbors ngh,
372  explosion_neighbors here,
373  explosion_neighbors there ) {
374  if( ( ngh & here ) != N_NO_NEIGHBORS ) {
375  auto other = neighbors.find( pos );
376  if( other != neighbors.end() ) {
377  other->second.neighborhood = ( other->second.neighborhood | there ) ^ there;
378  }
379  }
380  };
381 
382  // Find all neighborhoods
383  for( auto &pr : neighbors ) {
384  const tripoint &pt = pr.first;
385  explosion_neighbors &ngh = pr.second.neighborhood;
386 
387  set_neighbors( pt + point_west, ngh, N_WEST, N_EAST );
388  set_neighbors( pt + point_east, ngh, N_EAST, N_WEST );
389  set_neighbors( pt + point_north, ngh, N_NORTH, N_SOUTH );
390  set_neighbors( pt + point_south, ngh, N_SOUTH, N_NORTH );
391  }
392 
393  // We need to save the layers because we will draw them in reverse order
394  std::list< std::map<tripoint, explosion_tile> > layers;
395  while( !neighbors.empty() ) {
396  std::map<tripoint, explosion_tile> layer;
397  bool changed = false;
398  // Find a layer that can be drawn
399  for( const auto &pr : neighbors ) {
400  if( pr.second.neighborhood != all_neighbors ) {
401  changed = true;
402  layer.insert( pr );
403  }
404  }
405  if( !changed ) {
406  // An error, but a minor one - let it slide
407  return;
408  }
409  // Remove the layer from the area to process
410  for( const auto &pr : layer ) {
411  const tripoint &pt = pr.first;
412  const explosion_neighbors ngh = pr.second.neighborhood;
413 
414  unset_neighbor( pt + point_west, ngh, N_WEST, N_EAST );
415  unset_neighbor( pt + point_east, ngh, N_EAST, N_WEST );
416  unset_neighbor( pt + point_north, ngh, N_NORTH, N_SOUTH );
417  unset_neighbor( pt + point_south, ngh, N_SOUTH, N_NORTH );
418  neighbors.erase( pr.first );
419  }
420 
421  layers.push_front( std::move( layer ) );
422  }
423 
424 #if defined(TILES)
425  if( !use_tiles ) {
426  draw_custom_explosion_curses( *g, layers );
427  return;
428  }
429 
430  explosion_animation anim;
431  // We need to draw all explosions up to now
432  std::map<tripoint, explosion_tile> combined_layer;
433 
434  shared_ptr_fast<game::draw_callback_t> explosion_cb =
435  make_shared_fast<game::draw_callback_t>( [&]() {
436  tilecontext->init_custom_explosion_layer( combined_layer );
437  } );
438  g->add_draw_callback( explosion_cb );
439 
440  for( const auto &layer : layers ) {
441  combined_layer.insert( layer.begin(), layer.end() );
442  if( is_layer_visible( layer ) ) {
443  anim.progress();
444  }
445  }
446 
447  tilecontext->void_custom_explosion();
448 #else
449  draw_custom_explosion_curses( *g, layers );
450 #endif
451 }
