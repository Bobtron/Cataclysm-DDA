13 /**
14  * @name Enumeration (de)serialization to/from string.
15  *
16  * @ref enum_to_string converts an enumeration value to a string (which can be written to JSON).
17  * The result must be an non-empty string.
18  *
19  * This function must be implemented somewhere for each enumeration type for
20  * which conversion is required.
21  *
22  * Such enums must also specialize enum_traits to specify their 'last' member.
23  *
24  * @ref string_to_enum converts the string value back into an enumeration value. The input
25  * is expected to be one of the outputs of @ref enum_to_string. If the given string does
26  * not match an enumeration, an @ref InvalidEnumString is thrown.
27  *
28  * @code string_to_enum<E>(enum_to_string<E>(X)) == X @endcode must yield true for all values
29  * of the enumeration E.
30  */

42 template<typename E>
43 std::unordered_map<std::string, E> build_enum_lookup_map()
44 {
45  static_assert( std::is_enum_v<E>, "E should be an enum type" );
46  static_assert( has_enum_traits<E>::value, "enum E needs a specialization of enum_traits" );
47  std::unordered_map<std::string, E> result;
48 
49  using Int = std::underlying_type_t<E>;
50  constexpr Int max = static_cast<Int>( enum_traits<E>::last );
51 
52  for( Int i = 0; i < max; ++i ) {
53  E e = static_cast<E>( i );
54  auto inserted = result.emplace( enum_to_string( e ), e );
55  if( !inserted.second ) {
56  cata_fatal( "repeated enum string %s (%d and %d)", inserted.first->first,
57  static_cast<Int>( inserted.first->second ), i );
58  }
59  }
60 
61  return result;
62 }
