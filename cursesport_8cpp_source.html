15  * Whoever cares, btw. not my base design, but this is how it works:
16  * In absent of a native curses library, this is a simple implementation to
17  * store the data that would be given to the curses system.
18  * It is later displayed using code in sdltiles.cpp. This file only contains
19  * the curses interface.
20  *
21  * The struct WINDOW is the base. It acts as the normal curses window, having
22  * width/height, current cursor location, current coloring (background/foreground)
23  * and the actual text.
24  * The text is split into lines (curseline), which contains cells (cursecell).

221 // Get a sequence of Unicode code points, store them in target
222 // return the display width of the extracted string.
223 static int fill( const char *&fmt, int &len, std::string &target )
224 {
225  const char *const start = fmt;
226  int dlen = 0; // display width
227  const char *tmpptr = fmt; // pointer for UTF8_getch, which increments it
228  int tmplen = len;
229  while( tmplen > 0 ) {
230  const uint32_t ch = UTF8_getch( &tmpptr, &tmplen );
231  // UNKNOWN_UNICODE is most likely a (vertical/horizontal) line or similar
232  const int cw = ch == UNKNOWN_UNICODE ? 1 : mk_wcwidth( ch );
233  if( cw > 0 && dlen > 0 ) {
234  // Stop at the *second* non-zero-width character
235  break;
236  }
237  if( cw == -1 && start == fmt ) {
238  // First char is a control character: they only disturb the screen,
239  // so replace it with a single space (e.g. instead of a '\t').
240  // Newlines at the begin of a sequence are handled in printstring
241  target.assign( " ", 1 );
242  len = tmplen;
243  fmt = tmpptr;
244  return 1; // the space
245  } else if( cw == -1 ) {
246  // Control character but behind some other characters, finish the sequence.
247  // The character will either by handled by printstring (if it's a newline),
248  // or by the next call to this function (replaced with a space).
249  break;
250  }
251  fmt = tmpptr;
252  dlen += cw;
253  }
254  target.assign( start, fmt - start );
255  len -= target.length();
256  return dlen;
257 }

270 //The core printing function, prints characters to the array, and sets colors
271 static void printstring( cata_cursesport::WINDOW *win, const std::string &text )
272 {
273  using cata_cursesport::cursecell;
274  win->draw = true;
275  int len = text.length();
276  if( len == 0 ) {
277  return;
278  }
279  const char *fmt = text.c_str();
280  // avoid having an invalid cursorx, so that cur_cell will only return nullptr
281  // when the bottom of the window has been reached.
282  if( win->cursor.x >= win->width ) {
283  if( newline( win ) == 0 ) {
284  return;
285  }
286  }
287  if( win->cursor.y >= win->height || win->cursor.x >= win->width ) {
288  return;
289  }
290  if( win->cursor.x > 0 && win->line[win->cursor.y].chars[win->cursor.x].ch.empty() ) {
291  // start inside a wide character, erase it for good
292  win->line[win->cursor.y].chars[win->cursor.x - 1].ch.assign( " " );
293  }
294  while( len > 0 ) {
295  if( *fmt == '\n' ) {
296  if( newline( win ) == 0 ) {
297  return;
298  }
299  fmt++;
300  len--;
301  continue;
302  }
303  cursecell *curcell = cur_cell( win );
304  if( curcell == nullptr ) {
305  return;
306  }
307  const int dlen = fill( fmt, len, curcell->ch );
308  if( dlen >= 1 ) {
309  curcell->FG = win->FG;
310  curcell->BG = win->BG;
311  addedchar( win );
312  }
313  if( dlen == 1 ) {
314  // a wide character was converted to a narrow character leaving a null in the
315  // following cell ~> clear it
316  cursecell *seccell = cur_cell( win );
317  if( seccell && seccell->ch.empty() ) {
318  seccell->ch.assign( ' ', 1 );
319  }
320  } else if( dlen == 2 ) {
321  // the second cell, per definition must be empty
322  cursecell *seccell = cur_cell( win );
323  if( seccell == nullptr ) {
324  // the previous cell was valid, this one is outside of the window
325  // --> the previous was the last cell of the last line
326  // --> there should not be a two-cell width character in the last cell
327  curcell->ch.assign( ' ', 1 );
328  return;
329  }
330  seccell->FG = win->FG;
331  seccell->BG = win->BG;
332  seccell->ch.erase();
333  addedchar( win );
334  // Have just written a wide-character into the last cell, it would not
335  // display correctly if it was the last *cell* of a line
336  if( win->cursor.x == 1 ) {
337  // So make that last cell a space, move the width
338  // character in the first cell of the line
339  seccell->ch = curcell->ch;
340  curcell->ch.assign( 1, ' ' );
341  // and make the second cell on the new line empty.
342  addedchar( win );
343  cursecell *thicell = cur_cell( win );
344  if( thicell != nullptr ) {
345  thicell->ch.erase();
346  }
347  }
348  }
349  if( win->cursor.y >= win->height ) {
350  return;
351  }
352  }
353 }
