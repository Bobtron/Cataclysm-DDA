91 {

94  []( size_t prev, const auto & kv ) {

97  for( std::pair<const int, std::list<item_reference>> &kv : active_items ) {
98  // Rely on iteration logic to make sure the number is sane.
99  int num_to_process = kv.second.size() / kv.first;
100  std::list<item_reference>::iterator it = kv.second.begin();
101  for( ; it != kv.second.end() && num_to_process >= 0; ) {
102  if( it->item_ref ) {

106  } else {
