167  BPGRAPH_MAXCOLS );

172  r.insert( r.end(), BPGRAPH_MAXCOLS - r.size(), " " );

659  // Bodypart not present on character
660  if( !!id->parent_bp && !u.has_part( *id->parent_bp, body_part_filter::equivalent ) ) {

662  //~ 1$ = 2nd person pronoun (You), 2$ = body part (left arm)
663  _( "%1$s do not have a %2$s." ) :
664  //~ 1$ = name of character, 2$ = body part (left arm)
665  _( "%1$s does not have a %2$s." ), u.disp_name( false, true ),
666  id->parent_bp->obj().name.translated() );

669  txt.insert( txt.begin(), center_text_pos( txt, 0, width ), ' ' );

671  } else {
672  ret.emplace_back( width, ' ' );
673  }
674  }
675  return ret;
676  }
677 
678  const bool hflip = !!id->mirror;
679  const bodygraph_id &rid = get_bg_rows( id );
680  for( int i = 0; static_cast<size_t>( i ) < rid->rows.size() && i < height; i++ ) {
681  std::string ret_row;
682  int j = hflip ? rid->rows[i].size() - 1 : 0;
683  for( int x = 0 ; x < width && j < BPGRAPH_MAXCOLS && j >= 0; hflip ? j-- : j++, x++ ) {
684  std::string sym = id->fill_sym.empty() ? rid->rows[i][j] : id->fill_sym;

686  const bodygraph_part *bgp = nullptr;

691  if( u.has_part( bp, body_part_filter::equivalent ) ) {
692  missing_section = false;

697  missing_section = false;
698  }
699  }

701  }
702  if( rid->rows[i][j] == " " ) {
703  sym = " ";
704  }
705  ret_row.append( fragment_cb( bgp, sym ) );
