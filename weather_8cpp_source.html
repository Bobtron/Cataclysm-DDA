636 units::temperature_delta get_local_windchill( units::temperature temperature, double humidity,
637  double wind_mph )
638 {
639  double windchill_k = 0;
640 
641  const double temperature_c = units::to_celsius( temperature );
642 
643  if( temperature_c < 10 ) {
644  /// Model 1, cold wind chill (only valid for temps below 50F/10C)
645  /// Is also used as a standard in North America.
646 
647  if( wind_mph < 3 ) {
648  // This model fails when wind is less than 3 mph
649  return units::from_kelvin_delta( 0 );
650  }
651 
652  // Temperature is removed at the end, because get_local_windchill is meant to calculate the difference.
653  // Source : http://en.wikipedia.org/wiki/Wind_chill#North_American_and_United_Kingdom_wind_chill_index
654  double wind_km_per_h = wind_mph * 0.44704 * 3.6;
655  windchill_k = 13.12 + 0.6215 * temperature_c - 11.37 * std::pow( wind_km_per_h,
656  0.16 ) + 0.3965 * temperature_c * std::pow( wind_km_per_h, 0.16 ) - temperature_c;
657  } else {
658  /// Model 2, warm wind chill
659 
660  // Source : http://en.wikipedia.org/wiki/Wind_chill#Australian_Apparent_Temperature
661  // Convert to meters per second.
662 
663  // Cap the vapor pressure term to 50C of extra heat, as this term
664  // otherwise grows logistically to an asymptotic value of about 2e7
665  // for large values of temperature. This is presumably due to the
666  // model being designed for reasonable ambient temperature values,
667  // rather than extremely high ones.
668  double wind_meters_per_sec = wind_mph * 0.44704;
669  windchill_k = 0.33 * std::min<float>( 150.00, humidity / 100.00 * 6.105 *
670  std::exp( 17.27 * temperature_c / ( 237.70 + temperature_c ) ) ) - 0.70 *
671  wind_meters_per_sec - 4.00;
672  }
673 
674  return units::from_kelvin_delta( windchill_k );
675 }
