124 int Character::item_reload_cost( const item &it, const item &ammo, int qty ) const
125 {
126  if( ammo.is_ammo() || ammo.is_frozen_liquid() || ammo.made_of_from_type( phase_id::LIQUID ) ) {
127  qty = std::max( std::min( ammo.charges, qty ), 1 );
128  } else if( ammo.is_ammo_container() ) {
129  int min_clamp = 0;
130  // find the first ammo in the container to get its charges
131  ammo.visit_items( [&min_clamp]( const item * it, item * ) {
132  if( it->is_ammo() ) {
133  min_clamp = it->charges;
134  return VisitResponse::ABORT;
135  }
136  return VisitResponse::NEXT;
137  } );
138 
139  qty = clamp( qty, min_clamp, 1 );
140  } else {
141  // Handle everything else as magazines
142  qty = 1;
143  }
144 
145  // No base cost for handling ammo - that's already included in obtain cost
146  // We have the ammo in our hands right now
147  int mv = item_handling_cost( ammo, true, 0, qty );
148 
149  if( ammo.has_flag( flag_MAG_BULKY ) ) {
150  mv *= 1.5; // bulky magazines take longer to insert
151  }
152 
153  if( !it.is_gun() && !it.is_magazine() ) {
154  return mv + 100; // reload a tool or sealable container
155  }
156 
157  /** @EFFECT_GUN decreases the time taken to reload a magazine */
158  /** @EFFECT_PISTOL decreases time taken to reload a pistol */
159  /** @EFFECT_SMG decreases time taken to reload an SMG */
160  /** @EFFECT_RIFLE decreases time taken to reload a rifle */
161  /** @EFFECT_SHOTGUN decreases time taken to reload a shotgun */
162  /** @EFFECT_LAUNCHER decreases time taken to reload a launcher */
163 
164  int cost = 0;
165  if( it.is_gun() ) {
166  cost = it.get_reload_time();
167  } else if( it.type->magazine ) {
168  cost = it.type->magazine->reload_time * qty;
169  } else {
170  cost = it.obtain_cost( ammo );
171  }
172 
173  skill_id sk = it.is_gun() ? it.type->gun->skill_used : skill_gun;
174  mv += cost / ( 1.0f + std::min( get_skill_level( sk ) * 0.1f, 1.0f ) );
175 
176  if( it.has_flag( flag_STR_RELOAD ) ) {
177  /** @EFFECT_STR reduces reload time of some weapons */
178  mv -= get_str() * 20;
179  }
180 
181  return std::max( static_cast<int>( std::round( mv * get_modifier(
