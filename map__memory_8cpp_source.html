258 {

260  cata_assert( p1.x() <= p2.x() && p1.y() <= p2.y() );
261

263  tripoint_abs_sm sm_p2 = coord_pair( p2 ).sm + point_south_east;
264 
265  tripoint_abs_sm sm_pos = sm_p1;
266  point_rel_sm sm_size = sm_p2.xy() - sm_p1.xy();
267 
268  if( sm_pos.z() == cache_pos.z() ) {
269  inclusive_rectangle<point_abs_sm> rect(
270  point_abs_sm( cache_pos.xy() ),
271  point_abs_sm( cache_pos.xy() + cache_size ) );
272  if( rect.contains( sm_p1.xy() ) && rect.contains( sm_p2.xy() ) ) {
273  return false;
274  }
275  }
276 
277  dbg( D_INFO ) << "Preparing memory map for area: pos: " << sm_pos << " size: " << sm_size;
278 
279  cache_pos = sm_pos;
280  cache_size = sm_size.raw();
281  cached.clear();
282  // Loop through each z-level in vision range
283  for( int z = std::max( sm_pos.z() - fov_3d_z_range, -OVERMAP_DEPTH );
284  z <= std::min( sm_pos.z() + fov_3d_z_range, OVERMAP_HEIGHT ); z++ ) {
285  cached[z].reserve( static_cast<std::size_t>( cache_size.x ) * cache_size.y );
