1012 void Character::reach_attack( const tripoint &p, int forced_movecost )
1013 {

1016  /** @EFFECT_MELEE >5 allows WHIP_DISARM technique */
1017  if( weapon.has_flag( flag_WHIP ) && ( get_skill_level( skill_melee ) > 5 ) && one_in( 3 ) ) {
1018  force_technique = WHIP_DISARM;

1023 
1024  creature_tracker &creatures = get_creature_tracker();
1025  Creature *critter = creatures.creature_at( p );
1026  // Original target size, used when there are monsters in front of our target
1027  const int target_size = critter != nullptr ? static_cast<int>( critter->get_size() ) : 2;
1028  // Reset last target pos
1029  last_target_pos = std::nullopt;
1030  // Max out recoil
1031  recoil = MAX_RECOIL;

1033  // Weariness handling
1034  // 1 / mult because mult is the percent penalty, in the form 1.0 == 100%
1035  const float weary_mult = 1.0f / exertion_adjusted_move_multiplier( EXTRA_EXERCISE );
1036  int move_cost = attack_speed( weapon ) * weary_mult;
1037  float skill = std::min( 10.0f, get_skill_level( skill_melee ) );
1038  int t = 0;
1039  map &here = get_map();
1040  std::vector<tripoint> path = line_to( pos(), p, t, 0 );
1041  path.pop_back(); // Last point is our critter
1042  for( const tripoint &path_point : path ) {
1043  // Possibly hit some unintended target instead
1044  Creature *inter = creatures.creature_at( path_point );
1045  /** @EFFECT_MELEE decreases chance of hitting intervening target on reach attack */
1046  if( inter != nullptr &&
1047  !x_in_y( ( target_size * target_size + 1 ) * skill,
1048  ( inter->get_size() * inter->get_size() + 1 ) * 10 ) ) {
1049  // Even if we miss here, low roll means weapon is pushed away or something like that
1050  if( inter->has_effect( effect_pet ) || ( inter->is_npc() &&
1051  inter->as_npc()->is_friendly( get_player_character() ) ) ) {
1052  if( query_yn( _( "Your attack may cause accidental injury, continue?" ) ) ) {
1053  critter = inter;
1054  break;
1055  } else {
1056  return;
1057  }
1058  }
1059  critter = inter;
1060  break;
1061  } else if( here.impassable( path_point ) &&
1062  // Fences etc. Spears can stab through those
1063  !( weapon.has_flag( flag_SPEAR ) &&
1064  here.has_flag( ter_furn_flag::TFLAG_THIN_OBSTACLE, path_point ) ) ) {
1065  /** @ARM_STR increases bash effects when reach attacking past something */
1066  here.bash( path_point, get_arm_str() + weapon.damage_melee( damage_bash ) );
1067  handle_melee_wear( get_wielded_item() );
1068  mod_moves( forced_movecost >= 0 ? -forced_movecost : -move_cost );
1069  return;
1070  }
1071  }

1074  add_msg_if_player( _( "You swing at the air." ) );
1075 
1076  const ma_technique miss_recovery = martial_arts_data->get_miss_recovery( *this );

1169

1171  // Fresh (skill-less) 8/8/8/8, unarmed:
1172  // 50%, 49%, 25%; ~1/16 guaranteed critical + ~1/8 if roll>dodge*1.5

1188  stat_crit_chance * skill_crit_chance +
