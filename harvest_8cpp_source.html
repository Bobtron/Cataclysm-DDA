193 void harvest_list::check_consistency()
194 {
195  for( const harvest_list &hl : get_all() ) {
196  const std::string hl_id = hl.id.c_str();
197  auto error_func = [&]( const harvest_entry & entry ) {
198  std::string errorlist;
199  if( entry.type.is_null() ) {
200  // Type id is null
201  if( item::type_is_defined( itype_id( entry.drop ) ) ) {
202  return errorlist;
203  }
204  errorlist += "null type";
205  } else if( !entry.type.is_valid() ) {
206  // Type id is invalid
207  errorlist += "invalid type \"" + entry.type.str() + "\"";
208  } else if( ( entry.type->is_item_group() &&
209  !item_group::group_is_defined( item_group_id( entry.drop ) ) ) ||
210  ( !entry.type->is_item_group() && !item::type_is_defined( itype_id( entry.drop ) ) ) ) {
211  // Specified as item_group but no such group exists, or
212  // specified as single itype but no such itype exists
213  errorlist += entry.drop;
214  }
215  return errorlist;
216  };
217  const std::string errors = enumerate_as_string( hl.entries_.begin(), hl.entries_.end(),
218  error_func );
219  if( !errors.empty() ) {
220  debugmsg( "Harvest list %s has invalid entry: %s", hl_id, errors );
221  }
222  if( !hl.leftovers.is_valid() ) {
223  debugmsg( "Harvest id %s has invalid leftovers: %s", hl.id.c_str(), hl.leftovers.c_str() );
224  }
225  }

253 std::string harvest_list::describe( int at_skill ) const
254 {
255  if( empty() ) {
256  return "";
257  }
258 
259  return enumerate_as_string( entries().begin(), entries().end(),
260  [at_skill]( const harvest_entry & en ) {
261  float min_f = en.base_num.first;
262  float max_f = en.base_num.second;
263  if( at_skill >= 0 ) {
264  min_f += en.scale_num.first * at_skill;
265  max_f += en.scale_num.second * at_skill;
266  } else {
267  max_f = en.max;
268  }
269  // TODO: Avoid repetition here by making a common harvest drop function
270  int max_drops = std::min<int>( en.max, std::round( std::max( 0.0f, max_f ) ) );
271  int min_drops = std::max<int>( 0.0f, std::round( std::min( min_f, max_f ) ) );
272  if( max_drops <= 0 ) {
273  return std::string();
274  }
275 
276  std::string ss;
277  ss += "<bold>" + item::nname( itype_id( en.drop ), max_drops ) + "</bold>";
278  // If the number is unspecified, just list the type
279  if( max_drops >= 1000 && min_drops <= 0 ) {
280  return ss;
281  }
282  ss += ": ";
283  if( min_drops == max_drops ) {
284  ss += string_format( "<stat>%d</stat>", min_drops );
285  } else if( max_drops < 1000 ) {
286  ss += string_format( "<stat>%d-%d</stat>", min_drops, max_drops );
