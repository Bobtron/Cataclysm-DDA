90 {
91  // Bypass set_field_age() so we don't reset decay_time;
92  age += 1_turns;
93  if( type.obj().half_life > 0_turns && get_field_age() > 0_turns ) {
94  // Legacy handling for fire because it's weird and complicated.

147 /*
148 Function: add_field
149 Inserts the given field_type_id into the field list for a given tile if it does not already exist.
150 Returns false if the field_type_id already exists, true otherwise.
151 If the field already exists, it will return false BUT it will add the intensity/age to the current values for upkeep.
152 If you wish to modify an already existing field use find_field and modify the result.
153 Intensity defaults to 1, and age to 0 (permanent) if not specified.
154 */
155 bool field::add_field( const field_type_id &field_type_to_add, const int new_intensity,
156  const time_duration &new_age )
157 {
158  // sanity check, we don't want to store fd_null
159  if( !field_type_to_add ) {
160  return false;
161  }
162  auto it = _field_type_list->find( field_type_to_add );
163  if( it != _field_type_list->end() ) {
164  //Already exists, but lets update it. This is tentative.
165  int prev_intensity = it->second.get_field_intensity();
166  if( !it->second.is_field_alive() ) {
167  it->second.set_field_age( new_age );
168  prev_intensity = 0;
169  }
170  it->second.set_field_intensity( prev_intensity + new_intensity );
171  return false;
172  }
173  if( !_displayed_field_type ||
174  field_type_to_add.obj().priority >= _displayed_field_type.obj().priority ) {
175  _displayed_field_type = field_type_to_add;
176  }
177  _field_type_list[field_type_to_add] = field_entry( field_type_to_add, new_intensity, new_age );
