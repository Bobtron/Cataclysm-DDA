179 static void dpad_changes( task_t &task, const std::array<int, 16> &m, Uint32 now, int old_state,
180  int new_state )
181 {
182  // get rid of unneeded bits
183  old_state &= 0b1111;
184  new_state &= 0b1111;
185 
186  if( old_state == new_state ) {
187  return;
188  }
189 
190  switch( new_state ) {
191  case SDL_HAT_LEFTUP:
192  case SDL_HAT_LEFTDOWN:
193  case SDL_HAT_RIGHTUP:
194  case SDL_HAT_RIGHTDOWN:
195  send_input( m[new_state] );
196  schedule_task( task, now + repeat_delay, m[new_state], new_state );
197  break;
198  case SDL_HAT_CENTERED: {
199  if( task.counter == 1 ) {
200  switch( old_state ) {
201  // detecting quick CENTER -> Left|Up|Right|Down -> CENTER transition
202  case SDL_HAT_UP:
203  case SDL_HAT_DOWN:
204  case SDL_HAT_RIGHT:
205  case SDL_HAT_LEFT:
206  send_input( task.button );
207  }
208  }
209  cancel_task( task );
210  break;
211  }
212  case SDL_HAT_UP:
213  case SDL_HAT_DOWN:
214  case SDL_HAT_RIGHT:
215  case SDL_HAT_LEFT:
216  schedule_task( task, now + diagonal_detect_delay, m[new_state], new_state );
217  task.counter += old_state;
218  break;
219  }

222 void handle_axis_event( SDL_Event &event )
223 {
224  if( event.type != SDL_CONTROLLERAXISMOTION ) {
225  return;
226  }
227 
228  int axis = event.caxis.axis;
229  int value = event.caxis.value;
230  Uint32 now = event.caxis.timestamp;
231 
232  // check triggers
233  int idx = one_of_two( triggers_axis, axis );
234  if( idx >= 0 ) {
235  int state = triggers_state[idx];
236  int button = triggers_map[idx];
237  task_t &task = all_tasks[triggers_task_index + idx];
238  if( !state && value > triggers_threshold + error_margin ) {
239  send_input( button );
240  triggers_state[idx] = 1;
241  schedule_task( task, now + repeat_delay, button, 1 );
242  }
243  if( state && value < triggers_threshold - error_margin ) {
244  triggers_state[idx] = 0;
245  cancel_task( task );
246  }
247  return;
248  }
249 
250  // check sticks
251  for( int i = 0; i < max_sticks; ++i ) {
252  int idx = one_of_two( sticks_axis[i], axis );
253  if( idx >= 0 ) {
254  int old_state = sticks_state[i];
255  int new_state = old_state;
256  task_t &task = all_tasks[sticks_task_index + i];
257 
258  // calculate dpad state for the analog sticks, simulating joystick hat state (4 bits)
259  if( idx ) { // vertical stick movement
260  if( !( old_state & 0b0100 ) && value > sticks_threshold + error_margin ) {
261  new_state |= 0b0100; // turn on bit _x__
262  } else if( ( old_state & 0b0100 ) && value < sticks_threshold - error_margin ) {
263  new_state &= 0b1011; // turn off bit _x__
264  } else if( !( old_state & 0b0001 ) && value < -sticks_threshold - error_margin ) {
265  new_state |= 0b0001; // turn on bit ___x
266  } else if( ( old_state & 0b0001 ) && value > -sticks_threshold + error_margin ) {
267  new_state &= 0b1110; // turn off bit ___x
268  }
269  } else { // horizontal stick movement
270  if( !( old_state & 0b0010 ) && value > sticks_threshold + error_margin ) {
271  new_state |= 0b0010; // turn on bit __x_
272  } else if( ( old_state & 0b0010 ) && value < sticks_threshold - error_margin ) {
273  new_state &= 0b1101; // turn off bit __x_
274  } else if( !( old_state & 0b1000 ) && value < -sticks_threshold - error_margin ) {
275  new_state |= 0b1000; // turn on bit x___
276  } else if( ( old_state & 0b1000 ) && value > -sticks_threshold + error_margin ) {
277  new_state &= 0b0111; // turn off bit x___
278  }
279  }
280 
281  sticks_state[i] = new_state;
282  dpad_changes( task, sticks_map[i], now, old_state, new_state );
283  }
284  }
