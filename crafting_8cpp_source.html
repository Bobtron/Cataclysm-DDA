168 {
169  // negative is bright, 0 is just bright enough, positive is dark, +7.0f is pitch black
170  float darkness = fine_detail_vision_mod( p ) - 4.0f;
171  if( darkness <= 0.0f ) {
172  return 1.0f; // it's bright, go for it
173  }
174  bool rec_blind = rec.has_flag( flag_BLIND_HARD ) || rec.has_flag( flag_BLIND_EASY );
175  if( darkness > 0 && !rec_blind ) {
176  return 0.0f; // it's dark and this recipe can't be crafted in the dark
177  }
178  if( rec.has_flag( flag_BLIND_EASY ) ) {
179  // 100% speed in well lit area at skill+0
180  // 25% speed in pitch black at skill+0
181  // skill+2 removes speed penalty
182  if( rec.skill_used ) {

184  2 - exceeds_recipe_requirements( rec ) ) / 2.0f;
185  } else { // If there's no skill involved there shouldn't be any speed reduction due to not having levels in a non existent skill.
186  return 1.0f;

188  }

190  exceeds_recipe_requirements( rec ) >= 2 ) ) {
191  // 100% speed in well lit area at skill+2
192  // 25% speed in pitch black at skill+2
193  // skill+8 removes speed penalty
194  if( rec.skill_used ) {
195  return 1.0f - ( darkness / ( 7.0f / 0.75f ) ) * std::max( 0,
196  8 - exceeds_recipe_requirements( rec ) ) / 6.0f;

822 static item_location place_craft_or_disassembly(
823  Character &ch, item &craft, std::optional<tripoint> target )
824 {
825  item_location craft_in_world;
826 
827  // Check if we are standing next to a workbench. If so, just use that.
828  float best_bench_multi = 0.0f;
829  map &here = get_map();
830  for( const tripoint &adj : here.points_in_radius( ch.pos(), 1 ) ) {
831  if( here.dangerous_field_at( adj ) ) {
832  continue;
833  }
834  if( const cata::value_ptr<furn_workbench_info> &wb = here.furn( adj ).obj().workbench ) {
835  if( wb->multiplier > best_bench_multi ) {
836  best_bench_multi = wb->multiplier;
837  target = adj;
838  }
839  } else if( const std::optional<vpart_reference> vp = here.veh_at(
840  adj ).part_with_feature( "WORKBENCH", true ) ) {
841  if( const std::optional<vpslot_workbench> &wb_info = vp->part().info().workbench_info ) {
842  if( wb_info->multiplier > best_bench_multi ) {
843  best_bench_multi = wb_info->multiplier;
844  target = adj;
845  }
846  } else {
847  debugmsg( "part '%s' with WORKBENCH flag has no workbench info", vp->part().name() );
848  }
849  }
850  }
851 
852  // Crafting without a workbench
853  if( !target ) {
854  if( !ch.has_two_arms_lifting() ) {
855  craft_in_world = set_item_map_or_vehicle( ch, ch.pos(), craft );
856  } else if( !ch.has_wield_conflicts( craft ) || ch.is_npc() ) {
857  // NPC always tries wield craft first
858  if( std::optional<item_location> it_loc = wield_craft( ch, craft ) ) {
859  craft_in_world = *it_loc;
860  } else {
861  // This almost certainly shouldn't happen
862  put_into_vehicle_or_drop( ch, item_drop_reason::tumbling, {craft} );
863  }
864  } else {
865  enum option : int {
866  WIELD_CRAFT = 0,
867  DROP_CRAFT,
868  STASH,
869  DROP
870  };
871 
872  uilist amenu;
873  amenu.text = string_format( pgettext( "in progress craft", "What to do with the %s?" ),
874  craft.display_name() );
875 
876  amenu.addentry( WIELD_CRAFT, ch.can_unwield( *ch.get_wielded_item() ).success(),
877  '1', _( "Dispose of your wielded %s and start working." ), ch.get_wielded_item()->tname() );
878  amenu.addentry( DROP_CRAFT, true, '2', _( "Put it down and start working." ) );
879  const bool can_stash = ch.can_pickVolume( craft ) &&
880  ch.can_pickWeight( craft, !get_option<bool>( "DANGEROUS_PICKUPS" ) );
881  amenu.addentry( STASH, can_stash, '3', _( "Store it in your inventory." ) );
882  amenu.addentry( DROP, true, '4', _( "Drop it on the ground." ) );
883 
884  amenu.query();
885  const option choice = amenu.ret == UILIST_CANCEL ? DROP : static_cast<option>( amenu.ret );
886  switch( choice ) {
887  case WIELD_CRAFT: {
888  if( std::optional<item_location> it_loc = wield_craft( ch, craft ) ) {
889  craft_in_world = *it_loc;
890  } else {
891  // This almost certainly shouldn't happen
892  put_into_vehicle_or_drop( ch, item_drop_reason::tumbling, {craft} );
893  }
894  break;
895  }
896  case DROP_CRAFT: {
897  craft_in_world = set_item_map_or_vehicle( ch, ch.pos(), craft );
898  break;
899  }
900  case STASH: {
901  set_item_inventory( ch, craft );
902  break;
903  }
904  case DROP: {
905  put_into_vehicle_or_drop( ch, item_drop_reason::deliberate, {craft} );
906  break;
907  }
908  }
909  }
910  } else {
911  // We have a workbench, put the item there.
912  craft_in_world = set_item_map_or_vehicle( ch, *target, craft );
913  }
914 
915  if( !craft_in_world ) {
916  ch.add_msg_if_player( _( "Wield and activate the %s to start working." ), craft.tname() );
917  }
918 
919  return craft_in_world;

1023  proficiency_id proficiency;

1026  };
1027

1029  all_crafters.push_back( this );

1031  bool player_gained_prof = false;

1035  if( !p->_proficiencies->has_learned( prof.id ) &&
1036  prof.id->can_learn() &&
1037  p->_proficiencies->has_prereqs( prof.id ) ) {
1038  learn_subject subject{
1039  prof.id,
1040  prof.learning_time_mult / prof.time_multiplier,
1041  prof.max_experience
1042  };
1043  subjects.push_back( subject );
1044  }
1045  }
1046

1048  player_gained_prof = false;
1049  continue;
1050  }
1051  const time_duration learn_time = time / subjects.size();
1052 
1053  bool gained_prof = false;
1054  for( const learn_subject &subject : subjects ) {

1058  // Other characters who know the proficiency help you learn faster
1059  helper_bonus = 2;
1060  }
1061  }
1062  gained_prof |= p->practice_proficiency( subject.proficiency,
1063  subject.time_multiplier * learn_time * helper_bonus,
1064  subject.max_experience );
1065  }
1066  // This depends on the player being the last element in the iteration
1067  player_gained_prof = gained_prof;
1068  }

1073 float Character::get_recipe_weighted_skill_average( const recipe &making ) const
1074 {
1075  float secondary_skill_total = 0;
1076  int secondary_difficulty = 0;
1077  for( const std::pair<const skill_id, int> &count_secondaries : making.required_skills ) {
1078  // the difficulty of each secondary skill, count_secondaries.second, adds weight:
1079  // skills required at a higher level count more.
1080  secondary_skill_total += get_skill_level( count_secondaries.first ) * count_secondaries.second;
1081  secondary_difficulty += count_secondaries.second;
1082  }
1083  add_msg_debug( debugmode::DF_CRAFTING,
1084  "For craft %s, has %g secondary skills with difficulty sum %d", making.ident().str(),
1085  secondary_skill_total, secondary_difficulty );
1086  // The primary required skill counts extra compared to the secondary skills, before factoring in the
1087  // weight added by the required level.
1088  const float weighted_skill_average =
1089  ( ( 2.0f * std::max( making.difficulty,
1090  1 ) * get_skill_level( making.skill_used ) ) + secondary_skill_total ) /
1091  // No DBZ
1092  std::max( 1.f, ( 2.0f * std::max( making.difficulty, 1 ) + secondary_difficulty ) );
1093  add_msg_debug( debugmode::DF_CRAFTING, "Weighted skill average: %g", weighted_skill_average );
1094 
1095  float total_skill_modifiers = 0.0f;
1096 
1097  // farsightedness can impose a penalty on electronics and tailoring success
1098  // This should be changed to a json-defined penalty by skill (vision_penalty) in skills.json
1099  if( has_flag( json_flag_HYPEROPIC ) && !worn_with_flag( flag_FIX_FARSIGHT ) &&
1100  !has_effect( effect_contacts ) &&
1101  !has_effect( effect_transition_contacts ) ) {
1102  float vision_penalty = 0.0f;
1103  if( making.skill_used == skill_electronics ) {
1104  vision_penalty = 2.0f;
1105  } else if( making.skill_used == skill_tailor ) {
1106  vision_penalty = 1.0f;
1107  }
1108  total_skill_modifiers -= vision_penalty;
1109  }
1110 
1111  // Mutations can define specific skill bonuses and penalties. Gotta include those.
1112  for( const trait_id &mut : get_mutations() ) {
1113  for( const std::pair<const skill_id, int> &skill_bonuses : mut->craft_skill_bonus ) {
1114  if( making.skill_used == skill_bonuses.first ) {
1115  total_skill_modifiers += skill_bonuses.second * 1.0f;
1116  }
1117  }
1118  }
1119 
1120  // TO DO: Attribute role should also be data-driven either in skills.json or in the recipe itself.
1121  // For now let's just use Intelligence. For the average intelligence of 8, give +2. Inc/dec by 0.25 per stat point.
1122  // This ensures that at parity, where skill = difficulty, you have a roughly 85% chance of success at average intelligence.
1123  total_skill_modifiers += int_cur / 4.0f;
1124  add_msg_debug( debugmode::DF_CRAFTING, "Total skill modifiers: %g (+%g from int)",
1125  total_skill_modifiers, int_cur / 4.f );
1126 
1127  // Missing proficiencies penalize skill level
1128  for( const recipe_proficiency &recip : making.proficiencies ) {
1129  if( !recip.required && !has_proficiency( recip.id ) ) {

1139 Character::craft_roll_data Character::recipe_success_roll_data( const recipe &making ) const
1140 {
1141  // We're going to use a sqrt( sum of squares ) method here to give diminishing returns for more low level helpers.
1142  float player_weighted_skill_average = get_recipe_weighted_skill_average( making );
1143  bool negative = player_weighted_skill_average < 0;
1144  player_weighted_skill_average = std::pow( player_weighted_skill_average, 2 );
1145  if( negative ) {
1146  player_weighted_skill_average *= -1;
1147  }
1148 
1149  float helpers_weighted_average = 0.0f;
1150  for( const Character *guy : get_crafting_helpers() ) {
1151  // can they actually craft this recipe?
1152  bool has_all_skills = guy->get_skill_level( making.skill_used ) >= making.difficulty;
1153  if( has_all_skills ) {
1154  for( const std::pair<const skill_id, int> &secondary_skills : making.required_skills ) {
1155  if( guy->get_skill_level( secondary_skills.first ) < secondary_skills.second ) {
1156  has_all_skills = false;
1157  }
1158  }
1159  }
1160  if( has_all_skills && !making.character_has_required_proficiencies( *guy ) ) {
1161  has_all_skills = false;
1162  }
1163  if( has_all_skills ) {
1164  const float helper_skill_average = std::max( guy->get_recipe_weighted_skill_average( making ),
1165  0.0f );
1166  helpers_weighted_average += std::pow( helper_skill_average, 2 );
1167  add_msg_if_player( m_info, _( "%s helps with crafting…" ), guy->name );
1168  }
1169  }
1170 
1171  // Use a sum of squares to determine the final weighted average.
1172  negative = false;
1173  float summed_skills = player_weighted_skill_average + helpers_weighted_average;
1174  if( summed_skills < 0 ) {
1175  negative = true;
1176  // Store that it was negative to reapply later, then make it positive because we need a sqrt.
1177  summed_skills *= -1;
1178  }
1179  float weighted_skill_average = sqrt( summed_skills );
1180  if( negative ) {
1181  weighted_skill_average *= -1;
1182  }
1183  add_msg_debug( debugmode::DF_CHARACTER, "Skill combined with followers: %f",
1184  weighted_skill_average );
1185 
1186  int secondary_difficulty = 0;
1187  int secondary_level_count = 0;
1188  for( const std::pair<const skill_id, int> &count_secondaries : making.required_skills ) {
1189  secondary_level_count += count_secondaries.second;
1190  secondary_difficulty += std::pow( count_secondaries.second, 2 );
1191  }
1192 
1193  // The actual difficulty of the skill check is equal to:
1194  // (2*(primary skill difficulty)^2 + (sum of secondary difficulties)) / (2*(primary skill difficulty) + number of secondary skills)
1195  // This is basically just a hard to read weighted average formula: the fact that the primary difficulty gets squared
1196  // and then divided out again makes it a bit messy. Sorry, less mathy friends.
1197  const float final_difficulty =
1198  ( 2.0f * making.difficulty * making.difficulty + 1.0f * secondary_difficulty ) /
1199  // NO DBZ
1200  std::max( 1.f, ( 2.0f * making.difficulty + 1.0f * secondary_level_count ) );
1201  add_msg_debug( debugmode::DF_CHARACTER, "Final craft difficulty: %f", final_difficulty );
1202 
1203  // in the future we might want to make the standard deviation vary depending on some feature of the recipe.
1204  // For now, it varies only depending on your skill relative to the recipe's difficulty.
1205  float crafting_stddev = 2.0f;
1206  if( final_difficulty > weighted_skill_average ) {
1207  // Increase the standard deviation by 0.33 for every point below the difficulty your skill level is.
1208  // This makes the rolls more random and "swingy" at low levels, based more on luck.
1209  crafting_stddev += ( final_difficulty - weighted_skill_average ) / 3;
1210  }
1211  if( final_difficulty < weighted_skill_average ) {
1212  // decrease the standard deviation by 0.25 for every point above the difficulty your skill level is, to a cap of 1.
1213  // This means that luck plays less of a role the more overqualified you are.
1214  crafting_stddev -= std::min( ( weighted_skill_average - final_difficulty ) / 4, 1.0f );
1215  }
1216 
1217  // Let's just be careful, I don't want to touch a negative stddev
1218  crafting_stddev = std::max( crafting_stddev, 1.f );
1219 
1220  craft_roll_data ret;
1221  ret.center = weighted_skill_average;
1222  ret.stddev = crafting_stddev;
1223  ret.final_difficulty = final_difficulty + 1;
1224  if( has_trait( trait_DEBUG_CNF ) ) {
1225  ret.center = 2.f;
1226  ret.stddev = std::numeric_limits<decltype( ret.stddev )>::epsilon();
1227  ret.final_difficulty = 0.f;
1228  }
1229  return ret;
1230 }
