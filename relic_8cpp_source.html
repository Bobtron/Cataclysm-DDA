314 {
315  const bool time = activation_time == 0_seconds;
316  const bool regen_ammo = regenerate_ammo && item_can_not_load_ammo( parent );
317  const bool has_max_charges = !regenerate_ammo && charges >= max_charges && max_charges != 0;
318  if( time || regen_ammo || has_max_charges ) {
319  return;

470 void relic::try_recharge( item &parent, Character *carrier, const tripoint &pos )
471 {
472  if( charge.regenerate_ammo && item_can_not_load_ammo( parent ) ) {
473  return;
474  }
475  if( !charge.regenerate_ammo && charge.charges >= charge.max_charges ) {
476  return;
477  }
478  if( !can_recharge( parent, carrier ) ) {
479  return;
480  }
481 
482  switch( charge.type ) {
483  case relic_recharge_type::NONE: {
484  return;
485  }
486  case relic_recharge_type::PERIODIC: {
487  charge.accumulate_charge( parent );
488  return;
489  }
490  case relic_recharge_type::SOLAR_SUNNY: {
491  if( can_recharge_solar( parent, carrier, pos ) &&
492  get_weather().weather_id->light_modifier >= 0 ) {
493  charge.accumulate_charge( parent );
494  }
495  return;
496  }
497  case relic_recharge_type::LUNAR : {
498  if( can_recharge_lunar( parent, carrier, pos ) &&
499  get_moon_phase( calendar::turn ) >= MOON_NEW &&
500  get_moon_phase( calendar::turn ) <= MOON_WANING_CRESCENT ) {
501  charge.accumulate_charge( parent );
502  }
503  return;
504  }
505  case relic_recharge_type::FULL_MOON : {
506  if( can_recharge_lunar( parent, carrier, pos ) &&
507  get_moon_phase( calendar::turn ) == MOON_FULL ) {
508  charge.accumulate_charge( parent );
509  }
510  return;
511  }
512  case relic_recharge_type::NEW_MOON : {
513  if( can_recharge_lunar( parent, carrier, pos ) &&
514  get_moon_phase( calendar::turn ) == MOON_NEW ) {
515  charge.accumulate_charge( parent );
516  }
517  return;
518  }
519  case relic_recharge_type::SOLAR_CLOUDY: {
520  if( can_recharge_solar( parent, carrier, pos ) &&
521  get_weather().weather_id->light_modifier < 0 ) {
522  charge.accumulate_charge( parent );
523  }
524  return;
525  }
526  case relic_recharge_type::NUM: {
527  debugmsg( "Attempted to recharge relic with invalid recharge type" );
528  return;
529  }
530  }
531 }

653  const itype_id &it_id ) const
654 {
655  relic ret;
656  int num_attributes = 0;
657  int negative_attribute_power = 0;
658  const bool is_armor = item( it_id ).is_armor();
659 
660  while( rules.max_attributes > num_attributes && rules.power_level > ret.power_level( id ) ) {
661  switch( *type_weights.pick() ) {
662  case relic_procgen_data::type::active_enchantment: {
663  const relic_procgen_data::enchantment_active *active = active_procgen_values.pick();
664  if( active != nullptr ) {
665  fake_spell active_sp;
666  active_sp.id = active->activated_spell;
667  active_sp.level = rng( active->min_level, active->max_level );
668  int power = power_level( active_sp );
669  if( power < 0 ) {
670  if( rules.max_negative_power > negative_attribute_power ) {
671  break;
672  }
673  negative_attribute_power += power;
674  }
675  num_attributes++;
676  ret.add_active_effect( active_sp );
677  }
678  break;
679  }
680  case relic_procgen_data::type::passive_enchantment_add: {
681  const relic_procgen_data::enchantment_value_passive<int> *add = passive_add_procgen_values.pick();
682  if( add != nullptr ) {
683  enchant_cache ench;
684  int value = rng( add->min_value, add->max_value );
685  if( value == 0 ) {
686  break;
687  }
688  ench.add_value_add( add->type, value );
689  int negative_ench_attribute = power_level( ench );
690  if( negative_ench_attribute < 0 ) {
691  if( rules.max_negative_power > negative_attribute_power ) {
692  break;
693  }
694  negative_attribute_power += negative_ench_attribute;
695  }
696  if( add->ench_has ) {
697  ench.set_has( add->ench_has );
698  } else if( is_armor ) {
699  ench.set_has( enchantment::has::WORN );
700  } else {
701  ench.set_has( enchantment::has::HELD );
702  }
703  num_attributes++;
704  ret.add_passive_effect( ench );
705  }
706  break;
707  }
708  case relic_procgen_data::type::passive_enchantment_mult: {
709  const relic_procgen_data::enchantment_value_passive<float> *mult =
710  passive_mult_procgen_values.pick();
711  if( mult != nullptr ) {
712  enchant_cache ench;
713  float value = rng_float( mult->min_value, mult->max_value );
714  ench.add_value_mult( mult->type, value );
715  int negative_ench_attribute = power_level( ench );
716  if( negative_ench_attribute < 0 ) {
717  if( rules.max_negative_power > negative_attribute_power ) {
718  break;
719  }
720  negative_attribute_power += negative_ench_attribute;
721  }
722  if( mult->ench_has ) {
723  ench.set_has( mult->ench_has );
724  } else if( is_armor ) {
725  ench.set_has( enchantment::has::WORN );
726  } else {
727  ench.set_has( enchantment::has::HELD );
728  }
729  num_attributes++;
730  ret.add_passive_effect( ench );
731  }
732  break;
733  }
734  case relic_procgen_data::type::hit_me: {
735  const relic_procgen_data::enchantment_active *active = passive_hit_me.pick();
736  if( active != nullptr ) {
737  fake_spell active_sp;
738  active_sp.id = active->activated_spell;
739  active_sp.level = rng( active->min_level, active->max_level );

754  ench.set_has( enchantment::has::HELD );
755  }
756  num_attributes++;
757  ret.add_passive_effect( ench );
758  }
759  break;
760  }
761  case relic_procgen_data::type::hit_you: {
762  const relic_procgen_data::enchantment_active *active = passive_hit_you.pick();
763  if( active != nullptr ) {
764  fake_spell active_sp;
765  active_sp.id = active->activated_spell;
766  active_sp.level = rng( active->min_level, active->max_level );
767  enchant_cache ench;

777  ench.set_has( active->ench_has );

785  }
786  break;
787  }
788  case relic_procgen_data::type::last: {
789  debugmsg( "ERROR: invalid relic procgen type" );
790  break;
791  }
792  }
793  }
794 
795  if( ret.has_activation() ) {
796  const relic_charge_template *charge = charge_values.pick();
797  if( charge != nullptr ) {

807  resonance.set_has( enchantment::has::HELD );

809  }
