175  const int sight_points = you.overmap_sight_range( g->light_level( you.posz() ) );
176 
177  // i scans across width, with 0 in the middle(ish)
178  // -(w/2) ... w-(w/2)-1

196  std::pair<std::string, nc_color> sym_color = display::overmap_note_symbol_color( note_text );
197  ter_sym = sym_color.first;
198  ter_color = sym_color.second;
199  } else if( !seen ) {
200  // Always gray # for unseen
201  ter_sym = "#";
202  ter_color = c_dark_gray;
203  } else if( overmap_buffer.has_vehicle( omp ) ) {
204  ter_color = c_cyan;
205  ter_sym = overmap_buffer.get_vehicle_ter_sym( omp );
206  } else {
207  // Otherwise, get symbol and color appropriate for the terrain
208  const oter_id &cur_ter = overmap_buffer.ter( omp );
209  ter_sym = cur_ter->get_symbol();
210  if( overmap_buffer.is_explored( omp ) ) {
211  ter_color = c_dark_gray;
212  } else {
213  ter_color = cur_ter->get_color();
214  }
215  }
216  if( !drew_mission && targ.xy() == omp.xy() ) {
217  // If there is a mission target, and it's not on the same
218  // overmap terrain as the player character, mark it.
219  // TODO: Inform player if the mission is above or below
220  drew_mission = true;
221  if( i != 0 || j != 0 ) {
222  ter_color = red_background( ter_color );
223  }
224  }
225  // TODO: Build colorized string instead of writing directly to window
226  if( i == 0 && j == 0 ) {

475 void panel_manager::deserialize( const JsonArray &ja )
476 {
477  JsonObject joLayouts = ja.get_object( 0 );
478 
479  current_layout_id = joLayouts.get_string( "current_layout_id" );
480  if( layouts.find( current_layout_id ) == layouts.end() ) {
481  // Layout id updated between loads.
482  // Shouldn't happen unless custom sidebar id's were modified or removed.
483  joLayouts.allow_omitted_members();
484  current_layout_id = "labels";
485  return;
486  }
487 
488  for( JsonObject joLayout : joLayouts.get_array( "layouts" ) ) {
489  std::string layout_id = joLayout.get_string( "layout_id" );
490  const auto &cur_layout = layouts.find( layout_id );
491  if( cur_layout == layouts.end() ) {
492  joLayout.allow_omitted_members();
493  continue;
494  }
495 
496  auto &layout = cur_layout->second.panels();
497  auto it = layout.begin();
498 
499  for( JsonObject joPanel : joLayout.get_array( "panels" ) ) {
500  std::string id = joPanel.get_string( "name" );
501  bool toggle = joPanel.get_bool( "toggle" );
502 
503  for( auto it2 = layout.begin() + std::distance( layout.begin(), it ); it2 != layout.end(); ++it2 ) {
504  if( it2->get_id() == id ) {
505  if( it->get_id() != id ) {
506  window_panel panel = *it2;
507  layout.erase( it2 );
508  it = layout.insert( it, panel );
509  }
510  it->toggle = toggle;
511  ++it;
512  break;
513  }
514  }
515  }
516  }
517  if( ja.size() > 1 ) {
518  ja.throw_error( "panel_manager expects one object" );
519  }
520 }
