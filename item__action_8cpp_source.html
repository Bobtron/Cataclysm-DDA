115 item_action_map item_action_generator::map_actions_to_items( Character &you,
116  const std::vector<item *> &pseudos, const bool use_player_inventory ) const
117 {
118  std::set< item_action_id > unmapped_actions;
119  for( const auto &ia_ptr : item_actions ) { // Get ids of wanted actions
120  unmapped_actions.insert( ia_ptr.first );
121  }
122 
123  item_action_map candidates;
124  std::vector< item * > items;
125  // Default behavior
126  if( use_player_inventory ) {
127  items = you.inv_dump();
128  items.reserve( items.size() + pseudos.size() );
129  items.insert( items.end(), pseudos.begin(), pseudos.end() );
130  } else {
131  items = pseudos;
132  }
133 
134  std::unordered_set< item_action_id > to_remove;
135  for( item *i : items ) {
136  if( !i->item_has_uses_recursive() ) {
137  continue;
138  }
139 
140  for( const item_action_id &use : unmapped_actions ) {
141  // Actually used item can be different from the "outside item"
142  // For example, sheathed knife
143  item *actual_item = i->get_usable_item( use );
144  if( actual_item == nullptr ) {
145  continue;
146  }
147 
148  const use_function *func = actual_item->get_use( use );
149  if( !( func && func->get_actor_ptr() &&
150  func->get_actor_ptr()->can_use( you, *actual_item, you.pos() ).success() ) ) {
151  continue;
152  }
153 
154  if( !actual_item->ammo_sufficient( &you, use ) ) {
155  continue;
156  }
157 
158  // Don't try to remove 'irremovable' toolmods
159  if( actual_item->is_toolmod() && use == STATIC( item_action_id( "TOOLMOD_ATTACH" ) ) &&
160  actual_item->has_flag( STATIC( flag_id( "IRREMOVABLE" ) ) ) ) {
161  continue;
162  }
163 
164  // Prevent drinking frozen liquids that have specific use actions (ex: ALCOHOL)
165  if( actual_item->is_frozen_liquid() ) {
166  continue;
167  }
168 
169  // Add to usable items if it needs less charges per use or has less charges
170  auto found = candidates.find( use );
171  bool better = false;
172  if( found == candidates.end() ) {
173  better = true;
174  } else {
175  if( actual_item->ammo_required() > found->second->ammo_required() ) {
176  continue; // Other item consumes less charges
177  }
178 
179  if( found->second->ammo_remaining() > actual_item->ammo_remaining() ) {
180  better = true; // Items with less charges preferred
181  }
182  }
183 
184  if( better ) {
185  candidates[use] = actual_item;
186  if( actual_item->ammo_required() == 0 ) {
187  to_remove.insert( use );
188  }
189  }
190  }
191 
192  for( const item_action_id &r : to_remove ) {
193  unmapped_actions.erase( r );
194  }
195  }
196
