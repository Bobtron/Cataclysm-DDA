884 ter_furn_id forest_biome::pick() const
885 {
886  // Iterate through the biome components (which have already been put into sequence), roll for the
887  // one_in chance that component contributes a feature, and if so pick that feature and return it.
888  // If a given component does not roll as success, proceed to the next feature in sequence until
889  // a feature is picked or none are picked, in which case an empty feature is returned.
890  const ter_furn_id *result = nullptr;
891  for( const forest_biome_component &pr : biome_components ) {
892  if( one_in( pr.chance ) ) {
893  result = pr.types.pick();
894  break;
895  }
896  }
897 
898  if( result == nullptr ) {
899  return ter_furn_id();
900  }
901 
902  return *result;
903 }
