92 /**
93  * Returns a random entry in the container.
94  * The container must have a `size()` function and must support iterators as usual.
95  * For empty containers it returns the given default object.
96  * `C` is the container type,
97  * `D` is the type of the default value (which may differ from the return type)
98  * `V` is the type of the elements in the container.
99  * Note that this function does not return a reference because the default value could be
100  * a temporary object that is not valid after this function has left:
101  * \code random_entry( vect, std::string("default") ); \endcode
102  */
103 template<typename C, typename D, typename V = typename C::value_type>
104 inline V random_entry( const C &container, D default_value )
105 {
106  if( container.empty() ) {
107  return default_value;
108  }
109  auto iter = container.begin();
110  std::advance( iter, rng( 0, container.size() - 1 ) );
111  return *iter;
112 }
113 /**
114  * Same as above but returns a reference to the chosen entry (if the container
115  * is not empty) or an empty `optional` object (if the container is empty).
116  * This function handles empty containers without requiring an instance of the

118  */
