712  map &here = get_map();

715  }
716  if( !src_set.empty() ) {
717  _inv.form_from_zone( target_map, src_set, nullptr, false );
718  }
719  /*
720  * something of a hack: add the resources we know the camp has
721  * the hacky part is that we're adding resources based on the camp's flags, which were
722  * generated based on upgrade missions, instead of having resources added when the
723  * map changes
724  */
725  // make sure the array is empty
726  fuels.clear();
727  for( const itype_id &fuel_id : fuel_types ) {
728  basecamp_fuel bcp_f;
729  bcp_f.ammo_id = fuel_id;
730  fuels.emplace_back( bcp_f );

734 
735  for( const tripoint_abs_ms &abs_ms_pt : src_set ) {
736  const tripoint &pt = target_map.getlocal( abs_ms_pt );
737  if( target_map.accessible_items( pt ) ) {
738  for( const item &i : target_map.i_at( pt ) ) {
739  for( basecamp_fuel &bcp_f : fuels ) {
740  if( bcp_f.ammo_id == i.typeId() ) {
741  bcp_f.available += i.charges;
742  break;
743  }
744  }
745  }
746  }
747 
748  }
749  for( basecamp_resource &bcp_r : resources ) {
750  bcp_r.consumed = 0;
751  item camp_item( bcp_r.fake_id, calendar::turn_zero );
752  camp_item.set_flag( STATIC( flag_id( "PSEUDO" ) ) );
753  if( !bcp_r.ammo_id.is_null() ) {
754  for( basecamp_fuel &bcp_f : fuels ) {
755  if( bcp_f.ammo_id == bcp_r.ammo_id ) {
756  if( bcp_f.available > 0 ) {
757  bcp_r.available = bcp_f.available;
758  camp_item = camp_item.ammo_set( bcp_f.ammo_id, bcp_f.available );
759  }
760  break;
761  }
762  }
763  }
764  _inv.add_item( camp_item );
765  }

768  // but we're already that with the resources above. The resources are stored in expansions
769  // rather than in a common pool to allow them to apply only to their respective expansion
770  // in the future.
