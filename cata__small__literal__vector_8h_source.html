149  void reserve_exact( size_t new_capacity ) {

151  if( capacity_ <= kInlineCount ) {

154  }

211  // The whole small_literal_vector has alignas(T*) so that the alignment requirements of heap_
212  // are never violated, even if you allocate several of these in a row. However we use pragma pack
213  // So there are no padding bytes inserted into the union to bring it to a multiple of sizeof(T*)
214  // so that capacity_ and len_ can be tightly packed adjacent to it. Their alignments will still
215  // be legal because the compiler will insert padding before capacity_ if necessary.
216  // We put T *heap_ as the first entry in the union to avoid default initializing the whole inline storage.
217 #pragma pack(push, 1)
218  union alignas( T ) {
219  T *heap_;
220  // NOLINTNEXTLINE(modernize-avoid-c-arrays)
221  T inline_[ kInlineCount ];
222  };
223 #pragma pack(pop)
224  // capacity_ > kInlineCount => on heap.
225  SizeT capacity_;
226  SizeT len_;
