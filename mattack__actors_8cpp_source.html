528  // Recalculate the ray each step
529  // We can't depend on either the target position being constant (obviously),
530  // but neither on z pos staying constant, because we may want to shift the map mid-pull
531  const units::angle dir = coord_to_angle( target->pos(), z.pos() );
532  tileray tdir( dir );
533  tdir.advance();
534  pt.x = target->posx() + tdir.dx();
535  pt.y = target->posy() + tdir.dy();
536  //Cancel the grab if the space is occupied by something
537  if( !g->is_empty( pt ) ) {
538  break;
539  }
540 
541  if( foe != nullptr ) {
542  if( foe->in_vehicle ) {
543  here.unboard_vehicle( foe->pos() );
544  }
545 
546  if( foe->is_avatar() && ( pt.x < HALF_MAPSIZE_X || pt.y < HALF_MAPSIZE_Y ||
547  pt.x >= HALF_MAPSIZE_X + SEEX || pt.y >= HALF_MAPSIZE_Y + SEEY ) ) {
548  g->update_map( pt.x, pt.y );
549  }
550  }
551 
552  target->setpos( pt );
553  pull_range--;
554  if( animate ) {
555  g->invalidate_main_ui_adaptor();
556  inp_mngr.pump_events();
557  ui_manager::redraw_invalidated();
558  refresh_display();
559  }
560  }
561  // If we're in a vehicle after being dragged, board us onto it
562  // This prevents us from being run over by our own vehicle if we're dragged out of it
563  if( foe != nullptr && !foe->in_vehicle &&
564  here.veh_at( pt ).part_with_feature( VPFLAG_BOARDABLE, true ) ) {
565  here.board_vehicle( pt, foe );
566  }
567  // The monster might drag a target that's not on it's z level
568  // So if they leave them on open air, make them fall
569  here.creature_on_trap( *target );
570 
571  target->add_msg_player_or_npc( msg_type, grab_data.pull_msg_u, grab_data.pull_msg_npc, mon_name,
572  target->disp_name() );
573  } else if( grab_data.pull_chance > -1 ) {
574  // We failed the pull chance roll, return false to select a different attack
575  add_msg_debug( debugmode::DF_MATTACK, "Pull chance roll failed.", grab_data.pull_chance );
576  return -1;
577  }
578 
579  if( grab_data.grab_effect != effect_null ) {
580  if( foe ) {
581  z.add_grab( bp_id.id() );
582  add_msg_debug( debugmode::DF_MATTACK, "Added grabbing on %s",
583  bp_id->name );
584  // Add grabbed - permanent, removal handled in try_remove_grab on move/wait
585  target->add_effect( grab_data.grab_effect, 1_days, bp_id, true, eff_grab_strength );
586  } else {
587  // Monsters don't have limb scores, no need to target limbs
588  target->add_effect( grab_data.grab_effect, 1_days, body_part_bp_null, true, eff_grab_strength );
589  z.add_effect( effect_grabbing, 1_days, true, 1 );
590  }
591  }
592 
593  // Drag stuff
594  if( grab_data.drag_distance > 0 ) {
595  int distance = grab_data.drag_distance;
596  while( distance > 0 ) {
597  // Start with the opposite square
598  tripoint opposite_square = z.pos() - ( target->pos() - z.pos() );
599  // Keep track of our neighbors (no leaping)
600  std::set<tripoint> neighbors;
601  for( const tripoint &trp : here.points_in_radius( z.pos(), 1 ) ) {
602  if( trp != z.pos() && trp != target->pos() ) {
603  neighbors.insert( trp );
604  }
605  }
606  // Check where we get to consider dragging
607  std::set<tripoint> candidates;
608  for( const tripoint &trp : here.points_in_radius( opposite_square,
609  grab_data.drag_deviation ) ) {
610  if( trp != z.pos() && trp != target->pos() ) {
611  candidates.insert( trp );
612  }
613  }
614  // Select a random square from the options
615  std::set<tripoint> intersect;
616  std::set_intersection( neighbors.begin(), neighbors.end(), candidates.begin(), candidates.end(),
617  std::inserter( intersect, intersect.begin() ) );
618  std::set<tripoint>::iterator intersect_iter = intersect.begin();
619  std::advance( intersect_iter, rng( 0, intersect.size() - 1 ) );
620  tripoint target_square = random_entry<std::set<tripoint>>( intersect );
621  if( z.can_move_to( target_square ) ) {
622  monster *zz = target->as_monster();
623  tripoint zpt = z.pos();
624  z.move_to( target_square, false, false, grab_data.drag_movecost_mod );
625  if( !g->is_empty( zpt ) ) { //Cancel the grab if the space is occupied by something
626  return 0;
627  }
628  if( target->is_avatar() && ( zpt.x < HALF_MAPSIZE_X ||
629  zpt.y < HALF_MAPSIZE_Y ||
630  zpt.x >= HALF_MAPSIZE_X + SEEX || zpt.y >= HALF_MAPSIZE_Y + SEEY ) ) {
631  g->update_map( zpt.x, zpt.y );
632  }
633  if( foe != nullptr ) {
634  if( foe->in_vehicle ) {
635  here.unboard_vehicle( foe->pos() );
636  }
637  foe->setpos( zpt );
638  if( !foe->in_vehicle && here.veh_at( zpt ).part_with_feature( VPFLAG_BOARDABLE, true ) ) {
639  here.board_vehicle( zpt, foe );
640  }
641  } else {
642  zz->setpos( zpt );
643  }
644  target->add_msg_player_or_npc( m_bad, _( "You are dragged behind the %s!" ),
645  _( "<npcname> gets dragged behind the %s!" ), z.name() );
646  if( animate ) {
647  g->invalidate_main_ui_adaptor();
648  inp_mngr.pump_events();
649  ui_manager::redraw_invalidated();
650  refresh_display();
651  }
652  } else {
653  target->add_msg_player_or_npc( m_good, _( "You resist the %s as it tries to drag you!" ),
654  _( "<npcname> resist the %s as it tries to drag them!" ), z.name() );
655  return 0;
656  }
657  distance--;
658  if( grab_data.drag_grab_break_distance > 0 && foe ) {
659  // Attempt to break the drag if we stepped the appropriate amount of distance
660  if( ( grab_data.drag_distance - distance ) % grab_data.drag_grab_break_distance == 0 ) {
661  if( foe->try_remove_grab() ) {
662  return 1;
663  } else {
664  target->set_moves( 0 );
665  }
666  }
667  }
668  }
669  }
670  return 1;
671 }

674 {
675  Creature *target = find_target( z );
676  if( target == nullptr ) {
677  return false;
678  }
679 
680  if( has_condition ) {
681  dialogue d( get_talker_for( &z ), get_talker_for( *target ) );
682  if( !condition( d ) ) {
683  add_msg_debug( debugmode::DF_MATTACK, "Attack conditionals failed" );
684  return false;
685  }
686  }
687 
688  z.mod_moves( -move_cost );
689 
690  const std::string mon_name = get_player_character().sees( z.pos() ) ?
691  z.disp_name( false, true ) : _( "Something" );
692 
693  // Add always-applied self effects
694  for( const mon_effect_data &eff : self_effects_always ) {
695  if( x_in_y( eff.chance, 100 ) ) {
696  z.add_effect( eff.id, time_duration::from_turns( rng( eff.duration.first, eff.duration.second ) ),
697  eff.permanent,
698  rng( eff.intensity.first, eff.intensity.second ) );
699  target->add_msg_if_player( m_mixed, eff.message, mon_name );
700  }
701  }
702 
703  // Dodge check
704 
705  const int acc = accuracy >= 0 ? accuracy : z.type->melee_skill;
706  int hitspread = target->deal_melee_attack( &z, dice( acc, 10 ) );
707 
708  // Pick body part
709  bodypart_str_id bp_hit = body_parts.empty() ?
710  target->select_body_part( hitsize_min, hitsize_max, attack_upper, hitspread ).id() :
711  *body_parts.pick();
712 
713  bodypart_id bp_id = bodypart_id( bp_hit );
714  game_message_type msg_type = target->is_avatar() ? m_warning : m_info;
715 
716  add_msg_debug( debugmode::DF_MATTACK, "Accuracy %d, hitspread %d, dodgeable %s", acc, hitspread,
717  dodgeable ? "true" : "false" );
718 
719  if( z.has_flag( mon_flag_HIT_AND_RUN ) ) {
720  z.add_effect( effect_run, 4_turns );
721  }
722 
723  if( uncanny_dodgeable && target->uncanny_dodge() ) {
724  game_message_type msg_type = target->is_avatar() ? m_warning : m_info;
725  sfx::play_variant_sound( "mon_bite", "bite_miss", sfx::get_heard_volume( z.pos() ),
726  sfx::get_heard_angle( z.pos() ) );
727  target->add_msg_player_or_npc( msg_type, miss_msg_u,
728  get_option<bool>( "LOG_MONSTER_ATTACK_MONSTER" ) ? miss_msg_npc : translation(),
729  z.name(), body_part_name_accusative( bp_id ) );
730  return true;
731  }
732 
733  if( dodgeable ) {
734  if( hitspread < 0 ) {
735  sfx::play_variant_sound( "mon_bite", "bite_miss", sfx::get_heard_volume( z.pos() ),
736  sfx::get_heard_angle( z.pos() ) );
737  target->add_msg_player_or_npc( msg_type, miss_msg_u,
738  get_option<bool>( "LOG_MONSTER_ATTACK_MONSTER" ) ? miss_msg_npc : translation(),
739  mon_name, body_part_name_accusative( bp_id ) );
740  return true;
741  }
742  }
743 
744  // We need to do some calculations in the main function - we might mutate bp_hit
745  // But first we need to handle exclusive grabs etc.
746  std::optional<bodypart_id> grabbed_bp_id;
747  if( is_grab ) {
748  int eff_grab_strength = grab_data.grab_strength == -1 ? z.get_grab_strength() :
749  grab_data.grab_strength;
750  add_msg_debug( debugmode::DF_MATTACK,
751  "Grab attack targeting bp %s, grab strength %d, pull chance %d", bp_id->name,
752  eff_grab_strength, grab_data.pull_chance );
753 
754  // If we care iterate through grabs one by one, fail if we can't break one
755  if( target->has_effect_with_flag( json_flag_GRAB ) && grab_data.exclusive_grab ) {
756  add_msg_debug( debugmode::DF_MATTACK, "Exclusive grab, begin filtering" );
757  map &here = get_map();
758  const tripoint_range<tripoint> &surrounding = here.points_in_radius( target->pos(), 1, 0 );
759  creature_tracker &creatures = get_creature_tracker();
760 
761  for( const effect &eff : target->get_effects_with_flag( json_flag_GRAB ) ) {
762  monster *grabber = nullptr;
763  // Iterate through the target's surroundings to find the grabber of this grab
764  for( const tripoint loc : surrounding ) {
765  monster *mon = creatures.creature_at<monster>( loc );
766  if( mon && mon->has_effect_with_flag( json_flag_GRAB_FILTER ) && mon->attack_target() == target ) {
767  if( target->is_monster() || ( !target->is_monster() &&
768  mon->is_grabbing( eff.get_bp().id() ) ) ) {
769  grabber = mon;
770  break;
771  }
772  }
773  }
774  // Ignore our own grab
775  if( grabber == &z ) {
776  add_msg_debug( debugmode::DF_MATTACK, "Grabber %s is the attacker, continue grab filtering",
777  grabber->name() );
778  continue;
779  }
780  // Roll to remove anybody else's grab on our target
781  if( !x_in_y( eff_grab_strength / 2.0f, eff.get_intensity() ) ) {
782  target->add_msg_player_or_npc( msg_type,
783  _( "%1s tries to drag you, but something holds you in place!" ),
784  _( "%1s tries to drag <npcname>, but something holds them in place!" ),
785  z.disp_name( false, true ) );
786  return true;
787 
788  } else {
789  target->remove_effect( eff.get_id(), eff.get_bp() );
790  }
791  }
792  }
793 
794  // No second grabs on monsters (exclusive grabs got rolled before this)
795  if( target->is_monster() && target->has_effect_with_flag( json_flag_GRAB ) ) {
796  add_msg_debug( debugmode::DF_MATTACK, "Target monster already grabbed, attack failed silently" );
797  return false;
798  }
799 
800  // Filter out already-grabbed bodyparts (after attempting to remove them, in any case)
801  // Unless we're not actually trying to grab
802  if( target->has_effect_with_flag( json_flag_GRAB, bp_id ) &&
803  grab_data.grab_effect != effect_null ) {
804  // Iterate through eligable targets to look for a non-grabbed one, fail if none are found
805  add_msg_debug( debugmode::DF_MATTACK, "Target bodypart %s already grabbed",
806  bp_id->name );
807  for( const bodypart_id bp : target->get_all_eligable_parts( hitsize_min, hitsize_max,
808  attack_upper ) ) {
809  if( target->has_effect_with_flag( json_flag_GRAB, bp ) ) {
810  add_msg_debug( debugmode::DF_MATTACK, "Alternative target bodypart %s already grabbed",
811  bp->name );
812  continue;
813  } else {
814  bp_hit = bp.id();
815  bp_id = bp;
816  add_msg_debug( debugmode::DF_MATTACK, "Retargeted to ungrabbed %s",
817  bp->name );
818  }
819  }
820  // Couldn't find an eligable limb, fail loudly
821  if( target->has_effect_with_flag( json_flag_GRAB, bp_id ) ) {
822  target->add_msg_player_or_npc( msg_type, miss_msg_u, miss_msg_npc, mon_name,
823  body_part_name_accusative( bp_id ) );
824  return true;
825  }
826  }
827  int result = do_grab( z, target, bp_id );
828  if( result < 0 ) {
829  return false;
830  } else if( result == 0 ) {
831  return true;
832  }
833  grabbed_bp_id = bp_id;
834  }
835 
836  // Damage instance calculation
837  damage_instance damage = damage_max_instance;
838  double multiplier = rng_float( min_mul, max_mul );
839  add_msg_debug( debugmode::DF_MATTACK, "Damage multiplier %.1f (range %.1f - %.1f)", multiplier,
840  min_mul, max_mul );
841  damage.mult_damage( multiplier );
842 
843  // Block our hit
844  if( blockable ) {
845  target->block_hit( &z, bp_id, damage );
846  }
847 
848  // Take damage
849  dealt_damage_instance dealt_damage = target->deal_damage( &z, bp_id, damage );
850  dealt_damage.bp_hit = bp_id;
851 
852  // On hit effects
853  target->on_hit( &z, bp_id );
854 
855  // Apply onhit self effects
856  for( const mon_effect_data &eff : self_effects_onhit ) {
857  if( x_in_y( eff.chance, 100 ) ) {
858  z.add_effect( eff.id, time_duration::from_turns( rng( eff.duration.first, eff.duration.second ) ),
859  eff.permanent,
860  rng( eff.intensity.first, eff.intensity.second ) );
861  target->add_msg_if_player( msg_type, eff.message, mon_name );
862  }
863  }
864  int damage_total = dealt_damage.total_damage();
865  add_msg_debug( debugmode::DF_MATTACK, "%s's melee_attack did %d damage", z.name(), damage_total );
866  if( damage_total > 0 ) {
867  on_damage( z, *target, dealt_damage );
868  } else {
869  sfx::play_variant_sound( "mon_bite", "bite_miss", sfx::get_heard_volume( z.pos() ),
870  sfx::get_heard_angle( z.pos() ) );
871  target->add_msg_player_or_npc( msg_type, no_dmg_msg_u,
872  get_option<bool>( "LOG_MONSTER_ATTACK_MONSTER" ) ? no_dmg_msg_npc : translation(),
873  mon_name, body_part_name_accusative( grabbed_bp_id.value_or( bp_id ) ) );
874  if( !effects_require_dmg ) {
875  for( const mon_effect_data &eff : effects ) {
876  if( x_in_y( eff.chance, 100 ) ) {
877  const bodypart_id affected_bp = eff.affect_hit_bp ? bp_id : eff.bp.id();
878  if( !( effects_require_organic && affected_bp->has_flag( json_flag_BIONIC_LIMB ) ) ) {
879  target->add_effect( eff.id, time_duration::from_turns( rng( eff.duration.first,
880  eff.duration.second ) ), affected_bp, eff.permanent, rng( eff.intensity.first,
881  eff.intensity.second ) );
882  }
883  }
884  }
885  }
886  }
887  if( throw_strength > 0 ) {
888  if( g->fling_creature( target, coord_to_angle( z.pos(), target->pos() ),
889  throw_strength ) ) {
890  target->add_msg_player_or_npc( msg_type, throw_msg_u,
891  get_option<bool>( "LOG_MONSTER_ATTACK_MONSTER" ) ? throw_msg_npc : translation(),
892  mon_name );
893 
894  // Items strapped to you may fall off as you hit the ground
895  // when you break out of a grab you have a chance to lose some things from your pockets
896  // that are hanging off your character
897  if( target->is_avatar() ) {
898  std::vector<item_pocket *> pd = target->as_character()->worn.grab_drop_pockets();
899  // if we have items that can be pulled off
900  if( !pd.empty() ) {
901  // choose an item to be ripped off
902  int index = rng( 0, pd.size() - 1 );
903  // the chance the monster knocks an item off
904  int chance = rng( 0, z.type->melee_sides * z.type->melee_dice );
905  // the chance the pocket resists
906  int sturdiness = rng( 0, pd[index]->get_pocket_data()->ripoff );
907  // the item is ripped off your character
908  if( sturdiness < chance ) {
909  float path_distance = rng_float( 0, 1.0 );
910  tripoint vector = target->pos() - z.pos();
911  vector = tripoint( vector.x * path_distance, vector.y * path_distance, vector.z * path_distance );
912  pd[index]->spill_contents( z.pos() + vector );
913  add_msg( m_bad, _( "As you hit the ground something comes loose and is knocked away from you!" ) );
914  popup( _( "As you hit the ground something comes loose and is knocked away from you!" ) );
915  }
916  }
917  }
918  }
919  }
920  return true;
