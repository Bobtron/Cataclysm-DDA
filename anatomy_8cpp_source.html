264 bodypart_id anatomy::select_blocking_part( const Creature *blocker, bool arm, bool leg,
265  bool nonstandard ) const
266 {
267  weighted_float_list<bodypart_id> block_scores;
268  for( const bodypart_id &bp : cached_bps ) {
269  float block_score = bp->get_limb_score( limb_score_block );
270  if( const Character *u = dynamic_cast<const Character *>( blocker ) ) {
271  block_score = u->get_part( bp )->get_limb_score( limb_score_block );
272  // Weigh shielded bodyparts higher
273  block_score *= u->worn_with_flag( flag_BLOCK_WHILE_WORN, bp ) ? 5 : 1;
274  }
275 
276  // Filter out nonblocking / broken limbs
277  if( block_score == 0 ) {
278  add_msg_debug( debugmode::DF_MELEE, "BP %s discarded, no blocking score",
279  body_part_name( bp ) );
280  continue;
281  }
282 
283  // Always blocking limbs block always (if they have a block score)
284  if( bp->has_flag( json_flag_ALWAYS_BLOCK ) ) {
285  block_scores.add( bp, block_score );
286  add_msg_debug( debugmode::DF_MELEE, "BP %s always blocks",
287  body_part_name( bp ) );
288  continue;
289  }
290 
291  // Can we block with our normal boring arm?
292  if( bp->has_type( body_part_type::type::arm ) && !bp->has_flag( json_flag_NONSTANDARD_BLOCK ) &&
293  !arm ) {
294  add_msg_debug( debugmode::DF_MELEE, "BP %s discarded, no arm blocks allowed",
295  body_part_name( bp ) );
296  continue;
297  // Can we block with our normal boring legs?
298  } else if( bp->has_type( body_part_type::type::leg ) &&
299  !bp->has_flag( json_flag_NONSTANDARD_BLOCK ) && !leg ) {
300  add_msg_debug( debugmode::DF_MELEE, "BP %s discarded, no leg blocks allowed",
301  body_part_name( bp ) );
302  continue;
303  // Can we block with our non-normal non-arms/non-legs?
304  } else if( ( ( !bp->has_type( body_part_type::type::arm ) &&
305  !bp->has_type( body_part_type::type::leg ) ) || bp->has_flag( json_flag_NONSTANDARD_BLOCK ) ) &&
306  !nonstandard ) {
307  add_msg_debug( debugmode::DF_MELEE, "BP %s discarded, no nonstandard blocks allowed",
308  body_part_name( bp ) );
309  continue;
310  }
311 
312  block_scores.add( bp, block_score );
313  }
314

317  add_msg_debug( debugmode::DF_MELEE, "%s = %.3f", pr.obj.obj().name, pr.weight );
318  }
319 
320  const bodypart_id *ret = block_scores.pick();

353  const double range_max, const double value ) const
354 {

356
