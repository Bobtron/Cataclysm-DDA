168 void gates::open_gate( const tripoint_bub_ms &pos )
169 {
170  const gate_id gid = get_gate_id( pos );
171 
172  if( !gates_data.is_valid( gid ) ) {
173  return;
174  }
175 
176  const gate_data &gate = gates_data.obj( gid );
177 
178  bool open = false;
179  bool close = false;
180  bool fail = false;
181 
182  map &here = get_map();
183  for( const point &wall_offset : four_adjacent_offsets ) {
184  const tripoint_bub_ms wall_pos = pos + wall_offset;
185 
186  if( !gate.is_suitable_wall( wall_pos ) ) {
187  continue;
188  }
189 
190  for( const point &gate_offset : four_adjacent_offsets ) {
191  const tripoint_bub_ms gate_pos = wall_pos + gate_offset;
192 
193  if( gate_pos == pos ) {
194  continue; // Never comes back
195  }
196 
197  if( !open ) { // Closing the gate...
198  tripoint_bub_ms cur_pos = gate_pos;
199  while( here.ter( cur_pos ) == gate.floor.id() ) {
200  fail = !g->forced_door_closing( cur_pos, gate.door.id(), gate.bash_dmg ) || fail;
201  close = !fail;
202  cur_pos += gate_offset;
203  }
204  }
205 
206  if( !close ) { // Opening the gate...
207  tripoint_bub_ms cur_pos = gate_pos;
208  while( true ) {
209  const ter_id ter = here.ter( cur_pos );
210 
211  if( ter == gate.door.id() ) {
212  here.ter_set( cur_pos, gate.floor.id() );
213  open = !fail;
214  } else if( ter != gate.floor.id() ) {
215  break;
216  }
217  cur_pos += gate_offset;
218  }
219  }
220  }
221  }
222 
223  if( get_player_view().sees( pos ) ) {
224  if( open ) {
225  add_msg( gate.open_message );
226  } else if( close ) {
227  add_msg( gate.close_message );
228  } else if( fail ) {
229  add_msg( gate.fail_message );
230  } else {
231  add_msg( _( "Nothing happens." ) );
232  }
233  }
