154 void material_type::check() const
155 {
156  if( name().empty() ) {
157  debugmsg( "material %s has no name.", id.c_str() );
158  }
159  if( _dmg_adj.size() < 4 ) {
160  debugmsg( "material %s specifies insufficient damaged adjectives.", id.c_str() );
161  }
162  if( _salvaged_into && ( !item::type_is_defined( *_salvaged_into ) || _salvaged_into->is_null() ) ) {
163  debugmsg( "invalid \"salvaged_into\" %s for %s.", _salvaged_into->c_str(), id.c_str() );
164  }
165  if( !item::type_is_defined( _repaired_with ) ) {
166  debugmsg( "invalid \"repaired_with\" %s for %s.", _repaired_with.c_str(), id.c_str() );
167  }
168 
169  if( _wind_resist && ( *_wind_resist > 100 || *_wind_resist < 0 ) ) {
170  debugmsg( "Wind resistance outside of range (100%% to 0%%, is %d%%) for %s.", *_wind_resist,
171  id.str() );
172  }
173 
174  if( _repair_difficulty && ( _repair_difficulty > 10 || _repair_difficulty < 0 ) ) {
175  debugmsg( "Repair difficulty out of skill range (0 to 10, is %d) for %s.", _repair_difficulty,
176  id.str() );
177  }
178 
179  for( const auto &dt : _resistances.resist_vals ) {
180  if( !dt.first.is_valid() ) {
181  debugmsg( "Invalid resistance type \"%s\" for material %s", dt.first.c_str(), id.c_str() );
182  }
183  }
184 
185  for( const damage_type &dt : damage_type::get_all() ) {
186  bool type_defined =
187  std::find( _res_was_loaded.begin(), _res_was_loaded.end(), dt.id ) != _res_was_loaded.end();
188  if( dt.material_required && !type_defined ) {
189  debugmsg( "material %s is missing required resistance for \"%s\"", id.c_str(), dt.id.c_str() );
190  }
191  }
