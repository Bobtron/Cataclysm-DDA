329 food_summary stomach_contents::digest( const Character &owner, const needs_rates &metabolic_rates,
330  int five_mins, int half_hours )
331 {
332  food_summary digested;
333  stomach_digest_rates rates = get_digest_rates( metabolic_rates, owner );
334 
335  // Digest water, but no more than in stomach.
336  digested.water = std::min( water, rates.water * five_mins );
337  water -= digested.water;
338 
339  // If no half-hour intervals have passed, we only process water, so bail out early.
340  if( half_hours == 0 ) {
341  return digested;
342  }
343 
344  // Digest solids, but no more than in stomach.
345  digested.solids = std::min( contents, rates.solids * half_hours );
346  contents -= digested.solids;
347 
348  // Digest kCal -- use min_kcal by default, but no more than what's in stomach,
349  // and no less than percentage_kcal of what's in stomach.
350  int kcal_fraction = std::lround( nutr.kcal() * rates.percent_kcal );
351  digested.nutr.calories = half_hours * clamp<int64_t>( rates.min_calories, kcal_fraction * 1000,
352  nutr.calories );
353 
354  // Digest vitamins just like we did kCal, but we need to do one at a time.
355  for( const std::pair<const vitamin_id, int> &vit : nutr.vitamins() ) {
356  if( vit.first->type() != vitamin_type::DRUG ) {
357  int vit_fraction = std::lround( vit.second * rates.percent_vitamin );
358  digested.nutr.set_vitamin( vit.first, half_hours * clamp( rates.min_vitamin, vit_fraction,
359  vit.second ) );
360  }
361  // drug vitamins are absorbed to the blood instantly after the first stomach step.
362  // this makes the drug vitamins easier to balance (no need to account for slow trickle-ing in of the drug)
363  else if( vit.first->type() == vitamin_type::DRUG && stomach ) {
364  digested.nutr.set_vitamin( vit.first, vit.second );
365  }
366  }
367 
368  nutr -= digested.nutr;
369  return digested;
370 }

379 stomach_digest_rates stomach_contents::get_digest_rates( const needs_rates &metabolic_rates,
380  const Character &owner ) const
381 {
382  stomach_digest_rates rates;
383  if( stomach ) {
384  // The stomach is focused on passing material on to the guts.
385  // 3 hours to empty in 30 minute increments
386  rates.solids = capacity( owner ) / 6;
387  rates.water = 250_ml; // Water is special, passes very quickly, in 5 minute intervals
388  rates.min_vitamin = 1;
389  rates.percent_vitamin = 1.0f / 6.0f;
390  rates.min_calories = 5000;
391  rates.percent_kcal = 1.0f / 6.0f;
392  } else {
393  // The guts are focused on absorption into the body, we don't care about passing rates.
394  // Solids rate doesn't do anything impactful here so just make it big enough to avoid overflow.
395  rates.solids = 250_ml;
396  rates.water = 250_ml;
397  // Explicitly floor it, because casting it to an int will do so anyways
398  rates.min_calories = std::floor( metabolic_rates.kcal / 24.0 * metabolic_rates.hunger * 1000 );
399  rates.percent_kcal = 0.05f * metabolic_rates.hunger;
400  rates.min_vitamin = std::round( 100.0 / 24.0 * metabolic_rates.hunger );
401  rates.percent_vitamin = 0.05f * metabolic_rates.hunger;
402  }
403  return rates;
404 }
