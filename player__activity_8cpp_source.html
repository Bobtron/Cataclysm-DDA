262  int consume_moves = get_auto_consume_moves( you, false );

264  moves_total += consume_moves;

267  }
268  }

272  if( type->based_on() == based_on_type::TIME ) {
273  if( moves_left >= 100 ) {
274  moves_left -= 100 * activity_mult;
275  you.set_moves( 0 );
276  } else {
277  you.mod_moves( -you.get_moves() * moves_left / 100 );
278  moves_left = 0;
279  }
280  } else if( type->based_on() == based_on_type::SPEED ) {

283  you.set_moves( 0 );

287  }

292  // or just now reloaded back into it, and their activity target might
293  // be still unloaded, can cause infinite loops.
294  set_to_null();
295  you.drop_invalid_inventory();
296  return;
297  }

299  you.set_activity_level( exertion_level() );

302  // if we have an EOC defined in json do that

304  if( type->do_turn_EOC->type == eoc_type::ACTIVATION ) {
305  type->do_turn_EOC->activate( d );

308  }

310  if( type.is_null() ) {
311  return;
312  }
313  }
314 
315  // This might finish the activity (set it to null)
316  if( actor ) {
317  actor->do_turn( *this, you );

321  }

324  // adjusted stamina because
325  // autotravel doesn't reduce stamina after do_turn()
326  // it just sets a destination, clears the activity, then moves afterwards

339  you.add_msg_if_player( _( "You pause for a moment to catch your breath." ) );
340  }
341 
342  auto_resume = true;
343  player_activity new_act( ACT_WAIT_STAMINA, to_moves<int>( 5_minutes ) );
344  new_act.values.push_back( you.get_stamina_max() );

346  uilist tired_query;

349  tired_query.addentry( 2, true, 'm', _( "Maybe later." ) );
350  tired_query.addentry( 3, true, 'f', _( "Finish it." ) );

364  }
365  }

370  }

374  }

378  // type is ACT_NULL below.

380  // if we have an EOC defined in json do that
381  dialogue d( get_talker_for( you ), nullptr );
382  if( type->completion_EOC->type == eoc_type::ACTIVATION ) {
383  type->completion_EOC->activate( d );

386  }

391  } else {

393  // "Finish" is never a misnomer for any activity without a finish function
394  set_to_null();
395  }
396  }
397  }

399  // Make sure data of previous activity is cleared
400  you.activity = player_activity();
401  you.resume_backlog_activity();
