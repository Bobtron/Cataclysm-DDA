359  float wrap_width, bool *is_selected, bool *is_focused, bool *is_hovered )
360 {
361  ImGui::PushID( text.c_str() );
362  ImGuiID itemId = GImGui->CurrentWindow->IDStack.back();
363  std::stack<nc_color> color_stack;
364  color_stack.push( color );
365  size_t chars_per_line = size_t( wrap_width );
366  if( chars_per_line == 0 ) {
367  chars_per_line = SIZE_MAX;
368  }
369 #if defined(WIN32) || defined(TILES)
370  size_t char_width = size_t( ImGui::CalcTextSize( " " ).x );
371  chars_per_line /= char_width;
372 #endif
373  std::vector<std::string> folded_msg = foldstring( text, chars_per_line );
374 
375  for( const std::string &line : folded_msg ) {
376 
377  const auto color_segments = split_by_color( line );
378  if( is_selected != nullptr ) {
379  ImGui::Selectable( "", is_selected );
380  ImGui::SameLine( 0, 0 );
381  }
382  int i = 0;
383  for( auto seg : color_segments ) {
384  if( seg.empty() ) {
385  continue;
386  }
387 
388  if( seg[0] == '<' ) {
389  const color_tag_parse_result::tag_type type =
390  update_color_stack( color_stack, seg, report_color_error::yes );
391  if( type != color_tag_parse_result::non_color_tag ) {
392  seg = rm_prefix( seg );
393  }
394  }
395 
396  color = color_stack.empty() ? color : color_stack.top();
397  if( i++ != 0 ) {
398  ImGui::SameLine( 0, 0 );
399  }
400 #if !(defined(TILES) || defined(WIN32))
401  int pair_id = color.get_index();

405  if( color.is_bold() ) {

407  }
408  RGBTuple &rgbCol = rgbPalette[palette_index];
409  ImGui::TextColored( { static_cast<float>( rgbCol.Red / 255. ), static_cast<float>( rgbCol.Green / 255. ),

565  } else if( cached_bounds.x >= 0 && cached_bounds.y >= 0 ) {
566  ImGui::SetNextWindowPos( { cached_bounds.x, cached_bounds.y } );
567  }
568  if( cached_bounds.h > 0 || cached_bounds.w > 0 ) {

570  }

572  draw_controls();
