1030 bool Character::roll_bad_mutation( const mutation_category_id &categ ) const
1031 {
1032  // We will never have worse odds than this no matter our instability
1033  float MAX_BAD_CHANCE = 0.67;
1034  // or, if we have Robust, cap it lower.
1035 
1036  bool ret = false;
1037 
1038  // The following values are, respectively, the total number of non-bad traits in a category and
1039  int muts_max = get_total_nonbad_in_category( categ );
1040  // how many good mutations we have in total. Mutations which don't belong to the tree we're mutating towards count double for this value. Starting traits don't count at all.
1041  int insta_actual = get_instability_per_category( categ );
1042 
1043  if( insta_actual == 0 ) {
1044  add_msg_debug( debugmode::DF_MUTATION, "No mutations yet, no bad mutations allowed" );
1045  return ret;
1046  } else {
1047  // When we have a total instability score equal to the number of non-bad mutations in the tree, our odds of good/bad are 50/50.
1048  float chance = 0.5 * static_cast<float>( insta_actual ) / static_cast<float>( muts_max );
1049  chance = std::min( chance, MAX_BAD_CHANCE );
1050  ret = rng_float( 0, 1 ) < chance;
1051  add_msg_debug( debugmode::DF_MUTATION,
1052  "%s is the instability category chosen, which has %d total good traits. Adjusted instability score for the category is %d, giving a chance of bad mut of %.3f.",
1053  categ.c_str(), muts_max, insta_actual, chance );
1054  return ret;
1055  }

1058 void Character::mutate( const int &true_random_chance, bool use_vitamins )
1059 {
1060  // Determine the highest mutation category
1061  mutation_category_id cat;
1062  weighted_int_list<mutation_category_id> cat_list = get_vitamin_weighted_categories();
1063 
1064  bool select_mutation = is_avatar() && get_option<bool>( "SHOW_MUTATION_SELECTOR" );
1065 
1066  bool allow_good = false;
1067  bool allow_bad = false;
1068  bool allow_neutral = true;
1069  bool try_opposite = true;
1070 
1071  if( select_mutation ) {
1072  // Mutation selector overrides good / bad mutation rolls
1073  allow_good = true;
1074  allow_bad = true;
1075  try_opposite = false;
1076  }
1077 
1078  add_msg_debug( debugmode::DF_MUTATION, "mutate: true_random_chance %d",
1079  true_random_chance );
1080 
1081  if( true_random_chance > 0 && one_in( true_random_chance ) ) {
1082  add_msg_debug( debugmode::DF_MUTATION, "mutate: true_random roll succeeded",
1083  true_random_chance );
1084  cat = mutation_category_ANY;
1085  allow_good = true; // because i'm WILD YEAH
1086  allow_bad = true;
1087  allow_neutral = true;
1088  try_opposite = false;
1089  } else if( cat_list.get_weight() > 0 ) {
1090  cat = *cat_list.pick();
1091  cat_list.add_or_replace( cat, 0 );
1092  add_msg_debug( debugmode::DF_MUTATION, "Picked category %s", cat.c_str() );
1093  // Only decide if it's good or bad after we pick the category.
1094  if( roll_bad_mutation( cat ) ) {
1095  // If we picked bad, mutation can be bad or neutral.
1096  allow_bad = true;
1097  } else {
1098  // Otherwise, can be good or neutral.
1099  allow_good = true;
1100  }
1101  } else {
1102  // This is fairly direct in explaining why it fails - hopefully it'll help folks to learn the system without needing to read docs
1103  add_msg_if_player( m_bad,
1104  _( "Your body tries to mutate, but it lacks a primer to do so and only contorts for a moment." ) );
1105  return;
1106  }
1107  // Genetic Downwards Spiral has special logic that makes every possible mutation negative
1108  // Positive mutations can still be gained as prerequisites to a negative, but every targeted mutation will be a negative one
1109  if( has_trait( trait_CHAOTIC_BAD ) ) {
1110  add_msg_debug( debugmode::DF_MUTATION, "mutate: Genetic Downward Spiral found, all bad traits" );
1111  allow_good = false;
1112  allow_bad = true;
1113  try_opposite = false;
1114  }
1115 
1116  std::vector<trait_id> valid; // Valid mutations
1117  std::vector<trait_id> dummies; // Dummy mutations
1118 
1119  do {
1120  // See if we should upgrade/extend an existing mutation...
1121  std::vector<trait_id> upgrades;
1122 
1123  // ... or remove one that is not in our highest category
1124  std::vector<trait_id> downgrades;
1125 
1126  const vitamin_id mut_vit = use_vitamins ? mutation_category_trait::get_category(
1127  cat ).vitamin : vitamin_id::NULL_ID();
1128 
1129  // For each mutation in category...
1130  for( const trait_id &traits_iter : mutations_category[cat] ) {
1131  const trait_id &base_mutation = traits_iter;
1132  const mutation_branch &base_mdata = traits_iter.obj();
1133 
1134  // ...those we don't have are valid.
1135  if( base_mdata.valid && is_category_allowed( base_mdata.category ) &&
1136  !has_trait( base_mutation ) && !base_mdata.dummy ) {
1137  add_msg_debug( debugmode::DF_MUTATION, "mutate: trait %s added to valid trait list",
1138  base_mdata.id.c_str() );
1139  valid.push_back( base_mdata.id );
1140  }
1141 
1142  // ...dummy traits are cached.
1143  if( base_mdata.dummy && is_category_allowed( base_mdata.category ) ) {
1144  add_msg_debug( debugmode::DF_MUTATION, "mutate: trait %s added to dummy trait list",
1145  base_mdata.id.c_str() );
1146  dummies.push_back( base_mdata.id );
1147  }
1148 
1149  // ...for those that we have...
1150  if( has_trait( base_mutation ) ) {
1151  // ...consider the mutations that replace it.
1152 
1153  for( const trait_id &mutation : base_mdata.replacements ) {
1154  if( mutation->valid &&
1155  mutation_ok( mutation, allow_good, allow_bad, allow_neutral, mut_vit ) &&
1156  mutation_is_in_category( mutation, cat ) ) {
1157  add_msg_debug( debugmode::DF_MUTATION,
1158  "mutate: trait %s found, replacement trait %s pushed to upgrades list", base_mutation.c_str(),
1159  mutation.c_str() );
1160  upgrades.push_back( mutation );
1161  }
1162  }
1163 
1164  // ...consider the mutations that add to it.
1165  for( const trait_id &mutation : base_mdata.additions ) {
1166  if( mutation->valid &&
1167  mutation_ok( mutation, allow_good, allow_bad, allow_neutral, mut_vit ) &&
1168  mutation_is_in_category( mutation, cat ) ) {
1169  add_msg_debug( debugmode::DF_MUTATION,
1170  "mutate: trait %s found, addition trait %s pushed to upgrades list",
1171  base_mutation.c_str(), mutation.c_str() );
1172  upgrades.push_back( mutation );
1173  }
1174  }
1175  }
1176  }
1177 
1178  // Remove anything we already have, that we have a child of, that
1179  // goes against our intention of a good/bad mutation, or that we lack resources for
1180  for( size_t i = 0; i < valid.size(); i++ ) {
1181  if( ( !mutation_ok( valid[i], allow_good, allow_bad, allow_neutral, mut_vit ) ) ||
1182  ( !valid[i]->valid ) ) {
1183  add_msg_debug( debugmode::DF_MUTATION, "mutate: trait %s removed from valid trait list",
