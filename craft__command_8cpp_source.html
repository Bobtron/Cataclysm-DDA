398 bool craft_command::safe_to_unload_comp( const item &it )
399 {
400  // Candle wax from candles should be consumed with the candle
401  if( it.is_container_empty() || it.typeId() == itype_candle ) {
402  return false;
403  }
404 
405  // Return true if item contains anything "real"
406  const std::function<bool( const item &i )> filter = []( const item & i ) {
407  return !i.has_flag( flag_ZERO_WEIGHT ) && !i.has_flag( flag_NO_DROP );
408  };
409  const bool valid = it.get_contents().has_any_with( filter, pocket_type::CONTAINER ) ||
410  it.get_contents().has_any_with( filter, pocket_type::MAGAZINE ) ||
411  it.get_contents().has_any_with( filter, pocket_type::MAGAZINE_WELL );
412  if( valid ) {
413  return true;
414  }
415 
416  // Don't try to unload items that shouldn't be outside their container
417  itype_id ammo = it.loaded_ammo().typeId();
418  if( ammo.is_null() ) {
419  return !it.empty_container();
420  } else if( ammo->has_flag( flag_ZERO_WEIGHT ) ||
421  ammo->has_flag( flag_NO_DROP ) ) {
422  return false;
423  }
424 
425  return true;
