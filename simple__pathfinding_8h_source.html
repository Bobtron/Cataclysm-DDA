73 /**
74  * Uses Greedy Best-First-Search to find a short path from source to destination [2D only].
75  * The search area is a rectangle with corners at (0,0) and max.
76  *
77  * @param source Starting point of path
78  * @param dest End point of path
79  * @param max Max permissible coordinates for a point on the path
80  * @param scorer function of (node &current, node *previous) that returns node_score.
81  */
82 template<typename Point, typename = std::enable_if_t<Point::dimension == 2>>
83 directed_path<Point> greedy_path( const Point &source, const Point &dest, const Point &max,
84  two_node_scoring_fn<Point> scorer )
85 {
86  directed_path<Point> res;
87  const two_node_scoring_fn<point> point_scorer
88  = [scorer]( directed_node<point> current, std::optional<directed_node<point>> prev ) {
89  std::optional<directed_node<Point>> prev_node;
90  if( prev ) {
91  prev_node = directed_node<Point>( Point( prev->pos ), prev->dir );
92  }
93  return scorer( directed_node<Point>( Point( current.pos ), current.dir ), prev_node );
94  };
95  directed_path<point> path = greedy_path( source.raw(), dest.raw(), max.raw(), point_scorer );
96  res.nodes.reserve( path.nodes.size() );
97  for( const auto &node : path.nodes ) {
98  res.nodes.emplace_back( Point( node.pos ), node.dir );
99  }
100  return res;
101 }
