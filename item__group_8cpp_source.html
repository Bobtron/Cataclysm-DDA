113  const std::optional<itype_id> &container_type, const std::optional<std::string> &container_variant,
114  const bool sealed,
115  time_point birthday, Item_spawn_data::overflow_behaviour on_overflow,
116  const std::string &context )
117 {
118  if( !container_type ) {
119  return;
120  }
121 
122  // Randomly permute the list of items so that when some don't fit it's
123  // not always the ones at the end which are rejected.

125  std::shuffle( items.end() - num_items, items.end(), rng_get_engine() );

128  if( container_variant ) {

134  if( ret.success() ) {

136  ctr.put_in( *it, pk_type );

139  ctr.force_insert_item( *it, pk_type );
140  } else {

143  debugmsg( "item %s could not be put in container %s when spawning item group %s: %s. "
144  "This can be resolved either by changing the container or contents "
145  "to ensure that they fit, or by specifying an overflow behaviour via "
146  "\"on_overflow\" on the item group.",
147  it->typeId().str(), container_type->str(), context, ret.str() );
148  break;
149  case Item_spawn_data::overflow_behaviour::spill:

152  case Item_spawn_data::overflow_behaviour::discard:
153  break;
154  case Item_spawn_data::overflow_behaviour::last:
155  debugmsg( "Invalid overflow_behaviour" );
156  break;
157  }
158  }

160  ctr.add_automatic_whitelist();
161  if( sealed ) {
162  ctr.seal();
163  }
164

166  items.reserve( items.size() + excess.size() + 1 );
