202 void multi_projectile_hit_message( Creature *critter, int hit_count, int damage_taken,
203  const std::string &projectile_name )
204 {
205  if( hit_count > 0 && get_player_character().sees( *critter ) ) {
206  // Building a phrase to summarize the fragment effects.

208  std::map<int, std::string> impact_count_descriptions = {
209  { 1, _( "a" ) }, { 2, _( "several" ) }, { 5, _( "many" ) },
210  { 20, _( "a large number of" ) }, { 100, _( "a huge number of" ) },
211  { std::numeric_limits<int>::max(), _( "an immense number of" ) }
212  };
213  std::string impact_count = std::find_if(
214  impact_count_descriptions.begin(), impact_count_descriptions.end(),
215  [hit_count]( const std::pair<int, std::string> &desc ) {
216  return desc.first >= hit_count;
217  } )->second;
218  std::string damage_description = ( damage_taken > 0 ) ?
219  string_format( _( "dealing %d damage" ), damage_taken ) :
220  _( "but they deal no damage" );
221  if( critter->is_avatar() ) {
222  //~ Phrase describing getting hit by multiple projectiles, i.e. "You are hit by many bomb fragments, dealing 50 damage."
223  add_msg( _( "You are hit by %s %s, %s." ), impact_count, projectile_name, damage_description );
224  } else if( critter->is_npc() ) {
225  //~ Phrase describing getting hit by multiple projectiles, i.e. "Fred Johnson is hit by many bomb fragments, dealing 50 damage."
226  critter->add_msg_if_npc( _( "<npcname> is hit by %s %s, %s." ), impact_count, projectile_name,
227  damage_description );
228  } else {
229  //~ Phrase describing getting hit by multiple projectiles, i.e. "The zombie is hit by many bomb fragments, dealing 50 damage."
230  add_msg( _( "%s is hit by %s %s, %s." ), critter->disp_name( false, true ), impact_count,
