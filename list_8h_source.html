369  void expand_capacity( const size_type new_capacity ) { // used by add_new and append
370  group_pointer_type const old_block = block_pointer;
371  block_pointer = LIST_ALLOCATE( group_allocator_type, group_allocator_pair, new_capacity, nullptr );
372 
373  if LIST_CONSTEXPR( std::is_trivially_copyable<node_pointer_type>::value &&
374  std::is_trivially_destructible<node_pointer_type>::value ) {
375  // Dereferencing here in order to deal with smart pointer situations i.e. obtaining the raw pointer from the smart pointer
376  // reinterpret_cast necessary to deal with GCC 8 warnings
377  std::memcpy( static_cast<void *>( &*block_pointer ), static_cast<void *>( &*old_block ),
378  sizeof( group ) * size );
379  } else if LIST_CONSTEXPR( std::is_move_constructible<node_pointer_type>::value ) {
380  std::uninitialized_copy( std::make_move_iterator( old_block ),
381  std::make_move_iterator( old_block + size ), block_pointer );

383  // If allocator supplies non-trivial pointers it becomes necessary to destroy the group. uninitialized_copy will not work in this context as the copy constructor for "group" is overridden in C++03/98. The = operator for "group" has been overridden to make the following work:

388  *( current_new_group++ ) = *current_group;
389

511  } else { // search for the node group which location_node is located within, using last_searched_group as a starting point and searching left and right. Try and find the closest node group with reusable erased-element locations along the way:

513  nullptr :

516  nullptr : last_searched_group;

524  last_searched_group = right;

526  }

530  if( closest_freelist_right != nullptr ) {
531  last_searched_group = right;
532  left_distance = right - closest_freelist_right;
533

535  return closest_freelist_right;
536  }
537 
538  freelist_group = closest_freelist_right;
539  } else {

541  left_distance = right - left;
542  }

545  const group_pointer_type end_group = ( ( right + left_distance ) > beyond_end_group ) ?
546  beyond_end_group : ( right + left_distance - 1 );

550  return right;

555  return freelist_group;
556  }

560  }

562  // location_node's group not found, but a reusable location found
563  if( right->free_list_head != nullptr ) {
564  if( ( closest_freelist_right == nullptr ) & ( closest_freelist_left == nullptr ) ) {
565  closest_freelist_left = right;
566  }
567 
568  closest_freelist_right = right;
569  }
570 
571  right_not_beyond_back = ( ++right < beyond_end_group );
572  }
573 
574  if( left_not_beyond_front ) {
575  if( ( location_node >= left->nodes ) && ( location_node < left->beyond_end ) ) {
576  if( left->free_list_head != nullptr ) {
577  last_searched_group = left;
578  return left;
579  }
580 
581  difference_type right_distance;
582

584  last_searched_group = left;
585  right_distance = closest_freelist_left - left;
586 
587  if( right_distance <= 2 ) {
588  return closest_freelist_left;
589  }

591  freelist_group = closest_freelist_left;
592  } else {
593  last_searched_group = left;
594  right_distance = right - left;
595  }
596 
597  // Otherwise find closest group with freelist:
598  const group_pointer_type end_group = ( ( left - right_distance ) < block_pointer ) ? block_pointer
599  - 1 : ( left - right_distance ) + 1;

608  return freelist_group;

617  closest_freelist_right = left;
618  }
619

621  }
