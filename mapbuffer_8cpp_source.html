151  // Submaps are generated in quads, so we know if we have one member of a quad,
152  // we have the rest of it, if that assumption is broken we have REAL problems.
153  const tripoint_abs_omt om_addr = project_to<coords::omt>( elem.first );

169  save_quad( dirname, quad_path, om_addr, submaps_to_delete,
170  delete_after_save || !inside_reality_bubble );

172  }

275 
276  if( !file_exist( quad_path ) ) {
277  // Fix for old saves where the path was generated using std::stringstream, which
278  // did format the number using the current locale. That formatting may insert
279  // thousands separators, so the resulting path is "map/1,234.7.8.map" instead
280  // of "map/1234.7.8.map".
281  std::ostringstream buffer;
282  buffer << om_addr.x() << "." << om_addr.y() << "." << om_addr.z()
283  << ".map";
284  cata_path legacy_quad_path = dirname / buffer.str();
285  if( file_exist( legacy_quad_path ) ) {
286  quad_path = std::move( legacy_quad_path );
287  }
288  }
289 
290  if( !read_from_file_optional_json( quad_path, [this]( const JsonValue & jsin ) {
291  deserialize( jsin );
292  } ) ) {
293  // If it doesn't exist, trigger generating it.
294  return nullptr;
295  }
296  // fill in uniform submaps that were not serialized
297  oter_id const oid = overmap_buffer.ter( om_addr );
298  generate_uniform_omt( project_to<coords::sm>( om_addr ), oid );
299  if( submaps.count( p ) == 0 ) {
