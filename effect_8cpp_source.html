286 void effect_type::extract_effect(
287  const std::array<std::optional<JsonObject>, 2> &j,
288  const std::string &effect_name,
289  const std::vector<std::pair<std::string, mod_action>> &action_keys )
290 {
291  std::unordered_map<uint32_t, modifier_value_arr> modifiers;
292 
293  for( uint32_t cur_mod_type = 0; cur_mod_type < static_cast<size_t>( mod_type::MAX );
294  cur_mod_type++ ) {
295  if( !j[cur_mod_type].has_value() ) {
296  continue;
297  }
298 
299  const JsonObject &base = *j[cur_mod_type];
300 
301  for( const auto &action_key : action_keys ) {
302  if( !base.has_array( action_key.first ) ) {
303  continue;
304  }
305 
306  JsonArray jsarr = base.get_array( action_key.first );
307  if( jsarr.empty() ) {
308  continue;
309  }
310 
311  mod_action action = action_key.second;
312  cata_assert( jsarr.size() < 0x100 );
313  for( size_t i = 0; i < jsarr.size(); i++ ) {
314  uint32_t key = get_effect_modifier_key( action, i );
315 
316  // Create default entry if doesn't exist
317  const auto &it = modifiers.emplace( key, default_modifier_values );
318 
319  // Update entry with the fetched value
320  it.first->second[cur_mod_type] = jsarr.get_float( i );
321  }
322  }
323  }
324 
325  if( modifiers.empty() ) {
326  return;
327  }
328 
329  mod_data[effect_name] = std::move( modifiers );
330 }
