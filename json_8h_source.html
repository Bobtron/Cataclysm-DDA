160 /* TextJsonIn
161  * ======
162  *
163  * The TextJsonIn class provides a wrapper around a std::istream,
164  * with methods for reading JSON data directly from the stream.
165  *
166  * TextJsonObject and TextJsonArray provide higher-level wrappers,
167  * and are a little easier to use in most cases,
168  * but have the small overhead of indexing the members or elements before use.
169  *
170  * Typical TextJsonIn usage might be something like the following:
171  *
172  * TextJsonIn jsin(myistream);
173  * // expecting an array of objects
174  * jsin.start_array(); // throws JsonError if array not found
175  * while (!jsin.end_array()) { // end_array returns false if not the end
176  * TextJsonObject jo = jsin.get_object();
177  * ... // load object using TextJsonObject methods
178  * }
179  *
180  * The array could have been loaded into a TextJsonArray for convenience.
181  * Not doing so saves one full pass of the data,
182  * as the element positions don't have to be read beforehand.
183  *
184  * If the JSON structure is not as expected,
185  * verbose error messages are provided, indicating the problem,
186  * and the exact line number and byte offset within the istream.
187  *
188  *
189  * Single-Pass Loading
190  * -------------------
191  *
192  * A TextJsonIn can also be used for single-pass loading,
193  * by passing off members as they arrive according to their names.
194  *
195  * Typical usage might be:
196  *
197  * TextJsonIn jsin(&myistream);
198  * // expecting an array of objects, to be mapped by id
199  * std::map<std::string,my_data_type> myobjects;
200  * jsin.start_array();
201  * while (!jsin.end_array()) {
202  * my_data_type myobject;
203  * jsin.start_object();
204  * while (!jsin.end_object()) {
205  * std::string name = jsin.get_member_name();
206  * if (name == "id") {
207  * myobject.id = jsin.get_string();
208  * } else if (name == "name") {
209  * myobject.name = _(jsin.get_string());
210  * } else if (name == "description") {
211  * myobject.description = _(jsin.get_string());
212  * } else if (name == "points") {
213  * myobject.points = jsin.get_int();
214  * } else if (name == "flags") {
215  * if (jsin.test_string()) { // load single string tag
216  * myobject.tags.insert(jsin.get_string());
217  * } else { // load tag array
218  * jsin.start_array();
219  * while (!jsin.end_array()) {
220  * myobject.tags.insert(jsin.get_string());
221  * }
222  * }
223  * } else {
224  * jsin.skip_value();
225  * }
226  * }
227  * myobjects[myobject.id] = myobject;
228  * }
229  *
230  * The get_* methods automatically verify and skip separators and whitespace.
231  *
232  * Using this method, unrecognized members are silently ignored,
233  * allowing for things like "comment" members,
234  * but the user must remember to provide an "else" clause,
235  * explicitly skipping them.
236  *
237  * If an if;else if;... is missing the "else", it /will/ cause bugs,
238  * so preindexing as a TextJsonObject is safer, as well as tidier.
239  */

964 /* TextJsonObject
965  * ==========
966  *
967  * The TextJsonObject class provides easy access to incoming JSON object data.
968  *
969  * TextJsonObject maps member names to the byte offset of the paired value,
970  * given an underlying TextJsonIn stream.
971  *
972  * It provides data by seeking the stream to the relevant position,
973  * and calling the correct TextJsonIn method to read the value from the stream.
974  *
975  *
976  * General Usage
977  * -------------
978  *
979  * TextJsonObject jo(jsin);
980  * std::string id = jo.get_string("id");
981  * std::string name = _(jo.get_string("name"));
982  * std::string description = _(jo.get_string("description"));
983  * int points = jo.get_int("points", 0);
984  * std::set<std::string> tags = jo.get_tags("flags");
985  * my_object_type myobject(id, name, description, points, tags);
986  *
987  * Here the "id", "name" and "description" members are required.
988  * TextJsonObject will throw a JsonError if they are not found,
989  * identifying the problem and the current position in the input stream.
990  *
991  * Note that "name" and "description" are passed to gettext for translating.
992  * Any members with string information that is displayed directly to the user
993  * will need similar treatment, and may also need to be explicitly handled
994  * in lang/extract_json_strings.py to be translatable.
995  *
996  * The "points" member here is not required.
997  * If it is not found in the incoming JSON object,
998  * it will be initialized with the default value given as the second parameter,
999  * in this case, 0.
1000  *
1001  * get_tags() always returns an empty set if the member is not found,
1002  * and so does not require a default value to be specified.
1003  *
1004  *
1005  * Member Testing and Automatic Deserialization
1006  * --------------------------------------------
1007  *
1008  * JsonObjects can test for member type with has_int(name) etc.,
1009  * and for member existence with has_member(name).
1010  *
1011  * They can also read directly into compatible data structures,
1012  * including sets, vectors, maps, and any class with a compatible
1013  * deserialize() routine.
1014  *
1015  * read() returns true on success, false on failure.
1016  *
1017  * TextJsonObject jo(jsin);
1018  * std::vector<std::string> messages;

1021  * }
1022  *
1023  *
1024  * Automatic error checking
1025  * ------------------------

1027  * By default, when a TextJsonObject is destroyed (or when you call finish) it will
1028  * check to see whether every member of the object was referenced in some way
1029  * (even simply checking for the existence of the member is sufficient).

1031  * If not all the members were referenced, then an error will be written to the
1032  * log (which in particular will cause the tests to fail).
1033  *
1034  * If you don't want this behavior, then call allow_omitted_members() before
1035  * the TextJsonObject is destroyed. Calling str() also suppresses it (on the basis
1036  * that you may be intending to re-parse that string later).
1037  */

1173  * =========
1174  *
1175  * The TextJsonArray class provides easy access to incoming JSON array data.
1176  *
1177  * TextJsonArray stores only the byte offset of each element in the TextJsonIn stream,
1178  * and iterates or accesses according to these offsets.
1179  *
1180  * It provides data by seeking the stream to the relevant position,
1181  * and calling the correct TextJsonIn method to read the value from the stream.
1182  *
1183  * Arrays can be iterated over,
1184  * or accessed directly by element index.
1185  *
1186  * Elements can be returned as standard data types,
1187  * or automatically read into a compatible container.
1188  *
1189  *
1190  * Iterative Access
1191  * ----------------
1192  *
1193  * TextJsonArray ja = jo.get_array("some_array_member");
1194  * std::vector<int> myarray;
1195  * while (ja.has_more()) {
1196  * myarray.push_back(ja.next_int());
1197  * }
1198  *
1199  * If the array member is not found, get_array() returns an empty array,
1200  * and has_more() is always false, so myarray will remain empty.
1201  *
1202  * next_int() and the other next_* methods advance the array iterator,
1203  * so after the final element has been read,
1204  * has_more() will return false and the loop will terminate.
1205  *
1206  * If the next element is not an integer,
1207  * TextJsonArray will throw a JsonError indicating the problem,
1208  * and the position in the input stream.
1209  *
1210  * To handle arrays with elements of indeterminate type,
1211  * the test_* methods can be used, calling next_* according to the result.
1212  *
1213  * Note that this style of iterative access requires an element to be read,
1214  * or else the index will not be incremented, resulting in an infinite loop.
1215  * Unwanted elements can be skipped with TextJsonArray::skip_value().
1216  *
1217  *
1218  * Positional Access
1219  * -----------------
1220  *
1221  * TextJsonArray ja = jo.get_array("xydata");
1222  * point xydata(ja.get_int(0), ja.get_int(1));
1223  *
1224  * Arrays also provide has_int(index) etc., for positional type testing,
1225  * and read(index, &container) for automatic deserialization.
1226  *
1227  *
1228  * Automatic Deserialization
1229  * -------------------------
1230  *
1231  * Elements can also be automatically read into compatible containers,
1232  * such as maps, sets, vectors, and classes using a deserialize routine,
1233  * using the read_next() and read() methods.
1234  *
1235  * TextJsonArray ja = jo.get_array("custom_datatype_array");
1236  * while (ja.has_more()) {
