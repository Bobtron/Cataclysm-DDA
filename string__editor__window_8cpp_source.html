163 point folded_text::codepoint_coordinates( const int cpt_idx, const bool zero_x ) const
164 {
165  if( lines.empty() ) {
166  return point_zero;
167  }
168  // find the line before the cursor position

170  []( const folded_line & l, const int p ) {
171  return l.cpts_end < p;
172  } );
173  if( it == lines.end() ) {
174  // past the last codepoint, shouldn't happen
175  return point_zero;
176  }
177  int y = std::distance( lines.begin(), it );
178  // if zero_x is true and the line is not the last line, cursor at the end of
179  // the line is moved to the start of the next line
180  if( zero_x && static_cast<size_t>( y ) + 1 < lines.size()
181  && cpt_idx == it->cpts_end ) {
182  return point( 0, y + 1 );
183  }
184  // otherwise, calculate the width until cpt_idx
185  int x = 0;
186  const char *src = it->str.c_str();
187  int bytes = it->str.length();
188  for( int i = 0; bytes > 0 && i < cpt_idx - it->cpts_start; ++i ) {
189  const uint32_t uc = UTF8_getch( &src, &bytes );
190  if( is_linebreak( uc ) ) {
191  x = 0;
192  ++y;
193  } else {
194  x += std::max( 0, mk_wcwidth( uc ) );
195  }
196  }
