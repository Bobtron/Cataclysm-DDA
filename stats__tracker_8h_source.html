33 // The stats_tracker is intended to keep a summary of events that have occurred.
34 // For each event_type it stores an event_multiset.
35 // Within the event_tracker, the events are partitioned according to their data
36 // (an event::data_type object, which is a map of keys to values).
37 // Within each partition, an event_summary is stored, which contains the first
38 // and last times such events were seen, and the number of them seen.
39 // The stats_tracker can be queried in various ways to get summary statistics
40 // about events that have occurred.

74  // count returns the number of events matching given criteria that have
75  // occurred.
76  // total returns the sum of some integer-valued field across every
77  // event satisfying certain criteria.
78  // maximum and minimum return the max and min respectively of some
79  // integer-valued field across every event satisfying certain criteria.
80  // For example, count might return the number of times the avatar has
81  // taken damage, total might return the total damage taken in all
82  // those cases, and maximum might return the single largest amount
83  // of damage taken in a single event across all those cases.
84  // The criteria takes the form of an event::data_type map specifying

149  template<typename Class, typename... FnArgs, typename... Args>
150  void send_to_all( void ( Class::*mem_fn )( FnArgs... ), Args &&... args ) const {
151  static_assert( std::is_base_of<Class, Watcher>::value,
152  "Watcher must be derived from Class" );
153  // Sending an event to a watcher can cause it to be erased, so we
154  // need to always ensure we have the next iterator prepared in
155  // advance.
156  auto current = watchers_.begin();

158  while( current != watchers_.end() ) {
159  auto next = current;
160  ++next;
161  ( static_cast<Watcher *>( *current )->*mem_fn )( args... );
162  current = next;
163  }
164  }
