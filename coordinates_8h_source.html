583 // project_remain returns a helper struct, intended to be used with std::tie
584 // to pull out the two components of the result.
585 // For example, when splitting a point:
586 // point_abs_sm val;
587 // point_abs_om quotient;
588 // point_om_sm remainder;
589 // std::tie( quotient, remainder ) = project_remain<coords::om>( val );
590 //
591 // If passing a tripoint to project_remain, you must choose exactly one of the
592 // quotient or remainder to get the z coordinate. Both of these should work:
593 // tripoint_abs_sm val;
594 //
595 // tripoint_abs_om quotient;
596 // point_om_sm remainder;
597 // std::tie( quotient, remainder ) = project_remain<coords::om>( val );
598 //
599 // point_abs_om quotient;
600 // tripoint_om_sm remainder;
601 // std::tie( quotient, remainder ) = project_remain<coords::om>( val );
602 template<scale ResultScale, origin Origin, scale SourceScale, bool InBounds>
603 inline quotient_remainder_point<Origin, ResultScale, SourceScale, InBounds> project_remain(
604  const coord_point<point, Origin, SourceScale, InBounds> &src )
605 {
606  constexpr int ScaleDown = map_squares_per( ResultScale ) / map_squares_per( SourceScale );
607  static_assert( ScaleDown > 0, "You can only project to coarser coordinate systems" );
608  constexpr static origin RemainderOrigin = origin_from_scale( ResultScale );
609  coord_point<point, Origin, ResultScale, InBounds> quotient = project_to< ResultScale>( src );
610  remainder_inbounds_t<point, RemainderOrigin, SourceScale, InBounds> remainder =
611  remainder_inbounds_t<point, RemainderOrigin, SourceScale, InBounds>::make_unchecked(
612  src.raw() - quotient.raw() * ScaleDown );
613 
614  return { quotient, remainder };
615 }
