86  return rule_state::WHITELISTED;

121 /**
122  * Iterate through every item inside the container to find items that match auto pickup rules.
123  * In most cases whitelisted items will be included and blacklisted one will be excluded however
124  * there are special cases. Below is an overview of selection rules for container auto pickup.
125  *
126  * Containers and items will **never** be picked up when:
127  *
128  * - they are owned by player and `AUTO_PICKUP_OWNED` game option is disabled.
129  * - they exceed volume or weight user auto pickup limitations.
130  * - they are blacklisted in auto pickup rules.
131  *
132  * Containers will **always** be picked up when:
133  *
134  * - the container is sealed.
135  * - there is any liquids stored in container pockets.
136  * - all items inside the container are marked for pickup.
137  *
138  * Containers will **never** be picked up when:
139  *
140  * - only batteries were selected and the container is battery powered.
141  * - the container is a non-whitelisted corpse.
142  *
143  * Items will **NOT** be picked up when:
144  *
145  * - the parent container is non-rigid and the item is not whitelisted.
146  *
147  * @param from item to search for items to auto pickup from.
148  * @return sequence of items to auto pickup from given container.
149  */
150 static std::vector<item_location> get_autopickup_items( item_location &from )
151 {
152  item *container_item = from.get_item();
153  // items sealed in containers should never be unsealed by auto pickup
154  bool force_pick_container = container_item->any_pockets_sealed();
155  bool pick_all_items = true;
156 
157  std::vector<item_location> result;
158  // do not auto pickup owned containers or items
159  if( !get_option<bool>( "AUTO_PICKUP_OWNED" ) &&
160  container_item->is_owned_by( get_player_character() ) ) {
161  return result;
162  }
163  std::list<item *> contents = container_item->all_items_top();
164  result.reserve( contents.size() );
165 
166  bool any_whitelisted = false;
167  std::list<item *>::iterator it;
168  for( it = contents.begin(); it != contents.end(); ++it ) {
169  item *item_entry = *it;
170  if( !within_autopickup_limits( item_entry ) ) {
171  pick_all_items = false;
172  continue;
173  }
174  const rule_state pickup_state = get_autopickup_rule( item_entry );
175  if( pickup_state == rule_state::WHITELISTED ) {
176  any_whitelisted = true;
177  if( !force_pick_container ) {
178  if( item_entry->is_container() ) {
179  // whitelisted containers should exclude contained blacklisted items
180  empty_autopickup_target( item_entry, from.position() );
181  } else if( item_entry->made_of_from_type( phase_id::LIQUID ) ) {
182  // liquid items should never be picked up without container
183  force_pick_container = true;
184  break;
185  }
186  // pick up the whitelisted item
187  result.emplace_back( from, item_entry );
188  }
189  } else if( !force_pick_container && item_entry->is_container() &&
190  !item_entry->is_container_empty() ) {
191  // get pickup list from nested item container
192  item_location location = item_location( from, item_entry );
193  std::vector<item_location> result_nested = get_autopickup_items( location );
194 
195  // container with content was NOT marked for pickup
196  if( result_nested.size() != 1 || result_nested[0] != location ) {
197  pick_all_items = false;
198  }
199  result.reserve( result_nested.size() + result.size() );
200  result.insert( result.end(), result_nested.begin(), result_nested.end() );
201  } else {
202  // skip not whitelisted items that are not containers with items
203  pick_all_items = false;
204  }
205  }
206  // all items in container were approved for pickup
207  if( !contents.empty() && ( pick_all_items || force_pick_container ) ) {
208  // only auto pickup corpses if they are whitelisted
209  // blacklisted containers should still have their contents picked up but themselves should be excluded.
210  // If all items inside blacklisted container match then just pickup the items without the container
211  rule_state pickup_state = get_autopickup_rule( container_item );
212  if( pickup_state == rule_state::BLACKLISTED || !any_whitelisted ||
213  ( container_item->is_corpse() && pickup_state != rule_state::WHITELISTED ) ) {
214  return result;
215  }
216  bool all_batteries = true;
217  bool powered_container = container_item->ammo_capacity( ammo_battery );
218  if( powered_container ) {
219  // when dealing with battery powered tools there should only be one pocket
220  // and one battery inside but this could change in future so account for that here
221  all_batteries = std::all_of( result.begin(), result.end(),
222  []( const item_location & il ) {
223  return il.get_item()->is_battery();
224  } );
225  }
226  bool batteries_from_tool = powered_container && all_batteries;
227  // make sure container is allowed to be picked up
228  // when picking up batteries from powered containers don't pick container
229  if( within_autopickup_limits( container_item ) && !batteries_from_tool ) {
230  result.clear();
231  result.push_back( from );
232  } else if( force_pick_container ) {
233  // when force picking never pick individual items
234  result.clear();
235  }
236  }
237  return result;
238 }

788 bool check_special_rule( const std::map<material_id, int> &materials, const std::string_view rule )
789 {
790  char type = ' ';
791  std::vector<std::string> filter;
792  if( rule[1] == ':' ) {
793  type = rule[0];
794  filter = string_split( rule.substr( 2 ), ',' );
795  }
796 
797  if( filter.empty() || materials.empty() ) {
798  return false;
799  }
800 
801  if( type == 'm' ) {
802  return std::any_of( materials.begin(),
803  materials.end(), [&filter]( const std::pair<material_id, int> &mat ) {
804  return std::any_of( filter.begin(), filter.end(), [&mat]( const std::string_view search ) {
805  return lcmatch( mat.first->name(), search );
806  } );
807  } );
808 
809  } else if( type == 'M' ) {
810  return std::all_of( materials.begin(),
811  materials.end(), [&filter]( const std::pair<material_id, int> &mat ) {
812  return std::any_of( filter.begin(), filter.end(), [&mat]( const std::string_view search ) {
813  return lcmatch( mat.first->name(), search );
814  } );
815  } );
816  }
817 
818  return false;
819 }

871 void rule_list::refresh_map_items( cache &map_items ) const
872 {
873  for( const rule &elem : *this ) {
874  if( elem.sRule.empty() || !elem.bActive ) {
875  continue;
876  }
877 
878  if( !elem.bExclude ) {
879  //Check include patterns against all itemfactory items
880  for( const itype *e : item_controller->all() ) {
881  const std::string &cur_item = e->nname( 1 );
882 
883  if( !check_special_rule( e->materials, elem.sRule ) && !wildcard_match( cur_item, elem.sRule ) ) {

888  map_items.temp_items[ cur_item ] = e;

895  !wildcard_match( map_item.first, elem.sRule ) ) {
