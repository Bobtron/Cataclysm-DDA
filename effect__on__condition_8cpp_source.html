232 static void process_eocs( queued_eocs &eoc_queue, std::vector<effect_on_condition_id> &eoc_vector,
233  dialogue &d )
234 {
235  static std::vector<queued_eocs::storage_iter> eocs_to_queue;
236  eocs_to_queue.clear();
237 
238  while( !eoc_queue.empty() &&
239  eoc_queue.top().time <= calendar::turn ) {
240  queued_eocs::storage_iter it = eoc_queue.queue.top();
241  queued_eoc &top = *it;
242  eoc_queue.queue.pop();
243 
244  dialogue nested_d{ d };
245  for( const auto &val : top.context ) {
246  nested_d.set_value( val.first, val.second );
247  }
248  bool activated = top.eoc->activate( nested_d );
249  if( top.eoc->type == eoc_type::RECURRING ) {
250  if( activated ) { // It worked so add it back
251  it->time = calendar::turn + next_recurrence( top.eoc, d );
252  eocs_to_queue.emplace_back( it );
253  } else {
254  if( !top.eoc->check_deactivate(
255  nested_d ) ) { // It failed but shouldn't be deactivated so add it back
256  it->time = calendar::turn + next_recurrence( top.eoc, d );
257  eocs_to_queue.emplace_back( it );
258  } else { // It failed and should be deactivated for now
259  eoc_vector.push_back( top.eoc );
260  eoc_queue.list.erase( it );
261  }
262  }
263  } else {
264  eoc_queue.list.erase( it );
265  }
266  }
267  for( queued_eocs::storage_iter &q_eoc : eocs_to_queue ) {
268  eoc_queue.queue.emplace( q_eoc );
269  }

315 {
316  bool retval = false;
317  if( require_callstack_check ) {
318  d.amend_callstack( "EOC: " + id.str() );
319  if( d.get_callstack().size() > 5000 ) {
320  if( query_yn( string_format( _( "Possible infinite loop in EOC %s. Stop execution?" ),

322  return false;

324  }

332  false_effect.apply( d_eoc );

334  // This works because if global is true then this is recurring and thus should only ever be passed containing the player
335  // Thus we just need to run the npcs.
