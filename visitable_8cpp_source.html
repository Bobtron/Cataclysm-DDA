98 static T sum_no_wrap( T a, T b )

800 static int charges_of_internal( const T &self, const M &main, const itype_id &id, int limit,
801  const std::function<bool( const item & )> &filter,
802  const std::function<void( int )> &visitor, bool in_tools )
803 {
804  int qty = 0;
805 
806  bool found_tool_with_UPS = false;
807  bool found_bionic_tool = false;
808  self.visit_items( [&]( const item * e, item * ) {
809  if( filter( *e ) &&
810  ( id == e->typeId() || ( in_tools && id == e->ammo_current() ) ||
811  ( id == itype_UPS && e->has_flag( flag_IS_UPS ) ) ) &&
812  !e->is_broken() ) {
813  if( id != itype_UPS ) {
814  if( e->count_by_charges() ) {
815  qty = sum_no_wrap( qty, e->charges );
816  } else {
817  qty = sum_no_wrap( qty, e->ammo_remaining( nullptr, true ) );
818  }
819  if( e->has_flag( STATIC( flag_id( "USE_UPS" ) ) ) ) {
820  found_tool_with_UPS = true;
821  } else if( e->has_flag( STATIC( flag_id( "USES_BIONIC_POWER" ) ) ) ) {
822  found_bionic_tool = true;
823  }
824  } else if( id == itype_UPS && e->has_flag( flag_IS_UPS ) ) {
825  qty = sum_no_wrap( qty, e->ammo_remaining( nullptr, true ) );
826  }
827  }
828  if( qty >= limit ) {
829  return VisitResponse::ABORT;
830  }
831  // recurse through nested containers if any
832  return e->is_container() ? VisitResponse::NEXT : VisitResponse::SKIP;
833  } );
834 
835  if( found_tool_with_UPS && qty < limit && get_player_character().has_active_bionic( bio_ups ) ) {
836  qty = sum_no_wrap( qty, static_cast<int>( units::to_kilojoule(
837  get_player_character().get_power_level() ) ) );
838  }
839 
840  if( found_bionic_tool ) {
841  qty = sum_no_wrap( qty, static_cast<int>( units::to_kilojoule(
842  get_player_character().get_power_level() ) ) );
843  }
844 
845  if( qty < limit && found_tool_with_UPS ) {
846  int used_ups = main.charges_of( itype_UPS, limit - qty );
847  qty += used_ups;
848  if( visitor ) {
849  visitor( used_ups );
850  }
851  }
852 
853  return std::min( qty, limit );
