716 {

718  popup( colorize( _( "Zones tied to vehicles cannot be displayed" ), c_magenta ) );
719  return;
720  }
721 
722  std::unique_ptr<tinymap> p_update_tmap = std::make_unique<tinymap>();
723  tinymap &update_tmap = *p_update_tmap;
724 
725  std::pair<tripoint_abs_ms, tripoint_abs_ms> bounds = get_corners( get_start_point(),
726  get_end_point() );

728  const tripoint_abs_ms end = bounds.second;

731  const tripoint_abs_omt omt_end_pos = coords::project_to<coords::omt>( end );

734  ( omt_start_pos ) ).raw() );
735  const tripoint_omt_ms end_remainder = tripoint_omt_ms( ( get_end_point() - coords::project_to
736  < coords::ms >
737  ( omt_end_pos ) ).raw() );
738

740 
741  field_type_str_id field = fd_null;
742  static const std::vector<zone_type> &all_zone_types = zone_type::get_all();
743  for( const zone_type &zone : all_zone_types ) {
744  if( zone.id == type ) {
745  field = zone.get_field();
746  break;
747  }
748  }

760  k > omt_start_pos.y() ? 0 : start_remainder.y(), start_remainder.z() );
761  end_ms = tripoint_omt_ms( i < omt_end_pos.x() ? SEEX * 2 - 1 : end_remainder.x(),
762  k < omt_end_pos.y() ? SEEY * 2 - 1 : end_remainder.y(), end_remainder.z() );

765  if( is_displayed ) {
766  update_tmap.add_field( pt, field, 1, time_duration::from_turns( 0 ), false );
