7 // enum_traits is a template you can specialize for your enum types. It serves
8 // two purposes:
9 // * Picking out the special "last" enumerator if you have one. This is used
10 // by various generic code to iterate over all the enumerators. Most notably
11 // it enables io::string_to_enum and thereby string-based json serialization.
12 // * Specifying that your enum is a flag enum, and therefore that you want
13 // bitwise operators to work for it. This saves everyone from implementing
14 // those operators independently.
15 //
16 // Usage examples:
17 //
18 // For a 'regular' enum
19 //
20 // enum ordinal {
21 // first,
22 // second,
23 // third,
24 // last
25 // };
26 //
27 // template<>
28 // struct enum_traits<ordinal> {
29 // static constexpr ordinal last = ordinal::last;
30 // };
31 //
32 // For a flag enum
33 //
34 // enum my_flags {
35 // option_one = 1 << 0,
36 // option_two = 1 << 1,
37 // };
38 //
39 // template<>
40 // struct enum_traits<ordinal> {
41 // static constexpr bool is_flag_enum = true;
42 // };
