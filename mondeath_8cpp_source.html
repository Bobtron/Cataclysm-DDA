92 static void scatter_chunks( const itype_id &chunk_name, int chunk_amt, monster &z, int distance,
93  int pile_size = 1 )
94 {
95  // can't have less than one item in a pile or it would cause an infinite loop
96  pile_size = std::max( pile_size, 1 );
97  // can't have more items in a pile than total items
98  pile_size = std::min( chunk_amt, pile_size );
99  distance = std::abs( distance );
100  const item chunk( chunk_name, calendar::turn );
101  map &here = get_map();
102  int placed_chunks = 0;
103  while( placed_chunks < chunk_amt ) {
104  bool drop_chunks = true;
105  tripoint tarp( z.pos() + point( rng( -distance, distance ), rng( -distance, distance ) ) );
106  const auto traj = line_to( z.pos(), tarp );
107 
108  for( size_t j = 0; j < traj.size(); j++ ) {
109  tarp = traj[j];
110  if( one_in( 2 ) && z.bloodType().id() ) {
111  here.add_splatter( z.bloodType(), tarp );
112  } else {
113  here.add_splatter( z.gibType(), tarp, rng( 1, j + 1 ) );
114  }
115  if( here.impassable( tarp ) ) {
116  here.bash( tarp, distance );
117  if( here.impassable( tarp ) ) {
118  // Target is obstacle, not destroyed by bashing,
119  // stop trajectory in front of it, if this is the first
120  // point (e.g. wall adjacent to monster), don't drop anything on it
121  if( j > 0 ) {
122  tarp = traj[j - 1];
123  } else {
124  drop_chunks = false;
125  }
126  break;
127  }
128  }
129  }
130  if( drop_chunks ) {
131  for( int i = placed_chunks; i < chunk_amt && i < placed_chunks + pile_size; i++ ) {
132  here.add_item_or_charges( tarp, chunk );
133  }
134  }
135  placed_chunks += pile_size;
136  }

139 item_location mdeath::splatter( monster &z )
140 {
141  const bool gibbable = !z.type->has_flag( mon_flag_NOGIB );
142 
143  const int max_hp = std::max( z.get_hp_max(), 1 );
144  const float overflow_damage = std::max( -z.get_hp(), 0 );
145  const float corpse_damage = 2.5 * overflow_damage / max_hp;
146 
147  const field_type_id type_blood = z.bloodType();
148  const field_type_id type_gib = z.gibType();
149 
150  map &here = get_map();
151  if( gibbable ) {
152  const auto area = here.points_in_radius( z.pos(), 1 );
153  int number_of_gibs = std::min( std::floor( corpse_damage ) - 1, 1 + max_hp / 5.0f );
154 
155  if( z.type->size >= creature_size::medium ) {
156  number_of_gibs += rng( 1, 6 );
157  sfx::play_variant_sound( "mon_death", "zombie_gibbed", sfx::get_heard_volume( z.pos() ) );
158  }
159 
160  for( int i = 0; i < number_of_gibs; ++i ) {
161  here.add_splatter( type_gib, random_entry( area ), rng( 1, i + 1 ) );
162  here.add_splatter( type_blood, random_entry( area ) );
163  }
164  }
165  // 1% of the weight of the monster is the base, with overflow damage as a multiplier
166  int gibbed_weight = rng( 0, std::round( to_gram( z.get_weight() ) / 100.0 *
167  ( overflow_damage / max_hp + 1 ) ) );
168  const int z_weight = to_gram( z.get_weight() );
169  // limit gibbing to 15%
170  gibbed_weight = std::min( gibbed_weight, z_weight * 15 / 100 );
171 
172  if( gibbable ) {
173  float overflow_ratio = overflow_damage / max_hp + 1.0f;
174  int gib_distance = std::round( rng( 2, 4 ) );
175  for( const harvest_entry &entry : *z.type->harvest ) {
176  // only flesh and bones survive.
177  if( entry.type == harvest_drop_flesh || entry.type == harvest_drop_bone ) {
178  // the larger the overflow damage, the less you get
179  const int chunk_amt =
180  entry.mass_ratio / overflow_ratio / 10 *
181  z.get_weight() / item::find_type( itype_id( entry.drop ) )->weight;
182  scatter_chunks( itype_id( entry.drop ), chunk_amt, z, gib_distance,
183  chunk_amt / ( gib_distance - 1 ) );
184  gibbed_weight -= entry.mass_ratio / overflow_ratio / 20 * to_gram( z.get_weight() );
185  }
186  }
187  if( gibbed_weight > 0 ) {
188  const itype_id &leftover_id = z.type->id->harvest->leftovers;
189  const int chunk_amount =
190  gibbed_weight / to_gram( item::find_type( leftover_id )->weight );
191  scatter_chunks( leftover_id, chunk_amount, z, gib_distance,
192  chunk_amount / ( gib_distance + 1 ) );
193  }
194  // add corpse with gib flag
195  item corpse = item::make_corpse( z.type->id, calendar::turn, z.unique_name, z.get_upgrade_time() );
196  if( corpse.is_null() ) {
197  return {};
198  }
199  // Set corpse to damage that aligns with being pulped
200  corpse.set_damage( 4000 );
201  corpse.set_flag( STATIC( flag_id( "GIBBED" ) ) );
202  if( z.has_effect( effect_no_ammo ) ) {
203  corpse.set_var( "no_ammo", "no_ammo" );
204  }
205  if( z.has_effect( effect_critter_underfed ) ) {
206  corpse.set_flag( STATIC( flag_id( "UNDERFED" ) ) );
207  }
208  return here.add_item_ret_loc( z.pos(), corpse );
209  }
210  return {};
