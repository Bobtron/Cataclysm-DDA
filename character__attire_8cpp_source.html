823 static void layer_item( std::map<bodypart_id, encumbrance_data> &vals, const item &it,
824  std::map<sub_bodypart_id, layer_level> &highest_layer_so_far, const Character &c )
825 {
826  body_part_set covered_parts = it.get_covered_body_parts();
827  for( const bodypart_id &bp : c.get_all_body_parts() ) {
828  if( !covered_parts.test( bp.id() ) ) {
829  continue;
830  }
831 
832  const std::vector<layer_level> item_layers = it.get_layer( bp );
833  int encumber_val = it.get_encumber( c, bp.id() );
834  int layering_encumbrance = clamp( encumber_val, 2, 10 );
835 
836  /*
837  * Setting layering_encumbrance to 0 at this point makes the item cease to exist
838  * for the purposes of the layer penalty system. (normally an item has a minimum
839  * layering_encumbrance of 2 )
840  * Personal layer items and semitangible items do not conflict.
841  */
842  if( it.has_flag( flag_SEMITANGIBLE ) ) {
843  encumber_val = 0;
844  layering_encumbrance = 0;
845  }
846  if( it.has_flag( flag_PERSONAL ) ) {
847  layering_encumbrance = 0;
848  }
849  for( layer_level item_layer : item_layers ) {
850  // do the sublayers of this armor conflict
851  bool conflicts = false;
852 
853  // check if we've already added conflict for the layer and body part since each sbp is check individually
854  std::map<layer_level, bool> bpcovered;
855 
856  // add the sublocations to the overall body part layer and update if we are conflicting
857  for( sub_bodypart_id &sbp : it.get_covered_sub_body_parts() ) {
858  if( std::count_if( bp->sub_parts.begin(),
859  bp->sub_parts.end(), [sbp]( const sub_bodypart_str_id & bpid ) {
860  return sbp.id() == bpid;
861  } ) == 0 ) {
862  // the sub part isn't part of the bodypart we are checking
863  continue;
864  }
865  // bit hacky but needed since we are doing one layer at a time
866  std::vector<layer_level> ll;
867  ll.push_back( item_layer );
868  if( !it.has_layer( ll, sbp ) ) {
869  // skip this layer and sbp if it doesn't cover it
870  continue;
871  }
872 
873  if( item_layer >= highest_layer_so_far[sbp] ) {
874  conflicts = vals[bp].add_sub_location( item_layer, sbp );
875  } else {
876  // if it is on a lower layer it conflicts for sure
877  conflicts = true;
878  }
879 
880  highest_layer_so_far[sbp] = std::max( highest_layer_so_far[sbp], item_layer );
881 
882  // Apply layering penalty to this layer, as well as any layer worn
883  // within it that would normally be worn outside of it.
884  for( layer_level penalty_layer = item_layer;
885  penalty_layer <= highest_layer_so_far[sbp]; ++penalty_layer ) {
886 
887  // make sure we haven't already found a subpart that covers and would cause penalty
888  if( !bpcovered[penalty_layer] ) {
889  vals[bp].layer( penalty_layer, layering_encumbrance, conflicts );
890  bpcovered[penalty_layer] = true;
891  }
892  }
893  }
894  }
895  vals[bp].armor_encumbrance += encumber_val;
896  }
897 }
898 
899 /*
900  * Encumbrance logic:
901  * Some clothing is intrinsically encumbering, such as heavy jackets, backpacks, body armor, etc.
902  * These simply add their encumbrance value to each body part they cover.
903  * In addition, each article of clothing after the first in a layer imposes an additional penalty.
904  * e.g. one shirt will not encumber you, but two is tight and starts to restrict movement.
905  * Clothes on separate layers don't interact, so if you wear e.g. a light jacket over a shirt,
906  * they're intended to be worn that way, and don't impose a penalty.
907  * The default is to assume that clothes do not fit, clothes that are "fitted" either
908  * reduce the encumbrance penalty by ten, or if that is already 0, they reduce the layering effect.
909  *
910  * Use cases:
911  * What would typically be considered normal "street clothes" should not be considered encumbering.
912  * T-shirt, shirt, jacket on torso/arms, underwear and pants on legs, socks and shoes on feet.
913  * This is currently handled by each of these articles of clothing
914  * being on a different layer and/or body part, therefore accumulating no encumbrance.
915  */
916 void outfit::item_encumb( std::map<bodypart_id, encumbrance_data> &vals,
917  const item &new_item, const Character &guy ) const
918 {
919 
920  // reset all layer data
921  vals = std::map<bodypart_id, encumbrance_data>();
922 
923  // Figure out where new_item would be worn
924  std::list<item>::const_iterator new_item_position = worn.end();
925  if( !new_item.is_null() ) {
926  // const_cast required to work around g++-4.8 library bug
927  // see the commit that added this comment to understand why
928  new_item_position =
929  const_cast<outfit *>( this )->position_to_wear_new_item( new_item );
930  }
931 
932  // Track highest layer observed so far so we can penalize out-of-order
933  // items
934  std::map<sub_bodypart_id, layer_level> highest_layer_so_far;
935 
936  for( auto w_it = worn.begin(); w_it != worn.end(); ++w_it ) {
937  if( w_it == new_item_position ) {
938  layer_item( vals, new_item, highest_layer_so_far, guy );
939  }
940  layer_item( vals, *w_it, highest_layer_so_far, guy );
941  }
942 
943  if( worn.end() == new_item_position && !new_item.is_null() ) {
944  layer_item( vals, new_item, highest_layer_so_far, guy );
945  }
946 
947  // make sure values are sane
948  for( const bodypart_id &bp : guy.get_all_body_parts() ) {
949  encumbrance_data &elem = vals[bp];
950 
951  for( const layer_details &cur_layer : elem.layer_penalty_details ) {
952  // only apply the layers penalty to the limb if it is conflicting
953  if( cur_layer.is_conflicting ) {
954  elem.layer_penalty += cur_layer.total;
955  }
956  }
957 
958  elem.armor_encumbrance = std::max( 0, elem.armor_encumbrance );
959 
960  // Add armor and layering penalties for the final values
961  elem.encumbrance += elem.armor_encumbrance + elem.layer_penalty;
962  }
963 }
