309

312  add_message( tut_lesson::LESSON_RECOIL );
313  }
314  break;
315 
316  case ACTION_CLOSE:
317  add_message( tut_lesson::LESSON_SMASH );
318  break;
319 
320  case ACTION_USE: {
321  if( player_character.has_amount( itype_grenade_act, 1 ) ) {
322  add_message( tut_lesson::LESSON_ACT_GRENADE );
323  }
324 
325  if( player_character.last_item == itype_codeine ) {
326  add_message( tut_lesson::LESSON_TOOK_PAINKILLER );
327  }
328  }
329  break;
330 
331  case ACTION_EAT:
332  if( player_character.last_item == itype_cig ) {
333  add_message( tut_lesson::LESSON_TOOK_CIG );
334  } else if( player_character.last_item == itype_water_clean ) {
335  add_message( tut_lesson::LESSON_DRANK_WATER );
336  }
337  break;
338 
339  case ACTION_WEAR: {
340  item it( player_character.last_item, calendar::turn_zero );
341  if( it.is_holster() ) {
342  add_message( tut_lesson::LESSON_HOLSTERS_ACTIVATE );
343  } else if( it.has_pocket_type( pocket_type::CONTAINER ) ) {
344  add_message( tut_lesson::LESSON_WORE_STORAGE );

349  add_message( tut_lesson::LESSON_WORE_ARMOR );
350  }
351 
352  }
353  }
354  break;
355 
356  /* fallthrough */
