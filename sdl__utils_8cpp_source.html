162 std::vector<SDL_Color> color_linear_interpolate( const SDL_Color &start_color,
163  const SDL_Color &end_color, const unsigned additional_steps )
164 {
165  const unsigned steps = additional_steps + 2;
166  std::vector<SDL_Color> colors;
167  colors.reserve( steps );
168  colors.push_back( start_color );
169 
170  const float step = 1.0f / ( steps - 1 );
171  for( unsigned i = 0; i < steps - 2; i++ ) {
172  const float percent = ( i + 1 ) * step;
173  const Uint8 r = static_cast<Uint8>( start_color.r + static_cast<Uint8>( percent *
174  ( end_color.r - start_color.r ) ) );
175  const Uint8 g = static_cast<Uint8>( start_color.g + static_cast<Uint8>( percent *
176  ( end_color.g - start_color.g ) ) );
177  const Uint8 b = static_cast<Uint8>( start_color.b + static_cast<Uint8>( percent *
178  ( end_color.b - start_color.b ) ) );
179  colors.push_back( { r, g, b, 255 } );
180  }
181  colors.push_back( end_color );
182  return colors;
183 }
