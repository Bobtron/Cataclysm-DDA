603 static bool has_cyclic_dependency( const trait_id &mid, std::vector<trait_id> already_visited )
604 {
605  if( contains_trait( already_visited, mid ) ) {
606  return true; // Circular dependency was found.
607  }
608 
609  already_visited.push_back( mid );
610 
611  // Perform Depth-first search
612  for( const trait_id &current_mutation : mid->prereqs ) {
613  if( has_cyclic_dependency( current_mutation->id, already_visited ) ) {
614  return true;
615  }
616  }
617 
618  for( const trait_id &current_mutation : mid->prereqs2 ) {
619  if( has_cyclic_dependency( current_mutation->id, already_visited ) ) {
620  return true;
621  }
622  }
623 
624  return false;
625 }

651 void mutation_branch::check_consistency()
652 {
653  for( const mutation_branch &mdata : get_all() ) {
654  const trait_id &mid = mdata.id;
655  const mod_id &trait_source = mdata.src.back().second;
656  const bool basegame_trait = trait_source.str() == "dda";
657  const std::optional<scenttype_id> &s_id = mdata.scent_typeid;
658  const std::map<species_id, int> &an_id = mdata.anger_relations;
659  for( const auto &style : mdata.initial_ma_styles ) {
660  if( !style.is_valid() ) {
661  debugmsg( "mutation %s refers to undefined martial art style %s", mid.c_str(), style.c_str() );
662  }
663  }
664  for( const std::string &type : mdata.types ) {
665  if( !mutation_type_exists( type ) ) {
666  debugmsg( "mutation %s refers to undefined mutation type %s", mid.c_str(), type );
667  }
668  }
669  if( mid->transform ) {
670  const trait_id tid = mid->transform->target;
671  if( !tid.is_valid() ) {
672  debugmsg( "mutation %s transform uses undefined target %s", mid.c_str(), tid.c_str() );
673  }
674  }
675  for( const std::pair<species_id, int> elem : an_id ) {
676  if( !elem.first.is_valid() ) {
677  debugmsg( "mutation %s refers to undefined species id %s", mid.c_str(), elem.first.c_str() );
678  }
679  }
680  if( s_id && !s_id.value().is_valid() ) {
681  debugmsg( "mutation %s refers to undefined scent type %s", mid.c_str(), s_id.value().c_str() );
682  }
683  // Suppress these onload warnings for overlapping mods
684  for( const trait_id &replacement : mdata.replacements ) {
685  const mutation_branch &rdata = replacement.obj();
686  bool suppressed = rdata.src.back().second != trait_source && !basegame_trait;
687  for( const mutation_category_id &cat : rdata.category ) {
688  if( std::find( mdata.category.begin(), mdata.category.end(), cat ) == mdata.category.end() &&
689  !suppressed ) {
690  debugmsg( "mutation %s lacks category %s present in replacement mutation %s", mid.c_str(),
691  cat.c_str(), replacement.c_str() );
692  }
693  }
694  }
695  for( const trait_id &addition : mdata.additions ) {
696  if( mdata.category.empty() ) {
697  break;
698  }
699  const mutation_branch &adata = addition.obj();
700  bool found = false;
701  bool suppressed = adata.src.back().second != trait_source && !basegame_trait;
702  for( const mutation_category_id &cat : adata.category ) {
703  found = found ||
704  std::find( mdata.category.begin(), mdata.category.end(), cat ) != mdata.category.end();
705  }
706  if( !found && !suppressed ) {
707  debugmsg( "categories in mutation %s don't match any category present in additive mutation %s",
708  mid.c_str(), addition.c_str() );
709  }
710  }
711 
712  for( const mutation_category_id &cat_id : mdata.category ) {
713  if( !mdata.prereqs.empty() ) {
714  bool found = false;
715  bool suppressed = false;
716  for( const trait_id &prereq_id : mdata.prereqs ) {
717  const mutation_branch &prereq = prereq_id.obj();
718  suppressed = suppressed || ( prereq.src.back().second != trait_source && !basegame_trait );
719  found = found ||
720  std::find( prereq.category.begin(), prereq.category.end(), cat_id ) != prereq.category.end();
721  }
722  if( !found && !suppressed ) {
723  debugmsg( "mutation %s is in category %s but none of its slot 1 prereqs have this category",
724  mid.c_str(), cat_id.c_str() );
725  }
726  }
727 
728  if( !mdata.prereqs2.empty() ) {
729  bool found = false;
730  bool suppressed = false;
731  for( const trait_id &prereq_id : mdata.prereqs2 ) {
732  const mutation_branch &prereq = prereq_id.obj();
733  suppressed = suppressed || ( prereq.src.back().second != trait_source && !basegame_trait );
734  found = found ||
735  std::find( prereq.category.begin(), prereq.category.end(), cat_id ) != prereq.category.end();
736  }
737  if( !found && !suppressed ) {
738  debugmsg( "mutation %s is in category %s but none of its slot 2 prereqs have this category",
739  mid.c_str(), cat_id.c_str() );
740  }
741  }
742  }
743 
744  for( const std::pair<const bodypart_str_id, resistances> &ma : mdata.armor ) {
745  for( const std::pair<const damage_type_id, float> &dt : ma.second.resist_vals ) {
746  if( !dt.first.is_valid() ) {
747  debugmsg( "Invalid armor type \"%s\" for mutation %s", dt.first.c_str(), mdata.id.c_str() );
748  }
749  }
750  }
751 
752  for( const mut_attack &atk : mdata.attacks_granted ) {
753  for( const damage_unit &dt : atk.base_damage.damage_units ) {
754  if( !dt.type.is_valid() ) {
755  debugmsg( "Invalid base_damage type \"%s\" for a mutation attack in mutation %s", dt.type.c_str(),
756  mdata.id.c_str() );
757  }
758  }
759  for( const damage_unit &dt : atk.strength_damage.damage_units ) {
760  if( !dt.type.is_valid() ) {
761  debugmsg( "Invalid strength_damage type \"%s\" for a mutation attack in mutation %s",
762  dt.type.c_str(), mdata.id.c_str() );
763  }
764  }
765  }
766 
767  // We need to display active mutations in the UI.
768  if( mdata.activated && !mdata.player_display ) {
769  debugmsg( "mutation %s is not displayed but set as active" );
770  }
771 
772  check_has_cyclic_dependency( mid );
773 
774  ::check_consistency( mdata.prereqs, mid, "prereqs" );
775  ::check_consistency( mdata.prereqs2, mid, "prereqs2" );
776  ::check_consistency( mdata.threshreq, mid, "threshreq" );
777  ::check_consistency( mdata.cancels, mid, "cancels" );
778  ::check_consistency( mdata.replacements, mid, "replacements" );
