180 struct arg_handler {
181  //! Handler function to be invoked when this argument is encountered. The handler will be
182  //! called with the number of parameters after the flag was encountered, along with the array
183  //! of following parameters. It must return an integer indicating how many parameters were
184  //! consumed by the call or -1 to indicate that a required argument was missing.
185  using handler_method = std::function<int ( int, const char ** )>;
186 
187  std::string_view flag; //!< The commandline parameter to handle (e.g., "--seed").
188  std::string_view param_documentation; //!< Human readable description of this arguments parameter.
189  std::string_view documentation; //!< Human readable documentation for this argument.
190  std::string_view help_group; //!< Section of the help message in which to include this argument.
191  int num_args; //!< How many further arguments are expected for this parameter (usually 0 or 1).
192  handler_method handler; //!< The callback to be invoked when this argument is encountered.
193 };

252 void process_args( const char **argv, int argc, const std::vector<arg_handler> &arg_handlers )
253 {
254  while( argc ) {
255  bool arg_handled = false;
256  for( const arg_handler &handler : arg_handlers ) {
257  if( handler.flag == argv[0] ) {
258  argc--;
259  argv++;
260  if( argc < handler.num_args ) {
261  std::cout << "Missing expected argument to command line parameter " << handler.flag << std::endl;
262  std::exit( 1 );
263  }
264  int args_consumed = handler.handler( argc, argv );
265  if( args_consumed < 0 ) {
266  printf( "Failed parsing parameter '%s'\n", *( argv - 1 ) );
267  std::exit( 1 );
268  }
269  argc -= args_consumed;
270  argv += args_consumed;
271  arg_handled = true;
272  break;
273  }
274  }
275  // Skip other options.
276  if( !arg_handled ) {
277  --argc;
278  ++argv;
279  }
280  }
