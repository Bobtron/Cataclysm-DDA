739 // Linear interpolation between start and end values.
740 // Constant outside of interval (x_start; x_end)
741 static float linear_interpolation( float x_start, float y_start, float x_end, float y_end, float x )
742 {
743  if( x < x_start ) {
744  return y_start;
745  }
746  if( x > x_end ) {
747  return y_end;
748  }
749  float interval = x_end - x_start;
750  float perc_to_end = ( x - x_start ) / interval;
751  return ( 1 - perc_to_end ) * y_start + perc_to_end * y_end;
752 }

772 void suffer::from_sunburn( Character &you, bool severe )
773 {
774  // Sunburn effects and albinism/datura occur about once per minute
775  if( !one_turn_in( 1_minutes ) ) {
776  return;
777  }
778 
779  // TODO: Could factor bodypart_exposure out of Character too
780  std::map<bodypart_id, float> bp_exposure = you.bodypart_exposure();
781 
782  enum Sunburn { None, Focus_Loss, Pain, Damage };
783 
784  auto heavy_sunburn = [severe, &you]( bodypart_id bp ) {
785  if( severe ) {
786  // Because hands and feet share an HP pool with arms and legs, and the mouth shares
787  // an HP pool with the head, those parts take an unfair share of damage in relation
788  // to the torso, which only has one part. Increase torso damage to balance this.
789  if( bp == bodypart_id( "torso" ) ) {
790  you.apply_damage( nullptr, bp, 2 );
791  } else {
792  you.apply_damage( nullptr, bp, 1 );
793  }
794  return Damage;
795  } else {
796  you.mod_pain( 1 );
797  return Pain;
798  }
799  };
800  auto medium_sunburn = [severe, &you] {
801  if( severe )
802  {
803  you.mod_pain( 1 );
804  return Pain;
805  } else
806  {
807  you.mod_focus( -1 );
808  return Focus_Loss;
809  }
810  };
811  auto light_sunburn = [severe, &you] {
812  if( severe )
813  {
814  you.mod_focus( -1 );
815  return Focus_Loss;
816  } else
817  {
818  return None;
819  }
820  };
821 
822  // Track body parts above the threshold
823  std::map<bodypart_id, Sunburn> affected_bodyparts;
824 
825  for( auto &bp_exp : bp_exposure ) {
826  bodypart_id bp = bp_exp.first;
827  float exposure = bp_exp.second;
828 
829  if( bp == bodypart_id( "eyes" ) ) {
830  // Sunglasses can keep the sun off the eyes.
831  if( you.has_flag( json_flag_GLARE_RESIST )
832  || you.worn_with_flag( flag_SUN_GLASSES )
833  || you.worn_with_flag( flag_BLIND ) ) {
834  continue;
835  }
836  // If no UV-/glare-protection gear is worn the eyes should be treated as unprotected
837  exposure = 1.0;
838  } else if( ( you.get_wielded_item() && you.get_wielded_item()->has_flag( flag_RAIN_PROTECT ) )
839  || ( ( bp == body_part_hand_l || bp == body_part_hand_r )
840  && you.worn_with_flag( flag_POCKETS )
841  && you.can_use_pockets() )
842  || ( bp == body_part_head
843  && you.worn_with_flag( flag_HOOD )
844  && you.can_use_hood() )
845  || ( bp == body_part_mouth
846  && you.worn_with_flag( flag_COLLAR )
847  && you.can_use_collar() ) ) {
848  // Eyes suffer even in the presence of the checks in this branch!
849  // Umbrellas can keep the sun off all bodyparts
850  // Pockets can keep the sun off your hands if you don't wield a too large item
851  // Hoods can keep the sun off your unencumbered head
852  // Collars can keep the sun off your unencumbered mouth
853  continue;
854  }
855 
856  float heavy_cumul_chance = heavy_eff_chance( exposure );
857  float medium_cumul_chance = heavy_cumul_chance + medium_eff_chance( exposure );
858  float light_cumul_chance = medium_cumul_chance + light_eff_chance( exposure );
859  float roll = rng_float( 0.0, 1.0 );
860 
861  Sunburn eff;
862  if( roll < heavy_cumul_chance ) {
863  eff = heavy_sunburn( bp );
864  } else if( roll < medium_cumul_chance ) {
865  eff = medium_sunburn( );
866  } else if( roll < light_cumul_chance ) {
867  eff = light_sunburn( );
868  } else {
869  // Do nothing. Assert that exposure is lower than 0.05 as above that point at least light_eff should always happen
870  if( exposure > 0.05 ) {
871  debugmsg( "No sunburn effect was applied although the bodypart %s is sufficiently exposed at %f exposure",
872  body_part_name( bp ), exposure );
873  };
874  eff = None;
875  }
876  affected_bodyparts.emplace( bp, eff );
877  }
878 
879  std::vector<std::string> affected_part_names;
880  std::unordered_set<bodypart_id> excluded_other_parts;
881 
882  if( affected_bodyparts.empty() ) {
883  return;
884  }
885 
886  Sunburn worst_effect = None;
887  bool contains_eyes = false;
888 
889  for( const std::pair<const bodypart_id, Sunburn> &exp_bp : affected_bodyparts ) {
890  const bodypart_id &bp = exp_bp.first;
891  const Sunburn effect = exp_bp.second;
892  if( excluded_other_parts.count( bp ) || effect < worst_effect ) {
893  continue;

900  }

940  };

945  //~ number of body parts (eyes count as 2 body parts)
946  //~ This message indicates damage to bodyparts through sunshine

948  "Your %s are bathed in sunlight. They feel like they are burning up.",
949  plurality ),
950  m_bad );
951  break;
952  case Pain:
953  //~ %s is a list of body parts. The plurality integer is the total
954  //~ number of body parts (eyes count as 2 body parts)
955  //~ This message indicates pain through sunshine
956  warn_and_wake_up( n_gettext( "The sunlight burns on your %s.",
957  "The sunlight burns on your %s.",
958  plurality ),
959  m_bad );

961  case Focus_Loss:
962  //~ %s is a list of body parts. The plurality integer is the total
963  //~ number of body parts (eyes count as 2 body parts)
964  //~ This message indicates focus loss through sunshine
965  warn_and_wake_up( n_gettext( "The sunlight on your %s irritates you.",
966  "The sunlight on your %s irritates you.",
967  plurality ),
968  m_bad );

970  case None:
971  break;
972  }

1947  const double global_temperature_mod = -1.0 + ( 2.0 * temperature_f / 100.0 );
1948
