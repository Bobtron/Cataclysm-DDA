85 void query_popup_impl::on_resized()
86 {
87  size_t frame_padding = size_t( ImGui::GetStyle().FramePadding.x * 2 );
88  size_t item_padding = size_t( ImGui::GetStyle().ItemSpacing.x );

91 
92  // Fold message text

94 
95  // Fold query buttons
96  const auto &folded_query = fold_query( parent->category, parent->pref_kbd_mode,
97  parent->options, max_line_width,
98  frame_padding + item_padding );
99 
100  // Calculate size of message part
101  msg_width = 0;
102  for( const auto &line : parent->folded_msg ) {
103  msg_width = std::max( msg_width,
104  get_text_width( remove_color_tags( line ) ) ); //utf8_width( line, true ) );
105  }
106  auto btn_padding = [&frame_padding, &item_padding]( size_t num_buttons ) {
107  return ( frame_padding * ( num_buttons - 1 ) ) + ( item_padding * ( num_buttons - 1 ) );
108  };
109  // Calculate width with query buttons
110  for( const auto &line : folded_query ) {
111  if( !line.empty() ) {
112  int button_width = 0;
113  for( const auto &opt : line ) {
114  button_width += get_text_width( remove_color_tags( opt ) );
115  }
116  // extra item padding needed here to account for space left at the beginning of the window by ImGui
117  msg_width = std::max( msg_width, button_width + btn_padding( line.size() ) + item_padding );
118  }
119  }
120  msg_width = std::min( msg_width, max_line_width );
121 
122  // Calculate height with query buttons & button positions
123  parent->buttons.clear();
124  size_t line_idx = 0;
125  if( !folded_query.empty() ) {
126  for( const auto &line : folded_query ) {
127  if( !line.empty() ) {
128  int button_width = 0;
129  for( const auto &opt : line ) {
130  button_width += get_text_width( remove_color_tags( opt ) );
131  }
132  button_width += btn_padding( line.size() );
133  // Right align.
134  // TODO: multi-line buttons
135  size_t button_x = button_width > int( msg_width ) ? size_t( 0 ) :
136  size_t( msg_width - button_width );
137  for( const auto &opt : line ) {
138  parent->buttons.emplace_back( opt, point( button_x, line_idx ) );
139  button_x += get_text_width( remove_color_tags( opt ) ) + frame_padding + item_padding;
140  }

229 std::vector<std::vector<std::string>> query_popup_impl::fold_query(
230  const std::string &category,
231  const keyboard_mode pref_kbd_mode,
232  const std::vector<query_popup::query_option> &options,
233  const int max_width, const int horz_padding )
234 {
235  input_context ctxt( category, pref_kbd_mode );
236 
237  std::vector<std::vector<std::string>> folded_query;
238  folded_query.emplace_back();
239 
240  int query_cnt = 0;
241  int query_width = 0;
242  for( const query_popup::query_option &opt : options ) {
243  const std::string &name = ctxt.get_action_name( opt.action );
244  const std::string &desc = ctxt.get_desc( opt.action, name, opt.filter );
245  const int this_query_width = get_text_width( remove_color_tags( desc ) ) + horz_padding;
246  ++query_cnt;
247  query_width += this_query_width;
248  if( query_width > max_width + horz_padding ) {
249  if( query_cnt == 1 ) {
250  // Each line has at least one query, so keep this query in the current line
251  folded_query.back().emplace_back( desc );
252  folded_query.emplace_back();
253  query_cnt = 0;
254  query_width = 0;
255  } else {
256  // Wrap this query to the next line
257  folded_query.emplace_back();
258  folded_query.back().emplace_back( desc );
259  query_cnt = 1;
260  query_width = this_query_width;
261  }
262  } else {
263  folded_query.back().emplace_back( desc );
264  }
265  }
266 
267  if( folded_query.back().empty() ) {
268  folded_query.pop_back();
269  }
270 
271  return folded_query;
272 }

296  ui_manager::redraw();
297

300  ctxt.register_action( "HELP_KEYBINDINGS" );

303  ctxt.register_action( "CONFIRM" );
304  for( const query_popup::query_option &opt : options ) {
305  ctxt.register_action( opt.action );
306  }
307  // Mouse movement and button
308  ctxt.register_action( "SELECT" );
309  ctxt.register_action( "MOUSE_MOVE" );
310  ctxt.register_action( "LEFT" );
311  ctxt.register_action( "RIGHT" );
312  }
313  if( anykey ) {

315  // Mouse movement, button, and wheel
316  ctxt.register_action( "COORDINATE" );
317  }
318  if( cancel ) {
319  ctxt.register_action( "QUIT" );
320  }
321 #if defined(WIN32) || defined(TILES)
322  ctxt.set_timeout( 50 );
323 #endif
324 
325  result res;
326  // Assign outside construction of `res` to ensure execution order
327  res.wait_input = !anykey;
328  do {
329  ui_manager::redraw();
330  res.action = ctxt.handle_input();
331  res.evt = ctxt.get_raw_input();
332 
333  // If we're tracking mouse movement
334  if( !options.empty() && res.action == "SELECT" && impl->get_mouse_selected_option() != -1 ) {
335  // Left-click to confirm selection
336  res.action = "CONFIRM";
337  cur = size_t( impl->get_mouse_selected_option() );
338  } else if( res.action == "CONFIRM" && impl->keyboard_selected_option != -1 ) {
339  cur = size_t( impl->keyboard_selected_option );
340  }
341  } while(
342  // Always ignore mouse movement
343  ( res.evt.type == input_event_t::mouse &&
344  res.evt.get_first_input() == static_cast<int>( MouseInput::Move ) ) ||
345  // Ignore window losing focus in SDL
346  ( res.evt.type == input_event_t::keyboard_char && res.evt.sequence.empty() ) ||
347  res.evt.type == input_event_t::timeout
348  );
349 
350  if( cancel && res.action == "QUIT" ) {
351  res.wait_input = false;
352  } else if( res.action == "CONFIRM" ) {
353  if( cur < options.size() ) {
354  res.wait_input = false;
355  res.action = options[cur].action;
356  }
357  } else if( res.action == "LEFT" ) {
358  if( impl->keyboard_selected_option > 0 ) {
359  impl->keyboard_selected_option--;
360  } else {
361  impl->keyboard_selected_option = short( buttons.size() - 1 );
362  }
363  } else if( res.action == "RIGHT" ) {
364  if( impl->keyboard_selected_option < short( buttons.size() - 1 ) ) {
365  impl->keyboard_selected_option++;
366  } else {
367  impl->keyboard_selected_option = 0;
368  }
369  } else if( res.action == "HELP_KEYBINDINGS" ) {
370  // Keybindings may have changed, regenerate the UI
371  std::shared_ptr<query_popup_impl> impl = p_impl.lock();
372  if( impl ) {
373  impl->on_resized();
374  }
375  //init();
376  } else {
377  for( size_t ind = 0; ind < options.size(); ++ind ) {
378  if( res.action == options[ind].action ) {
379  impl->keyboard_selected_option = ind;
380  if( options[ind].filter( res.evt ) ) {
381  res.wait_input = false;
382  break;
383  }
384  }
385  }
386  }
