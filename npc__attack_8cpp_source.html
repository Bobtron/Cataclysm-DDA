173 npc_attack_rating npc_attack_spell::evaluate_tripoint(
174  const npc &source, const Creature *target, const tripoint &location ) const
175 {
176  const spell &attack_spell = source.magic->get_spell( attack_spell_id );
177 
178  double total_potential = 0;
179 
180  creature_tracker &creatures = get_creature_tracker();
181  for( const tripoint &potential_target : calculate_spell_effect_area( attack_spell, location,
182  source ) ) {
183  Creature *critter = creatures.creature_at( potential_target );
184 
185  if( !critter ) {
186  // no critter? no damage! however, we assume fields are worth something
187  if( attack_spell_id->field ) {
188  dialogue d( get_talker_for( source ), nullptr );
189  total_potential += static_cast<double>( attack_spell.field_intensity( source ) ) /
190  static_cast<double>( attack_spell_id->field_chance.evaluate( d ) ) / 2.0;
191  }
192  continue;
193  }
194 
195  const Creature::Attitude att = source.attitude_to( *critter );
196  int damage = 0;
197  if( source.sees( *critter ) ) {
198  damage = attack_spell.dps( source, *critter );
199  }
200  const int distance_to_me = rl_dist( source.pos(), potential_target );
201  const bool friendly_fire = att == Creature::Attitude::FRIENDLY &&
202  !source.rules.has_flag( ally_rule::avoid_friendly_fire );
203  int attitude_mult = 3;
204  if( att == Creature::Attitude::FRIENDLY ) {
205  attitude_mult = -10;
206  } else if( att == Creature::Attitude::NEUTRAL || friendly_fire ) {
207  // hitting a neutral creature isn't exactly desired, but it's a lot less than a friendly.
208  // if friendly fire is on, we don't care too much, though if an available hit doesn't damage them it would be better.
209  attitude_mult = -1;
210  }
211  int potential = damage * attitude_mult - distance_to_me + 1;
212  if( target && critter == target ) {
213  potential *= npc_attack_constants::target_modifier;
214  }
215  if( damage >= critter->get_hp() ) {
216  potential *= npc_attack_constants::kill_modifier;
217  if( att == Creature::Attitude::NEUTRAL ) {
218  // we don't care if we kill a neutral creature in one hit
219  potential = std::abs( potential );
220  }
221  const Creature *source_ptr = &source;
222  if( att == Creature::Attitude::FRIENDLY || critter == source_ptr ) {
223  // however we under no circumstances want to kill an ally (or ourselves!)
224  return npc_attack_rating( std::nullopt, location );
225  }
226  }
227  total_potential += potential;
228  }
229  return npc_attack_rating( static_cast<int>( std::round( total_potential ) ), location );
230 }

722 npc_attack_rating npc_attack_throw::evaluate(
723  const npc &source, const Creature *target ) const
724 {

726  if( !can_use( source ) ) {
727  // please don't throw your pants...
728  return effectiveness;
729  }
730  const inventory &available_weapons = source.crafting_inventory( -1 );
731  if( &thrown_item == source.evaluate_best_weapon() &&
732  available_weapons.amount_of( thrown_item.typeId() ) <= 1 &&
733  available_weapons.charges_of( thrown_item.typeId() ) <= 1 ) {
734  // Don't throw if it's the best individual killy-thing we've got
735  return effectiveness;
736  }
737  const int penalty = base_penalty( source );
738  const bool throw_now = thrown_item.has_flag( flag_NPC_THROW_NOW );
739  // TODO: Should this be a field to cache the result?
740  const bool avoids_friendly_fire = source.rules.has_flag( ally_rule::avoid_friendly_fire );
741  creature_tracker &creatures = get_creature_tracker();
742  for( const tripoint &potential : targetable_points( source ) ) {
743 
744  // hot potato! HOT POTATO!
745  // Calculated for all targetable points, not just those with targets
746  if( throw_now ) {
747  // TODO: Take into account distance to allies too
748  const int distance_to_me = rl_dist( potential, source.pos() );
749  int result = npc_attack_constants::base_throw_now + distance_to_me;
750  if( !has_obstruction( source.pos(), potential, avoids_friendly_fire ) ) {
751  // More likely to pick a target tile that isn't obstructed
752  result += 100;
753  }
754  return npc_attack_rating( result, potential );
755  }
756 
757  if( Creature *critter = creatures.creature_at( potential ) ) {
758  if( source.attitude_to( *critter ) != Creature::Attitude::HOSTILE ) {
759  // no point in friendly fire!
760  continue;
761  }
762  npc_attack_rating effectiveness_at_point = evaluate_tripoint( source, target,
763  potential );
764  effectiveness_at_point -= penalty;
765  if( effectiveness_at_point > effectiveness ) {
766  effectiveness = effectiveness_at_point;
767  }
768  }
769  }
770  return effectiveness;
771 }

798 npc_attack_rating npc_attack_throw::evaluate_tripoint(
799  const npc &source, const Creature *target, const tripoint &location ) const
800 {
801  if( has_obstruction( source.pos(), location ) ) {
802  return npc_attack_rating( std::nullopt, location );
803  }
804  item single_item( thrown_item );
805  if( single_item.count_by_charges() ) {
806  single_item.charges = 1;
807  }
808 
809  Creature::Attitude att = Creature::Attitude::NEUTRAL;
810  const Creature *critter = get_creature_tracker().creature_at( location );
811  if( critter ) {
812  att = source.attitude_to( *critter );
813  }
814 
815  if( att != Creature::Attitude::HOSTILE ) {
816  // No point in throwing stuff at neutral/friendly targets
817  return npc_attack_rating( std::nullopt, location );
818  }
819 
820  if( source.rules.has_flag( ally_rule::avoid_friendly_fire ) &&
821  !source.wont_hit_friend( location, thrown_item, true ) ) {
822  // Avoid friendy fire
823  return npc_attack_rating( std::nullopt, location );
824  }
825 
826  const float throw_mult = throw_cost( source, single_item ) * source.speed_rating() / 100.0f;
827  const int damage = source.thrown_item_total_damage_raw( single_item );
828  float dps = damage / throw_mult;
829  const int distance_to_me = rl_dist( location, source.pos() );
830  float suitable_item_mult = -0.15f;
831  if( distance_to_me > 1 ) {
832  if( thrown_item.has_flag( flag_NPC_THROWN ) ) {
833  suitable_item_mult = 0.08f * std::min( distance_to_me - 1, 5 );
834  }
835  } else {
836  suitable_item_mult = -0.35f;
837  }
838 
839  double potential = dps * ( 1.0 + suitable_item_mult );
840  if( potential > 0.0f && critter && damage >= critter->get_hp() ) {
841  potential *= npc_attack_constants::kill_modifier;
842  }
843  if( potential > 0.0f && target && critter && target->pos() == critter->pos() ) {
844  potential *= npc_attack_constants::target_modifier;
845  }
846  return npc_attack_rating( static_cast<int>( std::round( potential ) ), location );
847 }
