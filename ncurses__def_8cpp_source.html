383 input_event input_manager::get_input_event( const keyboard_mode /*preferred_keyboard_mode*/ )
384 {
385  if( test_mode ) {
386  // input should be skipped in caller's code
387  throw std::runtime_error( "input_manager::get_input_event called in test mode" );
388  }
389 
390  int key = ERR;
391  input_event rval;
392  do {
393  previously_pressed_key = 0;
394  // flush any output
395  catacurses::doupdate();
396  key = getch();
397  if( key != ERR ) {
398  int newch;
399  // Clear the buffer of characters that match the one we're going to act on.
400  const int prev_timeout = input_timeout;
401  set_timeout( 0 );
402  do {
403  newch = wgetch( stdscr );
404  } while( newch != ERR && newch == key );
405  set_timeout( prev_timeout );
406  // If we read a different character than the one we're going to act on, re-queue it.
407  if( newch != ERR && newch != key ) {
408  ungetch( newch );
409  }
410  }
411  rval = input_event();
412  if( key == ERR ) {
413  if( input_timeout > 0 ) {
414  rval.type = input_event_t::timeout;
415  } else {
416  rval.type = input_event_t::error;
417  }
418  // ncurses mouse handling
419  } else if( key == KEY_RESIZE ) {
420  catacurses::resizeterm();
421  } else if( key == KEY_MOUSE ) {
422  MEVENT event;
423  if( getmouse( &event ) == OK ) {
424  rval.type = input_event_t::mouse;
425  rval.mouse_pos = point( event.x, event.y );
426  if( event.bstate & BUTTON1_CLICKED || event.bstate & BUTTON1_RELEASED ) {
427  rval.add_input( MouseInput::LeftButtonReleased );
428  } else if( event.bstate & BUTTON1_PRESSED ) {
429  rval.add_input( MouseInput::LeftButtonPressed );
430  } else if( event.bstate & BUTTON3_CLICKED || event.bstate & BUTTON3_RELEASED ) {
431  rval.add_input( MouseInput::RightButtonReleased );
432  // If curses version is prepared for a 5-button mouse, enable mousewheel
433 #if defined(BUTTON5_PRESSED)
434  } else if( event.bstate & BUTTON4_PRESSED ) {
435  rval.add_input( MouseInput::ScrollWheelUp );
436  } else if( event.bstate & BUTTON5_PRESSED ) {
437  rval.add_input( MouseInput::ScrollWheelDown );
438 #endif
439  } else {
440  rval.add_input( MouseInput::Move );
441  if( input_timeout > 0 ) {
442  // Mouse movement seems to clear ncurses timeout
443  set_timeout( input_timeout );
444  }
445  }
446  } else {
447  rval.type = input_event_t::error;
448  }
449  } else {
450  if( key == 127 ) { // == Unicode DELETE
451  previously_pressed_key = KEY_BACKSPACE;
452  return input_event( KEY_BACKSPACE, input_event_t::keyboard_char );
453  }
454  rval.type = input_event_t::keyboard_char;
455  rval.text.append( 1, static_cast<char>( key ) );
456  // Read the UTF-8 sequence (if any)
457  if( key < 127 ) {
458  // Single byte sequence
459  } else if( 194 <= key && key <= 223 ) {
460  rval.text.append( 1, static_cast<char>( getch() ) );
461  } else if( 224 <= key && key <= 239 ) {
462  rval.text.append( 1, static_cast<char>( getch() ) );
463  rval.text.append( 1, static_cast<char>( getch() ) );
464  } else if( 240 <= key && key <= 244 ) {
465  rval.text.append( 1, static_cast<char>( getch() ) );
466  rval.text.append( 1, static_cast<char>( getch() ) );
467  rval.text.append( 1, static_cast<char>( getch() ) );
468  } else {
469  // Other control character, etc. - no text at all, return an event
470  // without the text property
471  previously_pressed_key = key;
472  input_event tmp_event( key, input_event_t::keyboard_char );
473  imclient->process_input( &tmp_event );
474  return tmp_event;
475  }
476  // Now we have loaded an UTF-8 sequence (possibly several bytes)
477  // but we should only return *one* key, so return the code point of it.
478  const uint32_t cp = UTF8_getch( rval.text );
479  if( cp == UNKNOWN_UNICODE ) {
480  // Invalid UTF-8 sequence, this should never happen, what now?
481  // Maybe return any error instead?
482  previously_pressed_key = key;
483  return input_event( key, input_event_t::keyboard_char );
484  }
485  previously_pressed_key = cp;
486  // for compatibility only add the first byte, not the code point
487  // as it would conflict with the special keys defined by ncurses
488  rval.add_input( key );
489  }
490  imclient->process_input( &rval );
491  } while( key == KEY_RESIZE );
492 
493  return rval;
494 }
