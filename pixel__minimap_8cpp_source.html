121 // a texture pool to avoid recreating textures every time player changes their view
122 // at most 142 out of 144 textures can be in use due to regular player movement
123 // (moving from submap corner to new corner) with MAPSIZE = 11
124 // textures are dumped when the player moves more than one submap in one update
125 // (teleporting, z-level change) to prevent running out of the remaining pool
126 class pixel_minimap::shared_texture_pool
127 {
128  public:
129  explicit shared_texture_pool( const std::function<SDL_Texture_Ptr()> &generator ) {
130  const size_t pool_size = ( MAPSIZE + 1 ) * ( MAPSIZE + 1 );
131 
132  texture_pool.reserve( pool_size );
133  inactive_index.reserve( pool_size );
134 
135  for( size_t i = 0; i < pool_size; ++i ) {
136  texture_pool.emplace_back( generator() );
137  inactive_index.push_back( i );
138  }
139  }
140 
141  //reserves a texture from the inactive group and returns tracking info
142  SDL_Texture_Ptr request_tex( size_t &index ) {
143  if( inactive_index.empty() ) {
144  debugmsg( "Ran out of available textures in the pool." );
145  //shouldn't be happening, but minimap will just be default color instead of crashing
146  return nullptr;
147  }
148  index = inactive_index.back();
149  inactive_index.pop_back();
150  return std::move( texture_pool[index] );
151  }
152 
153  //releases the provided texture back into the inactive pool to be used again
154  //called automatically in the submap cache destructor
155  void release_tex( size_t index, SDL_Texture_Ptr &&ptr ) {
156  if( ptr ) {
157  inactive_index.push_back( index );
158  texture_pool[index] = std::move( ptr );
159  }
160  }
161 
162  private:
163  std::vector<SDL_Texture_Ptr> texture_pool;
164  std::vector<size_t> inactive_index;
165 };
