463 itype_id mtype::get_meat_itype() const
464 {

477  if( has_flag( mon_flag_HUMAN ) ) {
478  return itype_human_flesh;
479  } else if( has_flag( mon_flag_AQUATIC ) ) {
480  return itype_fish;
481  } else {
482  return itype_meat;
483  }
484  } else if( made_of( material_iflesh ) ) {
485  //In the future, insects could drop insect flesh rather than plain ol' meat.
486  return itype_meat;
487  } else if( made_of( material_veggy ) ) {
488  return itype_veggy;
489  } else if( made_of( material_bone ) ) {
490  return itype_bone;
491  }
492  }
