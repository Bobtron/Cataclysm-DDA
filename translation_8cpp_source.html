294 std::string translation::translated( const int num ) const
295 {
296  if( !needs_translation || raw.empty() ) {
297  return raw;
298  }
299  // Note1: `raw`, `raw_pl` and `ctxt` are effectively immutable for caching purposes:
300  // in the places where they are changed, cache is explicitly invalidated
301  // Note2: if `raw_pl` is defined, `num` becomes part of the "cache key"
302  // otherwise `num` is ignored (for both translation and cache)
303  if( cached_language_version != detail::get_current_language_version() ||
304  ( raw_pl && cached_num != num ) || !cached_translation ) {
305  cached_language_version = detail::get_current_language_version();
306  cached_num = num;
307 
308  if( !ctxt ) {
309  if( !raw_pl ) {
310  cached_translation = cata::make_value<std::string>( detail::_translate_internal( raw ) );
311  } else {
312  cached_translation = cata::make_value<std::string>(
313  n_gettext( raw.c_str(), raw_pl->c_str(), num ) );
314  }
315  } else {
316  if( !raw_pl ) {
317  cached_translation = cata::make_value<std::string>( pgettext( ctxt->c_str(), raw.c_str() ) );
318  } else {
319  cached_translation = cata::make_value<std::string>(
320  npgettext( ctxt->c_str(), raw.c_str(), raw_pl->c_str(), num ) );
321  }
322  }
323  }
324  return *cached_translation;
