329 void advanced_inventory_pane::scroll_page( int linesPerPage, int offset )
330 {
331  // only those two offsets are allowed
332  cata_assert( offset == -1 || offset == +1 );
333  if( items.empty() ) {
334  return;
335  }
336  const int size = static_cast<int>( items.size() );
337 
338  advanced_inventory_pagination old_pagination( linesPerPage, *this );
339  for( int i = 0; i <= index; i++ ) {
340  old_pagination.step( i );
341  }
342 
343  // underflow
344  if( old_pagination.page + offset < 0 ) {
345  if( index > 0 ) {
346  // scroll to top of first page
347  index = 0;
348  } else {
349  // scroll wrap
350  index = size - 1;
351  }
352  return;
353  }
354 
355  int previous_line = -1; // matching line one up from our line
356  advanced_inventory_pagination new_pagination( linesPerPage, *this );
357  for( int i = 0; i < size; i++ ) {
358  new_pagination.step( i );
359  // right page
360  if( new_pagination.page == old_pagination.page + offset ) {
361  // right line
362  if( new_pagination.line == old_pagination.line ) {
363  index = i;
364  return;
365  }
366  // one up from right line
367  if( new_pagination.line == old_pagination.line - 1 ) {
368  previous_line = i;
369  }
370  }
371  }
372  // second-best matching line
373  if( previous_line != -1 ) {
374  index = previous_line;
375  return;
376  }
377 
378  // overflow
379  if( index < size - 1 ) {
380  // scroll to end of last page
381  index = size - 1;
382  } else {
383  // scroll wrap
384  index = 0;
385  }
386 }
